285  uint32_t *tranID,
286  const boost::system::error_code &ec )
287 {
288  PUSH_FUNC;
289 
290  if ( ec.value() != 0 )
291  {
292  SVR_INFO( 4, "Aborted timeout: [%s], idle map size:%zu", ec.message().c_str(), idle_msgs.size() );
293  return;
294  }
295 
296  // Check whether the deadline has passed. We compare the deadline against
297  // the current time since a new asynchronous operation may have moved the
298  // deadline before this actor had a chance to run.
299  if ( idle_msgs.count( *tranID ) > 0 )
300  {
301  if ( timeout->expires_at() <= boost::asio::deadline_timer::traits_type::now() )
302  {
303  SVR_DEBUG( 4, "expired timeout id:%d msg_count:%zu map_size:%zu",
304  *tranID, idle_msgs.count( *tranID ), idle_msgs.size() );
305 
306  CMercuryCdsHandle::msg_map_t::iterator it =
307  idle_msgs.find( *tranID );
308 
309  try
310  {
311  it->second->err( px::proto::TIME_OUT,
312  "Request Timeout exceeded." );
313  }
314  catch ( msg_err &e )
315  {
316  it->second->respond();
317  }
318  idle_msgs.erase( *tranID );
319  timeout->cancel();
320  }
321  else
322  {
323  // Put the actor back to sleep.
324  timeout->async_wait( boost::bind( &CMercuryCdsHandle::check_timeout,
325  this, timeout, tranID,
326  boost::asio::placeholders::error ) );
327  }
328  }
