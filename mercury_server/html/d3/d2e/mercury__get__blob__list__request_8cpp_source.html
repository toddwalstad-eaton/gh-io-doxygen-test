150 /* Function to filter one recursively iterated entry,
151  * and add it to the response if it passes the filtering.
152  * Filtering two ways:
153  * 1) Any hidden files or folders (beginning with '.' dot) will be excluded from the response
154  * 2) Only files or folders passing the real expression pattern match will be included.
155  *
156  * The reason to break this into its own function is because when the try-catch catches
157  * an exception, it will unwind the stack, and so this couldn't be done inside the for loop below.
158  */
159 static void filter_one_path( const fs::path &p, const fs::file_status &st, const px::string &real_folder,
160  const std::string &pattern, bool folders_only,
161  const std::string &prepend_path, px::proto::GetBlobListResponse *response )
162 {
163  PUSH_FUNC;
164 
165  try
166  {
167  auto filename_path = p.filename();
168  auto filename = filename_path.c_str();
169  auto pathname = p.c_str();
170 
171  if ( filename[0] == '.' )
172  {
173  // This is a hidden file/folder; exclude it
174  SVR_DEBUG( 8, "Skipping %s - hidden folder/file type", pathname );
175  }
176  else if ( fnmatch( pattern.c_str(), filename, FNM_NOESCAPE | FNM_PATHNAME ) == 0 )
177  {
178  if ( !( ( !folders_only && fs::is_regular_file( st ) ) ||
179  ( folders_only && fs::is_directory( st ) ) ) )
180  {
181  SVR_DEBUG( 8, "Skipping %s - wrong folder/file type", pathname );
182  return;
183  }
184  // Need some minimal level of OS access, or we might as well not
185  // tell the caller about this one.
186  if ( ( folders_only && ( access( pathname, R_OK | X_OK ) != 0 ) ) ||
187  ( ( !folders_only ) && ( access( pathname, R_OK ) != 0 ) ) )
188  {
189  SVR_DEBUG( 8, "Skipping %s - inaccessible to this process", pathname );
190  }
191  std::string resp_uri( prepend_path );
192  // These filenames are just the last part of the path (folder or file)
193  // We need to provide the folder path(s) we have recursed into
194  std::string abs_fname( p.parent_path().c_str() );
195  abs_fname += "/";
196  abs_fname += filename;
197  SVR_DEBUG( 8, " abs_fname %s", abs_fname.c_str() );
198  // Remove the real_folder portion, plus the slash char, and add to any prepended part
199  resp_uri += abs_fname.substr( real_folder.length() );
200  response->add_uri( resp_uri );
201  SVR_DEBUG( 8, "Added URI %s", resp_uri.c_str() );
202  }
203  else
204  {
205  SVR_DEBUG( 8, "Skipping %s - did not match pattern", pathname );
206  }
207  }
208  catch ( const fs::filesystem_error &e )
209  {
210  // In the recursive version of GetBlobList, we can be more forgiving of
211  // access problems (like authorization rights).
212  // Just log a message and continue.
213  SVR_DEBUG( 5, "Boost FS Error for recursive GetBlobListRequest: %s \n Skipping...", e.what() );
214  }

220  bool recurse, bool folders_only, const std::string &prepend_path,
221  px::proto::GetBlobListResponse *response )
222 {
223  PUSH_FUNC;
224 
225  try
226  {
227  fs::path base_folder( real_folder.c_str() );
228 
229  SVR_DEBUG( 6, "Filtering BinaryBlob base path %s %s", real_folder.c_str(), recurse ? "recursively" : "" );
230 
231  if ( recurse )
232  {
233  // Get sub-directories under base_folder
234  px::CRecursiveFind dir_set( base_folder.string() );
235  std::set<px::CRecursiveFind::file_info_t> dirs;
236  dir_set.get_dirs( dirs );

240  fs::path path( filepath );

242  fs::file_status sym = fs::symlink_status( path, ec );

244  {

246  }
247  }
248

250  px::CRecursiveFind file_set( base_folder.string() );
251  std::set<px::CRecursiveFind::file_info_t> files;
252  file_set.get_regular_files( files, pattern );
253  for ( auto file_it : files )
254  {
255  std::string filepath = std::get<px::CRecursiveFind::GET_FILEPATH>( file_it );
256  fs::path path( filepath );
257  boost::system::error_code ec;
258  fs::file_status sym = fs::symlink_status( path, ec );
259  if ( ec == boost::system::errc::success )
260  {
261  filter_one_path( path, sym, real_folder, pattern, folders_only, prepend_path, response );
262  }
263  }
264  }
265  else
266  {
267  // Non-recursive
268  fs::directory_iterator end_it;
269  boost::system::error_code ec;
270  for ( fs::directory_iterator it( base_folder, ec ); it != end_it; it.increment( ec ) )
271  {
272  if ( ec == boost::system::errc::success )
273  {
274  filter_one_path( it->path(), it->symlink_status(), real_folder, pattern, folders_only, prepend_path, response );
275  }
276  else if ( ( ec == boost::system::errc::no_such_file_or_directory ) ||
277  ( ec == boost::system::errc::permission_denied ) )
278  {
279  // ignore - these can happen and that's ok
280  }
281  else
282  {
283  SVR_DEBUG( 5, "Boost FS Iter Error for GetBlobListRequest: %s", ec.message().c_str() );
284  }
285  ec.clear();
286  }
287  }
288  }
289  catch ( const fs::filesystem_error &e )
290  {
291  // In GetBlobList, we can be more forgiving of access problems (like authorization rights).
292  // Log a message and exit with as much response as we've built up before the filesystem error.
293  SVR_DEBUG( 5, "Boost FS Error for GetBlobListRequest: %s", e.what() );
294  }
