238 void CHandleApiIds::get_apis_from_permissions( const px::CBitset<px::permissions> &permissions,
239  std::vector<px::mercury_msg_api_t> &api_ids )
240 {
241  PUSH_FUNC;
242 
243  // Clear the vector and then start adding APIs
244  api_ids.clear();
245  // Here the "id" is both index into __api_perm_map and the API ID that we will report
246  for ( size_t id = 1; id < px::proto::ApiId_ARRAYSIZE; id++ )
247  {
248  if ( CHandleApiIds::check_api_for_permissions( permissions, ( px::mercury_msg_api_t ) id ) )
249  {
250  // Then we found an authorized API ID; return it
251  api_ids.push_back( ( px::mercury_msg_api_t ) id );
252  }
253  }
254 
255  SVR_DEBUG( 5, "Found %zu supported Mercury Message APIs.", api_ids.size() );
256 }

258 /* Checks if the given API is permitted (possibly) for the given Permissions.
259  */
260 bool CHandleApiIds::check_api_for_permissions( const px::CBitset<px::permissions> &permissions, px::mercury_msg_api_t api_id )
261 {
262  PUSH_FUNC;
263 
264  for ( size_t j = 0; j < NUM_PERMS; j++ )
265  {
266  px::permissions test_perm = __api_perm_map[api_id][j];
267  if ( test_perm == _zed )
268  {
269  continue;
270  }
271  if ( permissions.get( test_perm ) )
272  {
273  // Then we found an authorized API ID
274  return ( true );
275  }
276  // Second chance: if the permission is "per_user", then everyone is a winner
277  // (The CDS can sort this out later)
278  if ( test_perm == px::permissions::per_user )
279  {
280  // We let each user have this one
281  return ( true );
282  }
283  }
284 
285  return ( false );
286 }

698 bool CHandleApiIds::get_api_id( std::shared_ptr<px::proto::Request> request, px::mercury_msg_api_t &api_id )
699 {
700  PUSH_FUNC;
701 
702  bool bOK = false; // Until we get a match
703 
704  /* Walk through our table until we find the right "has_xxx" function,
705  * which tells us which API ID this Request message has.
706  */
707  for ( size_t idx = 0; idx < px::proto::ApiId_ARRAYSIZE; idx++ )
708  {
709  px::proto::Request req = *request;
710  Has_xxx func_ptr = api_map[idx].func;
711  if ( func_ptr == nullptr )
712  {
713  continue; // Unused or unimplemented entry
714  }
715  if ( ( req.*func_ptr )() )
716  {
717  // Then this is a hit! Return this entry's ID as the API ID.
718  api_id = api_map[idx].api_id;
719  bOK = true;
720  SVR_DEBUG( 6, "Found Message type: %s in the Request", api_map[idx].name );
721  break;
722  }
723  }
724  if ( !bOK )
725  {
726  SVR_WARN( 2, "Did not find a supported message type in the Request" );
727  }
728 
729  return ( bOK );
730 }

732 // Determine if the message in this request, given by its API ID, is actually present.
733 bool CHandleApiIds::validate_api_id( std::shared_ptr<px::proto::Request> request, px::mercury_msg_api_t api_id )
734 {
735  PUSH_FUNC;
736 
737  bool bOK = false; // Until we get a match
738  // First check the api_id for range
739  uint8_t idx = ( uint8_t ) api_id;
740  if ( idx == 0 )
741  {
742  // Less serious warning; the MTT does this all the time.
743  SVR_NOTICE( 5, "Received no API ID (=0) in the header." );
744  }
745  if ( idx >= px::proto::ApiId_ARRAYSIZE )
746  {
747  SVR_WARN( 2, "Received an API ID (%u) that is invalid: out of range.", api_id );
748  }
749  else
750  {
751  px::proto::Request req = *request;
752  Has_xxx func_ptr = api_map[idx].func;
753  if ( func_ptr == nullptr )
754  {
755  SVR_WARN( 2, "Received an API ID (%u = %s) for an unimplemented message type.", api_id, api_map[idx].name );
756  }
757  else if ( ( req.*func_ptr )() )
758  {
759  // Then this is correct, and as expected.
760  SVR_DEBUG( 6, "Found valid %s Message in the Request", api_map[idx].name );
761  bOK = true;
762  }
763  else
764  {
765  SVR_WARN( 2, "Received an API ID (%u) that is invalid: no such message found in the Request.", api_id );
766  }
767  }
768 
769  return ( bOK );
770 }
