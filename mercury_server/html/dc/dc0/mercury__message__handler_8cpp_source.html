111 // Post an error response, based on the CDS response status, with the optional error message.
112 void CMercuryMessageHandler::err_status( px::cds::RequestStatus resp_status, const px::string &msg )
113 {
114  PUSH_FUNC;
115 
116  px::proto::ErrorCode error = px::proto::ERROR_NONE;
117  switch ( resp_status )
118  {
119  case px::cds::RequestStatus::FAILED:
120  // Finished, bad -> generic error (we don't know)
121  error = px::proto::OTHER_ERROR;
122  break;
123 
124  case px::cds::RequestStatus::QUEUE_FULL:
125  // Queue too full to fit response. Might work later. -> Busy
126  error = px::proto::TOO_BUSY;
127  break;
128 
129  case px::cds::RequestStatus::RESPONSE_TOO_BIG:
130  // Response is itself too large to fit in an empty
131  // queue, it will never succeed. -> Too much data
132  error = px::proto::TOO_MUCH_DATA;
133  break;
134 
135  case px::cds::RequestStatus::TIMEOUT:
136  // Expiration time has passed -> Timeout
137  error = px::proto::TIME_OUT;
138  break;
139 
140  case px::cds::RequestStatus::PERMISSION_DENIED:
141  // Client session or user not allowed to do this -> Not Authorized
142  error = px::proto::NOT_AUTHORIZED;
143  break;
144 
145  case px::cds::RequestStatus::BAD_PARAMETER:
146  // A parameter was invalid, missing, or just bad.
147  error = px::proto::INVALID_PARAMETER;
148  break;
149 
150  case px::cds::RequestStatus::EMPTY_RESULT:
151  // Unexpectedly empty results - should have had something here
152  error = px::proto::EMPTY_RESULT;
153  break;
154 
155  case px::cds::RequestStatus::NOT_ABLE_TO_EXECUTE:
156  // Could not execute; normally because too busy,
157  // but could also be NOT_SUPPORTED
158  error = px::proto::TOO_BUSY;
159  break;
160 
161  default:
162  // Leave set to px::proto::ERROR_NONE
163  break;
164  }
165  // Now call the normal err() handler
166  err( error, msg );
167 }
