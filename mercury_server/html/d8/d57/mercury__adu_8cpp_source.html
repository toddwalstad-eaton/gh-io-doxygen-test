104

111  else

115  assert( retval.second <= end );

122  {

124  retval.first = boost::logic::indeterminate;
125  // Reset the start pointer, too
126  retval.second = start;
127  return ( retval );
128  }
129

145  }

201 void CMercuryAdu::serialize( px::asio::io_buf_t &buf )
202 {
203  PUSH_FUNC;
204 
205  std::vector<uint8_t> msg;
206  uint32_t msg_size;
207 
208  SVR_FDEBUG( 8, "Serializing ADU" );
209 
210  if ( hg_header._message_type == px::mercury_msg_format_t::HG_MSG_PROTOBUF )
211  {
212  msg_size = pdu.msg_out->ByteSize();
213  msg.resize( msg_size );
214  bool success = pdu.msg_out->SerializeWithCachedSizesToArray( msg.data() );
215  if ( success )
216  {
217  SVR_DEBUG( 8, "Success Serializing Protobuf to PDU msg (payload size = %d)",
218  msg_size );
219  }
220  else
221  {
222  // This would be bad, but does it ever happen??
223  SVR_ERR( 1, "Failed to serialize protobuf message." );
224  }
225  }
226  else
227  {
228  px::string json_out = pdu.serialize();
229  msg_size = json_out.length();
230  msg.insert( msg.end(), json_out.begin(), json_out.end() );
231  SVR_DEBUG( 8, "Success Serializing JSON msg (payload size = %d)", msg_size );
232  json_out.clear();
233  }
234  if ( has_hg_header )
235  {
236  hg_header._message_size = msg_size;
237  // This adjusts the serializing per version
238  hg_header.serialize( buf );
239  // Now append the message bytes
240  buf.insert( buf.end(), msg.begin(), msg.end() );
241  }
242  else
243  {
244  // Swap is faster than copying; no header involved
245  buf.swap( msg );
246  }
247 }
