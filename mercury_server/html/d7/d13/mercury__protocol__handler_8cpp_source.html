122 bool CMercuryProtocolHandler::parse_request()
123 {
124  PUSH_FUNC;
125 
126  assert( _adu );
127  assert( _publish_adu );
128  assert( _transactor != nullptr );
129  px::asio::CIoTransactor &transactor = *_transactor;
130 
131  px::asio::iobuf_const_pair_t rbuf( transactor.get_rbuf() );
132 
133  SVR_DEBUG( 4, "Parsing incoming request" );
134 
135  // Parse the input buffer
136  px::asio::parse_result_t result( _adu->parse( rbuf.first, rbuf.second ) );
137  assert( result.second <= rbuf.second );
138 
139  if ( boost::logic::indeterminate( result.first ) )
140  {
141  // Buffer is incomplete
142  SVR_DEBUG( 7, "Buffer incomplete..." );
143  chunk_size = _adu->request_size;
144  return ( false );
145  }
146 
147  // The input is no longer necessary; discard [this portion of] the
148  // read buffer
149  transactor.clear_rbuf( result.second );
150 
151  if ( !result.first )
152  {
153  SVR_DEBUG( 3, "Error encountered while parsing ADU" );
154  // since the parsing failed, we do not have the request_id.
155  // we do not try to read the request_id from the incorrect msg, because
156  // it can be totally wrong.
157  // in this case, the mercury 1.4/1.5 spec says that we should return zero
158  // as the request id.
159  // todo: set the ErrorCode to INVALID_MSG.
160  err( px::proto::ErrorCode::OTHER_ERROR,
161  0,
162  "Failed to parse the request." );
163 
164  return ( true );
165  }
166  // With each successful parse, update the parameters of the publish ADU
167  // with the most results of the most recent parse
168  _publish_adu->has_hg_header = _adu->has_hg_header;
169  _publish_adu->hg_header = _adu->hg_header;
170 
171  // reset chunk size for next request
172  chunk_size = 128;
173 
174  // Execute the just-parsed request, using the transactor's write buffer
175  // as a space to construct our response
176  execute();
177 
178  return ( true );
179 }

185 {
186  PUSH_FUNC;
187 
188  const char *str = "unknown";

190  // quit signal.
191  if ( _transactor->get_quit_status() )
192  {
193  SVR_INFO( 3, "Exiting the Handle requet function as received quit signal!!! Doing nothing" );
194  exit( EXIT_SUCCESS );
195  }
196 
197  //given mercury protobuf message. create a message handler to perform
198  //request.
199  if ( pdu.msg->requests_size() > 0 )
200  {
201  if ( !_adu->username.empty() )
202  {
203  SVR_DEBUG( 6, "Process %d requests from %s", pdu.msg->requests_size(), _adu->username.c_str() );
204  }
205  else
206  {
207  SVR_DEBUG( 6, "Process %d requests from %s", pdu.msg->requests_size(), session_user().c_str() );
208  }
209 
210  uint32_t first_req_id = 0;
211  bool first_req_id_set = false;
212 
213  for ( int idx = 0; idx < pdu.msg->requests_size(); idx++ )
214  {
215  // If missed in above check for quit signal processing,
216  // checking in this for loop logic, again.
217  if ( _transactor->get_quit_status() )
218  {
219  SVR_INFO( 3, "Exiting the Handle requet function as received quit signal!!! Doing nothing" );
220  exit( EXIT_SUCCESS );
221  }
222 
223  SVR_INFO( 6, "Handle request[%d]", idx );
224  auto request = std::make_shared<px::proto::Request>( pdu.msg->requests( idx ) );
225 
226  if ( unlikely( !request->has_request_id() ) )
227  {
228  err( px::proto::INVALID_ID, 0, "Missing request id." );
229  continue;
230  }
231 
232  if ( pdu.msg->has_bundled() )
233  {
234  // If we say we are bundling but it is not supported
235  if ( pdu.msg->bundled() && !pdu.unbundling_support )
236  {
237  err( px::proto::INVALID_PARAMETER, request->request_id(),
238  "Protocol transport does not support unbundling." );
239  continue;
240  }
241  if ( pdu.msg->bundled() )
242  {
243  pdu.bundled_request_ids.emplace( request->request_id() );
244  }
245  }
246 
247  if ( !first_req_id_set )
248  {
249  first_req_id_set = true;
250  first_req_id = request->request_id();
251  add_new_adu( first_req_id );
252  }
253  _adu_req_ids.emplace( std::make_pair( request->request_id(), first_req_id ) );
254 
255  // Determine the Mercury Request API ID for this Request
256  px::mercury_msg_api_t api_id = px::mercury_msg_api_t::API_UNDEFINED;
257  // First, see if the header provided it
258  if ( ( idx == 0 ) && _adu->has_hg_header )
259  {
260  if ( CHandleApiIds::validate_api_id( request, _adu->hg_header._message_api ) )
261  {
262  // The provided API ID is good.
263  api_id = _adu->hg_header._message_api;
264  }
265  }
266  if ( api_id == px::mercury_msg_api_t::API_UNDEFINED )
267  {
268  // Second chance: lookup the message type in the protobuf request
269  if ( !CHandleApiIds::get_api_id( request, api_id ) )
270  {
271  // Failed; this message can't be found or isn't supported
272  err( px::proto::NOT_SUPPORTED, request->request_id(), "Unsupported message type" );
273  // Loop back; maybe there are more requests
274  continue;
275  }
276  }
277 
278  /* Call track_session() each time through here, as long as we got a valid message
279  * If this is a login request, we are provisionally creating a session entry,
280  * and will fill in the session ID and user_perms with a good response in CLogin::post()
281  * If a later request, track_session will ensure that we get the session ID and permissions
282  * Would like to see that some authorization happened before calling this, but don't see a
283  * good way to get to that, so proceeding like this: */
284  // Will determine username by other means, inside track_session
285  track_session();
286 
287  // If not yet logged in, and this message is not a login request,
288  // return not authenticated
289  if ( _perms.empty() && ( api_id != px::mercury_msg_api_t::LOGIN ) )
290  {
291  err( px::proto::NOT_AUTHENTICATED, request->request_id(), "This connection has not yet logged in" );
292  continue;
293  }
294 
295  // Now the preliminary security check: does this session permit this Request to be handled?
296  // (The CDS may make a further determination, but this is the first-level check.)
297  if ( !CHandleApiIds::check_api_for_permissions( _perms, api_id ) )
298  {
299  px::string err_msg( "Message: " );
300  err_msg += CHandleApiIds::get_msg_name( api_id );
301  err_msg += " is Not Authorized for user: ";
302  err_msg += session_user();
303  // ToDo: Audit log this message, since the CDS won't see it to log for us
304  err( px::proto::NOT_AUTHORIZED, request->request_id(), err_msg );
305  continue;
306  }
307  std::shared_ptr<CMercuryMessageHandler> msg_hndlr;
308 
309  try
310  {
311  // SVR_FDEBUG(5, "User here is %s@%s", session_user().c_str(), session_source().c_str() );
312  msg_hndlr = CHandleApiIds::make_msg_handler( shared_from_this(), request, api_id );
313  }
314  catch ( const std::bad_weak_ptr &e )
315  {
316  if ( e.what() != NULL )
317  {
318  SVR_INFO( 0, "Invalid Message Handler error %s", e.what() );
319  }
320  else
321  {
322  SVR_INFO( 0, "Invalid Message Handler error %s", str );
323  }
324  }
325 
326  // Have to make sure we got a valid Message Handler first
327  if ( !msg_hndlr.get() )
328  {
329  err( px::proto::NOT_SUPPORTED, request->request_id(), "Request cannot be handled." );
330  continue;
331  }
332  //Begin execution of the message handler
333  try
334  {
335  msg_hndlr->execute();
336  if ( cds_msg_posted )
337  {
338  auto &cds = get_cds_handle();
339  cds.add_idle_msg( msg_hndlr );
340  cds_msg_posted = false;
341  }
342  }
343  catch ( const responded &e )
344  {
345  // We already responded; nothing further to do here.
346  }
347  catch ( const msg_err &e )
348  {
349  SVR_DEBUG( 2, "Request failed to execute. %s",
350  e.what() != nullptr ? e.what() : "unknown error" );
351  msg_hndlr->respond();
352  }
353  catch ( const msg_warn &e )
354  {
355  SVR_DEBUG( 2, "Request failed to execute. %s",
356  e.what() != nullptr ? e.what() : "unknown warning" );
357  msg_hndlr->respond();
358  }
359  catch ( const conn_err &e )
360  {
361  SVR_DEBUG( 2, "Request failed to submit to CDS caused by connection error. %s",
362  e.what() != nullptr ? e.what() : "unknown error" );
363  err( px::proto::OTHER_ERROR, request->request_id(),
364  "Failed to submit to CDS, which might be down." );
365  }
366  catch ( const cds_err &e )
367  {
368  SVR_DEBUG( 2, "Request failed to submit to CDS. %s",
369  e.what() != nullptr ? e.what() : "unknown error" );
370  err( px::proto::TOO_BUSY, request->request_id(), "Failed to submit to CDS" );
371  }
372  catch ( ... )
373  {
374  SVR_DEBUG( 2, "Unknown message handler execution error" );
375  err( px::proto::OTHER_ERROR, request->request_id(), "Unknown message handler execution error" );
376  }
377  } // end of requests loop
378  }
379  else
380  {
381  err( px::proto::OTHER_ERROR, 0, "Empty request." );
382  }

676 void CMercuryProtocolHandler::fill_out_role( const px::string &role_name, const px::CBitset<px::permissions> &user_perms,
677  px::proto::Role *proto_role )
678 {
679  PUSH_FUNC;
680 
681  proto_role->set_name( role_name.c_str() );
682 
683  // Currently enums are the same between px::proto::Permission and
684  // px::permissions.
685  for ( uint32_t i = ( uint32_t )px::proto::Permission::DATA_READ;
686  i <= ( uint32_t )px::proto::Permission::PER_USER; ++i )
687  {
688  if ( user_perms.get( ( px::permissions ) i ) )
689  {
690  proto_role->add_perms( ( px::proto::Permission ) i );
691  }
692  }
693 
694  // One advantage of instantiating this is for the assert checking in the Ctor.
695  // Otherwise, get_apis_from_permissions() could be a static function.
696  CHandleApiIds api_handler;
697  std::vector<px::mercury_msg_api_t> api_ids;
698  api_handler.get_apis_from_permissions( user_perms, api_ids );
699  for ( const px::mercury_msg_api_t &hg_api_id : api_ids )
700  {
701  // Similarly, the API IDs map one-to-one from px to protobufs
702  proto_role->add_apis( ( px::proto::ApiId ) hg_api_id );
703  }
704 }
