168 session_map_itr_t CSessionManager::close_session( session_map_itr_t sess_itr )
169 {
170  PUSH_FUNC;
171 
172  if ( sess_itr != _session_map.end() )
173  {
174  // "first" is the Session ID
175  px::string session_id = sess_itr->first;
176  auto &sess = sess_itr->second;
177  px::string protocol = sess.protocol();
178  px::string source = sess.source();
179  px::string who = sess.who();
180  // Setup the Info for Audit Logging
181  toolkit::CCeeLogMsg cee_info( CEE_ACT_close, CEE_STS_success,
182  who, source, APP_NAME );
183  AUDIT_LOG_MSG( AUDIT_LOG_SESSION, LOG_INFO, cee_info, "%s User %s@%s Logged Off",
184  protocol.c_str(), who.c_str(), source.c_str() );
185 
186  // Remove this session from the CDS
187  px::cds::logout_session( px::cds::CClient::get(), session_id );
188  // And from our tracker set
189  sess_itr = _session_map.erase( sess_itr );
190 
191  // If the protocol is WebSocket, check for any WebSocket users left
192  // before turning off the Web LED.
193  if ( protocol == _web_led_protocol )
194  {
195  session_map_itr_t itr;
196  unsigned int web_users = 0;
197  for ( itr = _session_map.begin(); itr != _session_map.end(); ++itr )
198  {
199  if ( itr->second.protocol() == _web_led_protocol )
200  {
201  ++web_users;
202  break; // only care about if this is zero or not
203  }
204  }
205  if ( web_users > 0 )
206  {
207  set_led( true );
208  }
209  else
210  {
211  set_led( false );
212  }
213  }
214 
215  // Remove the source to session ID mapping
216  _session_id_map.erase( source );
217 
218  SVR_INFO( 3, "Removed %s session for %s (tok=%s)", protocol.c_str(), source.c_str(), session_id.c_str() );
219  SVR_INFO( 6, "Remaining map counts - Sessions: %zu, IDs: %zu", _session_map.size(), _session_id_map.size() );
220  }
221  return ( sess_itr );
222 }
