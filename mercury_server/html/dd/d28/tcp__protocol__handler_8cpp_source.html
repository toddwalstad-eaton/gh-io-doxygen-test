78  }
79  // Parse - AND execute - the request:
80  bool done = parse_request();
81  if ( done )
82  {
83  // Get the Mercury Header, and, if no error, get its version number

85  if ( _adu->has_hg_header && _adu->hg_header._error_message.empty() )
86  {
87  // This is a weak indication of parsing success, but we'll take it
88  _header_version = _adu->hg_header._header_version;
89  SVR_DEBUG( 8, "Setting header version to %u", _header_version );
90  }
91  }
92  return ( done );

115 {
116  PUSH_FUNC;
117 
118  assert( _transactor != nullptr );
119 
120  if ( _closed )
121  {
122  SVR_DEBUG( 4, "Dropping response due to closed handler" );
123  return;
124  }
125  if ( unlikely( !_transactor ) )
126  {
127  SVR_FERR( 0, "Missing ADU or transactor, dropping response" );
128  return;
129  }
130  bool del_from_adus = true;
131  uint32_t first_req_id = 0;
132  CTcpAdu *tcpadu = nullptr;
133  try
134  {
135  first_req_id = _adu_req_ids.at( resp.request_id() );
136  _adu_req_ids.erase( resp.request_id() );
137  tcpadu = px::poly_cast<CTcpAdu *>( _adus.at( first_req_id ).get() );
138  }
139  catch ( const std::out_of_range &oor )
140  {
141  // No request found - may have been a message with more
142  // than 1 request but bundled = false.
143  tcpadu = new CTcpAdu;
144  tcpadu->has_hg_header = _publish_adu->has_hg_header;
145  tcpadu->hg_header = _publish_adu->hg_header;
146  tcpadu->pdu.msg->add_requests();
147  tcpadu->pdu.msg->set_bundled( false );
148  del_from_adus = false;
149  }
150 
151  px::proto::Response *response = tcpadu->pdu.msg_out->add_responses();
152  auto &req_response = *( response->mutable_request_response() );
153  req_response.CopyFrom( resp );
154  SVR_DEBUG( 4, "Respond: %s", req_response.DebugString().c_str() );
155  auto request_size = tcpadu->pdu.msg->requests_size();
156  auto response_size = tcpadu->pdu.msg_out->responses_size();
157 
158  bool bundled = false;
159  if ( tcpadu->pdu.msg->has_bundled() )
160  {
161  bundled = tcpadu->pdu.msg->bundled();
162  if ( bundled )
163  {
164  if ( resp.has_request_id() )
165  {
166  tcpadu->pdu.bundled_request_ids.erase( resp.request_id() );
167  }
168  else
169  {
170  // No request ID, just bail and send what we have
171  tcpadu->pdu.bundled_request_ids.clear();
172  }
173  }
174  }
175 
176  // we need to handle both req/res and unsolicited push (req is empty)
177  if ( !bundled ||
178  ( ( request_size <= response_size ) && ( response_size != 0 ) && ( tcpadu->pdu.bundled_request_ids.size() == 0 ) ) )
179  {
180  std::shared_ptr<px::asio::io_buf_t> pbuf( new px::asio::io_buf_t() );
181 
182  try
183  {
184  tcpadu->CMercuryAdu::serialize( *pbuf );
185  }
186  catch ( ... )
187  {
188  SVR_ERR( 1, "TCP Serialize fail" );
189  //TODO: output exact error, push error to connection
190  }
191

193  pbuf->size() );

196 
197  _transactor->write( pbuf );
198

201  {

203  }
204  else

206  delete tcpadu;
207  }
208  }
209  else

212  request_size, response_size );
