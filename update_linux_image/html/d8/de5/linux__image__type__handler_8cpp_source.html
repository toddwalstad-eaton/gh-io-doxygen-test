119 bool LinuxImageTypeHandler::do_update_steps( px::id_t /* dev_id */,
120  const CFwUtil::image_update_params_t & /* update_params */ )
121 {
122  PUSH_FUNC;
123 
124  // Get target names and new partion.
125  _fw_util._new_partition = _fw_util.get_target_names( _fw_util._current_partition );
126  SYSLOG_INFO( 2, " _fw_util.get_target_names: current partition %u, new partition %u", _fw_util._current_partition, _fw_util._new_partition );
127 
128  // Iterate through update process. Update OOB after each step. Bail on failure.
129  bool reboot = false;
130  bool success = true; // can only be invalidated.
131  size_t part_trigger = 0;
132  int rc;
133  std::stringstream result;
134 
135  // Loop through the update steps vector
136  for ( auto const &step: _linux_steps_vec )
137  {
138  // OOB step update - here at the beginning of the step.
139  _iroob.set_next_step( step.id );
140 
141  // Loop through the sub-step update vector for this step
142  for ( auto const &substep: step.substeps )
143  {
144  rc = _fw_util.do_update_step( substep, reboot, part_trigger );
145  if ( rc )
146  {
147  // Bail - Error while performing substep update.
148  result << "Error occurred while performing '" << substep << "' update..";
149  SYSLOG_NOTICE( 0, "%s", result.str().c_str() );
150  success = false;
151  /* BUT! If this was merely a script or copy failure, don't block the
152  * switching of the parition and reboot. We know that the linux and rootfs
153  * update succeeded if part_trigger is >= 3 from those update steps. */
154  if ( part_trigger < 3 )
155  {
156  _iroob.done_failure( step.id.c_str(), result.str() );
157  // Do something with 'reboot' variable value.
158  return ( success );
159  }
160  }
161  }
162  }
163 
164  // If the kernel, dtb, image.sums and rootfs have been successfully updated,
165  // switch the active partion and call for a reboot
166  // (In the case of KVM, there is no DTB, so we accept 3+ here as good enough)
167  if ( part_trigger >= 3 )
168  {
169  reboot = true;
170  rc = _fw_util.switch_partition( _fw_util._new_partition );
171  SYSLOG_INFO( 2, " update: switch_partition result [%d]", rc );
172 
173  if ( rc )
174  {
175  // Bail - Failed to switch partitions.
176  success = false;
177  result << "Failed to switch from partion " << _fw_util._current_partition << " to partition " << _fw_util._new_partition;
178  SYSLOG_NOTICE( 0, "%s", result.str().c_str() );
179  _iroob.done_failure( "Switch partition", result.str() );
180  // Do something with 'reboot' variable value.
181  }
182  else
183  {
184  std::string fname( REBOOT_TOUCH_FILE );
185  rc = CFwUtil::touch( fname );
186  // Ignoring the return here
187  SYSLOG_INFO( 2, " update: touch %s: result [%d]", REBOOT_TOUCH_FILE, rc );
188  // This touch file prevents another update from happening without a reboot (at least, that's its purpose)
189  // But it isn't watched by configsubscriber, so doesn't in itself call for a reboot.
190  // But we'll leave it to the iot_agent to implement reboot upon our return here.
191  }
192  }
193 
194  return ( success );
195 }

255  int rc;

257  // Tic the OOB up by one step
258  _iroob.set_next_step( update_params.image_type );
259 
260  // Already validated this folder
261  std::string copy_path( update_params.product_arg );
262  for ( auto &entry : update_params.type_file_list )
263  {
264  std::string type, fname;
265  std::tie( type, fname ) = entry;
266  // We treat "copy" and "resource" the same way
267  if ( ( type == ID_COPY ) || ( type == ID_RESOURCE ) )
268  {
269  SYSLOG_DEBUG( 3, " update: Extracting/Copying file '%s' to [%s]", fname.c_str(), copy_path.c_str() );
270  rc = _fw_util.extract_resource( fname, copy_path );
271  if ( rc )
272  {
273  SYSLOG_ERR( 0, "update failure when extracting [%s/%s], result [%d]", copy_path.c_str(), fname.c_str(), rc );
274  return ( false );
275  }
276  SYSLOG_INFO( 3, " update: Extracted [%s/%s], result [%d]", copy_path.c_str(), fname.c_str(), rc );
277  }
278  else if ( type == ID_SCRIPT )
279  {
280  SYSLOG_DEBUG( 3, " update: Running script '%s' ", fname.c_str() );
281  // Providing copy_path here to satisfy the arguments, but it isn't used anymore.
282  rc = _fw_util.run_script( fname, copy_path, _g_options.user );
283  if ( rc )
284  {
285  SYSLOG_ERR( 0, "update failure for script named [%s], result [%d]", fname.c_str(), rc );
286  return ( false );
287  }
288  SYSLOG_INFO( 3, " update: run_script named [%s], result [%d]", fname.c_str(), rc );
289  }
290  }
291 
292  return ( true );
293 }
