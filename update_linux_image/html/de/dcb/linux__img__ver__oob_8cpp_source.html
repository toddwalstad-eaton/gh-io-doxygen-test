24 // Utility function to determine the LinuxImage version from the device data.
25 px::string LinuxImageVersionOob::get_img_version( const px::CDevice &cds_dev )
26 {
27  PUSH_FUNC;
28 
29  // For the image version that we will use to compare versions, use "software_version" field
30  // But exclude the 7 or 8-char hash at the end, if present, eg "RI7-181-g25242f9"
31  // (This is the result of "git describe", where RI7 is tag, 181 is the commit number
32  // under that tag, and the hash is the "uniquely abbreviated commit object")
33  // The ImageVersionOob::construct_oob_tag() function follows this same flow to abbreviate the version.
34  px::string img_version( cds_dev.software_version ); // assume will take as is, by default.
35  px::string::size_type idx = cds_dev.software_version.rfind( '-' );
36  if ( ( idx != px::string::npos ) &&
37  ( idx >= ( cds_dev.software_version.size() - 9 ) ) &&
38  ( idx < ( cds_dev.software_version.size() - 6 ) ) )
39  {
40  img_version = cds_dev.software_version.substr( 0, idx );
41  }
42  return ( img_version );
43 }
