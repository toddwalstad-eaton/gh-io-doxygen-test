147 bool CBacnetIpHandler::parse()
148 {
149  PUSH_FUNC;
150 
151  // Get a pointer to our transactor
152  assert( _transactor != nullptr );
153  px::asio::CUdpListener *udp_listener = px::poly_cast<px::asio::CUdpListener *>( _transactor );
154  assert( udp_listener != nullptr );
155 
156  px::asio::iobuf_const_pair_t range = udp_listener->get_rbuf();
157  int received_bytes = ( int ) std::distance( range.first, range.second );
158 
159  if ( received_bytes == 0 )
160  {
161  SVR_DEBUG( 8, "Received No bytes; going back to listening" );
162  return ( false );
163  }
164 
165  /* One initial check: is this the signature of a BACnet/IP packet? */
166  if ( ( *range.first != BVLL_TYPE_BACNET_IP ) || ( received_bytes > MAX_MPDU ) )
167  {
168  SVR_NOTICE( 2, "Received non-BACnet/IP or too much (%d, cf %d) data; dropping input.", received_bytes, MAX_MPDU );
169  // Clear the read buffer
170  udp_listener->clear_rbuf( range.second );
171  // Go back to listening
172  return ( false );
173  }
174 
175  // We were probably created inside CBacnetIpHandler::clone()
176  // So save (update, in fact) a pointer to this for the local bip_send_pdu() function.
177  _this_ip_handler = this;
178  _force_unicast_response = false;
179 
180  // Copy out our data, to this point, but only enough for the longest header
181  uint8_t pdu[MAX_MPDU] = {0, 0}; // a little initialization to keep valgrind quiet
182  size_t i = 0;
183  for ( auto pos = range.first; pos != range.second; ++pos )
184  {
185  pdu[i++] = *pos;
186  if ( i > MAX_BVLC_HEADER ) // Stop after the longest header length
187  {
188  break;
189  }
190  }
191 
192  // Setup the call to bvlc_for_non_bbmd(), etc, as in bip_receive()
193  struct sockaddr_in sin;
194  uint16_t source_port;
195  memset( &sin, 0, sizeof( struct sockaddr_in ) );
196  boost::asio::ip::address source_addr = udp_listener->get_source_address( source_port );
197  // Convert the port number back to network byte order
198  sin.sin_port = htons( source_port );
199  if ( source_addr.is_v4() )
200  {
201  // Get the IPv4 address, make sure it is back in network order
202  sin.sin_addr.s_addr = htonl( source_addr.to_v4().to_ulong() );
203  }
204  BACNET_ADDRESS src_bac_addr; /* source address - who should get the response */
205  memset( &src_bac_addr, 0, sizeof( BACNET_ADDRESS ) );
206  src_bac_addr.net = src_bac_addr.len = 0; // until and unless we learn otherwise
207  /* We want data in src_bac_addr.mac[] to be in network order. */
208  src_bac_addr.mac_len = BIP_MAC_LEN;
209  memcpy( &src_bac_addr.mac[0], &sin.sin_addr.s_addr, sizeof( sin.sin_addr.s_addr ) );
210  memcpy( &src_bac_addr.mac[4], &sin.sin_port, 2 );
211  SVR_DEBUG( 8, "Received %d bytes from %s :", received_bytes, source_addr.to_string().c_str() );
212  SVR_DBGDUMP( 8, pdu, received_bytes );
213  SVR_DEBUG( 8, "Source MAC address (network byte order): " );
214  SVR_DBGDUMP( 8, src_bac_addr.mac, BIP_MAC_LEN );
215 
216  if ( bvlc_for_non_bbmd( &src_bac_addr, pdu, received_bytes ) > 0 )
217  {
218  /* Response already Handled, usually with a NACK. */
219  SVR_DEBUG( 2, "BVLC response sent or dropped!" );
220  // We are done with the input and can clear it away now
221  udp_listener->clear_rbuf( range.second );
222  return ( true );
223  }
224 
225  // This function code was locally stored by the bvlc_for_non_bbmd() function
226  uint8_t bvlc_function = bvlc_get_function_code(); /* aka, pdu[1] */
227 
228  uint16_t pdu_len = 0; // If we end up with this as 0, clear the rbuf
229  uint16_t hdr_len = 0;
230  BACNET_IP_ADDRESS my_addr = { 0 };
231  /* decode the length of the PDU - length is inclusive of BVLC */
232  decode_unsigned16( &pdu[2], &pdu_len );
233 
234  if ( ( pdu_len <= 4 ) || ( pdu_len > MAX_MPDU ) )
235  {
236  pdu_len = 0;
237  SVR_NOTICE( 1, "PDU too large or small (%hu). Discarded!", pdu_len );
238  }
239  // Last chance check to go back for more:
240  else if ( pdu_len > received_bytes )
241  {
242  SVR_DEBUG( 7, "Incomplete BIP packet; expected length (%hu) > received (%d)", pdu_len, received_bytes );
243  return ( false );
244  }
245  else
246  {
247  if ( ( bvlc_function == BVLC_ORIGINAL_UNICAST_NPDU ) ||
248  ( bvlc_function == BVLC_ORIGINAL_BROADCAST_NPDU ) )
249  {
250  hdr_len = MAX_UNICAST_HEADER;
251  }
252  else if ( bvlc_function == BVLC_FORWARDED_NPDU )
253  {
254  hdr_len = MAX_BVLC_HEADER;
255  memcpy( &sin.sin_addr.s_addr, &pdu[4], sizeof( sin.sin_addr.s_addr ) );
256  memcpy( &sin.sin_port, &pdu[8], 2 );
257  // Update who we send back to, too:
258  memcpy( &src_bac_addr.mac[0], &sin.sin_addr.s_addr, sizeof( sin.sin_addr.s_addr ) );
259  memcpy( &src_bac_addr.mac[4], &sin.sin_port, 2 );
260  }
261  // Then, for either Unicast or BVLC packets:
262  if ( ( sin.sin_addr.s_addr == bip_get_addr( &my_addr ) ) &&
263  ( sin.sin_port == bip_get_port() ) )
264  {
265  /* ignore messages from me */
266  pdu_len = 0;
267  SVR_DEBUG( 4, "I was the source. Discarded!" );
268  }
269  else
270  {
271  /* FIXME: check destination address */
272  /* see if it is broadcast or for us */
273  /* subtract off the header */
274  pdu_len -= hdr_len;
275  SVR_DEBUG( 7, "NPDU length = %hu", pdu_len );
276  /* Copy out the remaining PDU bytes, after the header */
277  auto pos = range.first;
278  pos += hdr_len; // increment the pointer to beyond the header
279  for ( i = 0; pos != range.second; ++pos )
280  {
281  pdu[i++] = *pos;
282  }
283  }
284  }
285 
286  // We are done with the input in rbuf[] and can clear it away now
287  udp_listener->clear_rbuf( range.second );
288  // clear this flag that bip_send_pdu() should then set
289  _sending_response = false;
290 
291  /* process */
292  if ( pdu_len > 0 )
293  {
294  /*********************************************************************************
295  * All this, just to Setup and print a decoding of the message type
296  * (but it's worth it!)
297  */
298  const char *bvlc_descr = bactext_bvlc_function( bvlc_function );
299  // Borrowing code from npdu_handler()
300  // Just to figure out what our offset is
301  // we don't care about dest or src_bac_addr, so set them to nullptr's
302  BACNET_NPDU_DATA npdu_data;
303  // npdu_decode returns -1 on error, so check for that
304  int offset = npdu_decode( pdu, nullptr, nullptr, &npdu_data );
305  uint8_t service_choice = 0;
306  bool is_confirmed_broadcast = false;
307  // uint8_t invoke_id;
308  if ( ( offset <= 0 ) || ( offset + MAX_NPDU_TRAILER_BYTES > pdu_len ) )
309  {
310  SVR_NOTICE( 2, "Error decoding npdu." );
311  }
312  else if ( npdu_data.network_layer_message )
313  {
314  SVR_INFO( 5, "%s Network Layer Message: %s", bvlc_descr,
315  bactext_network_layer_msg_name( npdu_data.network_message_type ) );
316  }
317  else // ie, if ( offset + MAX_NPDU_TRAILER_BYTES >= pdu_len )
318  {
319  switch ( pdu[offset] & 0xF0 )
320  {
321  case PDU_TYPE_CONFIRMED_SERVICE_REQUEST:
322  // invoke_id = pdu[offset +1];
323  // Assume no segmentation
324  service_choice = pdu[offset + CONFIRMED_SERVICE_CHOICE_INDEX];
325  SVR_INFO( 5, "%s Confirmed Service Request: %s", bvlc_descr,
326  bactext_confirmed_service_name( service_choice ) );
327  // _bip_core->add_bacnet_message( invoke_id, sin );
328 
329  // Further check if its a broadcast request and raise the flag for the same.
330  // AS per the spec, we should not send response to the confirmed broadcasts requests.
331  if ( bvlc_function == BVLC_ORIGINAL_BROADCAST_NPDU )
332  {
333  is_confirmed_broadcast = true;
334  }
335  break;
336  case PDU_TYPE_UNCONFIRMED_SERVICE_REQUEST:
337  service_choice = pdu[offset + UNCONFIRMED_SERVICE_CHOICE_INDEX];
338  SVR_INFO( 5, "%s Unconfirmed Service Request: %s", bvlc_descr,
339  bactext_unconfirmed_service_name( service_choice ) );
340  break;
341  default:
342  SVR_INFO( 5, "%s Other PDU Type: 0x%02X", bvlc_descr, pdu[offset] );
343  break;
344  }
345  }
346  /*******************************************************************************
347  * End of message printout
348  */
349 
350  // Now make the call, routed or not
351 #ifdef IS_GATEWAY
352  // Check the request type before making the call
353  // If its a confirmed broadcast request which is a demand not permitted to make
354  // then we should not process it further and shall be ignored as per the spec.
355  if ( !is_confirmed_broadcast )
356  {
357  routing_npdu_handler( &src_bac_addr, _DNET_list, pdu, pdu_len );
358  }
359  else
360  {
361  SVR_INFO( 5, "Ignore the Broadcast Confirmed Service Request: %s", bactext_confirmed_service_name( service_choice ) );
362  }
363 #else
364  if ( !is_confirmed_broadcast )
365  {
366  npdu_handler( &src_bac_addr, pdu, pdu_len );
367  }
368  else
369  {
370  SVR_INFO( 5, "Ignore the Broadcast Confirmed Service Request: %s", bactext_confirmed_service_name( service_choice ) );
371  }
372 
373 #endif
374  }
375 
376  if ( !_sending_response )
377  {
378  // Could also be that the response was already sent, as for I-Am responses to bcast Who-Is
379  SVR_DEBUG( 6, "Silently dropping request; no response" );
380  return ( false ); // ie, go back to listening
381  }
382  else
383  {
384  return ( true );
385  }

480 /* Encode the string(s) requested by the array_index in ReadProperty.
481  * Follows the rules for array access:
482  * - might want just the array size
483  * - might want the full array of strings (if there is enough room to encode them)
484  * - might want one specific string (indexed by array location)
485  */
486 int CBacnetIpHandler::encode_string_array( BACNET_READ_PROPERTY_DATA *rpdata, int maxApdu, const std::vector<px::string> &texts )
487 {
488  PUSH_FUNC;
489 
490  int len, apdu_len = 0;
491  uint8_t *apdu = rpdata->application_data;
492  BACNET_CHARACTER_STRING char_string;
493  uint i, count = texts.size();
494  assert( apdu != NULL );
495  assert( count > 0 );
496 
497  if ( rpdata->array_index == 0 )
498  {
499  /* Array element zero is the number of elements in the array */
500  apdu_len = encode_application_unsigned( apdu, count );
501  }
502  else if ( rpdata->array_index == ( uint32_t ) BACNET_ARRAY_ALL )
503  {
504  /* if no index was specified, then try to encode the entire list */
505  /* into one packet. */
506  for ( i = 0; i < count; i++ )
507  {
508  characterstring_init_ansi( &char_string, texts[i].c_str() );
509  /* Ensure that this does not go beyond MAX_APDU length!
510  * Encoded length will be string length + 2 header bytes
511  * unless strlen > 253, then 4 header byes */
512  int est_len = ( int ) characterstring_length( &char_string ) + 3;
513  if ( est_len > ( 253 + 3 ) )
514  {
515  est_len += 2;
516  }
517  /* add it if we have room */
518  if ( ( apdu_len + est_len ) < maxApdu )
519  {
520  len = encode_application_character_string( &apdu[apdu_len], &char_string );
521  apdu_len += len;
522  }
523  else
524  {
525  // Can't fit; signal an Abort for the RP or RPM return
526  SVR_NOTICE( 2, "Failed encode_string_array() for text #%u of %u, length %d > %d",
527  ( i + 1 ), count, ( apdu_len + est_len ), maxApdu );
528  rpdata->error_class = ERROR_CLASS_SERVICES;
529  rpdata->error_code = ERROR_CODE_ABORT_SEGMENTATION_NOT_SUPPORTED;
530  apdu_len = -1;
531  break;
532  }
533  }
534  }
535  else
536  {
537  if ( rpdata->array_index <= count )
538  {
539  i = rpdata->array_index - 1;
540  characterstring_init_ansi( &char_string, texts[i].c_str() );
541  apdu_len = encode_application_character_string( apdu, &char_string );
542  }
543  else
544  {
545  rpdata->error_class = ERROR_CLASS_PROPERTY;
546  rpdata->error_code = ERROR_CODE_INVALID_ARRAY_INDEX;
547  apdu_len = -1;
548  }
549  }
550  return ( apdu_len );
551 }

576  * ToDo: do we need to retain persistently, as in demo code??
577  * ToDo: or keep track of success / failure to reach our BBMD -
578  * more importantly, if we don't get an ACK back here because
579  * the BBMD can't be reached (we can't tell otherwise) */
580  BACNET_BVLC_RESULT BVLC_Result_Code = BVLC_RESULT_SUCCESSFUL_COMPLETION;
581 
582  /* This is the result of our foreign device registration */
583  decode_unsigned16( &npdu[4], &result_code );
584  BVLC_Result_Code = ( BACNET_BVLC_RESULT ) result_code;
585  SVR_DEBUG( 2, "BVLC Result: Rec'd Result Code=%d; not sending any response.", BVLC_Result_Code );
586 
587  result_code = 1; // a dummy code that will indicate that we handled this
588  /* Don't send any actual response */
589  bvlc_send_no_response();
590  // This will complete the cycle and set us back to listening
591  }
592  break;
593  case BVLC_WRITE_BROADCAST_DISTRIBUTION_TABLE:
594  result_code = BVLC_RESULT_WRITE_BROADCAST_DISTRIBUTION_TABLE_NAK;
595  break;
596  case BVLC_READ_BROADCAST_DIST_TABLE:
597  result_code = BVLC_RESULT_READ_BROADCAST_DISTRIBUTION_TABLE_NAK;
598  break;
599  case BVLC_READ_BROADCAST_DIST_TABLE_ACK:
600  result_code = 2; // dummy code
601  break;
602  case BVLC_REGISTER_FOREIGN_DEVICE:
603  result_code = BVLC_RESULT_REGISTER_FOREIGN_DEVICE_NAK;
604  break;
605  case BVLC_READ_FOREIGN_DEVICE_TABLE:
606  result_code = BVLC_RESULT_READ_FOREIGN_DEVICE_TABLE_NAK;
607  break;
608  case BVLC_READ_FOREIGN_DEVICE_TABLE_ACK:
609  result_code = 2; // dummy code
610  break;
611  case BVLC_DELETE_FOREIGN_DEVICE_TABLE_ENTRY:
612  result_code = BVLC_RESULT_DELETE_FOREIGN_DEVICE_TABLE_ENTRY_NAK;
613  break;
614  case BVLC_DISTRIBUTE_BROADCAST_TO_NETWORK:
615  result_code = BVLC_RESULT_DISTRIBUTE_BROADCAST_TO_NETWORK_NAK;
616  break;
617 
618  /* case BVLC_FORWARDED_NPDU: */
619  /* case BVLC_ORIGINAL_UNICAST_NPDU: */
620  /* case BVLC_ORIGINAL_BROADCAST_NPDU: */
621  default:
622  break;
623  }
624 
625  if ( result_code > 1 )
626  {
627  const char *bvlc_descr = bactext_bvlc_function( BVLC_Function_Code );
628  // Test if a NAK code was set above
629  // (all are BVLC_RESULT_WRITE_BROADCAST_DISTRIBUTION_TABLE_NAK (0x0010) or greater)
630  if ( result_code >= BVLC_RESULT_WRITE_BROADCAST_DISTRIBUTION_TABLE_NAK )
631  {
632  bvlc_send_result( src_addr, ( BACNET_BVLC_RESULT ) result_code );
633  SVR_DEBUG( 5, "Unsupported BVLC %s: Sending reply with NAK code=0x%04x", bvlc_descr, result_code );
634  }
635  else if ( result_code == 2 )
636  {
637  SVR_DEBUG( 5, "Unsupported BVLC %s; no response", bvlc_descr );
638  /* Don't send any actual response */
639  bvlc_send_no_response();
640  // This will complete the cycle and set us back to listening
641  }
642  // else if result_code == 1, already printed message
643  }
644  return ( result_code );
645 }

676 // Send a BACnet/IP packet to the given destination address.
677 int CBacnetIpHandler::send_bip_buf( BACNET_ADDRESS *dest, uint8_t *mtu, uint16_t mtu_len )
678 {
679  PUSH_FUNC;
680 
681  assert( ( mtu != nullptr ) && ( mtu[0] == BVLL_TYPE_BACNET_IP ) );
682 
683  // Get a pointer to our UdpListener transactor
684  assert( _transactor != nullptr );
685  px::asio::CUdpListener *udp_listener = px::poly_cast<px::asio::CUdpListener *>( _transactor );
686  assert( udp_listener != nullptr );
687 
688  // Create the write buffer; fill it with the bytes to send.
689  std::shared_ptr<px::asio::io_buf_t> wbuf = std::make_shared<px::asio::io_buf_t>( px::asio::io_buf_t() );
690  for ( size_t i = 0; i < mtu_len; i++ )
691  {
692  wbuf->push_back( mtu[i] );
693  }
694 
695  // Normally, we are responding to a request, and the UdpHandler (transactor) holds the destination
696  // address for this response. But if not, we use a different send function, supplying the destination address.
697  bool has_dest_addr = true;
698 
699  if ( ( bvlc_get_function_code() == BVLC_FORWARDED_NPDU ) && ( mtu[1] == BVLC_ORIGINAL_UNICAST_NPDU ) )
700  {
701  _sending_response = true;
702  has_dest_addr = false;
703  }
704  else if ( mtu[1] == BVLC_ORIGINAL_UNICAST_NPDU )
705  {
706  // Ensure that we are not broadcasting the response if unicasting back,
707  // even if request was broadcast (eg, Who-Is request)
708  // (In reality, this makes sure response is sent instead of just dropped)
709  udp_listener->set_want_response();
710  SVR_DEBUG( 5, "Writing %u bytes unicast via Transactor", mtu_len );
711  _sending_response = true;
712 
713  /* Check and see if this is a notification message (eg, COV), not a regular response
714  * To do that, we have to parse past the BVLC and NPDU headers. */
715  // BACNET_ADDRESS npdu_src, npdu_dest; -- use if we later want something out of these
716  BACNET_NPDU_DATA npdu_data;
717  // The npdu_decode is a little faster if we don't get the src and dest, so give null pointers for them
718  int offset = MAX_UNICAST_HEADER + npdu_decode( &mtu[MAX_UNICAST_HEADER], NULL, NULL, &npdu_data );
719  if ( !npdu_data.network_layer_message )
720  {
721  // Check the PDU's first byte
722  if ( mtu[offset] == PDU_TYPE_CONFIRMED_SERVICE_REQUEST )
723  {
724  // uint8_t InvokeID = mtu[offset+1];
725  // Check the service in the 4th byte of the PDU
726  if ( mtu[offset + CONFIRMED_SERVICE_CHOICE_INDEX] == SERVICE_CONFIRMED_COV_NOTIFICATION )
727  {
728  has_dest_addr = false;
729  }
730  }
731  else if ( mtu[offset] == PDU_TYPE_UNCONFIRMED_SERVICE_REQUEST )
732  {
733  // Check the service in the 2nd byte of the PDU
734  if ( mtu[offset + UNCONFIRMED_SERVICE_CHOICE_INDEX] == SERVICE_UNCONFIRMED_COV_NOTIFICATION )
735  {
736  has_dest_addr = false;
737  }
738  }
739  }
740  }
741  else if ( udp_listener->was_broadcast() )
742  {
743  SVR_DEBUG( 2, "Will not send any response to this broadcast request" );
744  // Remove the response bytes then
745  wbuf->clear();
746  }
747  else // if buf[1] == something other than BVLC_ORIGINAL_UNICAST_NPDU
748  {
749  SVR_DEBUG( 2, "Directed broadcast (?) of %u bytes via Transactor", mtu_len );
750  }
751 
752  if ( has_dest_addr )
753  {
754  // Normal reply
755  _transactor->write( wbuf );
756  }
757  else
758  {
759  // Initiated broadcast or notification
760  mtu_len = send_buf_to_dest( dest, wbuf, mtu_len );
761  }
762  // That write() call will complete the cycle and set us back to listening
763  return ( ( int ) mtu_len );
764 }
