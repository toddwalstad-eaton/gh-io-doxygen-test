157 // Handler for the derived class to supply values for its more special properties.
158 int CBacnetAnalogOutputObject::handle_read_property( BACNET_READ_PROPERTY_DATA *rpdata,
159  const px::CValue & /* cdsValue */, bool /* value_ok */ )
160 {
161  PUSH_FUNC;
162 
163  int apdu_len = -2; // default is that we don't handle this property
164  uint8_t *apdu = rpdata->application_data;
165 
166  switch ( rpdata->object_property )
167  {
168  case PROP_COV_INCREMENT:
169  apdu_len = encode_application_real( apdu, _cov_increment );
170  break;
171 
172  case PROP_MAX_PRES_VALUE:
173  if ( _max_val.has_value() )
174  {
175  apdu_len = encode_application_real( apdu, _max_val.value().as_float() );
176  SVR_INFO( 5, "Read Max_Pres_Value = %f for Object_Name %s", _max_val.value().as_float(), getObject_Name() );
177  }
178  break;
179 
180  case PROP_MIN_PRES_VALUE:
181  if ( _min_val.has_value() )
182  {
183  apdu_len = encode_application_real( apdu, _min_val.value().as_float() );
184  SVR_INFO( 5, "Read Min_Pres_Value = %f for Object_Name %s", _min_val.value().as_float(), getObject_Name() );
185  }
186  break;
187 
188  // We have no other "special properties" to manage in this class...
189  default:
190  // apdu_len already set to -2, so do nothing more
191  break;
192  }
193 
194  return ( apdu_len );
195 }

261 // Handler to get the priority-array value for this object type
262 int CBacnetAnalogOutputObject::get_priority_array_values( BACNET_READ_PROPERTY_DATA *rpdata,
263  const px::CValue &cdsValue, bool value_ok )
264 {
265  PUSH_FUNC;
266 
267  ( void )cdsValue;
268 
269  if ( !value_ok )
270  {
271  rpdata->error_class = ERROR_CLASS_PROPERTY;
272  rpdata->error_code = ERROR_CODE_READ_ACCESS_DENIED;
273  return ( -1 );
274  }
275 
276  size_t i = 0;
277  int len = 0;
278  int apdu_len = 0;
279  uint8_t *apdu = rpdata->application_data;
280  float prio_val = 0;
281  bool is_valid_prio_val = false;
282 
283  // Array element zero is the number of elements in the array
284  if ( rpdata->array_index == 0 )
285  {
286  apdu_len = encode_application_unsigned( &apdu[0], BACNET_MAX_PRIORITY );
287  }
288  // if no index was specified, then try to encode the entire list
289  // into one packet.
290  else if ( rpdata->array_index == BACNET_ARRAY_ALL )
291  {
292  for ( i = BACNET_MIN_PRIORITY; i <= BACNET_MAX_PRIORITY; i++ )
293  {
294  // Encode a real value if the value isn't equal to the cancel value AND the value at this index
295  // is supported. Else, encode null.
296  is_valid_prio_val = _priority_array->get_priority_value( i, prio_val );
297  if ( is_valid_prio_val && !is_almost_equal( prio_val, _priority_array->get_priority_cancel() ) )
298  {
299  len = encode_application_real( &apdu[apdu_len], prio_val );
300  }
301  else
302  {
303  len = encode_application_null( &apdu[apdu_len] );
304  }
305  // add it if we have room
306  if ( ( apdu_len + len ) < MAX_APDU )
307  {
308  apdu_len += len;
309  }
310  else
311  {
312  rpdata->error_class = ERROR_CLASS_SERVICES;
313  rpdata->error_code = ERROR_CODE_NO_SPACE_FOR_OBJECT;
314  apdu_len = BACNET_STATUS_ERROR;
315  break;
316  }
317  }
318  }
319  else
320  {
321  if ( rpdata->array_index <= BACNET_MAX_PRIORITY )
322  {
323  is_valid_prio_val = _priority_array->get_priority_value( rpdata->array_index, prio_val );
324  if ( is_valid_prio_val && !is_almost_equal( prio_val, _priority_array->get_priority_cancel() ) )
325  {
326  apdu_len = encode_application_real( &apdu[0], prio_val );
327  }
328  else
329  {
330  apdu_len = encode_application_null( &apdu[0] );
331  }
332  }
333  else
334  {
335  rpdata->error_class = ERROR_CLASS_PROPERTY;
336  rpdata->error_code = ERROR_CODE_INVALID_ARRAY_INDEX;
337  apdu_len = BACNET_STATUS_ERROR;
338  }
339  }
340 
341  return ( apdu_len );
342 }

374 void CBacnetAnalogOutputObject::update_optional_properties( const px::CChannelSettings &ch_settings )
375 {
376  const uint8_t old_prop_size = sizeof( AnalogOutputProps ) / sizeof( BacPropEntry_t );
377  uint8_t new_prop_size = old_prop_size;
378  uint8_t prop_index = 0;
379 
380  if ( ch_settings.maximum.is_initialized() )
381  {
382  _max_val = ch_settings.maximum;
383  new_prop_size++;
384  }
385  else
386  {
387  _max_val.reset();
388  }
389 
390  if ( ch_settings.minimum.is_initialized() )
391  {
392  _min_val = ch_settings.minimum;
393  new_prop_size++;
394  }
395  else
396  {
397  _min_val.reset();
398  }
399  // clear the old list and create a new list as channel settings has changed
400  if ( ( _analogOutputExtProps != nullptr ) && _extProps_is_local )
401  {
402  delete _analogOutputExtProps;
403  _analogOutputExtProps = 0;
404  _extProps_is_local = false;
405  }
406 
407  // Check if old and new sizes are same. If same, then no need to create local list. Point to the global list instead.
408  if ( old_prop_size == new_prop_size )
409  {
410  _analogOutputExtProps = AnalogOutputProps;
411  }
412  else
413  {
414  _analogOutputExtProps = new BacPropEntry_t[new_prop_size];
415 
416  if ( _analogOutputExtProps == 0 )
417  {
418  SVR_ERR( 2, "Could not create extended property array for %s", getObject_Name() );
419  _analogOutputExtProps = nullptr;
420  }
421  else
422  {
423  _extProps_is_local = true; // Indicating that we have locally alloated the list. SHould be released during destruction
424  // Copy all entries expect the last entry, as it will be added at the end
425  for ( prop_index = 0; prop_index < ( old_prop_size - 1 ); prop_index++ )
426  {
427  _analogOutputExtProps[prop_index] = AnalogOutputProps[prop_index];
428  }
429  if ( _max_val.has_value() )
430  {
431  _analogOutputExtProps[prop_index++] = AnalogOutputMaxProp;
432  }
433  if ( _min_val.has_value() )
434  {
435  _analogOutputExtProps[prop_index++] = AnalogOutputMinProp;
436  }
437  // Add the last entry
438  _analogOutputExtProps[prop_index] = AnalogOutputProps[( old_prop_size - 1 )];
439  }
440  }
441 
442  /* Now create the right pointer for the Optional Property List (integer array), based on the above. */
443  if ( _min_val.has_value() )
444  {
445  if ( _max_val.has_value() )
446  {
447  // This is the fullest list
448  _analogOutputOptionalProps = &MinMaxOptionalPropList[0];
449  }
450  else
451  {
452  _analogOutputOptionalProps = &MinNomaxOptionalPropList[0];
453  }
454  }
455  else if ( _max_val.has_value() )
456  {
457  _analogOutputOptionalProps = &MinMaxOptionalPropList[1];
458  }
459  else
460  {
461  _analogOutputOptionalProps = &MinNomaxOptionalPropList[1];
462  }
463 }

505 // Handles the write to present-value for this object type.
506 bool CBacnetAnalogOutputObject::handle_write_present_value( BACNET_WRITE_PROPERTY_DATA *wp_data,
507  BACNET_APPLICATION_DATA_VALUE &value )
508 {
509  PUSH_FUNC;
510 
511  // return value
512  bool status = false;
513 
514  assert( wp_data->error_class == ERROR_CLASS_PROPERTY );
515  assert( wp_data->error_code == ERROR_CODE_OTHER ); // set to this on entry
516 
517  // First, check the type for correctness and range check as appropriate,
518  // then store the value in our CValue for writing
519  px::CValue value_to_write_to_prio_array;
520  //
521  // For ""Real" property type like present value of Analog Output, "Double" type of data
522  // wont't be accepted and will be treated as incorrect data type. This change is needed to be done for
523  // certification purpose though it's not customer friendly approach and seems incorrect.
524  // Thus only REAL data type values will be accepted.
525  if ( value.tag == BACNET_APPLICATION_TAG_REAL )
526  {
527  if ( _min_val.has_value() )
528  {
529  // Perform range check before updating the present_value
530  if ( value.type.Real >= _min_val.value().as_float() )
531  {
532  value_to_write_to_prio_array = value.type.Real;
533  status = true;
534  }
535  else
536  {
537  wp_data->error_code = ERROR_CODE_VALUE_OUT_OF_RANGE;
538  SVR_DEBUG( 1, "Present value update failed for %s with error %s", getObject_Name(), bactext_error_code_name( wp_data->error_code ) );
539  status = false;
540  return ( status );
541  }
542  }
543  if ( _max_val.has_value() )
544  {
545  // Perform range check before updating the present_value
546  if ( value.type.Real <= _max_val.value().as_float() )
547  {
548  value_to_write_to_prio_array = value.type.Real;
549  status = true;
550  }
551  else
552  {
553  wp_data->error_code = ERROR_CODE_VALUE_OUT_OF_RANGE;
554  SVR_DEBUG( 1, "Present value update failed for %s with error %s", getObject_Name(), bactext_error_code_name( wp_data->error_code ) );
555  status = false;
556  return ( status );
557  }
558  }
559  // Range check has been performed and validated if corresponding channel in CDS has min/max defined thus update the value as it is.
560  value_to_write_to_prio_array = value.type.Real;
561  status = true;
562  }
563  else if ( value.tag == BACNET_APPLICATION_TAG_NULL )
564  {
565  // This is the cancel case; substitute the cancel value when writing here
566  value_to_write_to_prio_array = _priority_array->get_priority_cancel();
567  status = true;
568  }
569  else
570  {
571  SVR_ERR( 1, "Invalid BACnet value type %s", bactext_application_tag_name( value.tag ) );
572  wp_data->error_code = ERROR_CODE_INVALID_DATA_TYPE;
573  }
574  if ( status )
575  {
576  // Using priority arrays, set the channel value if necessary
577  size_t priority_to_set = ( size_t )wp_data->priority;
578  size_t priority_active = 0;
579  bool is_prio_set = _priority_array->set_priority_value( priority_to_set,
580  value_to_write_to_prio_array.as_float() );
581 
582  // Due to float equal comparisons not being particularly helpful, we have
583  // to do this on our own here.
584  if ( is_prio_set )
585  {
586  float highest_prio_val = _priority_array->get_relinquish_default();
587  std::vector<size_t> prio_array_list = _priority_array->get_priority_array_list();
588  for ( const auto &itr : prio_array_list )
589  {
590  float prio_val;
591  if ( _priority_array->get_priority_value( itr, prio_val ) )
592  {
593  if ( is_almost_equal( prio_val, _priority_array->get_priority_cancel() ) )
594  {
595  continue;
596  }
597  else
598  {
599  highest_prio_val = prio_val;
600  priority_active = itr;
601  break;
602  }
603  }
604  }
605 
606  if ( priority_active == 0 )
607  {
608  // No priority active, relinquish default is now active, use 17 as lowest priority
609  priority_active = 17;
610  }
611 
612  px::CValue value_to_write_to_cds( highest_prio_val );
613  // Execute command or write channel if needed
614  // (Here, just setting it to highest priority every time some value is entered in the priority array)
615  if ( _use_cmd )
616  {
617  status = _bip_core->_cds_handle->exec_output_cmd( _parent_id, mObject_CommandName,
618  value_to_write_to_cds, priority_active );
619  }
620  else
621  {
622  SVR_INFO( 4, "Setting channel %u to %f", ( uint32_t )_channel_id, value_to_write_to_cds.as_float() );
623  _bip_core->_cds_handle->set_channel_value( _channel_id, value_to_write_to_cds );
624 
625  // Determine whether this change constitutes a COV update or not
626  float diff = std::abs( value_to_write_to_cds.as_float() - _last_cov_value.as_float() );
627  if ( diff >= _cov_increment )
628  {
629  SVR_DEBUG( 7, "Updated value %f differs by >= %f from last published %f", value_to_write_to_cds.as_float(),
630  _cov_increment, _last_cov_value.as_float() );
631  // So set the flag that marks our desire to have a COV Notification sent
632  _has_cov_update = true;
633  }
634  status = true;
635  }
636  }
637  else
638  {
639  status = false;
640  }
641  }
642 
643  return ( status );
644 }
