196 // Handles the write to present-value for this object type.
197 bool CBacnetCharacterString::handle_write_present_value( BACNET_WRITE_PROPERTY_DATA *wp_data,
198  BACNET_APPLICATION_DATA_VALUE &value )
199 {
200  PUSH_FUNC;
201 
202  bool status = false; /* return value */
203 
204  // This is a simplified version of Analog_Value_Write_Property() and the rest
205  // Since we don't do object arrays, we don't worry about the object_index
206  // Since we don't do priority arrays, we don't worry about the priority
207  // Since we don't need to null out values in a priority array, we
208  // don't worry about BACNET_APPLICATION_TAG_NULL
209 
210  assert( wp_data->error_class == ERROR_CLASS_PROPERTY );
211  assert( wp_data->error_code == ERROR_CODE_OTHER ); // set to this on entry
212 
213  // Do not need to check _use_cmd for CharacterStrings
214 
215  // First, check the type for correctness and range check as appropriate,
216  // then store the value in our CValue for writing
217  px::CValue value_to_write;
218  if ( value.tag == BACNET_APPLICATION_TAG_CHARACTER_STRING )
219  {
220  // Just put a reasonable bound on string length or there's probably something wrong
221  if ( value.type.Character_String.length < MAX_BIP_CHAR_STRING )
222  {
223  // Have to cast to (const char *) or value_to_write gets the type wrong
224  value_to_write = ( const char * ) characterstring_value( &value.type.Character_String );
225  status = true;
226  }
227  else
228  {
229  wp_data->error_code = ERROR_CODE_VALUE_OUT_OF_RANGE;
230  }
231  }
232  else
233  {
234  wp_data->error_code = ERROR_CODE_INVALID_DATA_TYPE;
235  }
236 
237  if ( status )
238  {
239  if ( _use_cmd ) // if is a command - very unlikely for CharacterStrings
240  {
241  status = _bip_core->_cds_handle->exec_device_cmd(
242  _parent_id, mObject_CommandName, value_to_write );
243  SVR_INFO( 2, "Wrote %s to command %s", value_to_write.as_string().c_str(),
244  mObject_CommandName.c_str() );
245  }
246  else
247  {
248  _bip_core->_cds_handle->set_channel_value( _channel_id, value_to_write );
249  status = true; // fire-and-forget always presumed to work
250  SVR_INFO( 2, "Wrote %s to channel %" PRIu64 " (%s)", value_to_write.as_string().c_str(),
251  _channel_id, mObject_Name.c_str() );
252  }
253  }
254 
255  return ( status );
256 }
