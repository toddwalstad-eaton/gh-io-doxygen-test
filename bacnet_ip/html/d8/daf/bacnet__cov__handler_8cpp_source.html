93 // Timeout callback invoked when the _cov_timer expires.
94 // We don't have any use for the timer argument here
95 void BacnetCovHandler::timeout( const px::asio::CIoTimer & /* timer */ )
96 {
97  PUSH_FUNC;
98 
99  if ( ++_subsec_counter >= ( 1000 / _COV_TIMEBASE_MSECS ) )
100  {
101  // Then this is the one-second rollover
102  _subsec_counter = 0;
103  // Call our function to push the updates to their objects now
104  push_cov_updates();
105  // Call the bacnet-stack COV function to manage subscriptions, with this 1s update
106  handler_cov_timer_seconds( 1 );
107 
108  // See if we need to manage the Active Subscriptions list
109  if ( ++_sec_counter >= _CHECK_ACTIVE_SUBS_SECS )
110  {
111  _sec_counter = 0;
112  // Now do the garbage collection
113  check_active_subscriptions();
114  }
115  }
116  // Call the COV finite-state-machine task more often
117  handler_cov_task();
118 
119  // Call the timer engine at 100 msec rate for data retries and segmented transfer
120  px::CWallTime update_time = px::CWallTime::now();
121  time_t current_miliseconds = update_time.tv_nsec / 1000000;
122  // Skip the first call as we don't know the time difference.
123  if ( _tsm_engine_running )
124  {
125  time_t diff = current_miliseconds - _past_miliseconds;
126  if ( current_miliseconds < _past_miliseconds )
127  {
128  diff += 1000; //Since timer resets to 0 after 1000ms, 1000 is added to maintain appropriate difference
129  }
130  tsm_timer_milliseconds( diff );
131  }
132  _past_miliseconds = current_miliseconds;
133  _tsm_engine_running = true;
134 }

136 // Callback function to handle the channel value change notifications from the CDS.
137 void BacnetCovHandler::channel_values_callback( const std::map<px::id_t, px::CValue> &ch_map,
138  const px::CWallTime & /* update_time */ )
139 {
140  PUSH_FUNC;
141 
142  px::id_t ch_id;
143  px::CValue new_value;
144 
145  int lock_rv = pthread_rwlock_rdlock( &_actively_subscribed_map_rwlock );
146  if ( unlikely( lock_rv != 0 ) )
147  {
148  SVR_FERR( 0, "Subscribed map lock error %d, possibly missing some BACnet CoV updates", lock_rv );
149  return;
150  }
151  // SVR_DEBUG(9, "At start of %zu ch values CoV update, had %zu subscribed values", ch_map.size(), _actively_subscribed_map.size() );
152  // Copy (or update) these updates to _pending_channel_updates, but only if
153  // the channel is listed in _actively_subscribed_map
154  // First, check: is anyone subscribed?
155  if ( _actively_subscribed_map.empty() )
156  {
157  // No one interested; just return
158  }
159  // Second case: just a few subscribed points; iterate over that list
160  // instead of the changed map, as that will be quicker
161  else if ( _actively_subscribed_map.size() < ch_map.size() )
162  {
163  for ( auto active_pair : _actively_subscribed_map )
164  {
165  ch_id = active_pair.first;
166  auto pos = ch_map.find( ch_id );
167  if ( pos == ch_map.end() )
168  {
169  // No such entry for this channel
170  continue;
171  }
172  // First, just try to insert it into the map
173  {
174  std::unique_lock<std::mutex> _lock {_pending_channel_updates_mutex};
175  auto res_pair = _pending_channel_updates.insert( *pos );
176  if ( !res_pair.second )
177  {
178  // If that failed, it's because there was already an entry; update it instead
179  // first points to the right entry, then copy in the px::CValue from the update pair.
180  res_pair.first->second = pos->second;
181  }
182  }
183  }
184  SVR_DEBUG( 7, "At end of active subscriber search update" );
185  }
186  // Third case: lots of subscribed points, so iterate over the changed values map instead
187  else
188  {
189  for ( auto ch_pair : ch_map )
190  {
191  std::tie( ch_id, new_value ) = ch_pair;
192  // See if actively subscribed
193  if ( _actively_subscribed_map.find( ch_id ) == _actively_subscribed_map.end() )
194  {
195  // Nope, no one cares
196  continue;
197  }
198  {
199  std::unique_lock<std::mutex> _lock {_pending_channel_updates_mutex};
200  // First, just try to insert it into the map
201  auto res_pair = _pending_channel_updates.insert( ch_pair );
202  if ( !res_pair.second )
203  {
204  // If that failed, it's because there was already an entry; update it instead
205  // first points to the right entry, then copy in the px::CValue from the update pair.
206  res_pair.first->second = new_value;
207  }
208  }
209  }
210  SVR_DEBUG( 7, "At end of %zu values update", ch_map.size() );
211  }
212  ( void ) pthread_rwlock_unlock( &_actively_subscribed_map_rwlock );
213 }

216 void BacnetCovHandler::update_active_subscription( CBacnetObject *obj_ptr )
217 {
218  PUSH_FUNC;
219 
220  int lock_rv = pthread_rwlock_wrlock( &_actively_subscribed_map_rwlock );
221  if ( unlikely( lock_rv != 0 ) )
222  {
223  SVR_FERR( 0, "Map lock error %d, BACnet CoV subscriptions may be out of date", lock_rv );
224  return;
225  }
226  px::id_t ch_id = obj_ptr->get_cds_id();
227  px::CMonoTime tnow = px::CMonoTime::now();
228  auto pos = _actively_subscribed_map.find( ch_id );
229  if ( pos == _actively_subscribed_map.end() )
230  {
231  // Then this is a new entry
232  _actively_subscribed_map[ch_id] = std::make_pair( tnow, obj_ptr );
233  SVR_DEBUG( 5, "Added subscription for %s = %u", obj_ptr->getObject_Name(), ( uint32_t ) ch_id );
234  }
235  else
236  {
237  // Just update its time
238  pos->second.first = tnow;
239  SVR_DEBUG( 9, "Updated subscription for %s = %u", obj_ptr->getObject_Name(), ( uint32_t ) ch_id );
240  }
241  ( void ) pthread_rwlock_unlock( &_actively_subscribed_map_rwlock );
242 }

244 /* Function to periodically check the list of Active Subscriptions
245  * and remove any that have aged out */
246 void BacnetCovHandler::check_active_subscriptions()
247 {
248  PUSH_FUNC;
249 
250  // Compute the timestamp that determines old vs active
251  px::CMonoTime too_old = px::CMonoTime::now() - px::CMonoTime( _ACTIVE_SUBS_TIMEOUT_SECS );
252  int lock_rv = pthread_rwlock_wrlock( &_actively_subscribed_map_rwlock );
253  if ( unlikely( lock_rv != 0 ) )
254  {
255  SVR_FERR( 0, "Map lock error %d, BACnet CoV subscriptions may be around too long", lock_rv );
256  return;
257  }
258  for ( auto mpos = _actively_subscribed_map.begin(); mpos != _actively_subscribed_map.end();)
259  {
260  if ( mpos->second.first < too_old )
261  {
262  SVR_DEBUG( 5, "Removing subscription for %s = %u", mpos->second.second->getObject_Name(),
263  ( uint32_t ) mpos->first );
264  mpos = _actively_subscribed_map.erase( mpos );
265  }
266  else
267  {
268  ++mpos;
269  }
270  }
271  ( void ) pthread_rwlock_unlock( &_actively_subscribed_map_rwlock );
272 }

274 // Function to manage sending the list of pending updates to each object.
275 void BacnetCovHandler::push_cov_updates()
276 {
277  PUSH_FUNC;
278 
279  int lock_rv = pthread_rwlock_rdlock( &_actively_subscribed_map_rwlock );
280  if ( unlikely( lock_rv != 0 ) )
281  {
282  SVR_FERR( 0, "Map lock error %d, BACnet CoV notifications may be missed", lock_rv );
283  return;
284  }
285  // First, make doubly sure this list is empty
286  _swapped_channel_updates.clear();
287  // Now do the big swapparoo, so _swapped_channel_updates gets all the pending updates
288  // (and _pending_channel_updates is cleared and ready to accept new values)
289  {
290  std::unique_lock<std::mutex> _lock {_pending_channel_updates_mutex};
291  _swapped_channel_updates.swap( _pending_channel_updates );
292  } // end lock scope
293  if ( !_swapped_channel_updates.empty() )
294  {
295  for ( auto ch_val_pair : _swapped_channel_updates )
296  {
297  px::id_t ch_id;
298  px::CValue update_value;
299  std::tie( ch_id, update_value ) = ch_val_pair;
300  auto pos = _actively_subscribed_map.find( ch_id );
301  if ( unlikely( pos == _actively_subscribed_map.end() ) )
302  {
303  continue;
304  }
305  active_cov_obj_pair_t &time_obj_pair = pos->second;
306  // Now we have the pointer we need, to the object to get the update
307  time_obj_pair.second->update_cov_value( update_value );
308  }
309  SVR_DEBUG( 7, "Pushed %zu value updates", _swapped_channel_updates.size() );
310  _swapped_channel_updates.clear();
311  }
312  ( void ) pthread_rwlock_unlock( &_actively_subscribed_map_rwlock );
313 }
