207 {
208  PUSH_FUNC;
209 
210  bool success = false;
211  for ( auto itr : _priority_array_list )
212  {
213  if ( itr == priority )
214  {
215  try

247 template<class T> const T CBacnetPriorityArray<T>::get_highest_priority_value() const
248 {
249  PUSH_FUNC;
250 
251  // set the return value as relinquish default value; if we find
252  // a valid higher priority return it.
253  T ret_val = _relinquish_default;
254  for ( auto itr : _priority_array_list )
255  {
256  try
257  {
258  if ( _priority_array.at( itr ) != _priority_cancel )
259  {
260  ret_val = _priority_array.at( itr );
261  break;
262  }
263  }
264  catch ( std::out_of_range &oor )
265  {
266  // just continue to next priority if there's no value set here yet
267  continue;
268  }
269  catch ( ... )
270  {
271  SVR_ERR( 1, "Failed retrieving priority value" );
272  }
273  }
274 
275  return ( ret_val );
276 }

278 template<class T> ssize_t CBacnetPriorityArray<T>::get_highest_priority_index_active() const
279 {
280  PUSH_FUNC;
281 
282  // set the return value as -1 if no priorities are set;
283  // a valid higher priority index return it.
284  ssize_t ret_val = -1;
285  for ( const auto &itr : _priority_array_list )
286  {
287  try
288  {
289  if ( _priority_array.at( itr ) != _priority_cancel )
290  {
291  // Can safely cast as itr should never be greater than
292  // SSIZE_MAX, in fact should never be greater than 16
293  ret_val = ( ssize_t )itr;
294  break;
295  }
296  }
297  catch ( std::out_of_range &oor )
298  {
299  // just continue to next priority if there's no value set here yet
300  continue;
301  }
302  catch ( ... )
303  {
304  SVR_ERR( 1, "Failed retrieving priority value" );
305  }
306  }
