165 // Handler for the derived class to supply values for its more special properties.
166 // We don't need the cdsValue here.
167 int CBacnetMultiStateValue::handle_read_property( BACNET_READ_PROPERTY_DATA *rpdata,
168  const px::CValue & /* cdsValue */, bool /* value_ok */ )
169 {
170  PUSH_FUNC;
171 
172  int apdu_len = -2; // default is that we don't handle this property here
173  uint8_t *apdu = rpdata->application_data;
174 
175  switch ( rpdata->object_property )
176  {
177  case PROP_NUMBER_OF_STATES:
178  {
179  uint32_t count = _multistate_map.size();
180  apdu_len = encode_application_unsigned( apdu, count );
181  }
182  break;
183 
184  case PROP_STATE_TEXT:
185  {
186  std::vector<px::string> texts;
187  for ( auto entry : _multistate_map )
188  {
189  // The map is now ordered, so the BACnet data comes out correctly
190  texts.push_back( entry.second ); // get each string in turn
191  }
192 
193  if ( texts.size() )
194  {
195  apdu_len = CBacnetIpHandler::encode_string_array( rpdata,
196  rpdata->application_data_len,
197  texts );
198  }
199  else
200  {
201  apdu_len = -1;
202  }
203  }
204  break;
205 
206  default:
207  // apdu_len already set to -2, so do nothing more
208  break;
209  }
210 
211  return ( apdu_len );
212 }

214 // Handler to lookup the present-value for a MultiStateValue and do all necessary checking.
215 int CBacnetMultiStateValue::get_present_value( BACNET_READ_PROPERTY_DATA *rpdata,
216  const px::CValue &cdsValue, bool value_ok )
217 {
218  PUSH_FUNC;
219 
220  if ( !value_ok )
221  {
222  rpdata->error_class = ERROR_CLASS_PROPERTY;
223  rpdata->error_code = ERROR_CODE_READ_ACCESS_DENIED;
224  return ( -1 );
225  }
226 
227  int apdu_len = -1;
228  uint8_t *apdu = rpdata->application_data;
229 
230  // If this is a request for the Value, encode it
231  if ( rpdata->object_property == PROP_PRESENT_VALUE )
232  {
233  BACNET_APPLICATION_DATA_VALUE bac_value;
234  if ( encode_present_value( cdsValue, bac_value ) )
235  {
236  apdu_len = encode_application_unsigned( apdu, bac_value.type.Unsigned_Int );
237  }
238  }
239  if ( apdu_len < 1 )
240  {
241  SVR_NOTICE( 2, "No MultiState %s match for CDS Value %u",
242  getObject_Name(), cdsValue.as_uint32() );
243  }
244 
245  return ( apdu_len );
246 }

248 //Encodes the present-value for a Point object from its CDS value.
249 bool CBacnetMultiStateValue::encode_present_value( const px::CValue &cds_value, BACNET_APPLICATION_DATA_VALUE &bac_value )
250 {
251  PUSH_FUNC;
252 
253  bool bResult = false;
254  // Always uint for BACnet MultiStates (but CDS Values can be zero or negative)
255  uint32_t uiVal = 0;
256  // BACnet states count up from 1 in simple sequence (no gaps)
257  // So we need to increment uival for each value that we test until we reach the right one.
258  for ( auto entry : _multistate_map )
259  {
260  uiVal++;
261  // Cast both the mapped and CDS Value we are comparing to signed ints
262  if ( entry.first.as_int32() == cds_value.as_int32() )
263  {
264  SVR_DEBUG( 8, "For CDS Value %d, matched to BACnet MultiState %s Value %u (%s)",
265  cds_value.as_int32(), getObject_Name(), uiVal, entry.second.c_str() );
266  bResult = true;
267  break;
268  }
269  }
270 
271  bac_value.tag = BACNET_APPLICATION_TAG_UNSIGNED_INT;
272  bac_value.type.Unsigned_Int = uiVal;
273  bac_value.context_specific = false;
274  bac_value.next = NULL;
275  return ( bResult );
276 }

330 // Handles the write to present-value for this object type.
331 bool CBacnetMultiStateValue::handle_write_present_value( BACNET_WRITE_PROPERTY_DATA *wp_data,
332  BACNET_APPLICATION_DATA_VALUE &value )
333 {
334  PUSH_FUNC;
335 
336  bool status = false; /* return value */
337 
338  // This is a simplified version of Multistate_Output_Write_Property()
339  // Since we don't do object arrays, we don't worry about the object_index
340  // Since we don't do priority arrays, we don't worry about the priority
341  // Since we don't need to null out values in a priority array, we
342  // don't worry about BACNET_APPLICATION_TAG_NULL
343  assert( wp_data->error_class == ERROR_CLASS_PROPERTY );
344  assert( wp_data->error_code == ERROR_CODE_OTHER ); // set to this on entry
345 
346  if ( value.tag == BACNET_APPLICATION_TAG_UNSIGNED_INT )
347  {
348  // validation: check the value against the channel settings
349  // Always uint for BACnet MultiStates (not necessarily so for corresponding CDS Values)
350  uint32_t uival = 0;
351 
352  // BACnet states count up from 1 in simple sequence (no gaps)
353  // So we need to increment uival for each value that we test until we reach the right one.
354  for ( auto entry : _multistate_map )
355  {
356  uival++;
357  if ( uival == value.type.Unsigned_Int )
358  {
359  // Get the corresponding CDS value
360  SVR_DEBUG( 5, "Attempting to write CDS Value %d, matched to BACnet MultiState %s Value %u (%s)",
361  entry.first.as_int32(), getObject_Name(), uival, entry.second.c_str() );
362 
363  if ( _use_cmd )
364  {
365  // ToDo: Assumes the Device Write command is just to set the given channel
366  // (ie, nothing fancy like a C3 command here)
367  status = _bip_core->_cds_handle->exec_device_cmd( _parent_id,
368  mObject_CommandName, entry.first );
369  }
370  else
371  {
372  _bip_core->_cds_handle->set_channel_value( _channel_id, entry.first );
373  status = true;
374  }
375  if ( !status )
376  {
377  //wp_data->error_class = ERROR_CLASS_PROPERTY;
378  wp_data->error_code = ERROR_CODE_WRITE_ACCESS_DENIED; // most probable
379  SVR_NOTICE( 2, "Device command to write BACnet MultiState %s Value to %u (%s) failed.",
380  getObject_Name(), uival, entry.second.c_str() );
381  }
382  break;
383  }
384  }
385 
386  // if any of the above failed, report error as out of range
387  if ( !status )
388  {
389  //wp_data->error_class = ERROR_CLASS_PROPERTY;
390  wp_data->error_code = ERROR_CODE_VALUE_OUT_OF_RANGE;
391  }
392  }
393  else
394  {
395  //wp_data->error_class = ERROR_CLASS_PROPERTY;
396  wp_data->error_code = ERROR_CODE_INVALID_DATA_TYPE;
397  }
398 
399  if ( !status )
400  {
401  SVR_NOTICE( 2, "Attempt to write BACnet MultiState %s Value failed.", getObject_Name() );
402  }
403 
404  return ( status );
405 }
