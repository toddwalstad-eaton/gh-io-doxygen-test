164 // Handler for the derived class to supply values for its more special properties.
165 int CBacnetBinaryObject::handle_read_property( BACNET_READ_PROPERTY_DATA *rpdata,
166  const px::CValue & /* cdsValue */, bool /* value_ok */ )
167 {
168  PUSH_FUNC;
169 
170  int apdu_len = -2; // default is that we don't handle this property
171  uint8_t *apdu = rpdata->application_data;
172 
173  // Handle the class-specific properties here
174  BACNET_CHARACTER_STRING char_string;
175  switch ( rpdata->object_property )
176  {
177  case PROP_POLARITY:
178  // Force one simple choice here
179  apdu_len = encode_application_enumerated( apdu, POLARITY_NORMAL );
180  break;
181 
182  case PROP_ACTIVE_TEXT:
183  characterstring_init_ansi( &char_string, getChannelMultiStateString( true ).c_str() );
184  apdu_len = encode_application_character_string( apdu, &char_string );
185  break;
186 
187  case PROP_INACTIVE_TEXT:
188  characterstring_init_ansi( &char_string, getChannelMultiStateString( false ).c_str() );
189  apdu_len = encode_application_character_string( apdu, &char_string );
190  break;
191 
192  default:
193  // apdu_len already set to -2, so do nothing more
194  break;
195  }
196 
197  return ( apdu_len );
198 }

200 // Handler to check and encode the present-value for a Point object and do all necessary checking.
201 int CBacnetBinaryObject::get_present_value( BACNET_READ_PROPERTY_DATA *rpdata,
202  const px::CValue &cdsValue, bool value_ok )
203 {
204  PUSH_FUNC;
205 
206  if ( !value_ok )
207  {
208  rpdata->error_class = ERROR_CLASS_PROPERTY;
209  rpdata->error_code = ERROR_CODE_READ_ACCESS_DENIED;
210  return ( -1 );
211  }
212 
213  int apdu_len = -1;
214  uint8_t *apdu = rpdata->application_data;
215 
216  // If this is a request for the Value, encode it
217  if ( rpdata->object_property == PROP_PRESENT_VALUE )
218  {
219  // apdu_len = encode_value( cdsValue, BAPTG_BOOLEAN, apdu ) -- NO! not that, but this:
220  BACNET_APPLICATION_DATA_VALUE bac_value;
221  encode_present_value( cdsValue, bac_value );
222  apdu_len = encode_application_enumerated( apdu, bac_value.type.Enumerated );
223  }
224 
225  return ( apdu_len );

228 //Encodes the present-value for a Point object from its CDS value.
229 bool CBacnetBinaryObject::encode_present_value( const px::CValue &cds_value, BACNET_APPLICATION_DATA_VALUE &bac_value )
230 {
231  PUSH_FUNC;
232 
233  // For Binary Values, we simply take True as always Active,
234  // so Polarity = POLARITY_NORMAL
235  uint32_t uiVal = cds_value.as_bool() ? BINARY_ACTIVE : BINARY_INACTIVE;
236  // If was a numeric but maps to a Binary Object type, show a Notice,
237  // except if the UINT value is 0 or 1 - some should-be-bools are done as CDS Multistates
238  if ( cds_value.as_uint32() > 1 )
239  {
240  SVR_NOTICE( 1, "Ill-advised conversion from %u to Binary value for %s",
241  cds_value.as_uint32(), getObject_Name() );
242  }
243  bac_value.tag = BACNET_APPLICATION_TAG_ENUMERATED;
244  bac_value.type.Enumerated = uiVal;
245  bac_value.context_specific = false;
246  bac_value.next = NULL;
247  return ( true );
248 }

303 // Handles the write to present-value for this object type.
304 bool CBacnetBinaryObject::handle_write_present_value( BACNET_WRITE_PROPERTY_DATA *wp_data,
305  BACNET_APPLICATION_DATA_VALUE &value )
306 {
307  PUSH_FUNC;
308 
309  bool status = false; /* return value */
310 
311  // This is a simplified version of Analog_Value_Write_Property() and the rest
312  // Since we don't do object arrays, we don't worry about the object_index
313  // Since we don't do priority arrays, we don't worry about the priority
314  // Since we don't need to null out values in a priority array, we
315  // don't worry about BACNET_APPLICATION_TAG_NULL
316 
317  assert( wp_data->error_class == ERROR_CLASS_PROPERTY );
318  assert( wp_data->error_code == ERROR_CODE_OTHER ); // set to this on entry
319 
320  px::CValue value_to_write;
321  if ( value.tag == BACNET_APPLICATION_TAG_BOOLEAN )
322  {
323  value_to_write = value.type.Boolean;
324  status = true;
325  }
326  else if ( value.tag == BACNET_APPLICATION_TAG_ENUMERATED )
327  {
328  // priority = wp_data->priority; -- see Binary_Value_Write_Property() for
329  // details about handling priority, arrays, and NULLs
330  if ( value.type.Enumerated <= MAX_BINARY_PV )
331  {
332  value_to_write = value.type.Enumerated;
333  status = true;
334  }
335  else
336  {
337  wp_data->error_code = ERROR_CODE_VALUE_OUT_OF_RANGE;
338  }
339  }
340  else
341  {
342  wp_data->error_code = ERROR_CODE_INVALID_DATA_TYPE;
343  }
344 
345  if ( status )
346  {
347  if ( _use_cmd )
348  {
349  status = _bip_core->_cds_handle->exec_device_cmd(
350  _parent_id,
351  mObject_CommandName,
352  value_to_write );
353  }
354  else
355  {
356  _bip_core->_cds_handle->set_channel_value( _channel_id, value_to_write );
357  // Determine whether this change constitutes a COV update or not
358  if ( value_to_write != _last_cov_value )
359  {
360  SVR_DEBUG( 7, "Updated value %s differs from last published %s for %s", value_to_write.as_string().c_str(),
361  _last_cov_value.as_string().c_str(), getObject_Name() );
362  // So set the flag that marks our desire to have a COV Notification sent
363  _has_cov_update = true;
364  }
365  status = true;
366  }
367  }
368 
369  return ( status );
370 }
