110 /* Initializes the subdevice objects which can be created from the channel data. */
111 void CBacnetStructuredViewObject::initialize_struct_view_objects( const std::vector<px::CChannelSettings> &settings )
112 {
113  PUSH_FUNC;
114  bacObjPtr obj_ptr = NULL;
115  std::shared_ptr<CBacnetIpCore> bip_core = _parent_device->_bip_core;
116  assert( bip_core.get() );
117  const px::string &subdev_id = _px_subdevice.network_address;
118  // Loop through channel settings canonical names
119  SVR_DEBUG( 5, "Checking %zu Channels for BACnet Object mappings of subdevice %s",
120  settings.size(), subdev_id.c_str() );
121  size_t count = 0;
122  for ( const px::CChannelSettings &ch_stts : settings )
123  {
124  // get the MCL channel info, if any
125  auto *pmci = bip_core->_mcl_channel_config->get_mcl_channel_info( ch_stts.canonical_name );
126  if ( pmci != nullptr )
127  {
128  uint32_t obj_id = _base_id + pmci->bac_obj_id;
129 
130  // Optimize by checking if the channel object already exists and skipping in that case:
131  if ( _sv_objects.find( obj_id ) != _sv_objects.end() )
132  {
133  SVR_DEBUG( 8, "-- Found existing SV BACnet point obj for channel %" PRIu64 ": %s",
134  ch_stts.id(), ch_stts.canonical_name.c_str() );
135  continue;
136  }
137 
138  // Compose a unique Object Name for this channel by adding the Subdevice # to the MCL name
139  // Then use the regular create() factory function so we can specify the non-standard fields
140  px::string sub_obj_name = ch_stts.canonical_name + subdev_id;
141 
142  if ( _parent_device.get() )
143  {
144  obj_ptr = CBacnetPointObject::create(
145  pmci->bac_obj_type, // BACnet object type
146  obj_id, // BACnet object id
147  sub_obj_name.c_str(), // BACnet object name
148  ch_stts.flags.get( px::channel_flag::is_writable ), // writable
149  ch_stts.value_type.val(), // value type
150  _parent_device.get(),
151  bip_core,
152  ch_stts.id(), NO_COV ); // channel id (CDS)
153  }
154  else
155  {
156  SVR_WARN( 1, "BACNet point object creation failed as the device object is invalid" );
157  return;
158  }
159 
160  if ( obj_ptr == NULL )
161  {
162  SVR_ERR( 2, "Failed to create object for bacObjPtr" );
163  }
164  else if ( obj_ptr )
165  {
166  SVR_DEBUG( 8, "Adding %s(%u) to map as %s",
167  sub_obj_name.c_str(), obj_id,
168  bactext_object_type_name( pmci->bac_obj_type ) );
169  // We used the regular create() function but need to save the channel settings of interest, too
170  CBacnetPointObject *po_ptr = dynamic_cast<CBacnetPointObject *>( obj_ptr.get() );
171  if ( likely( po_ptr != nullptr ) )
172  {
173  po_ptr->store_channel_settings( ch_stts );
174  }
175 
176  // Insert it in the map of maps, and our SV objects map
177  if ( _parent_device->AddObjectToMaps( obj_ptr ) )
178  {
179  count++;
180  if ( _sv_objects.find( obj_id ) == _sv_objects.end() )
181  {
182  _sv_objects[obj_id] = obj_ptr;
183  }
184  else
185  {
186  SVR_DEBUG( 5, "Updating existing %s in SV map for ObjID %u",
187  sub_obj_name.c_str(), obj_id );
188  _sv_objects[obj_id] = obj_ptr;
189  }
190  }
191  else
192  {
193  SVR_ERR( 2, "Failed to add %s to parent's maps", sub_obj_name.c_str() );
194  }
195  }
196  }
197  else
198  {
199  SVR_DEBUG( 4, "No BACnet mapping found for %s", ch_stts.canonical_name.c_str() );
200  }
201  }
202  SVR_DEBUG( 5, "Added %zu BACnet Objects to %s maps", count, _parent_device->getObject_Name() );
203 }

216 /* Static Utility function to return the calculated Object ID to use for a new
217  * StructuredView object, given its zero-based instance number. */
218 uint32_t CBacnetStructuredViewObject::calculate_object_id( uint8_t instance_number, size_t num_sub_devices )
219 {
220  PUSH_FUNC;
221 
222  // First, assume our guaranteed collision-free wide spacing
223  uint32_t spacing = CBacnetStructuredViewObject::_STRUCT_VIEW_SCALAR;
224  if ( num_sub_devices > CBacnetStructuredViewObject::_MAX_STRUCT_VIEWS )
225  {
226  spacing = CBacnetStructuredViewObject::_NARROWER_SCALAR;
227  if ( instance_number == 1 )
228  {
229  SVR_DEBUG( 2, "Exceeded the number of Structured Views (%u) for collision-free spacing; using %u spacing instead.",
230  CBacnetStructuredViewObject::_MAX_STRUCT_VIEWS, spacing );
231  }
232  }
233  else if ( instance_number > CBacnetStructuredViewObject::_MAX_STRUCT_VIEWS )
234  {
235  spacing = CBacnetStructuredViewObject::_NARROWER_SCALAR;
236  // The 8bit size limit on instance_number keeps rogue values still within reason here
237  SVR_INFO( 2, "Unexpectedly high instance number %uu (< %u); using less safe narrow spacing %u.",
238  instance_number, CBacnetStructuredViewObject::_MAX_STRUCT_VIEWS, spacing );
239  }
240  uint32_t obj_id = ( instance_number + 1 ) * spacing;
241  return ( obj_id );
242 }
