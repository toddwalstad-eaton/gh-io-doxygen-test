131 // Handler to get the present-value for a Point object and do all necessary checking.
132 // int CBacnetDateTimeValue::get_present_value( BACNET_READ_PROPERTY_DATA *rpdata,
133 // const px::CValue& cdsValue, bool value_ok )
134 int CBacnetDateTimeValue::get_present_value( BACNET_READ_PROPERTY_DATA *rpdata,
135  const px::CValue &cdsValue, bool value_ok )
136 {
137  PUSH_FUNC;
138 
139  if ( !value_ok )
140  {
141  rpdata->error_class = ERROR_CLASS_PROPERTY;
142  rpdata->error_code = ERROR_CODE_READ_ACCESS_DENIED;
143  return ( -1 );
144  }
145 
146  int apdu_len = -1;
147  uint8_t *apdu = rpdata->application_data;
148 
149  // If this is a request for the Value, encode it
150  if ( rpdata->object_property == PROP_PRESENT_VALUE )
151  {
152  if ( cdsValue.type().is_timestamp() )
153  {
154  struct tm tmbd; // broken-down time structure
155  px::CTimestamp px_ts = cdsValue.as_timestamp();
156  timeval tval = px_ts.to_timeval();
157  time_t tt = tval.tv_sec;
158  uint8_t hundredths = tval.tv_usec / 10000; // convert uSec to 100s
159  BACNET_DATE_TIME bdatetime;
160  gmtime_r( &tt, &tmbd );
161  datetime_set_values( &bdatetime, ( tmbd.tm_year + 1900 ), ( tmbd.tm_mon + 1 ),
162  tmbd.tm_mday, tmbd.tm_hour, tmbd.tm_min, tmbd.tm_sec, hundredths );
163  apdu_len = bacapp_encode_datetime( apdu, &bdatetime );
164  }
165  }
166 
167  return ( apdu_len );
168 }
