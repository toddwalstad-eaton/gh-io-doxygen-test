208 // Merge the given OOB data for the BACnet Device with our existing fields.
209 bool COOB_BACnetDevice::merge_from_oob_data( const px::proto::BACnetDevice &oob_dev )
210 {
211  PUSH_FUNC;
212 
213  bool needs_update = false;
214  // We will assume that the OOB data overrides any existing entries, in most cases
215 
216  if ( oob_dev.has_device_id() )
217  {
218  _cds_id = oob_dev.device_id();
219  }
220  if ( oob_dev.has_device_map_instance() )
221  {
222  _device_map_index = oob_dev.device_map_instance();
223  }
224  if ( oob_dev.has_device_object_id() )
225  {
226  object_id = oob_dev.device_object_id();
227  }
228  // For possible solution to ELMS-15752 BACnet epic file shows syntax error (missing objectName),
229  // also test for non-zero length
230  if ( oob_dev.has_device_object_name() && ( oob_dev.device_object_name().size() > 0 ) )
231  {
232  object_name = oob_dev.device_object_name().c_str();
233  }
234  if ( oob_dev.has_bacnet_mac_address() && ( oob_dev.bacnet_mac_address().size() > 0 ) )
235  {
236  set_mac_from_string( oob_dev.bacnet_mac_address().c_str(), false );
237  }
238  // Else will need an update whenever we get a valid MAC for this device
239 
240  if ( oob_dev.has_database_revision() && ( _database_revision <= oob_dev.database_revision() ) )
241  {
242  _database_revision = oob_dev.database_revision();
243  }
244  else
245  {
246  // Else the greater database-revision that we are holding needs to get written out
247  needs_update = true;
248  }
249  if ( oob_dev.has_apdu_timeout() )
250  {
251  _apdu_timeout = oob_dev.apdu_timeout();
252  }
253 
254  // Unpack the persisted_properties into our map (slam, not merge, for these)
255  for ( int i = 0; i < oob_dev.persisted_properties_size(); i++ )
256  {
257  const px::proto::BACnetPersistedProperty &prop_entry = oob_dev.persisted_properties( i );
258  // Reality check that we have the 3 pieces we must have
259  if ( !prop_entry.has_object_type() || !prop_entry.has_property_name() || !prop_entry.has_value() )
260  {
261  continue;
262  }
263  uint32_t object_type = 0;
264  if ( !bactext_object_type_index( prop_entry.object_type().c_str(), &object_type ) )
265  {
266  SVR_NOTICE( 2, "Failed lookup for object_type: %s", prop_entry.object_type().c_str() );
267  continue;
268  }
269  uint32_t object_id = 0;
270  if ( prop_entry.has_object_id() )
271  {
272  object_id = prop_entry.object_id();
273  }
274  uint32_t bac_property = 0;
275  if ( !bactext_property_index( prop_entry.property_name().c_str(), &bac_property ) )
276  {
277  SVR_NOTICE( 2, "Failed lookup for property name: %s", prop_entry.property_name().c_str() );
278  continue;
279  }
280  px::CValue prop_val;
281  prop_val.from( prop_entry.value() );
282 
283  // Use our function to add this entry
284  bool from_oob = true;
285  add_persisted_property( object_type, object_id, bac_property, prop_val, from_oob );
286  }
287 
288  return ( needs_update );
289 }

352 void COOB_BACnetDevice::add_persisted_property( uint32_t object_type, uint32_t object_id,
353  uint32_t bac_property, const px::CValue &value, bool from_oob )
354 {
355  PUSH_FUNC;
356 
357  oob_prop_value_t val_pair = std::make_pair( bac_property, value );
358  // Get this flag from our Settings class, unless we are processing OOB input
359  if ( _bac_ip_settings && _bac_ip_settings->_only_shared_persisted_properties && !from_oob )
360  {
361  object_id = 0;
362  SVR_DEBUG( 6, "Forcing property %u of object_type %u to be shared", bac_property, object_type );
363  }
364  uint32_t obj_identifier = ( object_type << 22 ) + object_id;
365  std::vector<oob_prop_value_t> props;
366  // Need non-const references here so we can insert the new value
367  oob_prop_map_t::iterator pos = _property_map.find( obj_identifier );
368  if ( pos == _property_map.end() )
369  {
370  // New entry in the map
371  props.push_back( val_pair );
372  _property_map.emplace( std::make_pair( obj_identifier, props ) );
373  }
374  else
375  {
376  bool found = false;
377  for ( oob_prop_value_t &oob_pair : pos->second )
378  {
379  if ( oob_pair.first == bac_property )
380  {
381  oob_pair.second = value;
382  found = true;
383  break;
384  }
385  }
386  if ( !found )
387  {
388  props.push_back( val_pair );
389  }
390  }
391  if ( object_id == 0 )
392  {
393  SVR_DEBUG( 5, "Added shared %s = %s for %s", bactext_property_name( bac_property ), value.as_string().c_str(),
394  bactext_object_type_name( object_type ) );
395  }
396  else
397  {
398  SVR_DEBUG( 5, "Added %s = %s for %s:%u", bactext_property_name( bac_property ), value.as_string().c_str(),
399  bactext_object_type_name( object_type ), object_id );
400  }
