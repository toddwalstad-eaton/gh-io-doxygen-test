121  // and fall through and continue with the next part

212 extern "C" bool Routed_Device_GetNext( BACNET_ADDRESS *dest, int *DNET_list, int *cursor )
213 {
214  PUSH_FUNC;
215 
216  int dnet = DNET_list[0]; /* Get the DNET of our virtual network */
217  int inst_num = *cursor;
218  bool bSuccess = false;
219  bool is_bcast = false; // Set this flag if we detect a broadcast in the destination
220 
221  /* First, see if the index is out of range.
222  * Eg, last call to GetNext may have been the last successful one.
223  * This is considered a failure and nothing will change.
224  */
225  if ( inst_num < 0 )
226  {
227  inst_num = -1;
228  }
229  /* Next, see if it's a BACnet broadcast.
230  * For these broadcasts, all Devices get a chance at it.
231  * Instead of checking all routed MAC bytes, just check the first port byte for 0.
232  */
233  else if ( ( dest->net == BACNET_BROADCAST_NETWORK ) ||
234  ( ( dest->net == dnet ) && ( ( dest->adr[4] == 0 ) || ( dest->len == 0 ) ) ) )
235  {
236  is_bcast = true;
237  if ( inst_num <= 1 )
238  {
239  // Is our gateway; "next" will be the first routed device
240  inst_num = GATEWAY_IDX;
241  bSuccess = true;
242  }
243  else
244  {
245  // Check validity of the given cursor.
246  bSuccess = global_core_ptr->_config->is_valid_instance_number( inst_num );
247  if ( !bSuccess )
248  {
249  inst_num = global_core_ptr->_config->get_next_instance_number( inst_num );
250  if ( inst_num > 1 )
251  {
252  bSuccess = true;
253  }
254  }
255  }
256  }
257  /* Next, see if addressing just the gateway directly (locally). */
258  else if ( dest->net == 0 )
259  {
260  /* Handle like a normal, non-routed access of the Gateway Device.
261  * Just make sure that there is no routed MAC address. */
262  // CBacnetDeviceObject* oob_gw = global_core_ptr->_config->get_bac_dev_obj_ptr( GATEWAY_IDX );
263  // if ( ( oob_gw != nullptr ) && ( memcmp( oob_gw->mac_address, mac_addr, len ) == 0 ) )
264  if ( dest->len == 0 )
265  {
266  /* Point at the gateway Device, and cursor will indicate done. */
267  inst_num = GATEWAY_IDX;
268  bSuccess = true;
269  }
270  }
271  /* Or if is our virtual DNET, check
272  * against each of our virtually routed Devices.
273  * If we get a match, have it handle the APDU.
274  * Broadcasts were handled above; this should only be directly addressed cases.
275  */
276  else if ( ( dest->net == dnet ) && ( dest->len == BIP_MAC_SIZE ) )
277  {
278  inst_num = global_core_ptr->_config->lookup_instance_by_mac_address( dest->adr, dest->len );
279  if ( inst_num > 0 )
280  {
281  bSuccess = true;
282  }
283  }
284 
285  if ( !bSuccess ) // current_device_instance will not be updated here
286  {
287  *cursor = -1;
288  SVR_DEBUG( 8, "Lookup failed for cursor instance %d", *cursor );
289  }
290  else
291  {
292  // At this point, inst_num is for the valid device, and pos points to the next device (or end)
293  set_current_device_idx( inst_num );
294  SVR_DEBUG( 8, "Device lookup succeeded for cursor %d -> instance %d", *cursor, inst_num );
295  if ( is_bcast )
296  {
297  // Get the "next" routed device instance for the cursor, for next time
298  *cursor = global_core_ptr->_config->get_next_instance_number( inst_num );
299  }
300  else
301  {
302  // We are all done; set cursor to -1 to stop coming back here
303  *cursor = -1;
304  }
305  }
306  return ( bSuccess );
307 }

377  }

381  }
382  }

384 
385  case SERVICE_CONFIRMED_DEVICE_COMMUNICATION_CONTROL:
386  /* If not the gateway device, we don't support DCC.
387  This is because it isn't required for the "virtual", non-really-BACnet devices.*/
388  if ( current_device_instance > GATEWAY_IDX )

394  }
395  else
396  {
397  len = 1; /* Non-zero return */
398  }
399  }
400  /* But we do ensure that we have a configured password, or we will reject it.
401  * This is also checked now in our patched version of h_dcc.c */
402 #if defined IS_GATEWAY && not defined DFLT_MAPPING
403  /* BUT! Don't check this for Jupiter LXI.
404  * So exclude this code in that case. */
405 #else
406  assert( global_core_ptr.get() );
407  assert( global_core_ptr->_settings.get() );
408  if ( global_core_ptr->_settings->dcc_password.empty() )
409  {
410  if ( apdu_buff != NULL )
411  {
412  len = bacerror_encode_apdu( apdu_buff, invoke_id,
413  SERVICE_CONFIRMED_DEVICE_COMMUNICATION_CONTROL,
414  ERROR_CLASS_SECURITY, ERROR_CODE_PASSWORD_FAILURE );
415  }
416  /* If apdu_buff is NULL, we are probably building PROP_PROTOCOL_SERVICES_SUPPORTED,
417  * and we don't want to say "no" here just because there is no password.
418  * So don't fail for that reason.
419  */
420  }
421  // Else, we'll let the BACnet library code decide if the password is correct
422 #endif
423  break;
424 
425  default:
426  /* Everything else is a pass, at this time. */
427  break;
428  }
429  return ( len );
