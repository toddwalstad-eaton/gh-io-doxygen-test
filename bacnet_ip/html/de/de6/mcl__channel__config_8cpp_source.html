168 void CMclChannelConfig::process_entries( json_spirit::mArray &entries )
169 {
170  PUSH_FUNC;
171  // "entries":[
172  // { "name": "(string)",
173  // "BacObjectType": "(int)_(string)",
174  // "BacObjectID": (int)
175  // },
176  // ...
177  // ]
178  for ( auto &entry : entries )
179  {
180  auto &obj = entry.get_obj();
181  px::string name = read_string_value( obj, px::string( "name" ) );
182  BACNET_OBJECT_TYPE type = get_obj_type( read_string_value( obj, px::string( "BacObjectType" ) ) );
183  int id = read_int_value( obj, px::string( "BacObjectID" ) );
184 
185  _mcl_channel_info_map.emplace( std::make_pair( name, CMclChannelInfo( type, id ) ) );
186 
187  // Now, look for the largest
188  if ( id > ( int ) _largest_id )
189  {
190  _largest_id = id;
191  }
192  }
193 }
