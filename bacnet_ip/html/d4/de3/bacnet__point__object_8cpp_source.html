215 /* Given this Point's Channel Settings, determine which (if any) need to be stored
216  * in the _field_map for later lookup.
217  * Will use the derived Point's property list for this purpose, looking for
218  * any with channel_setting "hint"s such as $channel_settings.description. */
219 void CBacnetPointObject::store_channel_settings( const px::CChannelSettings &ch_settings )
220 {
221  PUSH_FUNC;
222 
223  // Grap the objectlock first
224  get_property_array_access_lock();
225 
226  // Call the instance to update its property list based on latest optional property values in CDS channel settings
227  update_optional_properties( ch_settings );
228 
229  const BacPropEntry_t *propEntries = get_object_property_array();
230  if ( propEntries == nullptr )
231  {
232  SVR_ERR( 2, "Could not get acccess to the property array for %s", getObject_Name() );
233  release_property_array_access_lock();
234  return;
235  }
236  for ( size_t i = 0; propEntries[i].propID != PROP_ALL; i++ )
237  {
238  const BacPropEntry_t *prop = &propEntries[i];
239  // See if this property has has field hint (eg, $channel_settings.description)
240  px::CValue cds_value( 0 );
241  if ( ( prop->hint != nullptr ) && ( prop->hint[0] == '$' ) &&
242  ( strstr( prop->hint, "$channel_settings." ) != nullptr ) )
243  {
244  // Find the period, then take the text that is after that
245  px::string full_field( &prop->hint[1] ); // everything after the '$'
246  const char *pfield = index( prop->hint, '.' );
247  px::string field( ++pfield );
248  if ( field == "multistate_map" )
249  {
250  // This is a special case; the ChannelSettings holds a map for this,
251  // not just a CValue. Save that map
252  px::CMultistateMap::label_map_t &ref_map =
253  const_cast<px::CMultistateMap::label_map_t &>( ch_settings.multistate_map.get() );
254  /* Use C++17 merge() to efficiently move the data -- some fine day ...
255  * _multistate_map.merge( ch_settings.multistate_map.get() ); */
256  _multistate_map.insert( begin( ref_map ), end( ref_map ) );
257  if ( !_multistate_map.empty() )
258  {
259  SVR_DEBUG( 6, "Added %zu _multistate_map entries for Channel %s", _multistate_map.size(),
260  ch_settings.canonical_name.c_str() );
261  }
262  else
263  {
264  SVR_DEBUG( 2, "No _multistate_map entries found for Channel %s",
265  ch_settings.canonical_name.c_str() );
266  }
267  }
268  else// for regular CValue-based properties
269  {
270  try
271  {
272  if ( ch_settings.get_field_by_name( field, cds_value ) )
273  {
274  // Then we have a match; save it, using the full hint for lookup later
275  _field_map[full_field] = cds_value;
276  }
277  // Permit one common, important second chance here: MCL descriptions are often empty;
278  // in that case, substitute the channel "best" Name (eg, channel_settings.name)
279  else if ( strcmp( "$channel_settings.description", prop->hint ) == 0 )
280  {
281  cds_value = ch_settings.get_best_name( 0 ); // no length limit
282  _field_map[full_field] = cds_value;
283  SVR_DEBUG( 2, "Substituting best_name %s for description for Channel %s",
284  cds_value.as_string().c_str(), ch_settings.canonical_name.c_str() );
285  }
286  }
287  catch ( const px::generic_err & )
288  {
289  SVR_ERR( 2, "Generic exception caught during store_channel_settings" );
290  }
291  }
292  }
293  }
294  if ( !_field_map.empty() )
295  {
296  SVR_DEBUG( 6, "Added %zu _field_map entries for Channel %s", _field_map.size(),
297  ch_settings.canonical_name.c_str() );
298  }
299  release_property_array_access_lock();
300 }

304 bool CBacnetPointObject::get_property_by_field_name( const char *field_name, px::CValue &field_value )
305 {
306  PUSH_FUNC;
307 
308  bool bOK = false;
309  if ( ( _channel_id != 0 ) || ( _use_cmd ) )
310  {
311  // This point is based on a channel
312  // if ( strcmp( field_name, "value" ) == 0 )
313  // First, see if we have stored this Channel Setting field in our map (should be so)
314  auto pos = _field_map.find( field_name );
315  if ( pos != _field_map.end() )
316  {
317  bOK = true;
318  field_value = pos->second;
319  }
320  /* Had to stop doing this synchronous CDS access:
321  * Read the px::CChannel from the CDS, using _channel_id
322  * Then call get_field_by_name() on it
323  * px::CChannel px_channel;
324  * bOK = _bip_core->_cds_handle->get_px_channel( _channel_id, px_channel );
325  * if ( bOK ) // then take the next step
326  * bOK = px_channel.get_field_by_name( field_name, field_value );
327  */
328  if ( bOK )
329  {
330  // One more check: the CDS might have "succeeded" but returned an invalid value
331  if ( field_value.type() == px::datatype::invalid )
332  {
333  SVR_DEBUG( 4, "Invalid result for px::CChannel %s field %s", getObject_Name(), field_name );
334  bOK = false;
335  }
336  else
337  {
338  SVR_DEBUG( 6, "Looked up px::CChannel %s field %s: %s", getObject_Name(), field_name, field_value.as_string().c_str() );
339  }
340  }
341  else
342  {
343  SVR_DEBUG( 2, "Look up failed for px::CChannel %s field %s", getObject_Name(), field_name );
344  }
345  }
346  else
347  {
348  // This point is (we hope) based on a device property (eg, CharacterString Object)
349  // But first! If the field_name is "value", then we will substitute the object name
350  // (This is, at least for now, a workaround for the device's CharacterString objects)
351  const char *fd_name = field_name;
352  if ( strcmp( field_name, "value" ) == 0 )
353  {
354  fd_name = mObject_Name.c_str();
355  }
356 
357  bOK = _parent_dev->get_property_by_field_name( fd_name, field_value );
358  if ( bOK )
359  {
360  SVR_DEBUG( 6, "Looked up px::CDevice(%s) field %s: %s", _parent_dev->getObject_Name(),
361  fd_name, field_value.as_string().c_str() );
362  }
363  else
364  {
365  SVR_DEBUG( 2, "Look up failed for px::CDevice(%s) field %s", _parent_dev->getObject_Name(), fd_name );
366  }
367  }
368  return ( bOK );
369 }

371 /* Manages the property by calling handlers in this order until one handles it:
372  * 1) Calls ObjectReadPropertyFromList() to let the base CBacnetObject class
373  * handle the simple properties common to all objects
374  * 2) Prefetches a value from the CDS if this property has a field name hint
375  * 3) Calls handle_read_property() so the derived class can handle its
376  * special properties
377  * 4) Else, handles the simple properties common to the CBacnetPointObject types.
378  */
379 int CBacnetPointObject::Object_Read_Property( BACNET_READ_PROPERTY_DATA *rpdata )
380 {
381  PUSH_FUNC;
382 
383  int apdu_len = -1;
384  unsigned propsIdx;
385 
386  get_property_array_access_lock();
387  // Assign the right property list for this type of Object
388  const BacPropEntry_t *propsList = get_object_property_array();
389 
390  if ( propsList == nullptr )
391  {
392  release_property_array_access_lock();
393  SVR_ERR( 2, "Could not get acccess to the property array for %s", getObject_Name() );
394  return ( apdu_len );
395  }
396 
397  // Give the base class a chance to process common properties
398  apdu_len = ObjectReadPropertyFromList( rpdata, propsList );
399  if ( apdu_len >= -2 ) // If that worked, including errors, we're done here
400  {
401  release_property_array_access_lock();
402  return ( apdu_len );
403  }
404 
405  // Else, check for the Point-specific properties
406  // In this case, apdu_len holds the negative of the index into
407  // propsList[] for this Property's entry, in case it is needed.
408  propsIdx = -apdu_len;
409  const BacPropEntry_t *prop_entry = &propsList[propsIdx];
410 
411  // If this property has has field hint (eg, $value), then
412  // we can prefetch that Value from the CDS for processing below
413  // Unless it is $channel_settings.multistate_map - there is no prefetch for that case
414  bool value_ok = false;
415  px::CValue cds_value( 0 );
416  if ( ( prop_entry->hint[0] == '$' ) && ( strcmp( prop_entry->hint, "$channel_settings.multistate_map" ) != 0 ) )
417  {
418  const char *field_name = &prop_entry->hint[1];
419  // Look for "value" - a common case
420  if ( strcmp( field_name, "value" ) == 0 )
421  {
422  // If channel_id is 0, then this data comes from the Device, not a Channel
423  if ( _channel_id == 0 )
424  {
425  if ( _use_cmd )
426  {
427  if ( ( getObject_Type() == OBJECT_INTEGER_VALUE ) || ( getObject_Type() == OBJECT_POSITIVE_INTEGER_VALUE ) ||
428  ( getObject_Type() == OBJECT_ANALOG_VALUE ) || ( getObject_Type() == OBJECT_ANALOG_OUTPUT ) ||
429  ( getObject_Type() == OBJECT_BINARY_OUTPUT ) )
430  {
431  cds_value = ( uint32_t ) 0;
432  }
433  else
434  {
435  cds_value = false;
436  }
437  value_ok = true;
438  }
439  else
440  {
441  // This is a device property
442  // Ask our derived class or default function (above) if it can look up this field
443  value_ok = get_property_by_field_name( field_name, cds_value );
444  }
445  }
446  else
447  {
448  // Read the channel's present value
449  value_ok = _bip_core->_cds_handle->get_channel_value( _channel_id, cds_value );
450  }
451  }
452  else
453  {
454  // Ask our derived class if it can look up this field
455  value_ok = get_property_by_field_name( field_name, cds_value );
456  }
457  if ( value_ok )
458  {
459  SVR_DEBUG( 8, "Prefetch of hint %s:%s = %s", getObject_Name(), prop_entry->hint,
460  cds_value.as_string().c_str() );
461  }
462  else
463  {
464  SVR_WARN( 2, "Failed Prefetch of hint %s:%s", getObject_Name(), prop_entry->hint );
465  }
466  }
467  // Release the BACnet object property array access lock
468  release_property_array_access_lock();
469 
470  uint8_t *apdu = rpdata->application_data;
471 
472  // Now pass to the derived class, to see if it can handle this property
473  apdu_len = handle_read_property( rpdata, cds_value, value_ok );
474  if ( apdu_len >= -1 ) // If that worked, including errors, we're done here
475  {
476  return ( apdu_len );
477  }
478 
479  // Else, handle the common properties here
480  bool bStat;
481  uint32_t uiVal;
482  unsigned i;
483  BACNET_BIT_STRING bit_string;
484  uint32_t reliability_value = 0;
485  // Casting BACNET_PROPERTY_ID to uint to avoid warnings on our proprietary props
486  switch ( ( uint32_t ) rpdata->object_property )
487  {
488  case PROP_STATUS_FLAGS:
489  bitstring_init( &bit_string );
490  // Determine if we are InAlarm:
491  bStat = check_in_alarm_status();
492  bitstring_set_bit( &bit_string, STATUS_FLAG_IN_ALARM, bStat );
493 
494  // Bad .Quality is the same as Fault:
495  bStat = !check_point_quality();
496  if ( !bStat )
497  {
498  // Second chance looking at reliability if Quality is OK
499  uint32_t reliability_value;
500  if ( _bip_core->_config->get_bip_routed_device_map()->get_reliability_value( _channel_id, reliability_value ) )
501  {
502  // If the reliability routine has a valid result, and it is not No Fault, then we have a Fault
503  bStat = ( reliability_value != RELIABILITY_NO_FAULT_DETECTED );
504  }
505  }
506  bitstring_set_bit( &bit_string, STATUS_FLAG_FAULT, bStat );
507 
508  bitstring_set_bit( &bit_string, STATUS_FLAG_OVERRIDDEN, false );
509  bitstring_set_bit( &bit_string, STATUS_FLAG_OUT_OF_SERVICE, check_point_out_of_service() );
510  apdu_len = encode_application_bitstring( &apdu[0], &bit_string );
511  break;
512 
513  case PROP_EVENT_STATE:
514  bStat = !check_point_quality(); // True for Fault case
515  if ( bStat )
516  {
517  uiVal = EVENT_STATE_FAULT;
518  }
519  else
520  {
521  uiVal = EVENT_STATE_NORMAL;
522  }
523  apdu_len = encode_application_enumerated( apdu, uiVal );
524  break;
525 
526  case PROP_OUT_OF_SERVICE:
527  // Usually false, unless OutOfService exists and is set to true
528  apdu_len = encode_application_boolean( &apdu[0], check_point_out_of_service() );
529  break;
530 
531  case PROP_UNITS:
532  uiVal = UNITS_NO_UNITS; // a default fallback value
533  if ( value_ok && cds_value.type().is_string() )
534  {
535  px::string bacnet_style_unit;
536  bool retval = false;
537  //Map the MCL unit to BACnet style unit
538  retval = _bip_core->_mcl_channel_config->mcl_unit_map( cds_value.as_string().c_str(), bacnet_style_unit );
539  if ( retval )
540  {
541  bStat = bactext_engineering_unit_index( bacnet_style_unit.c_str(), &i );
542  }
543  else
544  {
545  bStat = bactext_engineering_unit_index( cds_value.as_string().c_str(), &i );
546  }
547  if ( bStat )
548  {
549  uiVal = i;
550  }
551  }
552  apdu_len = encode_application_enumerated( &apdu[0], uiVal );
553  break;
554 
555  case PROP_RELIABILITY:
556  // check for any real interesting reliability value mappings; else use the defaults
557  if ( _bip_core->_config->get_bip_routed_device_map()->get_reliability_value( _channel_id, reliability_value ) )
558  {
559  uiVal = reliability_value;
560  }
561  else
562  {
563  bStat = !check_point_quality(); // True for Fault case
564  if ( bStat )
565  {
566  uiVal = RELIABILITY_UNRELIABLE_OTHER;
567  }
568  else
569  {
570  uiVal = RELIABILITY_NO_FAULT_DETECTED;
571  }
572  }
573  apdu_len = encode_application_enumerated( &apdu[0], uiVal );
574  break;
575 
576  // Analog values are handled at the default Object level (non-custom)
577  case PROP_PRESENT_VALUE:
578  if ( !value_ok )// failed access or bad map of value
579  {
580  rpdata->error_class = ERROR_CLASS_PROPERTY;
581  rpdata->error_code = ERROR_CODE_VALUE_NOT_INITIALIZED; // best match of error code
582  }
583  else
584  {
585  apdu_len = get_present_value( rpdata, cds_value, value_ok );
586  // Normally, if wrong type, apdu_len is -1 at this point
587  if ( apdu_len == 0 ) // if encoding failed, assume was wrong type ?
588  {
589  rpdata->error_class = ERROR_CLASS_PROPERTY;
590  rpdata->error_code = ERROR_CODE_INVALID_DATA_TYPE;
591  }
592  }

659 bool CBacnetPointObject::Object_Write_Property( BACNET_WRITE_PROPERTY_DATA *wp_data )
660 {
661  PUSH_FUNC;
662 
663  bool status = false; /* return value */
664  int len = 0;
665  BACNET_APPLICATION_DATA_VALUE value;
666 
667  // Setup the error return; (almost?) always a property error, no default reason
668  wp_data->error_class = ERROR_CLASS_PROPERTY;
669  wp_data->error_code = ERROR_CODE_OTHER; // if this changes, then something set it
670 
671  // Screen out all but the few allowed properties
672  switch ( wp_data->object_property )
673  {
674  case PROP_PRESENT_VALUE:
675  case PROP_OUT_OF_SERVICE:
676  case PROP_COV_INCREMENT:
677  break;
678  default:
679  wp_data->error_code = ERROR_CODE_WRITE_ACCESS_DENIED;
680  return ( false );
681  }
682 
683  /* decode the some of the request */
684  len = bacapp_decode_application_data( wp_data->application_data,
685  wp_data->application_data_len, &value );
686  if ( len == 0 )
687  {
688  // One last screen: is there any payload to be processed?
689  if ( wp_data->application_data_len <= 0 )
690  {
691  SVR_DEBUG( 2, "Invalid Write-Property: no payload" );
692  wp_data->error_code = ERROR_CODE_REJECT_MISSING_REQUIRED_PARAMETER;
693  }
694  else
695  {
696  /* Bad decode, wrong tag or following required parameter missing */
697  wp_data->error_code = ERROR_CODE_INVALID_DATA_TYPE;
698  }
699  return ( false );
700  }
701  else if ( len < 0 )
702  {
703  /* error while decoding - a value larger than we can handle */
704  wp_data->error_code = ERROR_CODE_VALUE_OUT_OF_RANGE;
705  return ( false );
706  }
707 
708  /* Since we are not supporting the writing of array properties, use this: */
709  if ( wp_data->array_index != ( uint32_t ) BACNET_ARRAY_ALL )
710  {
711  wp_data->error_code = ERROR_CODE_PROPERTY_IS_NOT_AN_ARRAY; // or ERROR_CODE_INVALID_ARRAY_INDEX
712  return ( false );
713  }
714 
715  switch ( wp_data->object_property )
716  {
717  case PROP_PRESENT_VALUE:
718  if ( _is_writable )
719  {
720  status = handle_write_present_value( wp_data, value );
721  }
722  // else - caught below
723  // wp_data->error_code = ERROR_CODE_WRITE_ACCESS_DENIED;
724  // ToDo: someday - check mIsOutOfService and allow writing of
725  // the other types like AnalogInput
726  break;
727 
728  // We do allow PROP_OUT_OF_SERVICE to be written to if _allow_out_of_service is true
729  case PROP_OUT_OF_SERVICE:
730  if ( _allow_out_of_service ) // if this can be set
731  {
732  status =
733  WPValidateArgType( &value, BACNET_APPLICATION_TAG_BOOLEAN,
734  &wp_data->error_class, &wp_data->error_code );
735  if ( status )
736  {
737  _is_out_of_service = value.type.Boolean;
738  // Now set it in the CDS (non-persistently)
739 #ifdef TODO_UPDATE_FOR_NEW_CDS
740 
741  elem = _is_out_of_service;
742  CEvent::format_source_string( setSource, CDS_PROTOCOL_BACNET_IP, tk_string(),
743  inet_ntoa( glptr->sourceIpv4 ) );
744  if ( PCDERR_NOERROR == ObjectSetCdsLeaf( glptr->cdsApi, BACNET_ATTR( ATTRIBUTE_OUTOFSERVICE ),
745  elem, TRIBOOL_FALSE, setSource ) )
746 #endif
747  status = true;
748  }
749  }
750  break;
751 
752  case PROP_COV_INCREMENT:
753  if ( get_supports_cov() )
754  {
755  status = handle_write_property( wp_data, value );
756  }
757  // else - caught below
758  // wp_data->error_code = ERROR_CODE_WRITE_ACCESS_DENIED;
759  break;
760 
761  // We have no other "special properties" to manage in this class...
762 
763 
764  default:
765  wp_data->error_code = ERROR_CODE_WRITE_ACCESS_DENIED;
766  break;
767  }
768 
769  if ( !status && ( wp_data->error_code == ERROR_CODE_OTHER ) )
770  {
771  // If we reach this point, and no other error type has been set, it's not writable
772  wp_data->error_code = ERROR_CODE_WRITE_ACCESS_DENIED;
773  }
774  if ( !status )
775  {
776  // Let's log a message
777  SVR_NOTICE( 1, "Error = %s when writing to %s", bactext_error_code_name( wp_data->error_code ), getObject_Name() );
778  }
779 
780  return ( status );
781 }

873 /* Get the present value for a COV notification for this object, and fill out
874  * the two entries in the given value list with its value and state. */
875 bool CBacnetPointObject::get_cov_value_state( BACNET_PROPERTY_VALUE *value_list )
876 {
877  px::CValue obj_value;
878 
879  _has_cov_update = false; // Always clear this flag now
880  bool bResult = _supports_cov && _bip_core->_cds_handle->get_channel_value( _channel_id, obj_value );
881  if ( bResult )
882  {
883  // Borrowing code flow from demo/object/ai.c
884  if ( value_list )
885  {
886  value_list->propertyIdentifier = PROP_PRESENT_VALUE;
887  // Presume for non-array type; array types must override this method with a better implementation
888  value_list->propertyArrayIndex = BACNET_ARRAY_ALL;
889  bResult = encode_present_value( obj_value, value_list->value );
890  value_list->priority = BACNET_NO_PRIORITY;
891  // Now point to the second element, which hopefully exists
892  value_list = value_list->next;
893  }
894  if ( value_list && bResult )
895  {
896  // Update _last_cov_value with this published value for comparison next time
897  _last_cov_value = obj_value;
898 
899  value_list->propertyIdentifier = PROP_STATUS_FLAGS;
900  value_list->propertyArrayIndex = BACNET_ARRAY_ALL;
901  value_list->value.context_specific = false;
902  value_list->value.tag = BACNET_APPLICATION_TAG_BIT_STRING;
903  // Process like we do for case PROP_STATUS_FLAGS: above
904  bitstring_init( &value_list->value.type.Bit_String );
905  // Determine if we are InAlarm:
906  bitstring_set_bit( &value_list->value.type.Bit_String, STATUS_FLAG_IN_ALARM, check_in_alarm_status() );
907  // Bad .Quality is the same as Fault:
908  bitstring_set_bit( &value_list->value.type.Bit_String, STATUS_FLAG_FAULT, !check_point_quality() );
909  bitstring_set_bit( &value_list->value.type.Bit_String, STATUS_FLAG_OVERRIDDEN, false );
910  bitstring_set_bit( &value_list->value.type.Bit_String, STATUS_FLAG_OUT_OF_SERVICE, check_point_out_of_service() );
911  value_list->value.next = NULL;
912  value_list->priority = BACNET_NO_PRIORITY;
913  value_list->next = NULL;
914  }
915  }
916  if ( !bResult )
917  {
918  SVR_NOTICE( 2, "Failed to encode COV value (%s) for %s", obj_value.as_string().c_str(), getObject_Name() );
919  }
920  return ( bResult );
921 }
