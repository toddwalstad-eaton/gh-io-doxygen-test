41 // Creates the initial BACnet Devices for the gateway device and virtual devices.
42 void CBacnetIpConfig::initialize_bacnet_devices()
43 {
44  PUSH_FUNC;
45 
46  bool want_channels = true; // Yes, we definitely want channels
47 
48  // Bind the calls to our member functions, to use in the call to CdsHandler
49  using std::placeholders::_1;
50  using std::placeholders::_2;
51  px::asio::CCdsHandler::cds_device_callback_t f_add_device =
52  std::bind( &CBacnetIpConfig::add_device_callback, this, _1 );
53  px::asio::CCdsHandler::cds_device_callback_t f_update_device =
54  std::bind( &CBacnetIpConfig::update_device_callback, this, _1 );
55  px::asio::CCdsHandler::device_delete_callback_t f_delete_device =
56  std::bind( &CBacnetIpConfig::delete_device_callback, this, _1 );
57  px::asio::CCdsHandler::cds_device_callback_t f_device_state =
58  std::bind( &CBacnetIpConfig::device_state_callback, this, _1 );
59  px::asio::CCdsHandler::cds_ch_settings_callback_t f_ch_settings =
60  std::bind( &CBacnetIpConfig::channel_settings_callback, this, _1, _2 );
61 
62  SVR_DEBUG( 5, "Begin initial BACnet device discovery." );
63  // Then start discovering and adding our devices to our map of BACnet devices
64  _bip_core->_cds_handle->init_device_map( f_add_device, f_update_device,
65  f_delete_device, f_device_state, want_channels, f_ch_settings );
66  // Won't return until it has done the initial scanning of the device tree
67  SVR_DEBUG( 5, "Calling BACnet device map device_count()." );
68  size_t count = _bip_routed_device_map->device_count();
69  SVR_DEBUG( 5, "Finished initial BACnet device discovery; found %zu BACnet devices.", count );
70 }

72 /* Function invoked just before entering the run state. */
73 void CBacnetIpConfig::ready_to_run()
74 {
75  PUSH_FUNC;
76 
77  // Now enter the CDS subscriptions for device changes
78  _bip_core->_cds_handle->subscribe_device_callbacks();
79 
80  // Register our callback for channel value updates
81  using std::placeholders::_1;
82  using std::placeholders::_2;
83  px::asio::CCdsHandler::cds_ch_values_callback_t f_ch_values =
84  std::bind( &CBacnetIpConfig::channel_values_callback, this, _1, _2 );
85 
86  _bip_core->_cds_handle->init_channel_values_callback( f_ch_values );
87 
88 
89  // And kick off another round of static discovery, just to see
90  // if anything has changed since our initial discovery and before
91  // we were subscribed to get changes.
92  // The change signatures will be checked and we can avoid re-discovering unchanged devices.
93  // Start with the root device's data, and kick off the first round of static discovery
94  _bip_core->_cds_handle->do_static_discovery( PX_ID_ROOT );
95 
96  // Won't return until it has done the re-scanning of the device tree
97  SVR_DEBUG( 5, "Calling BACnet device map device_count()." );
98  size_t count = _bip_routed_device_map->device_count();
99  SVR_DEBUG( 5, "Finished rescanning for BACnet devices; found %zu BACnet devices.", count );
100 }
