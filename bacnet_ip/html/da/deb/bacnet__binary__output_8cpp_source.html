197 bool CBacnetBinaryOutputObject::encode_present_value( const px::CValue &cds_value, BACNET_APPLICATION_DATA_VALUE &bac_value )
198 {
199  PUSH_FUNC;
200 
201  // For Binary Values, we simply take True as always Active,
202  // so Polarity = POLARITY_NORMAL
203  uint32_t uiVal = cds_value.as_bool() ? BINARY_ACTIVE : BINARY_INACTIVE;
204  // If was a numeric but maps to a Binary Object type, show a Notice,
205  // except if the UINT value is 0 or 1 - some should-be-bools are done as CDS Multistates
206  if ( cds_value.as_uint32() > 1 )
207  {
208  SVR_NOTICE( 1, "Ill-advised conversion from %u to Binary value for %s",
209  cds_value.as_uint32(), getObject_Name() );
210  }
211  bac_value.tag = BACNET_APPLICATION_TAG_ENUMERATED;
212  bac_value.type.Enumerated = uiVal;
213  bac_value.context_specific = false;
214  bac_value.next = NULL;
215 
216  return ( true );

267 // Handler to get the priority-array value for this object type
268 int CBacnetBinaryOutputObject::get_priority_array_values( BACNET_READ_PROPERTY_DATA *rpdata,
269  const px::CValue &cdsValue, bool value_ok )
270 {
271  PUSH_FUNC;
272 
273  ( void )cdsValue;
274 
275  if ( !value_ok )
276  {
277  rpdata->error_class = ERROR_CLASS_PROPERTY;
278  rpdata->error_code = ERROR_CODE_READ_ACCESS_DENIED;
279  return ( -1 );
280  }
281 
282  size_t i = 0;
283  int len = 0;
284  int apdu_len = 0;
285  uint8_t *apdu = rpdata->application_data;
286  uint32_t prio_val = 0;
287  bool is_valid_prio_val = false;
288 
289  // Array element zero is the number of elements in the array
290  if ( rpdata->array_index == 0 )
291  {
292  apdu_len = encode_application_unsigned( &apdu[0], BACNET_MAX_PRIORITY );
293  }
294  // if no index was specified, then try to encode the entire list
295  // into one packet.
296  else if ( rpdata->array_index == BACNET_ARRAY_ALL )
297  {
298  for ( i = BACNET_MIN_PRIORITY; i <= BACNET_MAX_PRIORITY; i++ )
299  {
300  // Encode a bool value if the value isn't equal to the cancel value AND the value at this index
301  // is supported. Else, encode null.
302  is_valid_prio_val = _priority_array->get_priority_value( i, prio_val );
303  if ( is_valid_prio_val && ( prio_val != _priority_array->get_priority_cancel() ) )
304  {
305  len = encode_application_enumerated( &apdu[apdu_len], prio_val );
306  }
307  else
308  {
309  len = encode_application_null( &apdu[apdu_len] );
310  }
311  // add it if we have room
312  if ( ( apdu_len + len ) < MAX_APDU )
313  {
314  apdu_len += len;
315  }
316  else
317  {
318  rpdata->error_class = ERROR_CLASS_SERVICES;
319  rpdata->error_code = ERROR_CODE_NO_SPACE_FOR_OBJECT;
320  apdu_len = BACNET_STATUS_ERROR;
321  break;
322  }
323  }
324  }
325  else
326  {
327  if ( rpdata->array_index <= BACNET_MAX_PRIORITY )
328  {
329  is_valid_prio_val = _priority_array->get_priority_value( rpdata->array_index, prio_val );
330  if ( is_valid_prio_val && ( prio_val != _priority_array->get_priority_cancel() ) )
331  {
332  apdu_len = encode_application_enumerated( &apdu[0], prio_val );
333  }
334  else
335  {
336  apdu_len = encode_application_null( &apdu[0] );
337  }
338  }
339  else
340  {
341  rpdata->error_class = ERROR_CLASS_PROPERTY;
342  rpdata->error_code = ERROR_CODE_INVALID_ARRAY_INDEX;
343  apdu_len = BACNET_STATUS_ERROR;
344  }
345  }
346 
347  return ( apdu_len );
348 }

380 // Handles the write to present-value for this object type.
381 bool CBacnetBinaryOutputObject::handle_write_present_value( BACNET_WRITE_PROPERTY_DATA *wp_data,
382  BACNET_APPLICATION_DATA_VALUE &value )
383 {
384  PUSH_FUNC;
385 
386  // return value
387  bool status = false;
388 
389  assert( wp_data->error_class == ERROR_CLASS_PROPERTY );
390  assert( wp_data->error_code == ERROR_CODE_OTHER ); // set to this on entry
391 
392  // First, check the type for correctness and range check as appropriate,
393  // then store the value in our CValue for writing
394  px::CValue value_to_write_to_prio_array;
395  // As per the standard, the Present_value property of Binary Output object is of Enumerated data type
396  // and Boolean value's i.e. true/false can cause problem during encoding. Hence it shall be rejected.
397  // Only enumerated data type values shall be accepted.
398  if ( value.tag == BACNET_APPLICATION_TAG_ENUMERATED )
399  {
400  // priority = wp_data->priority; -- see Binary_Value_Write_Property() for
401  // details about handling priority, arrays, and NULLs
402  // ToDo: should we allow BINARY_NULL (255) to be used here ??
403  if ( ( value.type.Enumerated <= MAX_BINARY_PV ) || ( value.type.Enumerated == BINARY_NULL ) )
404  {
405  value_to_write_to_prio_array = value.type.Enumerated;
406  status = true;
407  }
408  else
409  {
410  SVR_ERR( 1, "Invalid BACnet value out of range %s", bactext_application_tag_name( value.tag ) );
411  wp_data->error_code = ERROR_CODE_VALUE_OUT_OF_RANGE;
412  }
413  }
414  else if ( value.tag == BACNET_APPLICATION_TAG_NULL )
415  {
416  // This is the cancel case; substitute BINARY_NULL when writing here
417  value_to_write_to_prio_array = BINARY_NULL;
418  status = true;
419  }
420  else
421  {
422  SVR_ERR( 1, "Invalid BACnet value type %s", bactext_application_tag_name( value.tag ) );
423  wp_data->error_code = ERROR_CODE_INVALID_DATA_TYPE;
424  }
425 
426  if ( status )
427  {
428  // Using priority arrays, set the channel value if necessary
429  size_t priority_to_set = ( size_t )wp_data->priority;
430  ssize_t priority_active = 0;
431  bool is_prio_set = _priority_array->set_priority_value( priority_to_set,
432  value_to_write_to_prio_array.as_uint32() );
433 
434  if ( is_prio_set )
435  {
436  uint32_t highest_prio_val = _priority_array->get_highest_priority_value();
437  priority_active = _priority_array->get_highest_priority_index_active();
438  if ( priority_active == -1 )
439  {
440  // No priority active, relinquish default is now active, use 17 as lowest priority
441  priority_active = 17;
442  }
443 
444  // We need to be writing a Boolean value here in all cases
445  px::CValue value_to_write_to_cds( ( bool )highest_prio_val );
446  // Execute command or write channel for highest priority value
447  if ( _use_cmd )
448  {
449  status = _bip_core->_cds_handle->exec_output_cmd( _parent_id, mObject_CommandName,
450  value_to_write_to_cds, priority_active );
451  }
452  else
453  {
454  SVR_INFO( 4, "Setting channel %u to %s", ( uint32_t )_channel_id, value_to_write_to_cds.as_string().c_str() );
455  _bip_core->_cds_handle->set_channel_value( _channel_id, value_to_write_to_cds );
456  // Now as the channel value has been updated, determine whether this change constitutes a COV update or not
457  if ( value_to_write_to_cds != _last_cov_value )
458  {
459  SVR_DEBUG( 7, "Updated value %s differs from last published %s for %s", value_to_write_to_cds.as_string().c_str(), _last_cov_value.as_string().c_str(), getObject_Name() );
460  // So set the flag that marks our desire to have a COV Notification sent
461  _has_cov_update = true;
462  }
463  status = true;
464  }
465  }
466  else
467  {
468  status = false;
469  }
470  }
471 
472  return ( status );
473 }
