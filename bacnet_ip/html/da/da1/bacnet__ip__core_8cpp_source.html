147 void CBacnetIpCore::execute()
148 {
149  PUSH_FUNC;
150  assert( _options ); // we've had problems long before now if this isn't valid!
151 
152  if ( !_settings->ready_to_operate() )
153  {
154  return;
155  }
156 
157  std::shared_ptr<px::asio::CIoManager> manager;
158  std::shared_ptr<CBacnetRouting> bac_routing;
159  size_t good_count = 0;
160  try
161  {
162  manager.reset( new px::asio::CIoManager( *this, *_cds_handle, _options->watchdog_timeout ) );
163 
164  /* First, instantiate the BACnet/IP UDP Protocol Handler we will use
165  * Give it a null transactor at this time; will make it real when cloning later
166  * But this is our one chance to enter the routed_network_number, since
167  * the CUdpListener will clone its own copy when constructed and use that to handle incoming requests.
168  */
169  _bip_handler.reset( new CBacnetIpHandler( getptr(), nullptr, _settings->routed_network_number ) );
170 
171  uint16_t internal_port = _settings->_bip_internal_port;
172  _udp_listener = px::asio::CUdpListener::create( *manager, *_bip_handler, internal_port );
173  manager->add( _udp_listener );
174  _bip_handler->update_transactor( _udp_listener.get() );
175 
176  // Start the task timer that will pace ... (TBD)
177  _msec_timer = px::asio::CMsecTimer::create( *manager, this, _CORE_TIMEBASE_MSECS );
178 
179  if ( _udp_listener.get() && _udp_listener->is_open() )
180  {
181  SVR_INFO( 1, "Opened local UDP Port %u to listen for BACnet/IP requests.", internal_port );
182  // Note: Port 0xBAC0 (47808) is the standard port for BACnet/IP messages. We will listen
183  // to a different port here (perhaps 57808) that will, due to Linux network configuration,
184  // receive messages port forwarded from the real port (47808). This scheme allows changing the
185  // real port (47808) to something else without code here needing to care. We always listen
186  // to the same "internal_port". External network magic takes care of passing messages from the
187  // real port to the internal port - or blocking them, as a firewall measure.
188 
189  // If we get here, we're good to proceed
190  good_count++;
191 
192  // Initialize to our starting routed network number; will have to update later if user changes it.
193  bac_routing.reset( new CBacnetRouting( getptr() ) );
194 
195  _bip_handler->init_bacnet_services();
196 
197  // Redo our device scan and subscribe to updates
198  _config->ready_to_run();
199 
200  // If the UDP Listener is open for business, broadcast our I-Ams
201  if ( _settings->ready_to_operate() )
202  {
203  _bip_handler->send_i_ams();
204  }
205 
206  // Start our BACnet/IP services and their timers
207 
208  // Start the timer for periodic BBMD registration, in case it is needed
209  // We will use a period of 1/3 the BBMD TTL to give us 3 chances to get
210  // a foreign registration message through before the BBMD's TTL timer expires for our registration
211  unsigned int msec_timeout = _settings->bbmd_time_to_live * 1000 / 3;
212  _bbmd_timer = px::asio::CMsecTimer::create( *manager, _bip_handler.get(), msec_timeout );
213 
214  // Create the COV handler, initialize it, and kick off its timer
215  _cov_handler.reset( new BacnetCovHandler( manager, getptr() ) );
216  _cov_handler->init_cov_services();
217  _cov_timer = px::asio::CMsecTimer::create( *manager, _cov_handler.get(), _cov_handler->_COV_TIMEBASE_MSECS );
218  }
219  else
220  {
221  SVR_ERR( 1, "Failed to open local UDP Port %u to listen for BACnet/IP requests.", internal_port );
222  }
223  }
224  catch ( const boost::system::system_error &e )
225  {
226  // TODO: handle this
227  SVR_ERR( 0, "System error encountered in setting up I/O manager: %s", e.what() );
228  good_count = 0; // ensure that we close out below
229  }
230 
231  if ( good_count == 0 )
232  {
233  // Our attempt to open the serial port(s) failed;
234  // Free up our manager and exit below
235  manager->quit();
236  manager.reset();
237  }
238 
239  if ( manager.get() ) // if we created a valid IO Manager
240  {
241  // Write a copy of our BACnetSettings and BACnetDatabase OOBs,
242  // if revised, given our discovered and configured devices
243  _settings->update_full_bacnet_oob();
244  _is_running = true;
245  // From here on out, any device additions or removals will need another update to be written
246 
247  try
248  {
249  SVR_DEBUG( 2, "Entering the run() loop" );
250  // This will block indefinitely, and do all the real work of
251  // the server
252  ready();
253  manager->run();
254  }
255  catch ( const boost::system::system_error &e )
256  {
257  // TODO: handle this
258  SVR_ERR( 0, "System error encountered in I/O manager" );
259  }
260 
261  _is_running = false;
262  SVR_DEBUG( 2, "Exiting the run() loop" );
263 
264  // Closeout, if we haven't already, and free our resources.
265  bac_routing.reset();
266  shutdown();
267  manager->quit();
268  manager.reset();
269  }
