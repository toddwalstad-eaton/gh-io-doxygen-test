292

295  auto_assign_base_id = bac_ntwk->auto_assign_base_id();

303  SVR_INFO( 5, "BACnet/IP base ID for auto-assigning set to: %u", auto_assign_base_id );

420 {

436  }
437

439  for ( const px::proto::BACnetDevice &bdproto : bacnet_db->virtual_devices() )

442  if ( !oob_bdev )
443  {
444  oob_bdev.reset( new COOB_BACnetDevice( bdproto, shared_from_this() ) );
445  assert( oob_bdev );
446  }
447  else
448  {
449  _database_needs_update |= oob_bdev->merge_from_oob_data( bdproto );
450  }
451  if ( auto_assign_bacnet_virtual_device_ids && ( auto_assign_base_id > 0 ) )
452  {
453  // Make sure that autoIDs are used; override any user-supplied
454  uint32_t desired_obj_id = auto_assign_base_id + oob_bdev->_device_map_index;
455  if ( oob_bdev->object_id != desired_obj_id )
456  {
457  // Uptick the database_revision counter for this change
458  oob_bdev->_database_revision++;
459  SVR_INFO( 2, "Revising instance %zu Object-ID from %u to %u, DB Rev = %u",
460  oob_bdev->_device_map_index, oob_bdev->object_id, desired_obj_id,
461  oob_bdev->_database_revision );
462  oob_bdev->object_id = desired_obj_id;
463  _database_needs_update = true;
464  }
465  }
466  _virtual_decommissioned_device_map[bdproto.device_id()] = oob_bdev;
467  }

492  if ( !_gateway || ( _gateway->object_id == 0 ) || ( _gateway->object_name.empty() ) )
493  {
494  // Create our COOB_BACnetDevice device (first time) and set about initializing it
495  _gateway.reset( new COOB_BACnetDevice( px_device.id(), inst_num ) );
496 
497  // If auto assigning the gateway, do that now, else use the existing default
498  if ( auto_assign_bacnet_gateway_id )
499  {
500  _default_gateway_id = auto_assign_base_id + GATEWAY_IDX;
501  }
502  _gateway->initialize_from_px_device( px_device, _default_gateway_id, _gateway_mac, BIP_MAC_LEN );
503  _database_needs_update = true;
504  }
505 
506  // Else, does the CDS have anything to add to our device settings from the OOB? No, nothing at this time.
507 
508  return ( _gateway );
509  }
510 
511  // Else, Handle non-gateway (ie, routed "virtual") devices
512 
513  // The device could have its OOB data created already.
514  // For example, a device becomes enabled from disabled or communicating from non-communicating.
515  // In such a case, we only need to find it in the proper map.
516  std::shared_ptr<COOB_BACnetDevice> oob_bdev = check_for_oob_device_in_map( px_device );
517  if ( oob_bdev )
518  {
519  // Found, so we don't need to update the CDS; just return
520  SVR_DEBUG( 8, "Looked up BACnetDevice instance %zu - %s", oob_bdev->_device_map_index, oob_bdev->object_name.c_str() );
521  return ( oob_bdev );
522  }
523 
524  // If we are here, we need to create the OOB data
525  inst_num = get_next_inst_num(); // This is qualified and available instance (or 0)
526  if ( inst_num == 0 )
527  {
528  // Then we can't add this device; bail out
529  SVR_FWARN( 1, "Failed to add Device %" PRIu64 " to " TAG_BIP_SETTINGS, dev_id );
530  return ( oob_bdev ); // Still an empty pointer
531  }
532 
533  // Then we will need to write out the BACnetDatabase for this device
534  _database_needs_update = true;
535  // Create a new instance; let it fill out the object name and MAC
536  uint32_t dev_obj_id = auto_assign_base_id + inst_num;
537  oob_bdev.reset( new COOB_BACnetDevice( dev_id, inst_num, dev_obj_id ) );
538  oob_bdev->initialize_from_px_device( px_device, dev_obj_id, mac_template, BIP_MAC_LEN, inst_num );
539 
540  // Now that the OOB Device is created, add it to BACnetSettings OOB maps
541  if ( is_commissioned_device( px_device ) )
542  {
543  _virtual_device_map[dev_id] = oob_bdev;
544  SVR_DEBUG( 8, "Added commissioned BACnetDevice %" PRIu64 " instance %zu - %s",
545  dev_id, oob_bdev->_device_map_index, oob_bdev->object_name.c_str() );
546  }
547  else
548  {
549  _virtual_decommissioned_device_map[dev_id] = oob_bdev;
550  SVR_DEBUG( 8, "Added decommissioned BACnetDevice %" PRIu64 " instance %zu - %s",
551  dev_id, oob_bdev->_device_map_index, oob_bdev->object_name.c_str() );
552  }
553  // Does the OOB need updating in the CDS now?
554  if ( _bip_core->is_running() )
555  {
556  update_full_bacnet_oob();
557  }
558  return ( oob_bdev );
559 }

586 // Gets the next instance number we should use, for a new virtual device instance.
587 size_t CBacnetIpSettings::get_next_inst_num()
588 {
589  PUSH_FUNC;
590 
591  bool done = false;
592  size_t starting_num = _last_inst_num;
593  do
594  {
595  done = true;
596  _last_inst_num++;
597  // (MAX_BACNET_IP_DEVICES+2) is as many devices as we can address in 8 bits,
598  // and 0 and 1 are reserved.
599  if ( _last_inst_num >= ( MAX_BACNET_IP_DEVICES + 2 ) )
600  {
601  _last_inst_num = 2; // Skip 0 and the gateway (1)
602  }
603  // If we wrap back to the starting number, we are done (and failed)
604  if ( _last_inst_num == starting_num )
605  {
606  SVR_WARN( 1, "No available slots to use for next instance number; cannot add this device" );
607  return ( 0 );
608  }
609 
610  // Do we have the next instance already?
611  for ( auto entry : _virtual_device_map )
612  {
613  if ( entry.second->_device_map_index == _last_inst_num )
614  {
615  done = false;
616  break;
617  }
618  }
619  for ( auto entry : _virtual_decommissioned_device_map )
620  {
621  if ( entry.second->_device_map_index == _last_inst_num )
622  {
623  done = false;
624  break;
625  }
626  }
627  }
628  while ( !done );
629  SVR_DEBUG( 8, "Allocating next instance number: %zu", _last_inst_num );
630  return ( _last_inst_num );
631 }

702  * writing the update to the CDS. */
703 void CBacnetIpSettings::update_full_bacnet_oob()
704 {

706 
707  if ( _settings_need_update )
708  {
709  // Get the BACnetSettings OOB for B/IP:
710  px::proto::toolkitStruct bac_settings;
711  px::proto::BACnetNetwork *bac_ntwk = bac_settings.mutable_bacnet_settings()->mutable_network();
712 
713  // Update the basic settings
714 
715  bac_ntwk->set_routed_network_number( routed_network_number );
716  bac_ntwk->set_auto_assign_base_id( auto_assign_base_id );
717  bac_ntwk->set_auto_assign_bacnet_gateway_id( auto_assign_bacnet_gateway_id );
718  bac_ntwk->set_auto_assign_bacnet_virtual_device_ids( auto_assign_bacnet_virtual_device_ids );
719  bac_ntwk->set_bbmd_enabled( bbmd_enabled );
720  bac_ntwk->set_bbmd_time_to_live( bbmd_time_to_live );
721  bac_ntwk->set_bbmd_ipv4_address( bbmd_ipv4_address.c_str() );
722 
723  // Make sure that the gateway and virtual_devices are cleared, since we never get them from here anymore
724  bac_ntwk->clear_gateway();
725  bac_ntwk->clear_virtual_devices();
726 
727  // For mac_template, we just store it as a colon-separated hex string
728  const char *pmt = COOB_BACnetDevice::get_mac_as_string( mac_template, BIP_MAC_LEN );
729  bac_ntwk->set_mac_template_virtual_devices( pmt );
730

732  if ( bbmd_services_enabled )
733  {
734  bac_ntwk->set_bbmd_services_enabled( bbmd_services_enabled );
735  } // Else, the protobuf default for this bool is false
736  if ( !dcc_password.empty() )
737  {
738  bac_ntwk->set_dcc_password( dcc_password.c_str() );
739  }
740  bac_ntwk->set_only_shared_persisted_properties( _only_shared_persisted_properties );
741 
742  SVR_DEBUG( 2, "Updating the BACnetSettings OOB" );
743  _bip_core->_cds_handle->set_oob( TAG_BIP_SETTINGS, bac_settings, true ); // persist = true
744  _settings_need_update = false;
745  }
746 
747  if ( _database_needs_update )

757  _gateway->initialize_oob_device_data( oob_dev );

761  for ( auto entry : _virtual_device_map )
762  {
763  if ( _bip_core->_cds_handle->get_map_device( entry.first, map_pos ) )
764  {
765  // Add this instance
766  oob_dev = bacnet_db->add_virtual_devices();
767  entry.second->initialize_oob_device_data( oob_dev );
768  }
769  }
770  for ( auto entry : _virtual_decommissioned_device_map )
771  {
772  if ( _bip_core->_cds_handle->get_map_device( entry.first, map_pos ) )
773  {
774  // Add this instance
775  oob_dev = bacnet_db->add_virtual_devices();
776  entry.second->initialize_oob_device_data( oob_dev );
777  }
778  }
779 
780  SVR_DEBUG( 2, "Updating the BACnetDatabase OOB for %zu active and %zu inactive devices",
781  _virtual_device_map.size(), _virtual_decommissioned_device_map.size() );
782  _bip_core->_cds_handle->set_oob( TAG_BIP_DATABASE, bacnet_db_oob, true ); // persist = true
783  _database_needs_update = false;

842 /* Static function to use the the ethernet iMAC to get the default gateway 22-bit Object ID
843  * from the iMAC's lowest 22 bits, and to also create the MAC Template
844  * for the routed devices from it. */
845 uint32_t CBacnetIpSettings::get_defaults_from_imac( uint8_t *mac_buf, size_t buf_len,
846  uint8_t *mac_template_buf )
847 {
848  PUSH_FUNC;
849 
850  assert( ( mac_buf != nullptr ) && ( mac_template_buf != nullptr ) );
851  assert( buf_len >= 6 );
852 
853  uint32_t obj_id = ( ( mac_buf[buf_len - 3] & 0x3F ) << 16 ) +
854  ( mac_buf[buf_len - 2] << 8 ) + mac_buf[buf_len - 1];
855 
856  if ( buf_len >= 6 )
857  {
858  mac_template_buf[0] = mac_buf[3];
859  mac_template_buf[1] = mac_buf[4];
860  mac_template_buf[2] = mac_buf[5];
861 
862  mac_template_buf[3] = 0; // we want this byte to be filled in with the instance numbers later
863  mac_template_buf[4] = 0xBA; // The "port" bytes, using default BACnet port number 47808
864  mac_template_buf[5] = 0xC0;
865 
866  uint8_t *mb = mac_buf; // much shorter names
867  uint8_t *mt = mac_template_buf;
868  SVR_DEBUG( 2, "Successfully converted iMAC %02X%02X%02X%02X%02X%02X to default ObjID = %u"
869  " and MAC template %02X:%02X:%02X:%02X:%02X:%02X",
870  mb[0], mb[1], mb[2], mb[3], mb[4], mb[5], obj_id,
871  mt[0], mt[1], mt[2], mt[3], mt[4], mt[5] );
872  }
873 
874  return ( obj_id );
875 }
