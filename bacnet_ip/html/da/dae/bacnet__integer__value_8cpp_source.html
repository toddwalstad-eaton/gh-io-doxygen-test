156 // Handler for the derived class to supply values for its more special properties.
157 int CBacnetIntegerValue::handle_read_property( BACNET_READ_PROPERTY_DATA *rpdata,
158  const px::CValue & /* cdsValue */, bool /* value_ok */ )
159 {
160  PUSH_FUNC;
161 
162  int apdu_len = -2; // default is that we don't handle this property here
163  uint8_t *apdu = rpdata->application_data;
164 
165  switch ( rpdata->object_property )
166  {
167  case PROP_COV_INCREMENT:
168  apdu_len = encode_application_unsigned( apdu, _cov_increment );
169  SVR_INFO( 5, "Read COV_Increment = %u for Object_Name %s", _cov_increment, getObject_Name() );
170  break;
171 
172  case PROP_MAX_PRES_VALUE:
173  if ( _max_val.has_value() )
174  {
175  apdu_len = encode_application_signed( apdu, _max_val.value().as_int32() );
176  SVR_INFO( 5, "Read Max_Pres_Value = %d for Object_Name %s", _max_val.value().as_int32(), getObject_Name() );
177  }
178  break;
179 
180  case PROP_MIN_PRES_VALUE:
181  if ( _min_val.has_value() )
182  {
183  apdu_len = encode_application_signed( apdu, _min_val.value().as_int32() );
184  SVR_INFO( 5, "Read Min_Pres_Value = %d for Object_Name %s", _min_val.value().as_int32(), getObject_Name() );
185  }
186  break;
187 
188  // We have no other "special properties" to manage in this class...
189  default:
190  // apdu_len already set to -2, so do nothing more
191  break;
192  }
193 
194  return ( apdu_len );
195 }

283 void CBacnetIntegerValue::update_optional_properties( const px::CChannelSettings &ch_settings )
284 {
285  PUSH_FUNC;
286  const uint8_t old_prop_size = sizeof( IntegerValueUnitsProps ) / sizeof( BacPropEntry_t );
287  uint8_t new_prop_size = old_prop_size;
288  uint8_t prop_index = 0;
289 
290  if ( ch_settings.maximum.is_initialized() )
291  {
292  _max_val = ch_settings.maximum;
293  new_prop_size++;
294  }
295  else
296  {
297  _max_val.reset();
298  }
299 
300  if ( ch_settings.minimum.is_initialized() )
301  {
302  _min_val = ch_settings.minimum;
303  new_prop_size++;
304  }
305  else
306  {
307  _min_val.reset();
308  }
309  // clear the old list and create a new list as channel settings has changed
310  if ( ( _integerValueUnitsExtProps != nullptr ) && _extProps_is_local )
311  {
312  delete _integerValueUnitsExtProps;
313  _integerValueUnitsExtProps = 0;
314  _extProps_is_local = false;
315  }
316 
317  // Check if old and new sizes are same. If same, then no need to create local list. Point to the global list instead.
318  if ( old_prop_size == new_prop_size )
319  {
320  _integerValueUnitsExtProps = IntegerValueUnitsProps;
321  }
322  else
323  {
324  _integerValueUnitsExtProps = new BacPropEntry_t[new_prop_size];
325 
326  if ( _integerValueUnitsExtProps == 0 )
327  {
328  SVR_ERR( 2, "Could not create extended property array for %s", getObject_Name() );
329  _integerValueUnitsExtProps = nullptr;
330  }
331  else
332  {
333  _extProps_is_local = true; // Indicating that we have locally alloated the list. SHould be released during destruction
334  // Copy all entries expect the last entry, as it will be added at the end
335  for ( prop_index = 0; prop_index < ( old_prop_size - 1 ); prop_index++ )
336  {
337  _integerValueUnitsExtProps[prop_index] = IntegerValueUnitsProps[prop_index];
338  }
339  if ( _max_val.has_value() )
340  {
341  _integerValueUnitsExtProps[prop_index++] = IntegerValueUnitsMaxProp;
342  }
343  if ( _min_val.has_value() )
344  {
345  _integerValueUnitsExtProps[prop_index++] = IntegerValueUnitsMinProp;
346  }
347  // Add the last entry
348  _integerValueUnitsExtProps[prop_index] = IntegerValueUnitsProps[( old_prop_size - 1 )];
349  }
350  }
351 
352  /* Now create the right pointer for the Optional Property List (integer array), based on the above. */
353  if ( _min_val.has_value() )
354  {
355  if ( _max_val.has_value() )
356  {
357  // This is the fullest list
358  _integerValueOptionalProps = &MinMaxOptionalPropList[0];
359  }
360  else
361  {
362  _integerValueOptionalProps = &MinNomaxOptionalPropList[0];
363  }
364  }
365  else if ( _max_val.has_value() )
366  {
367  _integerValueOptionalProps = &MinMaxOptionalPropList[1];
368  }
369  else
370  {
371  _integerValueOptionalProps = &MinNomaxOptionalPropList[1];
372  }
373 }

418 // Handles the write to present-value for this object type.
419 bool CBacnetIntegerValue::handle_write_present_value( BACNET_WRITE_PROPERTY_DATA *wp_data,
420  BACNET_APPLICATION_DATA_VALUE &value )
421 {
422  PUSH_FUNC;
423  bool status = false; /* return value */
424 
425  // This is a simplified version of Analog_Value_Write_Property() and the rest
426  // Since we don't do object arrays, we don't worry about the object_index
427  // Since we don't do priority arrays, we don't worry about the priority
428  // Since we don't need to null out values in a priority array, we
429  // don't worry about BACNET_APPLICATION_TAG_NULL
430 
431  assert( wp_data->error_class == ERROR_CLASS_PROPERTY );
432  assert( wp_data->error_code == ERROR_CODE_OTHER ); // set to this on entry
433 
434  // First, check the type for correctness and range check as appropriate,
435  // then store the value in our CValue for writing
436  px::CValue value_to_write;
437  if ( ( value.tag == BACNET_APPLICATION_TAG_UNSIGNED_INT ) && ( getObject_Type() == OBJECT_POSITIVE_INTEGER_VALUE ) )
438  {
439  if ( _min_val.has_value() )
440  {
441  // Perform range check before updating the present_value
442  if ( value.type.Unsigned_Int >= _min_val.value().as_uint32() )
443  {
444  value_to_write = value.type.Unsigned_Int;
445  status = true;
446  }
447  else
448  {
449  wp_data->error_code = ERROR_CODE_VALUE_OUT_OF_RANGE;
450  SVR_DEBUG( 1, "Present value update failed for %s with error %s", getObject_Name(), bactext_error_code_name( wp_data->error_code ) );
451  status = false;
452  return ( status );
453  }
454  }
455  if ( _max_val.has_value() )
456  {
457  // Perform range check before updating the present_value
458  if ( value.type.Unsigned_Int <= _max_val.value().as_uint32() )
459  {
460  value_to_write = value.type.Unsigned_Int;
461  status = true;
462  }
463  else
464  {
465  wp_data->error_code = ERROR_CODE_VALUE_OUT_OF_RANGE;
466  SVR_DEBUG( 1, "Present value update failed for %s with error %s", getObject_Name(), bactext_error_code_name( wp_data->error_code ) );
467  status = false;
468  return ( status );
469  }
470  }
471  // Range check has been performed and validated if corresponding channel in CDS has min/max defined thus update the value as it is.
472  value_to_write = value.type.Unsigned_Int;
473  status = true;
474  }
475  else if ( ( value.tag == BACNET_APPLICATION_TAG_SIGNED_INT ) && ( getObject_Type() == OBJECT_INTEGER_VALUE ) )
476  {
477  if ( _min_val.has_value() )
478  {
479  // Perform range check before updating the present_value
480  if ( value.type.Signed_Int >= _min_val.value().as_int32() )
481  {
482  value_to_write = value.type.Signed_Int;
483  status = true;
484  }
485  else
486  {
487  wp_data->error_code = ERROR_CODE_VALUE_OUT_OF_RANGE;
488  SVR_DEBUG( 1, "Present value update failed for %s with error %s", getObject_Name(), bactext_error_code_name( wp_data->error_code ) );
489  status = false;
490  return ( status );
491  }
492  }
493  if ( _max_val.has_value() )
494  {
495  // Perform range check before updating the present_value
496  if ( value.type.Signed_Int <= _max_val.value().as_int32() )
497  {
498  value_to_write = value.type.Signed_Int;
499  status = true;
500  }
501  else
502  {
503  wp_data->error_code = ERROR_CODE_VALUE_OUT_OF_RANGE;
504  SVR_DEBUG( 1, "Present value update failed for %s with error %s", getObject_Name(), bactext_error_code_name( wp_data->error_code ) );
505  status = false;
506  return ( status );
507  }
508  }
509  value_to_write = value.type.Signed_Int;
510  status = true;
511  }
512  else
513  {
514  wp_data->error_code = ERROR_CODE_INVALID_DATA_TYPE;
515  }
516 
517  if ( status )
518  {
519  if ( _use_cmd )
520  {
521  status = _bip_core->_cds_handle->exec_device_cmd(
522  _parent_id,
523  mObject_CommandName,
524  value_to_write );
525  }
526  else
527  {
528  _bip_core->_cds_handle->set_channel_value( _channel_id, value_to_write );
529  // Now as the channel value has been updated, determine whether this change constitutes a COV update or not
530  uint32_t diff = std::abs( value_to_write.as_int32() - _last_cov_value.as_int32() );
531  if ( diff >= _cov_increment )
532  {
533  SVR_DEBUG( 7, "Updated value %d differs by >= %u from last published %d for %s", value_to_write.as_int32(),
534  _cov_increment, _last_cov_value.as_int32(), getObject_Name() );
535  // So set the flag that marks our desire to have a COV Notification sent
536  _has_cov_update = true;
537  }
538  status = true;
539  }
540  }
541 
542  return ( status );
543 }
