76 // Reads the MqttBrokerSettings OOB object and puts it into a map of GenericTableEntry entries.
77 display_map_t CMqttBrokerConfCdsHandler::get_mqtt_display_map( bool want_disabled_items )
78 {
79  PUSH_FUNC;
80 
81  px::proto::toolkitStruct oob_data;
82  display_map_t dismap;
83  bool success = get_oob( _mqtt_data_tag, oob_data );
84  if ( success && ( oob_data.generic_config_table_size() > 0 ) )
85  {
86  for ( const px::proto::GenericConfigFileEntry &entry : oob_data.generic_config_table() )
87  {
88  if ( !want_disabled_items &&
89  ( ( entry.display_hint() == px::proto::DisplayHint::DISPLAY_DISABLED ) ||
90  ( entry.display_hint() == px::proto::DisplayHint::DISPLAY_HIDDEN_DISABLED ) ) )
91  {
92  SYSLOG_DEBUG( 5, "Skipping Disabled item %s", entry.key_label().c_str() );
93  }
94  else
95  {
96  bool inserted;
97  std::tie( std::ignore, inserted ) = dismap.insert( std::make_pair( entry.rank(), entry ) );
98  if ( inserted )
99  {
100  SYSLOG_DEBUG( 4, "Adding item [%u] %s", entry.rank(), entry.key_label().c_str() );
101  }
102  else
103  {
104  SYSLOG_INFO( 2, "Failed to Add item [%u] %s due to existing %u entry",
105  entry.rank(), entry.key_label().c_str(), entry.rank() );
106  }
107  }
108  }
109  }
110  else
111  {
112  SYSLOG_ERR( 0, "Input Failure: no generic_table data in %s", _mqtt_data_tag.c_str() );
113  }
114  return ( dismap );
115 }
