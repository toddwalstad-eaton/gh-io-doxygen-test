150 bool CMqttBrokerSettings::update_key_label( bool unset,
151  bool &oob_change,
152  std::string &key_label,
153  px::proto::toolkitStruct &oob_data )
154 {
155  PUSH_FUNC;
156 
157  std::vector<std::string> label_vec;
158  px::safe_line_split( label_vec, key_label, "," );
159 
160  if ( !label_vec.size() )
161  {
162  SYSLOG_ERR( 1, "Empty MQTT Broker key_label\n" );
163  return ( false );
164  }
165 
166  auto key_label_it = key_label_map.find( label_vec[0] );
167  if ( key_label_it == key_label_map.end() )
168  {
169  SYSLOG_ERR( 1, "Did not find MQTT Broker key_label %s in the key_label_map", label_vec[0].c_str() );
170  return ( false );
171  }
172 
173  SYSLOG_DEBUG( 4, " key_label [%s], rank %u\n", key_label_it->first.c_str(), key_label_it->second.rank );
174 
175  px::proto::GenericConfigFileEntry *entry;
176 
177  // Search by the GenericConfigFileEntry 'rank' attribute
178  for ( int j = 0; j < oob_data.generic_config_table().size(); j++ )
179  {
180  entry = oob_data.mutable_generic_config_table( j );
181  if ( !entry )
182  {
183  continue;
184  }
185 
186  if ( entry->has_rank() && ( entry->rank() == key_label_it->second.rank ) )
187  {
188  SYSLOG_DEBUG( 4, " entry->has_value %u, entry->has_rank %u, entry->rank %u\n",
189  entry->has_value(), entry->has_rank(), entry->rank() );
190 
191  // Handle entry value set/unset
192  // Filter out log_XXX types, as we do not want to change the value for these items.
193  if ( ( label_vec.size() > 1 ) && entry->has_value() && ( ( entry->rank() < LOG_RANK_MIN ) || ( entry->rank() > LOG_RANK_MAX ) ) )
194  {
195  px::proto::DataValue *value = entry->mutable_value();
196  SYSLOG_DEBUG( 4, " value [%s], has_str_val %u, has_uint32_val %u\n", label_vec[1].c_str(), value->has_str_val(), value->has_uint32_val() );
197 
198  entry->set_display_hint( key_label_it->second.disable );
199 
200  if ( unlikely( value == nullptr ) )
201  {
202  SYSLOG_ERR( 1, "Protobuf mutable_value failure, for key_label %s", key_label_it->first.c_str() );
203  }
204  else
205  {
206  if ( value->has_str_val() )
207  {
208  if ( label_vec[1] == CLEAR_KEY_VALUE )
209  {
210  value->set_str_val( "" );
211  }
212  else
213  {
214  value->set_str_val( label_vec[1] );
215  }
216  }
217  else if ( value->has_uint32_val() )
218  {
219  value->set_uint32_val( ( uint32_t )std::stoul( label_vec[1] ) );
220  }
221  else if ( value->has_int32_val() )
222  {
223  value->set_int32_val( ( int32_t )std::stol( label_vec[1] ) );
224  }
225  else if ( value->has_bool_val() )
226  {
227  value->set_bool_val( ( label_vec[1] == "true" ? true : false ) );
228  }
229  oob_change = true;
230  }
231  }
232 
233  // Handle entry enable/disable
234  if ( unset )
235  {
236  entry->set_display_hint( key_label_it->second.disable );
237  oob_change = true;
238  }
239  else
240  {
241  entry->set_display_hint( key_label_it->second.enable );
242  oob_change = true;
243  }
244  }
245  }
246  return ( true );
247 }

266  ofile << "# " << std::endl; // end line and flush
267

270

274  if ( entry.category() != present_category )
275  {

277  // Then this is a new category entry; give it a little heading
278  ofile << "\n";
279  ofile << "###########################################################################\n";
280  ofile << "# " << entry.category() << std::endl;
281  ofile << "###########################################################################\n";
282  }
283 
284  // Now show either the tool_tip or all lines of the description; add blank line in the latter case
285  if ( !show_item_details )
286  {
287  ofile << "# " << entry.tool_tip() << std::endl;
288  }
289  else
290  {
291  ofile << "\n";
292  for ( const std::string &line : entry.description() )
293  {
294  ofile << "# " << line << std::endl;
295  }
296  }
297 
298  // Then the payload for this entry.
299  // Disabled entries have been filtered out if not wanted, so if we find one,
300  // we show it here after starting the line with a comment char '#'
301  if ( ( entry.display_hint() == px::proto::DisplayHint::DISPLAY_DISABLED ) ||
302  ( entry.display_hint() == DH_HIDDEN_DISABLED ) )
303  {
304  ofile << "# ";
305  }
306  // Then always show the key plus a space
307  ofile << entry.key_label() << " ";
308  // We interrupt this program to make a few exceptions:
309  // if the key_label is "persistence_location", then use the command line parameter to set the location
310  if ( entry.key_label() == MQTT_PERSISTENCE_KEY )
311  {
312  ofile << mqtt_persist_db_path << std::endl;
313  }
314  // If "log_dest" and the -Ns option was given (!use_syslogd), then change to "stderr" and addition topic logging.
315  else if ( !use_syslogd && ( entry.key_label() == MQTT_LOG_DEST_KEY ) )
316  {
317  ofile << "stderr" << std::endl;
318  // Add additional log destination of topic.
319  ofile << entry.key_label() << " " << "topic" << std::endl;
320  }
321  // In case of syslog, add additional log destination of topic.
322  else if ( use_syslogd && ( entry.key_label() == MQTT_LOG_DEST_KEY ) )
323  {
324  ofile << "syslog" << std::endl;
325  ofile << entry.key_label() << " " << "topic" << std::endl;
326  }
327  // Else If the complex_text is filled out, use that instead of the others
328  else if ( !entry.complex_text().empty() )
329  {
330  ofile << entry.complex_text() << std::endl;
331  }
332  else
333  {
334  // If there is option text, show it next, then the value
335  if ( !entry.key_options().empty() )
336  {
337  ofile << entry.key_options() << " ";
338  }
339  if ( entry.has_value() )
340  {
341  try
342  {
343  // Convert value to our CValue type, so we can easily print it
344  px::CValue cds_val( entry.value() );
345  ofile << cds_val.as_string();
346  }
347  catch ( const px::generic_err &x )
348  {}
349  }
350  // In all cases, end the line
351  ofile << std::endl;
352  }
353  }
354 
355  // Finish with a blank line and a flush
356  ofile << "\n" << std::endl;
357 }
