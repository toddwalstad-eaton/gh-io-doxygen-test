194 unsigned int COptions::post_process( int argc, char *argv[] )
195 {
196  // Validate the operations; if they depend on something, check it out
197  // Set the selected_op to OP_UNDEFINED and print the error if the check fails
198  bool is_bad = false;
199  const char *missing_text = nullptr;
200 
201  switch ( selected_op )
202  {
203  case OP_CAPTURE_SNAPSHOT:
204  if ( output_file_name == "" )
205  {
206  is_bad = true;
207  missing_text = "The output file name for capture operation is not specified";
208  }
209  break;
210 
211  case OP_REPLAY_SNAPSHOT:
212  if ( input_file_name == "" )
213  {
214  is_bad = true;
215  missing_text = "The input file name for replay operation is not specified";
216  }
217  break;
218  default:
219  // Just let these pass
220  break;
221  }
222 
223  if ( is_bad )
224  {
225  if ( ( missing_text != nullptr ) && ( argc > 1 ) )
226  {
227  fprintf( stderr, "post_process error for %s: %s \n", argv[1],
228  missing_text );
229  }
230  selected_op = OP_UNDEFINED;
231  }
232 
233  if ( very_verbose ) // if selected -vv, set verbose to true as well
234  {
235  verbose = true;
236  }
237  //exit_pt:
238 
239 
240  //fprintf(stderr, "post_process: selected_op = %u\n",selected_op);
241 
242  return ( selected_op );
243 }

248 int COptions::process_include_options( const px::string &include_options )
249 {
250  int rv = 0;
251  px::string include_options_str = include_options;
252 
253  if ( include_options_str == "" )
254  {
255  if ( selected_op == COptions::OP_CAPTURE_SNAPSHOT )
256  {
257  include_options_str = INCLUDE_CAPTURE_DEFAULT;
258  }
259  else if ( selected_op == COptions::OP_REPLAY_SNAPSHOT )
260  {
261  include_options_str = INCLUDE_REPLAY_DEFAULT;
262  }
263  }
264 
265  std::vector<px::string> include_options_types = px::CUtil::split_string( include_options_str, ',' );
266 
267  for ( size_t index = 0; index < include_options_types.size(); ++index )
268  {
269  if ( include_options_types[index] == INCLUDE_CONFIG_OPTION )
270  {
271  include_oobs = true;
272  }
273  else if ( include_options_types[index] == INCLUDE_USERS_OPTION )
274  {
275  include_users = true;
276  }
277  else if ( include_options_types[index] == INCLUDE_DEVICES_OPTION )
278  {
279  include_devices = true;
280  }
281  else if ( include_options_types[index] == INCLUDE_STATES_OPTION )
282  {
283  include_states = true;
284  }
285  else if ( include_options_types[index] == INCLUDE_CHANNELS_OPTION )
286  {
287  include_channels = true;
288  }
289  else if ( include_options_types[index] == INCLUDE_MCL_OPTION )
290  {
291  include_mcl = true;
292  }
293  else if ( include_options_types[index] == INCLUDE_VALUES_OPTION )
294  {
295  include_values = true;
296  }
297  else
298  {
299  std::cerr << "Error: Invalid string '" << include_options_types[index] << "' found with '-i' option" << std::endl;
300  rv = 1;
301  break;
302  }
303  }
304 
305  if ( ( include_devices == false ) && ( include_states == true ) )
306  {
307  if ( verbose )
308  {
309  std::cout << "'states' option needs 'devices' option to be enabled." << std::endl;
310  }
311  rv = 1;
312  }
313 
314  if ( ( include_devices == false ) && ( include_channels == true ) )
315  {
316  if ( verbose )
317  {
318  std::cout << "'channels' option needs 'devices' option to be enabled." << std::endl;
319  }
320  rv = 1;
321  }
322 
323  if ( ( ( include_devices == false ) || ( include_channels == false ) ) && ( include_mcl == true ) )
324  {
325  if ( verbose )
326  {
327  std::cout << "'mcl' option needs 'devices' and 'channels' options to be enabled." << std::endl;
328  }
329  rv = 1;
330  }
331 
332  if ( ( ( include_devices == false ) || ( include_channels == false ) ) && ( include_values == true ) )
333  {
334  if ( verbose )
335  {
336  std::cout << "'values' option needs 'devices' and 'channels' options to be enabled." << std::endl;
337  }
338  rv = 1;
339  }
340  return ( rv );
341 }
