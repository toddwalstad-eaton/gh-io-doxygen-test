32 void CGpioCdsHandler::handle_response( cds::response_ptr_t resp )
33 {
34  PUSH_FUNC;
35 
36  //Something bad happened if we can't get to our core
37  assert( _gpio_core );
38 
39  SVR_DEBUG( 4, "-------------- CGpioCdsHandler::handle_response begin" );
40 
41  if ( ( !resp->is_subscription() ) && ( !resp->is_unsolicited() ) )
42  {
43  SVR_NOTICE( 1, "Unexpected CDS response published to us" );
44  SVR_DEBUG( 3, "Handle CDS request response end" );
45  }
46  else if ( resp->is_subscription() )
47  {
48  SVR_DEBUG( 3, "Handle subscribed CDS response begin" );
49 
50  std::shared_ptr<px::cds::CSubscriberNotify> sub_notif =
51  std::dynamic_pointer_cast<px::cds::CSubscriberNotify>( resp );
52  if ( unlikely( !sub_notif ) )
53  {
54  SVR_ERR( 1, "Could not unpack subscription notification" );
55  }
56  else if ( _ch_value_sub_id == 0 )
57  {
58  SVR_INFO( 5, "Nothing subscribed to here; dropping notification." );
59  }
60  else if ( sub_notif->subscriber_id != _ch_value_sub_id )
61  {
62  SVR_INFO( 5, "Unmatched subscription IDs; expected %u, got %u.",
63  ( uint32_t ) _ch_value_sub_id, ( uint32_t ) sub_notif->subscriber_id );
64  }
65  else if ( _fn_channel_value_callback && !sub_notif->channel_values.empty() )
66  {
67  // Should be just one entry, but easy enough to process in a loop
68  for ( auto &entry_pair : sub_notif->channel_values )
69  {
70  _fn_channel_value_callback( entry_pair.first, entry_pair.second );
71  }
72  }
73  else
74  {
75  SVR_INFO( 5, "Unhandled subscription type %u for SubID %u.",
76  ( uint32_t ) sub_notif->event, ( uint32_t ) sub_notif->subscriber_id );
77  }
78  }
79  else
80  {
81  SVR_DEBUG( 3, "Handle CDS unsolicited response begin" );
82  SVR_NOTICE( 1, "Unexpected CDS unsolicited message published to us" );
83  }
84  SVR_DEBUG( 4, "-------------- CGpioCdsHandler::handle_response end" );
85 }

194  settings.minimum = min;

198  break;
199  }
200 
201  // Using fire-and-forget, ignoring return value
202  submit( ucs_msg );
