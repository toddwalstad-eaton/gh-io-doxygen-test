61 bool CModbusProtocolHandler::parse_request()
62 {
63  PUSH_FUNC;
64 
65  assert( _adu );
66  assert( _transactor != nullptr );
67  px::asio::CIoTransactor &transactor = *_transactor;
68 
69  px::asio::iobuf_const_pair_t rbuf( transactor.get_rbuf() );
70 
71  SVR_DEBUG( 4, "Parsing incoming command" );
72 
73  // Parse the input buffer
74  px::asio::parse_result_t result( _adu->parse( rbuf.first, rbuf.second ) );
75  assert( result.second <= rbuf.second );
76 
77  //
78  // case 1. buffer is incomplete
79  //
80  if ( boost::logic::indeterminate( result.first ) )
81  {
82  // Buffer is incomplete
83  SVR_DEBUG( 7, "Buffer incomplete..." );
84  chunk_size = _adu->request_size;
85  return ( false );
86  }
87 
88  //
89  // case 2. buffer is complete
90  //
91 
92  // reset chunk size for next request
93  chunk_size = 128;
94 
95  // The input is no longer necessary; discard [this portion of] the
96  // read buffer
97  transactor.clear_rbuf( result.second );
98 
99  if ( !result.first )
100  {
101  SVR_DEBUG( 3, "Error encountered while parsing ADU" );
102  // This error leads to fd leak while modbus fuzz testing,
103  // using below flag closing the connection in io_transactor.cpp
104  // LNXTK-7068
105  transactor.set_parse_failed( false );
106  return ( true );
107  }
108 
109  // Execute the just-parsed request, using the transactor's write buffer
110  // as a space to construct our response
111  execute();
112 
113  return ( true );
114 }

119 void CModbusProtocolHandler::handle_request( CModbusPdu &pdu, CModbusDeviceConfig &cfg, bool nan_to_zero )
120 {
121  PUSH_FUNC;
122 
123  //TODO: Create ASIO service to encapsulate modbus functions so that they
124  //can be handled by a background asio service manager
125 
126  std::shared_ptr<CModbusFunctionHandler> func = nullptr;
127 
128  // Create a suitably-typed function handler object
129  //
130  // We need to allow reads even if a device is offline to
131  // support entity state channels. This also mimics the
132  // behavior of the old toolkit Modbus TCP Server,
133  switch ( pdu.function_code )
134  {
135  case MB_READ_COILS:
136  case MB_READ_DISCRETE_INPUTS:
137  func = CModbusReadDiscretes::create( *this, pdu, cfg );
138  break;
139  case MB_READ_HOLDING_REGISTERS:
140  case MB_READ_INPUT_REGISTERS:
141  func = CModbusReadRegisters::create( *this, pdu, cfg, nan_to_zero );
142  break;
143  case MB_WRITE_SINGLE_COIL:
144  case MB_WRITE_MULTIPLE_COILS:
145  if ( cfg.writes_enabled() )
146  {
147  func = CModbusWriteDiscretes::create( *this, pdu, cfg );
148  break;
149  }
150  case MB_WRITE_SINGLE_REGISTER:
151  case MB_WRITE_MULTIPLE_HOLDING_REGISTERS:
152  if ( cfg.writes_enabled() )
153  {
154  func = CModbusWriteRegisters::create( *this, pdu, cfg );
155  break;
156  }
157  default:
158  SVR_FDEBUG( 2, "Illegal function code: 0x%x", pdu.function_code );
159  err( MB_ILLEGAL_FUNCTION );
160  return;
161  }
162 
163  // Begin execution of the function handler
164  func->execute();
165 
166  //CDS handler needs to be aware of this message so it knows who to respond
167  //so we store it for later.
168  //TODO: handle exceptions from bad requests and CDS failures alike
169  if ( func->tranID != UINT32_MAX )
170  {
171  px::CMonoTime now_time( px::CMonoTime::now() );
172  auto &cds = get_cds_handle();
173  idle_transaction_ids.emplace_back( now_time, func->tranID );
174  cds.add_idle_fcn( func );
175 
176  // See if any are old enough to age out
177  while ( ( !idle_transaction_ids.empty() ) &&
178  ( ( now_time - idle_transaction_ids.front().first ).msec() > max_transaction_age_ms ) )
179  {
180  cds.remove_idle_fcn( idle_transaction_ids.front().second );
181  idle_transaction_ids.pop_front();
182  }
183  }
184 }
