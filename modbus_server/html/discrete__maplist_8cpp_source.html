163 std::vector<bool> CDiscreteMaplist::translate_vals( const std::unordered_set<cds::field_t> &fields )
164 const
165 {
166  PUSH_FUNC;
167 
168  CDiscreteResultSet results( _addrs );
169  px::id_t dev = fields.begin()->id;
170 
171  // Traverse our discrete mappings and CDS fields, and add them to the
172  // result set (for each leaf that was fetched successfully)
173  for ( const auto &range : _mappings )
174  {
175  for ( const auto &mapping : range )
176  {
177  SVR_FDEBUG( 5, "mapping.first.size() %zu, mapping.first.first %u, _addrs.first %u, _addrs.second %u",
178  mapping.first.size(), mapping.first.first, _addrs.first, _addrs.second );
179 
180  assert( mapping.second != nullptr );
181  assert( mapping.first.size() == 1 );
182  assert( mapping.first.first >= _addrs.first );
183  assert( mapping.first.first <= _addrs.second );
184 
185  const auto &param = *( mapping.second );
186  cds::field_t field;
187  if ( param.is_device_field() )
188  {
189  field.type = px::cds::field_message_type::device;
190  field.id = dev;
191  field.field = param.path();
192  }
193  else
194  {
195  field.type = px::cds::field_message_type::channel_value_by_name_group_id;
196  field.id = dev;
197  field.tag = param.path();
198  }
199 
200  auto ifield = fields.find( field );
201  assert( ifield != fields.end() );
202 
203  if ( ifield->value.type() == px::datatype::invalid )
204  {
205  SVR_DEBUG( 5, "Couldn't fetch value for param '%s': %s", ifield->tag.c_str(), ifield->value.as_string().c_str() );
206  continue;
207  }
208  else
209  {
210  SVR_DEBUG( 3, "param %s=%s ", ifield->tag.c_str(), ifield->value.as_string().c_str() );
211  results.add( mapping.first, param, *ifield );
212  }
213  }
214  }
215 
216  // Calculate and return the result value
217  return ( results.translate( _zero_fill ) );
218 }
