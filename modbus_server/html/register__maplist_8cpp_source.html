226 regvec_t CRegisterMaplist::translate_vals( const std::unordered_set<cds::field_t> &fields, bool nan_to_zero ) const
227 {
228  PUSH_FUNC;
229 
230  CRegisterResultSet results( _addrs );
231 
232  px::id_t dev = fields.begin()->id;
233 
234  // Traverse our register mappings and CDS values, and add them to the
235  // result set (for each value that was fetched successfully)
236  for ( const auto &range : _reg_mappings )
237  {
238  for ( const auto &mapping : range )
239  {
240  SVR_FDEBUG( 3, "1 mapping.first.first %d mapping.first.second %d _addrs.first %d", mapping.first.first, mapping.first.second, _addrs.first );
241  assert( mapping.first.second >= _addrs.first );
242  assert( mapping.second != nullptr );
243 
244  const auto &param = *( mapping.second );
245  cds::field_t field;
246  if ( param.is_device_field() )
247  {
248  field.type = px::cds::field_message_type::device;
249  field.id = dev;
250  field.field = param.path();
251  }
252  else
253  {
254  field.type = px::cds::field_message_type::channel_value_by_name_group_id;
255  field.id = dev;
256  field.tag = param.path();
257  }
258 
259  auto ifield = fields.find( field );
260  assert( ifield != fields.end() );
261 
262  if ( ifield->value.type() == px::datatype::invalid )
263  {
264  SVR_DEBUG( 5, "Couldn't fetch value for param '%s': %s", ifield->tag.c_str(), ifield->value.as_string().c_str() );
265  continue;
266  }
267  else
268  {
269  if ( nan_to_zero )
270  {
271  if ( ifield->value.type() == px::datatype::float32 )
272  {
273  float f = ifield->value.as_float();
274  if ( std::isnan( f ) )
275  {
276  ifield->value = 0.0;
277  SVR_DEBUG( 5, "Param %s float value is NaN - converted to zero", ifield->tag.c_str() );
278  }
279  }
280  else if ( ifield->value.type() == px::datatype::float64 )
281  {
282  double d = 0;
283  if ( std::isnan( d ) )
284  {
285  ifield->value = 0.0;
286  SVR_DEBUG( 5, "Param %s double value is NaN - converted to zero", ifield->tag.c_str() );
287  }
288  }
289  }
290  SVR_FDEBUG( 5, "Fetched value for param %s=%s ", ifield->tag.c_str(), ifield->value.as_string().c_str() );
291  }
292 
293  results.add( mapping.first, param, *ifield );
294  }
295  }
296 
297  // Calculate and return the result value
298  return ( results.translate( _zero_fill ) );
299 }
