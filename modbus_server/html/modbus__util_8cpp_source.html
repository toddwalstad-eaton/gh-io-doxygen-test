263 void CModbusUtil::get_files_from_dir( std::vector<px::string> &files,
264  const px::string &dir,
265  const px::string &extension,
266  const bool full_path,
267  const bool recursive )
268 {
269  std::set<px::string> files_set;
270  fs::path path( dir.c_str() );
271 
272  if ( recursive )
273  {
274  std::string path( dir.c_str() );
275  std::string pattern = std::string( "*" ) + std::string( extension.c_str() );
276  px::CRecursiveFind file_set( path );
277  std::set<px::CRecursiveFind::file_info_t> files;
278  file_set.get_regular_files( files, pattern );
279  for ( auto file_it : files )
280  {
281  std::string filename = std::get<px::CRecursiveFind::GET_FILENAME>( file_it );
282  std::string filepath = std::get<px::CRecursiveFind::GET_FILEPATH>( file_it );
283  if ( full_path )
284  {
285  files_set.insert( px::string( filepath.c_str() ) );
286  SVR_DEBUG( 4, " Map File: '%s'", filepath.c_str() );
287  }
288  else
289  {
290  files_set.insert( px::string( filename.c_str() ) );
291  SVR_DEBUG( 4, " Map File: '%s'", filename.c_str() );
292  }
293  }
294  }
295  else
296  {
297  if ( fs::exists( path ) && fs::is_directory( path ) )
298  {
299  for ( fs::directory_iterator it( path ); it != fs::directory_iterator(); ++it )
300  {
301  if ( fs::is_regular_file( *it ) && ( it->path().extension() == extension.c_str() ) )
302  {
303  if ( full_path )
304  {
305  files_set.insert( px::string( it->path().string().c_str() ) );
306  SVR_DEBUG( 4, " Map File: '%s'", it->path().string().c_str() );
307  }
308  else
309  {
310  files_set.insert( px::string( it->path().filename().c_str() ) );
311  SVR_DEBUG( 4, " Map File: '%s'", it->path().filename().c_str() );
312  }
313  }
314  }
315  }
316  }
317 
318  for ( auto entry : files_set )
319  {
320  files.push_back( entry );
321  }
322 }
