62 regvec_t CRegisterResultSet::translate( bool zero_fill ) const
63 {
64  PUSH_FUNC;
65 
66  regvec_t retval( _addrs.size(), 0 );
67  std::vector<bool> assigned;
68 
69  if ( !zero_fill )
70  {
71  assigned.resize( _addrs.size(), false );
72  }
73 
74  // For each mapping that we've accumulated (in ascending order of
75  // parameter priority)...
76  auto ientry = _entries.end();
77  auto istart = _entries.begin();
78 
79  while ( ientry != istart )
80  {
81  --ientry;
82  const auto &entry = *ientry;
83 
84  auto result( entry.translate() );
85  SVR_FDEBUG( 4, "result: %zu== %zu", result.size(), entry.addr().size() );
86 
87  assert( result.size() == entry.addr().size() );
88  assert( entry.addr().first <= _addrs.second );
89  assert( entry.addr().second >= _addrs.first );
90 
91  // Define start & end indices of this set of registers, and
92  // our main result set
93 
94  size_t result_start = 0;
95  if ( _addrs.first > entry.addr().first )
96  {
97  result_start += ( _addrs.first - entry.addr().first );
98  }
99 
100  size_t result_end = entry.addr().size();
101  if ( _addrs.second < entry.addr().second )
102  {
103  result_end -= ( entry.addr().second - _addrs.second );
104  }
105 
106  size_t retval_start = 0;
107  if ( _addrs.first < entry.addr().first )
108  {
109  retval_start += ( entry.addr().first - _addrs.first );
110  }
111 
112  size_t retval_end = _addrs.size();
113  if ( _addrs.second > entry.addr().second )
114  {
115  retval_end -= ( _addrs.second - entry.addr().second );
116  }
117 
118  assert( result_end > result_start );
119  assert( ( retval_end - retval_start ) == ( result_end - result_start ) );
120 
121  // Copy the intersecting portion of 'result' over the current
122  // values in 'retval'
123  memcpy( &( retval[retval_start] ), &( result[result_start] ), ( retval_end - retval_start ) * sizeof( retval[0] ) );
124 
125  if ( !zero_fill )
126  {
127  // Flag each of the applicable addresses as having
128  // been filled in
129  for ( size_t i = retval_start; i < retval_end; i++ )
130  {
131  assigned[i] = true;
132  }
133  }
134  }
135 
136  if ( !zero_fill )
137  {
138  // Ensure that all of the addresses were covered in this
139  // request, i.e. there are no holes
140  for ( size_t i = 0; i < assigned.size(); i++ )
141  {
142  if ( !assigned[i] )
143  {
144  SVR_FDEBUG( 2, "No mapping available at address %zu", i + _addrs.first );
145  throw map_err();
146  }
147  }
148  }
149 
150  // 'retval' should now be complete
151  return ( retval );
152 }

157 std::unordered_set<cds::field_t> CRegisterResultSet::fields( px::id_t id, const regvec_t &vals, modbus_err_t &err ) const
158 {
159  PUSH_FUNC;
160 
161  assert( vals.size() == _addrs.size() );
162 
163  std::vector<bool> isused( vals.size(), false );
164  std::unordered_set<cds::field_t> retval;
165 
166  // TODO: handle units on write
167 
168  // For each mapping that we've accumulated (in descending order of
169  // parameter priority)...
170  for ( const auto &entry : _entries )
171  {
172  // If this mapping extends beyond the bounds of the input
173  // register set, we can't perform a full translation, and so
174  // we skip it.
175  if ( ( entry.addr().first < _addrs.first ) || ( entry.addr().second > _addrs.second ) )
176  {
177  SVR_DEBUG( 3, "Skipping write to partial entry '%s'. ", entry.name().c_str() );
178  SVR_DEBUG( 6, " Skipping: entry.addr().first [%u], _addrs.first [%u], entry.addr().second [%u], _addrs.second [%u]", entry.addr().first, _addrs.first, entry.addr().second, _addrs.second );
179  continue;
180  }
181 
182  // See if there are any registers needed by this mapping which
183  // haven't already been used by another one
184  bool needed = false;
185  for ( size_t i = entry.addr().first; i <= entry.addr().second; i++ )
186  {
187  if ( !isused[i - _addrs.first] )
188  {
189  needed = true;
190  isused[i - _addrs.first] = true;
191  }
192  }
193 
194  if ( !needed )
195  {
196  SVR_DEBUG( 4, "Entry '%s' not needed for write", entry.name().c_str() );
197  continue;
198  }
199 
200  // Create a register array containing only those registers
201  // that apply to this mapping
202  size_t start = entry.addr().first - _addrs.first;
203  size_t len = entry.addr().size();
204 
205  assert( start + len <= _addrs.size() );
206 
207  regvec_t param_vals( vals.begin() + start, vals.begin() + start + len );
208 
209  // Check if this entry is a valid discrete Device Command
210  if ( entry.is_command_field() )
211  {
212  for ( auto param_val : param_vals )
213  {
214  SVR_FDEBUG( 1, "CDS ID %" PRIu64 ", value %u", id, param_val );
215  // If any device command has a zero value, generate an error result.
216  // Can't get any more fine grained if this is a multiple register write.
217  if ( !param_val )
218  {
219  err = MB_ILLEGAL_DATA_VALUE;
220  }
221  }
222  }
223 
224  // Translate the registers into a CDS value, as part of a leaf
225  // ready to be written to the CDS.
226  retval.insert( entry.field( id, param_vals ) );
227  }
228 
229  return ( retval );
230 }
