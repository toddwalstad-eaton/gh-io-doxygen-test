15 regaddr_t roundUp( regaddr_t numToRound, regaddr_t multiple )
16 {
17  if ( multiple == 0 )
18  {
19  return ( numToRound );
20  }
21  regaddr_t remainder = numToRound % multiple;
22  if ( remainder == 0 )
23  {
24  return ( numToRound );
25  }
26  SVR_DEBUG( 3, " roundUp: retval %u", numToRound + multiple - remainder );
27  return ( numToRound + multiple - remainder );
28 }

30 regaddr_t get_band_size( eds_range_limits_t &limits )
31 {
32  regaddr_t retval = SUBDEV_DEV_ADDRESS_SIZE;
33  regaddr_t max = ( limits.max_disc_addr > limits.max_reg_addr ) ? limits.max_disc_addr : limits.max_reg_addr;
34 
35  retval += ( max - limits.max_reg_addr ) + SUBDEV_DEV_BAND_END_SPACE;
36  SVR_DEBUG( 3, " get_band_size: max %u, retval before roundup %u", max, retval );
37  return ( roundUp( retval, SUBDEV_DEV_ROUNDUP_TO ) );

80 CModbusDeviceConfig::device_id_set_t CModbusDeviceConfig::setup_subdevices( std::unordered_set<px::CDevice> &subdevices,
81  px::string peds_file,
82  peds_map_store_t &peds_map_store )
83 {
84  PUSH_FUNC;
85 
86  device_id_set_t retvals;
87 
88  // Intialize sub-device associated data.
89  _subdevice_count = 0;
90  _subdevice_reg_map.clear();
91  _subdevice_disc_map.clear();
92 
93  // Bail out if there are no sub-devices, or the EDS map does not support sub-devices
94  if ( !subdevices.size() || ( _max_subdevices < 1 ) )
95  {
96  SVR_DEBUG( 2, " Error: No sub-devices under Device ID %" PRIu64 ".", _dev_id );
97  return ( retvals );
98  }
99 
100  // Check for valid PEDS file.
101  if ( peds_file.empty() )
102  {
103  SVR_DEBUG( 2, " Error: Missing sub-device ped file name." );
104  return ( retvals );
105  }
106 
107  // Get the canonical_model_name name of this sub-device.
108  px::string canonical_model_name( "" );
109  auto sub_it = subdevices.begin();
110  if ( sub_it != subdevices.end() )
111  {
112  canonical_model_name = sub_it->canonical_model_name.c_str();
113  }
114 
115  // Check for valid canonical_model_name name.
116  if ( canonical_model_name.empty() )
117  {
118  SVR_DEBUG( 2, " Error: Missing sub-device canonical_model_name name." );
119  return ( retvals );
120  }
121 
122  eds_range_limits_t dev_eds_limits = _peds_mapping->eds_limits;
123 
124  SVR_DEBUG( 2, " Device ID %" PRIu64 " has %zu sub-devices: canonical_model_name %s, max sub-devices %d, max channel address %u, sub-device peds_file %s",
125  _dev_id, subdevices.size(), canonical_model_name.c_str(), _max_subdevices, dev_eds_limits.max_reg_addr, peds_file.c_str() );
126 
127  //
128  // Now the complicated stuff:
129  // We are going to add N sub-device EDS mappings to the peds_map_store. Each mapping will
130  // be created using the "peds_file" name, but the actual name added to the peds_map_store
131  // will be an indexed ped name with the format "peds_fileN.ped" - where "N" is the index
132  // of this sub-device.
133  //
134  // Generate map of sub-device IDs & indexed PEDS file names,
135  // keyed to the sub-device "network_address" in the CDS.
136  //
137  boost::filesystem::path ped_path( peds_file.c_str() );
138  subdevice_info_map_t subdevice_info_map;
139  for ( const auto &subdevice : subdevices )
140  {
141  subdevice_info_t info;
142  info.dev_id = subdevice.id();
143  info.peds_file = ped_path.stem().c_str();
144  info.peds_file += subdevice.network_address.c_str();
145  info.peds_file += ped_path.extension().c_str();
146  uint32_t subdevice_index = ( uint16_t )strtoul( subdevice.network_address.c_str(), NULL, 0 );
147  subdevice_info_map.insert( std::make_pair( subdevice_index, info ) );
148  SVR_DEBUG( 3, " subdevice_index %u, info.dev_id %" PRIu64 ", info.peds_file %s", subdevice_index, info.dev_id, info.peds_file.c_str() );
149  }
150 
151  regaddr_t min_limit;
152  regaddr_t max_limit;
153 
154  //
155  // Add the Main Device address ranges and device info
156  // to the discrete and registers map stores.
157  //
158  try
159  {
160  // Add main device Channel & Discrete range info.
161  device_info_t dev_info;
162  dev_info.is_subdevice = false;
163  dev_info.index = 0;
164  dev_info.dev_id = _dev_id;
165  dev_info.peds_file = _peds_mapping->peds_file();
166  dev_info.peds_mapping = _peds_mapping;
167 
168  min_limit = dev_eds_limits.min_reg_addr;
169  max_limit = ( dev_eds_limits.max_disc_addr > dev_eds_limits.max_reg_addr ) ? dev_eds_limits.max_disc_addr : dev_eds_limits.max_reg_addr;
170 
171  SVR_DEBUG( 3, " ch_addr limits: min_limit %u max_limit %u, min_reg_addr %u, max_reg_addr %u, max_disc_addr %u",
172  min_limit, max_limit, dev_eds_limits.min_reg_addr, dev_eds_limits.max_reg_addr, dev_eds_limits.max_disc_addr );
173 
174  CAddressRange ch_addr( min_limit, max_limit );
175  _subdevice_reg_map.insert( std::make_pair( ch_addr, dev_info ) );
176  _subdevice_disc_map.insert( std::make_pair( ch_addr, dev_info ) );
177 
178  // Sanity check the Device and Command min/max limits.
179  min_limit = dev_eds_limits.min_dev_reg_addr;
180  max_limit = ( dev_eds_limits.max_cmd_disc_addr > dev_eds_limits.max_cmd_reg_addr ) ? dev_eds_limits.max_cmd_disc_addr : dev_eds_limits.max_cmd_reg_addr;
181 
182  max_limit = ( dev_eds_limits.max_dev_reg_addr > min_limit ) ? dev_eds_limits.max_dev_reg_addr : max_limit;
183 
184  SVR_DEBUG( 3, " dev_addr: limits min_limit %u max_limit %u, min_reg_addr %u, max_reg_addr %u, max_disc_addr %u",
185  min_limit, max_limit, dev_eds_limits.min_reg_addr, dev_eds_limits.max_reg_addr, dev_eds_limits.max_disc_addr );
186 
187  // Add main device Device & Command range info.
188  CAddressRange dev_addr( min_limit, max_limit );
189  _subdevice_reg_map.insert( std::make_pair( dev_addr, dev_info ) );
190  _subdevice_disc_map.insert( std::make_pair( dev_addr, dev_info ) );
191 
192  SVR_DEBUG( 2, " Main Device: eds_band_size %u, channel range (min %u max %u), device/cmd range (min %u max %u)",
193  dev_eds_limits.subdevice_band_size, ch_addr.first, ch_addr.second, dev_addr.first, dev_addr.second );
194  }
195  catch ( const px::generic_err &x )
196  {}
197 
198  //
199  // Set the starting address (map_offset) for the subdevice map.
200  // We default to START_SUBDEVICE_ADDR, but set it higher if any Main
201  // Device register/discrete addresses execeed START_SUBDEVICE_ADDR.
202  //
203  regaddr_t map_offset = START_SUBDEVICE_ADDR;
204  max_limit = ( dev_eds_limits.max_disc_addr > dev_eds_limits.max_reg_addr ) ? dev_eds_limits.max_disc_addr : dev_eds_limits.max_reg_addr;
205  if ( max_limit > map_offset )
206  {
207  map_offset = roundUp( max_limit + DEV_DEV_BAND_END_SPACE, DEV_DEV_ROUNDUP_TO );
208  }
209 
210  //
211  // Select the sub-device band size from the sub-device EDS setting,
212  // or from the calculated EDS mapping limits.
213  //
214  regaddr_t calc_band_size = 0;
215  eds_range_limits_t subdevice_eds_limits;
216  try
217  {
218  // Get the sub-device EDS map
219  CModbusDeviceConfig::peds_mapping_t subdevice_eds_map( new CPedsMapping( peds_file, "", true ) );
220  subdevice_eds_limits = subdevice_eds_map->eds_limits;
221  calc_band_size = get_band_size( subdevice_eds_limits );
222 
223  SVR_DEBUG( 2, " Sub Device: calc_band_size %u", calc_band_size );
224 
225  // Choose the best fit sub-device band size - prefer the value from the main device EDS
226  if ( ( dev_eds_limits.subdevice_band_size > 0 ) && ( dev_eds_limits.subdevice_band_size > calc_band_size ) )
227  {
228  calc_band_size = dev_eds_limits.subdevice_band_size;
229  SVR_DEBUG( 2, " Sub Device: Using EDS sub-device band space value %u", calc_band_size );
230  }
231  }
232  catch ( const px::generic_err &x )
233  {}
234 
235  // Sanity check for calc_band_size and map_offset.
236  if ( !calc_band_size || !map_offset )
237  {
238  SVR_DEBUG( 2, " Error: Could not set a sub-device band size or map offset." );
239  return ( retvals );
240  }
241 
242  for ( const auto &dev_index : subdevice_info_map )
243  {
244  SVR_DEBUG( 2, " Sub-device %u: ID %" PRIu64 ", map start address %u, peds_file %s",
245  dev_index.first,
246  dev_index.second.dev_id, map_offset, dev_index.second.peds_file.c_str() );
247 
248  // Unlikely, but check if we are at the ceiling of the alloted sub-device register
249  // range. If we ever support a big enough device that hits the limit, consider
250  // continuing sub-device register numbering above the command address range.
251  if ( ( map_offset + calc_band_size ) >= dev_eds_limits.min_dev_reg_addr )
252  {
253  SVR_ERR( 1, "Reached end of allowed sub-device Modbus TCP register range at sub-device number %u.", dev_index.first );
254  break;
255  }
256 
257  try
258  {
259  peds_mapping_t peds_mapping;
260  auto peds_mapping_it = peds_map_store.find( dev_index.second.peds_file );
261  if ( peds_mapping_it != peds_map_store.end() )
262  {
263  SVR_DEBUG( 2, " Indexed PEDS mapping %s was in central store", dev_index.second.peds_file.c_str() );
264  peds_mapping = peds_mapping_it->second;
265  }
266  else
267  {
268  // Create new EDS mapping if we cannot find it in the central store
269  SVR_DEBUG( 2, " Indexed PEDS mapping %s was not in central store,", dev_index.second.peds_file.c_str() );
270  SVR_DEBUG( 2, " create one based on peds_file %s and add it with map_offset %u", peds_file.c_str(), map_offset );
271  try
272  {
273  peds_mapping.reset( new CPedsMapping( peds_file, "", true, map_offset ) );
274  map_offset += calc_band_size;
275 
276  // Store the new EDS mapping in the central store
277  peds_map_store.insert( std::make_pair( dev_index.second.peds_file.c_str(), peds_mapping ) );
278  }
279  catch ( const px::generic_err & )
280  {
281  SYSLOG_ERR( 1, "Exception occured while creating Peds mapping for %s", peds_file.c_str() );
282  }
283  }
284 
285  SVR_DEBUG( 2, " Sub-device %u: map limits: min reg %u, max reg %u, min disc %u, max disc %u\n",
286  dev_index.first,
287  peds_mapping->eds_limits.min_reg_addr, peds_mapping->eds_limits.max_reg_addr,
288  peds_mapping->eds_limits.min_disc_addr, peds_mapping->eds_limits.max_disc_addr );
289 
290  // Update list of sub-device IDs to return.
291  retvals.insert( dev_index.second.dev_id );
292 
293  // Associate sub-device register and discrete ranges with device info & EDS mapping.
294  device_info_t dev_info;
295  dev_info.is_subdevice = true;
296  dev_info.index = dev_index.first;
297  dev_info.dev_id = dev_index.second.dev_id;
298  dev_info.peds_file = dev_index.second.peds_file;
299  dev_info.peds_mapping = peds_mapping;
300  if ( peds_mapping->eds_limits.max_reg_addr > peds_mapping->eds_limits.min_reg_addr )
301  {
302  CAddressRange addr( peds_mapping->eds_limits.min_reg_addr, peds_mapping->eds_limits.max_reg_addr );
303  _subdevice_reg_map.insert( std::make_pair( addr, dev_info ) );
304  }
305  if ( peds_mapping->eds_limits.max_disc_addr > peds_mapping->eds_limits.min_disc_addr )
306  {
307  CAddressRange addr( peds_mapping->eds_limits.min_disc_addr, peds_mapping->eds_limits.max_disc_addr );
308  _subdevice_disc_map.insert( std::make_pair( addr, dev_info ) );
309  }
310 
311  _subdevice_count++;
312  }
313  catch ( const px::generic_err &x )
314  {}
315  }
316 
317  return ( retvals );
318 }

324 {
325  PUSH_FUNC;
326 
327  subdevice_info_map_t retvals;
328 
329  //
330  // Pull unique sub-device info from the register and discrete stores
331  // and return as subdevice_info_map_t map.
332  //
333  for ( const auto &reg_info : _subdevice_reg_map )
334  {
335  if ( reg_info.second.is_subdevice )

342  }

344  {
345  if ( disc_info.second.is_subdevice )
346  {
347  auto it = retvals.find( disc_info.second.dev_id );
348  if ( it == retvals.end() )
349  {
350  subdevice_info_t info;
351  info.dev_id = disc_info.second.dev_id;
352  info.peds_file = disc_info.second.peds_file;
353  retvals.insert( std::make_pair( disc_info.second.index, info ) );
354  }
355  }
356  }
357 
358  return ( retvals );
359 }
