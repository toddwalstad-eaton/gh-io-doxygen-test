165 peds_ptr_t CType1EDS::create_peds_from_eds( const px::string &eds_file,
166  const px::string &peds_name,
167  const std::vector<px::string> &protocol_types,
168  const bool limit_value_size )
169 {
170  PUSH_FUNC;
171 
172  peds_ptr_t peds( new px::proto::Peds() );
173 
174  // Check if the EDS protocol types are in the list of passed in protocol_types
175  bool protocol_test_passed = false;
176  for ( size_t i = 0; i < protocol_types.size(); i++ )
177  {
178  for ( size_t j = 0; j < _protocol_types.size(); j++ )
179  {
180  if ( _protocol_types[j] == protocol_types[i] )
181  {
182  protocol_test_passed = true;
183  SVR_DEBUG( 4, "Found legal protocol type %s", _protocol_types[j].c_str() );
184  break;
185  }
186  }
187  if ( protocol_test_passed )
188  {
189  break;
190  }
191  }
192 
193  // Error out if protocol test failed.
194  if ( !protocol_test_passed )
195  {
196  SVR_WARN( 4, "EDS protocol type did not match any of the specified types" );
197  peds.reset();
198  return ( peds );
199  }
200 
201  peds->set_protocol( PED_PROTOCOL_TYPE_MODBUS );
202  peds->set_version( PED_VERSION_NUMBER );
203  peds->set_description( _description.c_str() );
204  peds->set_eds_file( eds_file.c_str() );
205 
206  if ( !peds_name.empty() )
207  {
208  peds->set_name( peds_name.c_str() );
209  }
210  else
211  {
212  peds->set_name( PEDS_NAME_EFFICIENT );
213  }
214 
215  for ( auto &m : _models )
216  {
217  SVR_FDEBUG( 4, "add_canonical_model_name %s", m.c_str() );
218  peds->add_canonical_model_name( m.c_str() );
219  }
220 
221  uint16_t reg_address = addr_fields.start_register_address;
222  uint16_t discrete_address = addr_fields.start_register_address;
223  uint16_t command_reg = addr_fields.start_cmd_addr;
224  uint16_t command_discrete = addr_fields.start_cmd_addr;
225  uint16_t device_address = addr_fields.start_dev_addr;
226  uint16_t entity_address = addr_fields.start_ent_addr;
227 
228  SVR_DEBUG( 4, "Adding %zu device properties to peds", _device_properties.size() );
229 
230  // Add device properties
231  bool is_subdevice = false;
232  px::proto::PedsEntry *entry = nullptr;
233  for ( auto &prop : _device_properties )
234  {
235  if ( prop.first == DEV_PROP_ROLE )
236  {
237  if ( prop.second == DEV_PROP_ROLE_SUBDEVICE )
238  {
239  is_subdevice = true;
240  }
241  continue;
242  }
243 
244  if ( prop.first != NO_PROP_VALUE )
245  {
246  entry = peds->add_entries();
247  entry->set_name( prop.first.c_str() );
248  entry->set_device( true );
249 
250  if ( MB_UINT32_TYPE == prop.second )
251  {
252  entry->set_address( device_address );
253  entry->set_size( DEV_PROP_UINT32_COUNT );
254  entry->set_value_type( MB_UINT32_TYPE );
255  device_address += DEV_PROP_UINT32_COUNT / 2;
256  }
257  else
258  {
259  entry->set_address( device_address );
260  entry->set_size( DEV_PROP_BYTE_COUNT );
261  entry->set_value_type( MB_STRING_TYPE );
262  device_address += DEV_PROP_BYTE_COUNT / 2;
263  }
264  }
265  else
266  {
267  device_address += DEV_PROP_BYTE_COUNT / 2;
268  }
269  }
270 
271  // Add entity properties
272  if ( !is_subdevice )
273  {
274  entry = peds->add_entries();
275  entry->set_name( ENTITY_OPER_STATE );
276  entry->set_address( entity_address );
277  entry->set_size( ENT_PROP_BYTE_COUNT );
278  entry->set_value_type( MB_UINT16_TYPE );
279  entity_address += ENT_PROP_BYTE_COUNT / 2;
280  }
281 
282  // Add channel properties
283  for ( auto &channel : _channels )
284  {
285  if ( ( channel.value_type == px::datatype::invalid ) || ( channel.value_type == px::datatype::binary ) )
286  {
287  SVR_DEBUG( 4, "Skip unsupported channel %s", channel.canonical_name.c_str() );
288  continue;
289  }
290 
291  // Down convert channel type if 64-bits
292  if ( limit_value_size )
293  {
294  limit_channel_datatype( channel );
295  }
296 
297  entry = peds->add_entries();
298 
299  entry->set_name( channel.canonical_name.c_str() );
300  if ( channel.value_type == px::datatype::boolean )
301  {
302  entry->set_discrete( true );
303  entry->set_address( discrete_address++ );
304  }
305  else
306  {
307  int count = CModbusUtil::get_register_count( channel.value_type );
308  entry->set_address( reg_address );
309 
310  if ( channel.value_type == px::datatype::timestamp )
311  {
312  // use default time-spec
313  entry->set_date_spec( MB_DEFAULT_DATETIME_SPEC );
314 
315  // number of registers depends on the spec
316  if ( ( entry->date_spec().size() ) % 2 != 0 )
317  {
318  throw px::generic_err( "invalid date time spec" );
319  }
320  count = entry->date_spec().size() / 2;
321  }
322 
323  /* If value type is "string" then use maximum channel string
324  * size (mcss) value for register calculation.
325  */
326  if ( channel.value_type == px::datatype::string )
327  {
328  count = mcss_pds;
329  }
330 
331  entry->set_size( 2 * count );
332 
333  px::CDataType dt( channel.value_type );
334  entry->set_value_type( dt.str() );
335 
336  reg_address += count;
337  }
338  }
339 
340  // Add command properties
341  for ( auto &command : _commands )
342  {
343  if ( command.automatic || command.hidden )
344  {
345  SVR_DEBUG( 4, "Skip automatic or hidden commands %s", command.canonical_name.c_str() );
346  continue;
347  }
348 
349  if ( command.value_type == px::datatype::binary )
350  {
351  SVR_DEBUG( 4, "Skip unsupported command %s", command.canonical_name.c_str() );
352  continue;
353  }
354 
355  entry = peds->add_entries();
356 
357  entry->set_name( command.canonical_name.c_str() );
358  entry->set_writable( true );
359  entry->set_command( true );
360 
361  SVR_DEBUG( 4, "Process command type: %d", command.external_value_type.compare( 0, 5, EDS_TAG_INCOM ) );
362  if ( ( command.value_type == px::datatype::boolean ) ||
363  ( command.external_value_type.compare( 0, 5, EDS_TAG_INCOM ) == 0 ) ||
364  !command.has_parameter_info )
365  {
366  entry->set_discrete( true );
367  entry->set_address( command_discrete++ );
368  }
369  else
370  {
371  int count = CModbusUtil::get_register_count( command.value_type );
372  entry->set_address( command_reg );
373 
374  if ( command.value_type == px::datatype::timestamp )
375  {
376  // use default time-spec
377  entry->set_date_spec( MB_DEFAULT_DATETIME_SPEC );
378 
379  // number of registers depends on the spec
380  if ( ( entry->date_spec().size() ) % 2 != 0 )
381  {
382  throw px::generic_err( "invalid date time spec" );
383  }
384  count = entry->date_spec().size() / 2;
385  }
386 
387  entry->set_size( 2 * count );
388 
389  px::CDataType dt( command.value_type );
390  entry->set_value_type( dt.str() );
391 
392  command_reg += count;
393  }
394  }
395 
396  // Use max_subdevices properrty to tetect if sub-device data is present
397  if ( max_subdevices > 0 )
398  {
399  if ( !subdevice_eds_file.empty() )
400  {
401  peds->set_subdevice_eds_file( subdevice_eds_file.c_str() );
402  }
403 
404  peds->set_max_subdevices( max_subdevices );
405 
406  if ( subdevice_band_size > 0 )
407  {
408  peds->set_subdevice_band_size( subdevice_band_size );
409  }
410  }
411 
412  SVR_DEBUG( 4, "Added %d total entries to peds for all models", peds->entries_size() );
413 
414  // Set PEDS type based on 'is_subdevice' value
415  if ( !is_subdevice )
416  {
417  peds->set_type( device_eds_type.c_str() );
418  }
419  else
420  {
421  device_eds_type = PEDS_TYPE_SUBDEVICE;
422  peds->set_type( PEDS_TYPE_SUBDEVICE );
423  }
424 
425  // Do some error checking here to make sure we have the minimum requirements for a peds file
426  if ( ( _device_properties.size() == 0 ) || ( peds->entries_size() == 0 ) )
427  {
428  peds.reset();
429  }
430 
431  return ( peds );
432 }

456 void CType1EDS::process_device_properties( json_spirit::mArray &properties )
457 {
458  PUSH_FUNC;
459  // "device":[
460  // {"canonical_name": "family","Value": "Meters"},
461  // ...
462  // ]
463 
464  //
465  // This is the list of Device Params we want to show customers.
466  // All other params from the Device protobuf will be hidden.
467  std::vector<px::string> dev_param_vec
468  {
469  { DEV_PROP_FAMILY },
470  { DEV_PROP_MODEL_SERIES },
471  { DEV_PROP_MODEL },
472  { DEV_PROP_NAME },
473  { DEV_PROP_CAN_NAME },
474  { DEV_PROP_NET_ADDRESS },
475  { DEV_PROP_SERIAL_NUM },
476  { DEV_PROP_VENDOR },
477  { DEV_PROP_ASSET_ID },
478  { DEV_PROP_LOCATION },
479  { DEV_PROP_CONTACT },
480  { DEV_PROP_HW_VER },
481  { DEV_PROP_SW_VER },
482  { DEV_PROP_SYMBOL_NAME },
483  { DEV_PROP_ROLE },
484  { DEV_PROP_WAVE_SUPP }
485  };
486 
487  for ( const auto &dev_param : dev_param_vec )
488  {
489  if ( ( dev_param == DEV_PROP_SERIAL_NUM ) ||
490  ( dev_param == DEV_PROP_HW_VER ) ||
491  ( dev_param == DEV_PROP_SW_VER ) ||
492  ( dev_param == DEV_PROP_ROLE ) ||
493  ( dev_param == DEV_PROP_WAVE_SUPP ) )
494  {
495  bool found_prop = false;
496  for ( auto &prop : properties )
497  {
498  auto &obj = prop.get_obj();
499  px::string prop_name = read_string_value( obj, px::string( EDS_TAG_CANONICAL_NAME ) );
500  px::string prop_val = read_string_value( obj, px::string( EDS_TAG_VALUE ) );
501  if ( dev_param == prop_name )
502  {
503  found_prop = true;
504  // Handle the waveform supported property differently
505  if ( dev_param == DEV_PROP_WAVE_SUPP )
506  {
507  // Display "waveform enabled" property instead of "waveform supported"
508  _device_properties.push_back( std::make_pair( DEV_PROP_WAVE_ENBLD, MB_UINT32_TYPE ) );
509  }
510  else
511  {
512  _device_properties.push_back( std::make_pair( dev_param, prop_val ) );
513  }
514  break;
515  }
516  }
517  if ( !found_prop )
518  {
519  _device_properties.push_back( std::make_pair( NO_PROP_VALUE, "" ) );
520  }
521  }
522  else
523  {
524  _device_properties.push_back( std::make_pair( dev_param, "" ) );
525  }
526  }

567 void CType1EDS::process_commands( json_spirit::mArray &commands )
568 {
569  PUSH_FUNC;
570 
571  // "command_list": [
572  // { "canonical_name": "cSARstEnergy",
573  // "Value": "5555",
574  // "Translation": "(code)",
575  // "Command": "WRITE_SINGLE_REGISTER",
576  // "value_type": "UINT32", **optional**
577  // },
578  // ...
579  // ]
580 
581  for ( json_spirit::mValue entry : commands )
582  {
583  auto cmd_obj = entry.get_obj();
584 
585  CCommandEntry cmd_entry;
586  cmd_entry.canonical_name = read_string_value( cmd_obj, px::string( EDS_TAG_CANONICAL_NAME ) );
587  cmd_entry.translation = read_string_value( cmd_obj, px::string( EDS_TAG_TRANSLATION ) );
588  cmd_entry.command_type = read_string_value( cmd_obj, px::string( EDS_TAG_COMMAND ) );
589  cmd_entry.automatic = false;
590  cmd_entry.hidden = false;
591  cmd_entry.has_parameter_info = false;
592 
593  if ( cmd_obj.find( EDS_TAG_COMMAND_INFO ) != cmd_obj.end() )
594  {
595  auto cmd_info = cmd_obj[EDS_TAG_COMMAND_INFO].get_array();
596  json_spirit::mObject::const_iterator itr;
597  cmd_entry.has_parameter_info = false;
598 
599  for ( json_spirit::mValue entry : cmd_info )
600  {
601  auto info_obj = entry.get_obj();
602  itr = info_obj.find( EDS_TAG_IS_AUTOMATIC );
603  if ( itr != info_obj.end() )
604  {
605  cmd_entry.automatic = itr->second.get_bool();
606  SVR_DEBUG( 5, "----- is_automatic %u", cmd_entry.automatic );
607  }
608 
609  itr = info_obj.find( EDS_TAG_IS_HIDDEN );
610  if ( itr != info_obj.end() )
611  {
612  cmd_entry.hidden = itr->second.get_bool();
613  SVR_DEBUG( 5, "----- is_hidden %u", cmd_entry.hidden );
614  }
615  }
616 
617  for ( json_spirit::mValue entry : cmd_info )
618  {
619  auto info_obj = entry.get_obj();
620  if ( info_obj.find( "ParameterInfo" ) != info_obj.end() )
621  {
622  cmd_entry.has_parameter_info = true;
623  break;
624  }
625  }
626  }
627 
628  // get the External value type
629  cmd_entry.external_value_type = read_string_value( cmd_obj, px::string( EDS_TAG_EXT_VALUE_TYPE ) );
630 
631  // get the CDS value type
632  cmd_entry.value_type = read_data_type( cmd_obj, px::string( EDS_TAG_VALUE_TYPE ) );
633  _commands.push_back( cmd_entry );
634  }
635 
636  SVR_DEBUG( 4, "Command count: %zu", _commands.size() );
637 }

639 void CType1EDS::process_pollblocks( json_spirit::mArray &pollblocks )
640 {
641  PUSH_FUNC;
642 
643  // "pollblocks":[
644  // { "name": "xyz",
645  // ...,
646  // "Dependency": "(code)" **optional**
647  // "data_block_translation": [ **optional**
648  // { "Command":"xyz",
649  // ...,
650  // "N_Objects": n,
651  // "value_type": "DOUBLE"
652  // },
653  // ...
654  // ]
655  // },
656  // ...
657  // ]
658 
659  for ( auto &entry : pollblocks )
660  {

711 void CType1EDS::process_channels( json_spirit::mArray &channels )
712 {
713  PUSH_FUNC;
714 
715  // "channel_list": [
716  // { "canonical_name": "sStatus",
717  // "Translation": "(code)",
718  // "value_type": "INT32", **optional**
719  // "multistate_maps":[ **optional**
720  // {"str_val":"Inactive","int32_val":0},
721  // {"str_val":"Operational","int32_val":1},
722  // ...
723  // ]
724  // },
725  // ...
726  // ]
727 
728  std::unordered_set<px::string> name_overrides;
729  for ( json_spirit::mValue entry : channels )
730  {
731  auto ch_obj = entry.get_obj();
732 
733  CChannelEntry ch_entry;
734 
735  auto name_override = read_string_value( ch_obj, px::string( EDS_TAG_ALT_NAME ) );
736  if ( !name_override.empty() )
737  {
738  auto ret = name_overrides.emplace( name_override );
739  if ( ret.second )
740  {
741  ch_entry.canonical_name = name_override;
742  }
743  else
744  {
745  continue;
746  }
747  }
748  else
749  {
750  ch_entry.canonical_name = read_string_value( ch_obj, px::string( EDS_TAG_CANONICAL_NAME ) );
751  }
752 
753  ch_entry.translation = read_string_value( ch_obj, px::string( EDS_TAG_TRANSLATION ) );
754  //ch_entry.dependency = read_string_value(ch_obj, px::string(EDS_TAG_DEPENDENCY));
755 
756  if ( ch_obj.find( EDS_TAG_MULTISTATE_MAPS ) != ch_obj.end() )
757  {
758  auto multistate_maps = ch_obj[EDS_TAG_MULTISTATE_MAPS].get_array();
759  for ( auto &m : multistate_maps )
760  {
761  auto map_obj = m.get_obj();
762  std::unordered_map<px::string, int> map;
763  map.emplace( read_string_value( map_obj, px::string( EDS_TAG_STRVAL ) ), read_int_value( map_obj, px::string( EDS_TAG_INT32VAL ) ) );
764  ch_entry.multistate_maps = map;
765  }
766  }
767 
768  // get the value type
769  if ( ch_obj.find( EDS_TAG_VALUE_TYPE ) != ch_obj.end() )
770  {
771  ch_entry.value_type = read_data_type( ch_obj, px::string( EDS_TAG_VALUE_TYPE ) );
772  }
773  else if ( ch_entry.translation.size() > 0 )
774  {
775  // have to deduce it from xBlock using translation
776  //SVR_DEBUG(5, "--- %s has translation '%s'",
777  // ch_entry.canonical_name.c_str(),
778  // ch_entry.translation.c_str());
779  ch_entry.value_type = get_channel_datatype( ch_entry );
780  }
781  else
782  {
783  SVR_DEBUG( 4, "Skipped a channel (without value type): '%s'", ch_entry.canonical_name.c_str() );
784  continue;
785  }
786  _channels.push_back( ch_entry );
787  }
788 
789  // second pass to handle possible formula channels
790  for ( auto &channel : _channels )
791  {
792  if ( !channel.referenced_channels.size() == 0 )
793  {
794  continue;
795  }
796 
797  for ( auto &rc_name: channel.referenced_channels )
798  {
799  auto iter = std::find_if( _channels.begin(),
800  _channels.end(),
801  [&] ( const CChannelEntry &ch ){
802  return ( ch.canonical_name.compare( rc_name.c_str() ) == 0 );
803  }
804  );
805 
806  if ( iter != _channels.end() )
807  {
808  channel.value_type = iter->value_type;
809  }
810  }
811  }
812 
813  SVR_DEBUG( 4, "Channels count: %zu", _channels.size() );
814 }

863 std::pair<px::string, int> CType1EDS::get_xblock_and_index( px::string &translation, std::vector<px::string> &referenced )
864 {
865  PUSH_FUNC;
866 
867  std::pair<px::string, int> result;
868 
869  try
870  {
871  // use equation to get the number of values
872  // we only need to handle the simple case xBlock[n]
873  // In this case, there is only one value
874 
875  CPxValueHandler fHandler;
876  px::eq::CEquation equation( translation );
877  equation.test_evaluate( fHandler );
878 
879  if ( fHandler.vals.size() == 0 )
880  {
881  return ( result );
882  }
883 
884  for ( auto &val : fHandler.vals )
885  {
886  // get the xBlock name and index
887  // example xyz[n]: xblock=xyz, index=n
888  std::string tmp = px::to_basicstr( val.first );
889  size_t begin_bracket_pos = tmp.find( '[' );
890  size_t end_bracket_pos = tmp.find( ']' );
891 
892  if ( ( begin_bracket_pos != std::string::npos ) &&
893  ( end_bracket_pos != std::string::npos ) &&
894  ( end_bracket_pos - begin_bracket_pos > 1 ) )
895  {
896  result.first = tmp.substr( 0, begin_bracket_pos ).c_str();
897  std::string index_str = tmp.substr( begin_bracket_pos + 1, end_bracket_pos - begin_bracket_pos - 1 );
898  result.second = std::atoi( index_str.c_str() );
899  return ( result );
900  }
901  }
902 
903  // if we are here, the channel is a formal channel
904  for ( auto &val : fHandler.vals )
905  {
906  if ( val.first.size() > 1 )
907  {
908  referenced.push_back( val.first );
909  }
910  }
911  }
912  catch ( ... )
913  {
914  SVR_ERR( 5, "Error in process channel translation %s.", translation.c_str() );
915  }
916 
917  return ( result );
918 }
