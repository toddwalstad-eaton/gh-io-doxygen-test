381 {
382  // Validate the operations; if they depend on something, check it out
383  // Set the selected_op to OP_UNDEFINED and print the error if the check fails
384  bool is_bad = false;
385  const char *missing_text = nullptr;
386 
387  switch ( selected_op )
388  {
389  case OP_DEV_EDS_DIR_TO_PEDS_DIR: // -C option
390  if ( peds_conv_opts.primary_opts.size() < PRIMARY_OPTIONS_VEC_SIZE )
391  {
392  is_bad = true;
