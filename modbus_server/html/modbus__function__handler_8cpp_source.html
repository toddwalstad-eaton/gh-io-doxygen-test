52 CModbusFunctionHandler::~CModbusFunctionHandler()
53 {
54  PUSH_FUNC;
55 
56  // This destructor must be implemented to prevent
57  // memory leak when unique_ptr is deleted.
58 
59  if ( _prot.timer )
60  {
61  SVR_FDEBUG( 2, "Clear timer callback" );
62  // The function still owns the timer. We de-assign ourselves as
63  // its owner. If it's destroyed (because our shared ptr was
64  // the last one), then it doesn't matter. But if there are
65  // other copies floating around, the timer's expiration won't
66  // try to wrongly call us back.
67 
68  _prot.timer->assign( nullptr );
69  _prot.timer.reset();
70  }
71 }
