155  px::datatype get_value_type( int index )
156  {
157  if ( ( translations.size() > 0 ) && ( _data_types.size() == 0 ) )
158  {
159  for ( auto &entry : translations )
160  {
161  for ( int i = 0; i < entry.count; i++ )
162  {
163  _data_types.push_back( entry.value_type );
164  }
165  }
166  }
167 
168  if ( ( index < 0 ) || ( index >= int( _data_types.size() ) ) )
169  {
170  return ( px::datatype::invalid );
171  }
172 
173  return ( _data_types[index] );
174  }
