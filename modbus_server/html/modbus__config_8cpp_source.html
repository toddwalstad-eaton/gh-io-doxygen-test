789 void CModbusConfig::handle_oob_change( px::proto::toolkitStruct &update )

853  }
854  }
855  }
856  else
857  {

859  }
860 
861  if ( uid_change_occurred )
862  {
863  // make sure the new_uid is valid (this should be validated by UI)
864  // but we do check here just in case...
865  if ( ( new_uid < UNIT_ID_LO_LIMIT ) || ( new_uid > UNIT_ID_HI_LIMIT ) )
866  {

879  {

916  bool old_offline = tcp_it->second->offline;

926  if ( add_remove_device( true,
927  entry.did(),
928  new_uid,
929  new_peds_file,
930  "",
931  new_subdevice_peds_file,
932  old_writes_disabled,
933  old_fire_forget,
934  old_zero_fill,
935  old_offline ) )

940  if ( peds_mapping_it != _peds_map_store.end() )
