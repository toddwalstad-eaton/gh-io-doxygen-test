48  PUSH_FUNC;
49 
50  assert( offset < _str.size() );
51

53 
54  switch ( _str[offset] )
55  {
56  case 'd':
57  retval = broken_down_time.tm_mday;
58  SVR_INFO( 5, "d: %" PRIX16 "", retval );

60  case 'H':
61  retval = broken_down_time.tm_hour;
62  SVR_INFO( 5, "H: %" PRIX16 "", retval );
63  break;
64  case 'I':
65  retval = broken_down_time.tm_hour;
66  if ( ( retval == 0 ) || ( retval == 12 ) )
67  {
68  retval = 12;
69  }
70  else
71  {
72  retval = retval % 12;
73  }
74  SVR_INFO( 5, "I: %" PRIX16 "", retval );
75  break;
76  case 'j':
77  retval = broken_down_time.tm_yday + 1;
78  SVR_INFO( 5, "j: %" PRIX16 "", retval );
79  break;
80  case 'm':
81  retval = broken_down_time.tm_mon + 1;
82  SVR_INFO( 5, "m: %" PRIX16 "", retval );
83  break;
84  case 'M':
85  retval = broken_down_time.tm_min;
86  SVR_INFO( 5, "M: %" PRIX16 "", retval );
87  break;
88  case 'S':
89  retval = broken_down_time.tm_sec;
90  SVR_INFO( 5, "S: %" PRIX16 "", retval );
91  break;
92  case 'u':
93  retval = broken_down_time.tm_wday;
94  if ( retval == 0 )
95  {
96  retval = 7;
97  }
98  SVR_INFO( 5, "u: %" PRIX16 "", retval );
99  break;
100  case 'w':
101  retval = broken_down_time.tm_wday;
102  SVR_INFO( 5, "w: %" PRIX16 "", retval );
103  break;
104  case 'y':
105  // Year, no century
106  retval = broken_down_time.tm_year % 100;
107  SVR_INFO( 5, "y: %" PRIX16 "", retval );
108  break;
109  case 'Y':
110  // Year with century (add 1900 to broken-down time year)
111  retval = broken_down_time.tm_year + 1900;
112  SVR_INFO( 5, "Y: %" PRIX16 "", retval );
113  break;
114  case 'q':
115  retval = static_cast<int>( ( timestamp.tv_usec ) / 1000 );
116  SVR_INFO( 5, "q: %" PRIX16 "", retval );
117  break;
118  case 'Q':
119  retval = static_cast<int>( ( timestamp.tv_usec ) / 1000 + ( broken_down_time.tm_sec * 1000 ) );
120  SVR_INFO( 5, "Q: %" PRIX16 "", retval );
121  break;
122  default:
123  assert( false );
124  break;
125  }
126 
127  return ( retval );

140  assert( offset < _str.size() );
141 
142  switch ( _str[offset] )
143  {
144  case 'd':
145  broken_down_time.tm_mday = val;
146  break;
147  case 'H':
148  broken_down_time.tm_hour = val;
149  break;
150  case 'I':
151  if ( ( val == 0 ) || ( val == 12 ) )
152  {
153  val = 12;
154  }
155  else
156  {
157  val = val % 12;
158  }
159  broken_down_time.tm_hour = val;
160  break;
161  case 'j':
162  broken_down_time.tm_yday = val - 1;
163  break;
164  case 'm':
165  broken_down_time.tm_mon = val - 1;
166  break;
167  case 'M':
168  broken_down_time.tm_min = val;
169  break;
170  case 'S':
171  broken_down_time.tm_sec = val;
172  break;
173  case 'u':
174  if ( val == 0 )
175  {
176  val = 7;
177  }
178  broken_down_time.tm_wday = val;
179  break;
180  case 'w':
181  broken_down_time.tm_wday = val;
182  break;
183  case 'y':
184  // (Assume we're in the right century)
185  broken_down_time.tm_year /= 100;
186  broken_down_time.tm_year *= 100;
187  broken_down_time.tm_year += val;
188  break;
189  case 'Y':
190  broken_down_time.tm_year = val;
191  break;
192  case 'q':
193  timestamp.tv_usec = val * 1000;
194  break;
195  case 'Q':
196  timestamp.tv_usec = ( val % 1000 ) * 1000;
197  broken_down_time.tm_sec = val / 1000;
198  break;
199  default:
200  assert( false );
201  break;
202  }
