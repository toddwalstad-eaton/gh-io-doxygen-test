50 std::vector<bool> CDiscreteResultSet::translate( bool zero_fill ) const
51 {
52  PUSH_FUNC;
53 
54  std::vector<bool> retval( _addrs.size(), false );
55  std::vector<bool> assigned( _addrs.size(), false );
56 
57  // For each mapping that we've accumulated (in descending order of
58  // parameter priority)...
59  for ( const auto &entry : _entries )
60  {
61  assert( entry.addr().size() == 1 );
62  assert( entry.addr().first >= _addrs.first );
63  assert( entry.addr().first <= _addrs.second );
64 
65  size_t offset = entry.addr().first - _addrs.first;
66 
67  if ( !assigned[offset] )
68  {
69  assigned[offset] = true;
70  retval[offset] = entry.translate();
71  }
72  }
73 
74  if ( !zero_fill )
75  {
76  // Ensure that all of the addresses were covered in this
77  // request, i.e. there are no holes
78  for ( size_t i = 0; i < assigned.size(); i++ )
79  {
80  if ( !assigned[i] )
81  {
82  SVR_FDEBUG( 2, "No mapping available at address %zu", i + _addrs.first );
83  throw px::generic_err();
84  }
85  }
86  }
87 
88  // 'retval' should now be complete
89  return ( retval );
90 }
