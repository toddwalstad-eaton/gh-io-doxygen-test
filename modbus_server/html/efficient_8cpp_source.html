100  if ( !is_subdevice )

116  }

134  }

136  // Add channel properties using ordered map
137  for ( auto &channel : channels )
138  {
139  if ( ( channel.second.value_type == px::datatype::invalid ) || ( channel.second.value_type == px::datatype::binary ) )
140  {

146  entry->set_name( channel.second.canonical_name.c_str() );
147 
148  if ( channel.second.value_type == px::datatype::boolean )

150  entry->set_discrete( true );
151  entry->set_address( discrete_address++ );
152  }
153  else
154  {
155  int count = CModbusUtil::get_register_count( channel.second.value_type.val() );
156  entry->set_address( reg_address );
157 
158  if ( channel.second.value_type == px::datatype::timestamp )
159  {
160  // use default time-spec
161  entry->set_date_spec( MB_DEFAULT_DATETIME_SPEC );
162 
163  // number of registers depends on the spec
164  if ( ( entry->date_spec().size() ) % 2 != 0 )
165  {
166  throw px::generic_err( "invalid date time spec" );
167  }
168  count = entry->date_spec().size() / 2;
169  }
170  entry->set_size( 2 * count );
171 
172  px::CDataType dt( channel.second.value_type );
173  entry->set_value_type( dt.str() );
174 
175  reg_address += count;
176  }
177 
178  entry->set_writable( channel.second.flags.get( px::channel_flag::is_writable ) );
179  entry->set_command( false );
180 
181  SVR_FDEBUG( 9, "Added channel %s", channel.second.canonical_name.c_str() );
182  }
183 
184  // Add command properties
185  std::unordered_map<px::string, px::CDeviceCommand> commands;
186  try
187  {
188  commands = cds.list_device_commands( dev.id() );
189  }
190  catch ( const px::generic_err &e )
191  {
192  SYSLOG_ERR( 1, "create_peds_from_device: Exception error thrown by list_device_commands: %s", e.what() );
193  }
194 
195  for ( auto &command : commands )
196  {
197  if ( command.second.is_hidden )
198  {
199  SVR_DEBUG( 4, "Skip hidden command %s", command.second.name.c_str() );
200  continue;
201  }
202 
203  entry = peds->add_entries();
204  entry->set_name( command.second.name.c_str() );
205  entry->set_writable( true );
206  entry->set_command( true );
207  entry->set_discrete( true );
208  entry->set_address( command_discrete++ );
209 
210  SVR_FDEBUG( 9, "Added command %s", command.second.name.c_str() );
211  }
212 
213  SVR_DEBUG( 4, "Added %d total entries to peds for all models", peds->entries_size() );
214 
215  // Do some error checking here to make sure we have the minimum requirements for a peds file
216  if ( peds->entries_size() == 0 )
217  {
218  peds.reset();
219  }
220 
221  return ( peds );
