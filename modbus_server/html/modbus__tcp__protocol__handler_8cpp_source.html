96 void CModbusTcpProtocolHandler::execute()
97 {
98  PUSH_FUNC;
99 
100  assert( _adu );
101  CModbusTcpAdu &adu = px::poly_cast<CModbusTcpAdu &>( *_adu );
102 
103  SVR_FDEBUG( 5, "Executing command for UnitID %u function code 0x%02X", ( unsigned )adu.header.uid, adu.pdu.function_code );
104 
105  assert( _transactor != nullptr );
106  px::asio::CIoTransactor &transactor = *_transactor;
107 
108  if ( adu.header.pid != 0 )
109  {
110  // Modbus spec says we should ignore this, so we simply
111  // instruct the messenger to switch back into read mode
112  SVR_FDEBUG( 2, "Received request with non-Modbus PID (%u)", ( unsigned )adu.header.pid );
113 
114  transactor.read_more();
115  return;
116  }
117 
118  //px::asio::CServerCore &core=transactor.get_server_core();
119  CModbusConfig &cfg = get_config();
120 
121  if ( cfg.tcp_timeout() != 0 )
122  {
123  // Start an I/O timer right now, setting a timer at 90% of our
124  // max timeout (to allow for some processing delay)
125  size_t timeout = cfg.tcp_timeout() * 9 / 10;
126  SVR_DEBUG( 3, "Using timeout of %zu msec", timeout );
127 
128  timer = px::asio::CIoTimer::create( transactor, timeout );
129  }
130 
131  CModbusDeviceConfig *pcfg = cfg.tcp_config( adu.header.uid );
132  if ( pcfg == nullptr )
133  {
134  SVR_FDEBUG( 5, "Received request for unknown UnitID %u", ( unsigned )adu.header.uid );
135  err( modbus_err_t::MB_GATEWAY_UNAVAILABLE );
136  return;
137  }
138 
139  if ( pcfg->offline )
140  {
141  SVR_FDEBUG( 2, "Device %u offline. Unable to process request", ( unsigned )adu.header.uid );
142  err( modbus_err_t::MB_GATEWAY_UNRESPONSIVE );
143  return;
144  }
145 
146  // That's all of the TCP-specific stuff; now we can utilize the
147  // general-purpose Modbus request handler
148  handle_request( adu.pdu, *pcfg, cfg.nan_to_zero() );
149 }
