127 void CMbSecConnection::handle_handshake( const boost::system::error_code &error )
128 {
129  PUSH_FUNC;
130 
131  px::string certs_role;
132  if ( !error )
133  {
134  handshake_in_progress = false;
135  SSL *this_ssl = _sock->native_handle();
136  if ( this_ssl != nullptr )
137  {
138  X509 *peer_cert = SSL_get_peer_certificate( this_ssl );
139  if ( peer_cert != nullptr )
140  {
141  char *name_buf = X509_NAME_oneline( X509_get_subject_name( peer_cert ), nullptr, 256 );
142  SYSLOG_INFO( 0, "Client presented certificate %s", name_buf );
143  free( name_buf );
144  ASN1_OBJECT *role_oid = OBJ_txt2obj( "1.3.6.1.4.1.3833.7777.1", 1 );
145  int role_idx = X509_get_ext_by_OBJ( peer_cert, role_oid, -1 );
146  ASN1_OBJECT_free( role_oid );
147  if ( role_idx != -1 )
148  {
149  X509_EXTENSION *role_ext = X509_get_ext( peer_cert, role_idx );
150  if ( role_ext != nullptr )
151  {
152  ASN1_OCTET_STRING *certs_role_asn = X509_EXTENSION_get_data( role_ext );
153  const unsigned char *role = certs_role_asn->data;
154  long role_len;
155  int tag, xclass;
156  ( void )ASN1_get_object( &role, &role_len, &tag, &xclass, certs_role_asn->length );
157  certs_role = px::string( ( const char * )role, ( size_t )role_len );
158  SYSLOG_INFO( 0, "Certificate has role %s", certs_role.c_str() );
159  }
160  }
161  X509_free( peer_cert );
162  }
163  }
164  if ( !certs_role.empty() )
165  {
166  if ( !set_role_token( certs_role ) )
167  {
168  SYSLOG_WARN( 0, "Could not set role %s, dropping connection",
169  certs_role.c_str() );
170  close();
171  remove();
172  return;
173  }
174  }
175  start();
176  }
177  else
178  {
179  close();
180  remove();
181  }
182 }

195 bool CMbSecConnection::set_role_token( const px::string &role )
196 {
197  PUSH_FUNC;
198 
199  px::cds::CGetRole gr;
200  gr.role_names.push_back( role );
201  std::shared_ptr<px::cds::CGetRole> gr_rv = std::dynamic_pointer_cast<px::cds::CGetRole>( px::cds::CClient::get()->submit_wait( gr ) );
202  if ( ( !gr_rv ) || gr_rv->role_defs.empty() ||
203  ( gr_rv->role_defs[0].first != role ) )
204  {
205  SYSLOG_WARN( 0, "Role %s not found", role.c_str() );
206  return ( false );
207  }
208  auto role_perms = gr_rv->role_defs[0].second;
209 
210  SYSLOG_DEBUG( 5, "Got permissions for role, logging in" );
211  // Get master token
212  px::cds::CCdsClientLogin mli;
213  mli.service = "mercury";
214  std::shared_ptr<px::cds::CCdsClientLogin> mli_rv = std::dynamic_pointer_cast<px::cds::CCdsClientLogin>( px::cds::CClient::get()->submit_wait( mli ) );
215  main_token = mli_rv->session_token;
216  auto my_perms = mli_rv->permissions;
217  SYSLOG_DEBUG( 5, "CDS daemon login returned token %s",
218  main_token.c_str() );
219 
220  // Get the mask - the permissions we want to drop:
221  // any of ours not set in the role
222  auto perms_mask = my_perms;
223  perms_mask.mask_clear( role_perms );
224 
225  // Get one with this set of permissions
226  px::cds::CSubsessionLogin li;
227  li.parent_token = main_token;
228  if ( li.parent_token.empty() )
229  {
230  SYSLOG_WARN( 0, "No CDS daemon token exists to make a subsession of" );
231  return ( false );
232  }
233  li.permissions_mask = perms_mask;
234  std::shared_ptr<px::cds::CSubsessionLogin> li_rv = std::dynamic_pointer_cast<px::cds::CSubsessionLogin>( px::cds::CClient::get()->submit_wait( li ) );
235  if ( ( !li_rv ) || ( li_rv->session_token.empty() ) )
236  {
237  SYSLOG_WARN( 0, "Subsession login failed" );
238  return ( false );
239  }
240  std::shared_ptr<CModbusProtocolHandler> ph = std::dynamic_pointer_cast<CModbusProtocolHandler>( _prot );
241  ph->token = li_rv->session_token;
242  SYSLOG_DEBUG( 5, "Login for this connection for role %s returned token %s, permissions %s",
243  role.c_str(), ph->token.c_str(),
244  px::permission::perms_to_short_str( li_rv->permissions ).c_str() );
245  return ( true );
246 }
