33 CModbusCdsHandle::CModbusCdsHandle( CModbusServer *core, std::vector<unsigned int> &port_id_vec, uint32_t wait )
34  : CCdsHandler( APP_NAME, wait ),
35  oob_uid_map_subscription( 0 ),
36  device_state_change_subcription( 0 ),
37  _core( core )
38 {
39  PUSH_FUNC;
40 
41  //Ask the CDS to manage our subscriptions for us on a per-port basis
42  px::cds::CManageSubscriberIDs manage;
43 
44  manage.new_ids_requested = MIN_SUBSCIBE_IDS; // Default is 1 for oob_uid_map plus 1 for device_state_change subscriptions
45  manage.reflexive = false;
46 
47  if ( !port_id_vec.size() )
48  {
49  port_ids.insert( CDS_PUBLISHER_ID ); // Default to the Publisher ID
50  manage.new_ids_requested++;
51  }
52  else
53  {
54  port_ids = check_ports( port_id_vec );
55  if ( port_ids.size() )
56  {
57  manage.new_ids_requested += port_ids.size();
58  }
59  else
60  {
61  // Sanity check
62  port_ids.insert( CDS_PUBLISHER_ID ); // Default to the Publisher ID
63  manage.new_ids_requested++;
64  }
65  }
66 
67  std::shared_ptr<px::cds::CManageSubscriberIDs> manage_resp = std::dynamic_pointer_cast<px::cds::CManageSubscriberIDs>( submit_wait( manage ) );
68  check_cds_result( manage_resp, "CManageSubscriberIDs" );
69  if ( manage_resp->new_ids.size() != manage.new_ids_requested )
70  {
71  throw px::generic_err( "Failed to get subscriptions to the CDS" );
72  }
73 
74  //SVR_INFO(4, "manage.new_ids_requested %u, port_ids.size() %u, manage_resp->new_ids.size() %u", manage.new_ids_requested, port_ids.size(), manage_resp->new_ids.size());
75 
76  size_t new_id_index = 0;
77  px::cds::CSubscribe subscribe;
78  for ( auto port_id : port_ids )
79  {
80  dev_change_subscription.insert( manage_resp->new_ids.at( new_id_index ) );
81  subscribe.subscriptions.emplace_back( manage_resp->new_ids.at( new_id_index ), px::sub::subscription_type::device_create_recursive, port_id );
82  subscribe.subscriptions.emplace_back( manage_resp->new_ids.at( new_id_index ), px::sub::subscription_type::device_delete_recursive, port_id );
83  subscribe.subscriptions.emplace_back( manage_resp->new_ids.at( new_id_index ), px::sub::subscription_type::device_state_recursive, port_id );
84  SVR_INFO( 2, "CDS change signature subscription: %" PRIu64 " tied to Port ID %" PRIu64 "", manage_resp->new_ids.at( new_id_index ), port_id );
85  new_id_index++;
86  }
87 
88  oob_uid_map_subscription = manage_resp->new_ids.at( new_id_index++ );
89  SVR_INFO( 2, "CDS UID map subscription: %" PRIu64 "", oob_uid_map_subscription );
90 
91  device_state_change_subcription = manage_resp->new_ids.at( new_id_index++ );
92  SVR_INFO( 2, "CDS state subscription: %" PRIu64 "", device_state_change_subcription );
93 
94  //subscribe to all add/delete device notifications
95  std::shared_ptr<px::cds::CSubscribe> sub_resp = std::dynamic_pointer_cast<px::cds::CSubscribe>( submit_wait( subscribe ) );
96  check_cds_result( sub_resp, "CSubscribe to device changes" );
97 }

229 void CModbusCdsHandle::handle_response( std::shared_ptr<px::cds::CRequestResponse> resp )
230 {
231  PUSH_FUNC;
232 
233  auto &cfg = _core->get_config();
234 
235  SVR_DEBUG( 2, "-------------- CModbusCdsHandle::handle_response" );
236 
237  if ( ( !resp->is_subscription() ) && ( !resp->is_unsolicited() ) )
238  {
239  //find message in map indexed by transID and post results
240  SVR_DEBUG( 3, "Handle CDS request response begin" );
241 
242  assert( resp->transaction_id != 0 );
243  CModbusCdsHandle::fcn_map_t::iterator it = idle_fcns.find( resp->transaction_id );
244 
245  if ( it == idle_fcns.end() )
246  {
247  // did not find the cached trans id, ignore the response.
248  // this can happen if the caller does not care the return, or a
249  // timeout occured and removed it.
250  SVR_INFO( 3, " CDS request response with un-tracked trans ID: %d size:%zu.", resp->transaction_id, idle_fcns.size() );
251  }
252  else
253  {
254  // validate the device has not been deleted while we were waiting
255  // to process this request.
256  std::shared_ptr<cds::getv_field_t> get_resp = std::dynamic_pointer_cast<cds::getv_field_t>( resp );
257  if ( get_resp.get() )
258  {
259  for ( auto &field: get_resp->fields )
260  {
261  if ( !cfg.valid_device( field.id ) )
262  {
263  resp->status = px::cds::RequestStatus::FAILED;
264  }
265  }
266  }
267  else
268  {
269  std::shared_ptr<cds::setv_field_t> set_resp = std::dynamic_pointer_cast<cds::setv_field_t>( resp );
270  if ( set_resp.get() )
271  {
272  for ( auto &field: set_resp->fields )
273  {
274  if ( !cfg.valid_device( field.id ) )
275  {
276  resp->status = px::cds::RequestStatus::FAILED;
277  }
278  }
279  }
280  }
281 
282  if ( !( *it ).second->post( resp ) )
283  {
284  // still expecting more responses, the request must be cached again.
285  SVR_DEBUG( 3, " New Post transaction-> %d", resp->transaction_id );
286  add_idle_fcn( ( *it ).second );
287  }
288  SVR_DEBUG( 3, " remove transaction-> %d", resp->transaction_id );
289  idle_fcns.erase( it );
290  }
291  SVR_DEBUG( 3, " Handle CDS request response end" );
292  SVR_DEBUG( 3, " Idle Map size: %zu", idle_fcns.size() );
293  }
294  else
295  {
296  SVR_DEBUG( 2, "Handle CDS unsolicited response begin" );
297  std::shared_ptr<px::cds::CSubscriberNotify> sub_resp = std::dynamic_pointer_cast<px::cds::CSubscriberNotify>( resp );
298 
299  //Something bad happened if we can't get to our core
300  assert( _core );
301 
302  if ( sub_resp->subscriber_id == oob_uid_map_subscription )
303  {
304  // Handle oob_uid_map_subscription
305 
306  SVR_DEBUG( 2, " CDS subscriber_id==oob_uid_map_subscription, oobs.size %zu, cleared_oobs.size %zu",
307  sub_resp->oobs.size(), sub_resp->cleared_oobs.size() );
308 
309  for ( auto &entry:sub_resp->oobs )
310  {
311  SVR_DEBUG( 2, "Handle OOB change in %s", entry.first.c_str() );
312  cfg.handle_oob_change( entry.second );
313  }
314 
315  for ( auto &entry:sub_resp->cleared_oobs )
316  {
317  //TODO: investigate std::stoull asa possible simplification
318  //scrap off device id from tag
319  std::stringstream ss;
320  size_t found = entry.find_last_of( "/" ) + 1;
321  ss << entry.substr( found );
322  px::id_t dev_id;
323  ss >> dev_id;
324 
325  if ( dev_id <= 0 )
326  {
327  SVR_FERR( 2, "Unknown tag format .. moving on" );
328  continue;
329  }
330 
331  SVR_DEBUG( 2, "Handle UID MAP removal tag: %s, dev: %" PRIu64 "", entry.c_str(), dev_id );
332  cfg.handle_removed_device( dev_id );
333  }
334  }
335  else if ( sub_resp->subscriber_id == device_state_change_subcription )
336  {
337  // Handle device_state_change_subcription
338 
339  SVR_DEBUG( 2, " CDS subscriber_id==device_state_change_subcription, realtime_states.size %zu, deleted_devices.size %zu",
340  sub_resp->realtime_states.size(), sub_resp->deleted_devices.size() );
341 
342  for ( auto &entry:sub_resp->realtime_states )
343  {
344  SVR_DEBUG( 2, " Handle Device: %" PRIu64 " state update", entry.first );
345  if ( entry.first < CDS_MIN_DEVICE_ID )
346  {
347  continue;
348  }
349 
350  cfg.handle_device_state_change( entry.first, entry.second );
351  }
352 
353  for ( auto &entry:sub_resp->deleted_devices )
354  {
355  SVR_DEBUG( 2, " Handle Deleted Device: %" PRIu64 "", entry );
356  if ( entry < CDS_MIN_DEVICE_ID )
357  {
358  continue;
359  }
360 
361  cfg.handle_removed_device( entry );
362  }
363  }
364  else
365  {
366  // Handle dev_change_subscription
367  auto it_subscription = dev_change_subscription.find( sub_resp->subscriber_id );
368  if ( it_subscription != dev_change_subscription.end() )
369  {
370  SVR_DEBUG( 2, " CDS subscriber_id==dev_change_subscription, realtime_states.size %zu, device_settings.size %zu, deleted_devices.size %zu",
371  sub_resp->realtime_states.size(), sub_resp->device_settings.size(), sub_resp->deleted_devices.size() );
372 
373  for ( auto &entry:sub_resp->realtime_states )
374  {
375  SVR_DEBUG( 2, " Handle Device: %" PRIu64 " state update", entry.first );
376  if ( entry.first < CDS_MIN_DEVICE_ID )
377  {
378  continue;
379  }
380 
381  cfg.handle_device_state_change( entry.first, entry.second );
382  }
383 
384  for ( auto &entry:sub_resp->device_settings )
385  {
386  SVR_DEBUG( 2, " Handle New Device: %" PRIu64 ", canonical_model_name %s, role %d",
387  entry.first, entry.second.canonical_model_name.c_str(), ( int )entry.second.role.val() );
388  if ( entry.second.parent )
389  {
390  SVR_DEBUG( 2, " parent %" PRIu64 "", *entry.second.parent );
391  }
392 
393  // Avoid device IDs that may be ports
394  if ( entry.first < CDS_MIN_DEVICE_ID )
395  {
396  continue;
397  }
398 
399  bool offline = false;
400  if ( entry.second.get_state().get( px::realtime_state::disabled ) ||
401  entry.second.get_state().get( px::realtime_state::not_communicating ) ||
402  entry.second.is_hidden )
403  {
404  offline = true;
405  }
406 
407  // Only process devices, sub-devices will be handled in the
408  // "device_state_change_subcription" section
409  if ( entry.second.role == px::device_role::device )
410  {
411  if ( !cfg.is_dynamic_mode() )
412  {
413  cfg.handle_new_device( entry.first, entry.second.canonical_model_name, entry.second.config_file, offline );
414  }
415  else
416  {
417  cfg.handle_new_dynamic_device( entry.first, offline );
418  }
419  }
420  }
421 
422  for ( auto &entry:sub_resp->deleted_devices )
423  {
424  SVR_DEBUG( 2, " Handle Deleted Device: %" PRIu64 "", entry );
425  if ( entry < CDS_MIN_DEVICE_ID )
426  {
427  continue;
428  }
429  cfg.handle_removed_device( entry );
430  }
431  }
432  else
433  {
434  SVR_DEBUG( 2, " Unexpected subscription response from CDS" );
435  }
436  }
437  }
438 }
