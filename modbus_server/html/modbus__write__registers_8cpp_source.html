34 bool CModbusWriteRegisters::verify_length( const CModbusPdu &pdu )
35 {
36  PUSH_FUNC;
37 
38  if ( pdu.function_code == MB_WRITE_SINGLE_REGISTER )
39  {
40  return ( pdu.data.size() >= MB_PDU_BYTES_EXPECTED );
41  }
42 
43  assert( pdu.function_code == MB_WRITE_MULTIPLE_HOLDING_REGISTERS );
44  if ( pdu.data.size() < MB_PDU_BYTES_EXPECTED_MAX )
45  {
46  return ( false );
47  }
48 
49  auto nbytes = pdu.data[4];
50  if ( ( pdu.data.size() - MB_PDU_BYTES_EXPECTED_MAX ) < nbytes )
51  {
52  return ( false );
53  }
54 
55  return ( true );
56 }

61 void CModbusWriteRegisters::execute()
62 {
63  PUSH_FUNC;
64 
65  // First, we parse out the command arguments
66 
67  _values.clear();
68 
69  if ( _pdu.function_code == MB_WRITE_SINGLE_REGISTER )
70  {
71  if ( _pdu.data.size() != MB_PDU_BYTES_EXPECTED )
72  {
73  SVR_FDEBUG( 2, "Bad PDU payload size: %zu bytes", _pdu.data.size() );
74  _prot.err( MB_ILLEGAL_DATA_VALUE );
75  return;
76  }
77 
78  px::asio::iobuf_const_iter_t ibuf = _pdu.data.begin();
79 
80  _addr = px::asio::CUtil::read_u16( ibuf );
81  _values.push_back( px::asio::CUtil::read_u16( ibuf ) );
82  }
83  else
84  {
85  assert( _pdu.function_code == MB_WRITE_MULTIPLE_HOLDING_REGISTERS );
86  if ( _pdu.data.size() < MB_PDU_BYTES_EXPECTED_MAX )
87  {
88  SVR_FDEBUG( 2, "PDU data too short: %zu bytes", _pdu.data.size() );
89  _prot.err( MB_ILLEGAL_DATA_VALUE );
90  return;
91  }
92 
93  px::asio::iobuf_const_iter_t ibuf = _pdu.data.begin();
94 
95  _addr = px::asio::CUtil::read_u16( ibuf );
96  auto nout = px::asio::CUtil::read_u16( ibuf );
97  auto nbytes = *ibuf++;
98 
99  SVR_FDEBUG( 2, "Function code MB_WRITE_MULTIPLE_HOLDING_REGISTERS: _addr %u, bytes/output count (%u / %u)", _addr, ( unsigned )nbytes, ( unsigned )nout );
100 
101  if ( ( nbytes != ( nout * 2 ) ) || ( nout == 0 ) || ( nout > 120 ) || ( ( _pdu.data.end() - ibuf ) != nbytes ) )
102  {
103  SVR_FDEBUG( 2, "Invalid byte/output count (%u / %u)", ( unsigned )nbytes, ( unsigned )nout );
104  _prot.err( MB_ILLEGAL_DATA_VALUE );
105  return;
106  }
107 
108  for ( size_t i = 0; i < nout; i++ )
109  {
110  _values.push_back( px::asio::CUtil::read_u16( ibuf ) );
111  }
112  }
113 
114  if ( ( ( size_t )( MB_MAX_REGISTER_ADDR - _addr ) ) < _values.size() )
115  {
116  SVR_FDEBUG( 2, "Address range too large" );
117  _prot.err( MB_ILLEGAL_DATA_ADDRESS );
118  return;
119  }
120 
121  if ( _values.empty() )
122  {
123  SVR_FDEBUG( 2, "Zero-length request" );
124  _prot.err( MB_ILLEGAL_DATA_VALUE );
125  return;
126  }
127 
128  SVR_FDEBUG( 2, "Before write_values() call - _values.size() %zu", _values.size() );
129 
130  //TODO: Next we create a maplist with the relevant parameter mappings
131  write_values();
132 }

137 bool CModbusWriteRegisters::post( response_ptr_t resp )
138 {
139  PUSH_FUNC;
140 
141  if ( _timeout )
142  {
143  return ( true );
144  }
145 
146  if ( resp->status != px::cds::RequestStatus::SUCCEEDED )
147  {
148  // Unknown error encountered in CDS thread; since the request
149  // can't be fulfilled, we'll just send back an error
150  _prot.err( MB_SLAVE_DEVICE_FAILURE );
151  return ( true );
152  }
153 
154  std::shared_ptr<cds::setv_field_t> cds_resp_write = std::dynamic_pointer_cast<cds::setv_field_t>( resp );
155 
156  //TODO: CDS set field would benefit from a return status on a per field basis
157 
158  if ( cds_resp_write )
159  {
160  cds::field_set_t &fields = cds_resp_write->fields;
161  cds::field_set_t status_req;
162  for ( auto &field : fields )
163  {
164  int hndl = field.value.as_int32();
165 
166  //failed to send the command
167  if ( hndl <= 0 )
168  {
169  _prot.err( MB_SLAVE_DEVICE_FAILURE );
170  return ( true );
171  }
172 
173  SVR_FDEBUG( 2, "Get status of command %s:%d", field.field.c_str(), hndl );
174  status_req.emplace( field.type, field.field, hndl );
175  }
176 
177  if ( _cfg.write_status_enabled() )
178  {
179  cds_submit_get( status_req );
180  return ( false );
181  }
182  }
183  else
184  {
185  std::shared_ptr<cds::getv_field_t> cds_resp_retval = std::dynamic_pointer_cast<cds::getv_field_t>( resp );
186  assert( cds_resp_retval );
187 
188  cds::field_set_t ongoing_req;
189  for ( auto &field : cds_resp_retval->fields )
190  {
191  switch ( field.value.as_int32() )
192  {
193  case static_cast<int>( px::device_command_status::in_progress ):
194  case static_cast<int>( px::device_command_status::scheduled ):
195  //TODO: We might want to delay reasking for status to no hammer the cds
196  SVR_FDEBUG( 2, "WRITE of XX command still in progress...going back for more" );
197  ongoing_req.emplace( field.type, field.field, field.id );
198  break;
199  case static_cast<int>( px::device_command_status::unknown ):
200  case static_cast<int>( px::device_command_status::failed ):
201  SVR_FDEBUG( 2, "WRITE of XX command failed" );
202  _prot.err( MB_SLAVE_DEVICE_FAILURE );
203  return ( true );
204 
205  break;
206  default:
207  break;
208  }
209  }
210 
211  if ( ongoing_req.size() > 0 )
212  {
213  cds_submit_get( ongoing_req );
214  return ( false );
215  }
216  }
217 
218  if ( _pdu.function_code == MB_WRITE_SINGLE_REGISTER )
219  {
220  // Preserve the PDU payload as our response
221  assert( _pdu.data.size() == MB_PDU_BYTES_EXPECTED );
222  }
223  else
224  {
225  assert( _pdu.function_code == MB_WRITE_MULTIPLE_HOLDING_REGISTERS );
226 
227  // Preserve the first 4 bytes of the PDU payload as our response
228  assert( _pdu.data.size() >= MB_PDU_BYTES_EXPECTED_MAX );
229  _pdu.data.resize( MB_PDU_BYTES_EXPECTED );
230  }
231 
232  // Command successful - return the response data
233  _prot.respond();
234  return ( true );
235 }

240 void CModbusWriteRegisters::write_values()
241 {
242  PUSH_FUNC;
243 
244  assert( !_values.empty() );
245 
246  CAddressRange addrs( _addr, _addr + _values.size() - 1 );
247 
248  _maplist = _cfg.get_registers( addrs );
249 
250  if ( !_maplist )
251  {
252  SVR_FDEBUG( 2, "Returned nulptr maplist" );
253  _prot.err( MB_ILLEGAL_DATA_ADDRESS );
254  return;
255  }
256 
257  modbus_err_t err = MB_NO_ERROR;
258  auto writes( _maplist->get_writes( _values, err ) );
259 
260  if ( writes.empty() )
261  {
262  SVR_FDEBUG( 2, "No parameters in requested range" );
263  _prot.err( MB_ILLEGAL_DATA_ADDRESS );
264  return;
265  }
266 
267  if ( MB_NO_ERROR != err )
268  {
269  SVR_FDEBUG( 2, "_maplist->get_writes() failed with error %d!", err );
270  _prot.err( err );
271  return;
272  }
273 
274  //TODO: Its a shame we can't use "set_field_value". For now we have to use
275  //the limited cds commands
276  cds_submit_set( writes );
277 
278  // Execution is done for now; when the command is complete, we'll be
279  // notified via post()
280 }
