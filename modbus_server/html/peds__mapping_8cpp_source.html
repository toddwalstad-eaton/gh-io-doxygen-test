90 void CPedsMapping::map( px::proto::Peds &peds )
91 {
92  PUSH_FUNC;
93 
94  assert( peds.entries_size() > 0 );
95 
96  if ( _processed )
97  {
98  return;
99  }
100 
101  eds_limits.min_dev_reg_addr = 0;
102  eds_limits.max_dev_reg_addr = 0;
103 
104  eds_limits.min_disc_addr = 0;
105  eds_limits.max_disc_addr = 0;
106  eds_limits.min_reg_addr = 0;
107  eds_limits.max_reg_addr = 0;
108 
109  eds_limits.min_cmd_disc_addr = 0;
110  eds_limits.max_cmd_disc_addr = 0;
111  eds_limits.min_cmd_reg_addr = 0;
112  eds_limits.max_cmd_reg_addr = 0;
113 
114  size_t id = 1;
115 
116  if ( _address_offset )
117  {
118  //
119  // Build address maps for Sub-Device with the following patterns:
120  //
121  // Register Address Map
122  // channels
123  // commands
124  // device properties
125  //
126  // Discrete Address Map
127  // channels
128  // commands
129  //
130  size_t reg_addr = _address_offset;
131  size_t disc_addr = _address_offset;
132 
133  // Add channel entries first
134  for ( auto entry: peds.entries() )
135  {
136  if ( !entry.has_device() && !entry.has_command() )
137  {
138  if ( entry.has_discrete() && ( entry.discrete() == true ) )
139  {
140  entry.set_address( disc_addr );
141  disc_addr += 1;
142  add_discrete_mapping( entry, id++ );
143  }
144  else
145  {
146  entry.set_address( reg_addr );
147  reg_addr += entry.size() / 2;
148  add_register_mapping( entry, id++ );
149  }
150  SVR_DEBUG( 4, " Sub-device CPedsMapping: channel %s: address %d, size %d", entry.name().c_str(), entry.address(), entry.size() );
151  }
152  }
153 
154  // Add command entries
155  for ( auto entry: peds.entries() )
156  {
157  if ( entry.has_command() && ( entry.command() == true ) )
158  {
159  if ( entry.has_discrete() && ( entry.discrete() == true ) )
160  {
161  entry.set_address( disc_addr );
162  disc_addr += 1;
163  add_discrete_mapping( entry, id++ );
164  }
165  else
166  {
167  entry.set_address( reg_addr );
168  reg_addr += entry.size() / 2;
169  add_register_mapping( entry, id++ );
170  }
171  SVR_DEBUG( 4, " Sub-device CPedsMapping: command %s: address %d, size %d", entry.name().c_str(), entry.address(), entry.size() );
172  }
173  }
174 
175  // Add device "name" and "location" last
176  for ( auto entry: peds.entries() )
177  {
178  if ( entry.has_device() && ( entry.device() == true ) )
179  {
180  if ( ( entry.name() == DEV_PROP_NAME ) || ( entry.name() == DEV_PROP_LOCATION ) )
181  {
182  entry.set_address( reg_addr );
183  reg_addr += entry.size() / 2;
184  add_register_mapping( entry, id++ );
185  SVR_DEBUG( 4, " Sub-device CPedsMapping: device %s: address %d, size %d", entry.name().c_str(), entry.address(), entry.size() );
186  }
187  }
188  }
189 
190  eds_limits.min_reg_addr = _address_offset;
191  eds_limits.max_reg_addr = reg_addr;
192  eds_limits.min_disc_addr = _address_offset;
193  eds_limits.max_disc_addr = disc_addr;
194  }
195  else
196  {
197  //
198  // Build address map for Main Device
199  //
200  bool is_first_dev = true;
201  bool is_first_ch_reg = true;
202  bool is_first_ch_disc = true;
203  bool is_first_cmd_reg = true;
204  bool is_first_cmd_disc = true;
205 
206  for ( const auto entry: peds.entries() )
207  {
208  if ( entry.has_device() && ( entry.device() == true ) )
209  {
210  if ( is_first_dev )
211  {
212  is_first_dev = false;
213  eds_limits.min_dev_reg_addr = entry.address();
214  }
215  add_register_mapping( entry, id++ );
216  eds_limits.max_dev_reg_addr = entry.address() + entry.size() / 2;
217  }
218  else if ( entry.has_command() && ( entry.command() == true ) )
219  {
220  if ( entry.has_discrete() && ( entry.discrete() == true ) )
221  {
222  if ( is_first_cmd_disc )
223  {
224  is_first_cmd_reg = false;
225  eds_limits.min_cmd_disc_addr = entry.address();
226  }
227  add_discrete_mapping( entry, id++ );
228  eds_limits.max_cmd_disc_addr = entry.address() + 1;
229  }
230  else
231  {
232  if ( is_first_cmd_reg )
233  {
234  is_first_cmd_reg = false;
235  eds_limits.min_cmd_reg_addr = entry.address();
236  }
237  add_register_mapping( entry, id++ );
238  eds_limits.max_cmd_reg_addr = entry.address() + entry.size() / 2;
239  }
240  }
241  else
242  {
243  if ( entry.has_discrete() && ( entry.discrete() == true ) )
244  {
245  if ( is_first_ch_disc )
246  {
247  is_first_ch_disc = false;
248  eds_limits.min_disc_addr = entry.address();
249  }
250  add_discrete_mapping( entry, id++ );
251  eds_limits.max_disc_addr = entry.address() + 1;
252  }
253  else
254  {
255  if ( is_first_ch_reg )
256  {
257  // Make sure the Entity channel(s) does not cause a false min address
258  if ( !( ( entry.address() >= START_DEVICE_ADDR ) && ( entry.address() <= START_COMMAND_ADDR ) ) )
259  {
260  is_first_ch_reg = false;
261  eds_limits.min_reg_addr = entry.address();
262  }
263  }
264  add_register_mapping( entry, id++ );
265  eds_limits.max_reg_addr = entry.address() + entry.size() / 2;
266  }
267  }
268  }
269  }
270 
271  SVR_DEBUG( 3, " CPedsMapping: min_reg_addr %u", eds_limits.min_reg_addr );
272  SVR_DEBUG( 3, " CPedsMapping: max_reg_addr %u", eds_limits.max_reg_addr );
273  SVR_DEBUG( 3, " CPedsMapping: min_disc_addr %u", eds_limits.min_disc_addr );
274  SVR_DEBUG( 3, " CPedsMapping: max_disc_addr %u", eds_limits.max_disc_addr );
275  SVR_DEBUG( 3, " CPedsMapping: min_dev_reg_addr %u", eds_limits.min_dev_reg_addr );
276  SVR_DEBUG( 3, " CPedsMapping: max_dev_reg_addr %u", eds_limits.max_dev_reg_addr );
277  SVR_DEBUG( 3, " CPedsMapping: min_cmd_reg_addr %u", eds_limits.min_cmd_reg_addr );
278  SVR_DEBUG( 3, " CPedsMapping: max_cmd_reg_addr %u", eds_limits.max_cmd_reg_addr );
279  SVR_DEBUG( 3, " CPedsMapping: min_cmd_disc_addr %u", eds_limits.min_cmd_disc_addr );
280  SVR_DEBUG( 3, " CPedsMapping: max_cmd_disc_addr %u", eds_limits.max_cmd_disc_addr );
281 
282  _processed = true;
283 }

370 CAddressRange CPedsMapping::get_addrs( const px::proto::PedsEntry &param, size_t naddrs )
371 {
372  PUSH_FUNC;
373 
374  assert( naddrs > 0 );
375  naddrs--;
376 
377  if ( !param.has_address() )
378  {
379  SVR_FERR( 1, "No address specified for mapping" );
380  throw px::generic_err();
381  }
382 
383  long addr = param.address();
384  if ( ( addr < 0 ) || ( addr > MB_MAX_ADDRESS ) )
385  {
386  SVR_FERR( 1, "Address %ld is invalid", addr );
387  throw px::generic_err();
388  }
389 
390  long endaddr = addr + naddrs;
391  if ( ( endaddr < addr ) || ( endaddr > MB_MAX_ADDRESS ) )
392  {
393  SVR_FERR( 1, "Address range (%ld, %ld) is invalid", addr, endaddr );
394  throw px::generic_err();
395  }
396 
397  return ( CAddressRange( ( regaddr_t )addr, ( regaddr_t )endaddr ) );
398 }
