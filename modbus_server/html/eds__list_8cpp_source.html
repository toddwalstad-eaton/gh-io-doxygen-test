151 
152  // Convert global_map elements to protobuf
153  for ( auto &global : _eds_sorted_list.global_map )
154  {

166  global.second.peds_file.c_str(),
167  global.second.description.c_str() );
168  }
169  }
170 
171  // Convert device_map elements to protobuf
172  for ( auto &device : _eds_sorted_list.device_map )
173  {
174  px::proto::EDSInfo *eds_info = eds_list->add_device_list();
175  eds_info->set_eds_file( device.first.c_str() );

189  {
190  peds_info->set_subdevice_peds_file( entry.second.subdevice_peds_file.c_str() );
191  }
192  SVR_DEBUG( 3, " name %s, %s, subdev %s, desc %s",
193  entry.first.c_str(),
194  entry.second.peds_file.c_str(),
195  entry.second.subdevice_peds_file.c_str(),
196  entry.second.description.c_str() );
197  }
198  }
199  }
200
