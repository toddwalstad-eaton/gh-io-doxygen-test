65 regvec_t CTimeMapping::translate( const px::CValue &val ) const
66 {
67  PUSH_FUNC;
68 
69  timeval tstamp;
70 
71  // Use either the timestamp of the value, or the value itself
72  if ( _use_timestamp )
73  {
74  SVR_FERR( 1, "_use_timestamp Feature not supported" );
75  throw px::generic_err();
76  //tstamp=val.get_time();
77  }
78  else
79  {
80  tstamp = val.as_timestamp().to_timeval();
81  }
82 
83  // Convert into broken-down time
84  struct tm broken_down_time;
85  if ( gmtime_r( &( tstamp.tv_sec ), &broken_down_time ) == nullptr )
86  {
87  SVR_FERR( 1, "gmtime() returned null" );
88  throw px::generic_err();
89  }
90 
91  size_t nregs = reg_count();
92  assert( nregs != 0 );
93  assert( _size != 0 );
94  assert( _spec );
95 
96  regvec_t retval( nregs );
97 
98  size_t i = 0;
99  size_t nbytes = 0;
100 
101  while ( nbytes < ( ( size_t )( _size - 1 ) ) )
102  {
103  assert( i < nregs );
104 
105  if ( _spec->double_specifier( nbytes ) )
106  {
107  // This register contains a full 16-bit integer value
108  // extracted from the date/time
109 
110  retval[i] = _spec->translate( nbytes, broken_down_time, tstamp );
111  }
112  else
113  {
114  // Each byte of this register contains a different
115  // 8-bit integer value extracted from the date/time
116 
117  retval[i] = ( ( _spec->translate( nbytes, broken_down_time, tstamp ) << 8 ) & 0xFF00 ) |
118  ( _spec->translate( nbytes + 1, broken_down_time, tstamp ) & 0x00FF );
119  }
120 
121  nbytes += 2;
122  i++;
123  }
124 
125  if ( nbytes != _size )
126  {
127  // We end with a half-register value
128 
129  assert( nbytes == ( size_t )( _size - 1 ) );
130  assert( i < nregs );
131 
132  retval[i] = ( _spec->translate( nbytes, broken_down_time, tstamp ) << 8 ) & 0xFF00;
133 
134  i++;
135  }
136 
137  assert( i == nregs );
138 
139  return ( retval );
140 }

145 px::CValue CTimeMapping::translate( const regvec_t &val ) const
146 {
147  PUSH_FUNC;
148 
149  if ( _use_timestamp )
150  {
151  // Can't set the timestamp of a value via an incoming WRITE
152  SVR_FWARN( 2, "Can't set a value's timestamp via Modbus" );
153  throw px::generic_err();
154  }
155 
156  size_t nregs = reg_count();
157  assert( nregs != 0 );
158  assert( _size != 0 );
159  assert( _spec );
160 
161  size_t i = 0;
162  size_t nbytes = 0;
163  struct tm broken_down_time;
164  timeval tstamp;
165 
166  // Pre-seed the timestamp using the current date/time, in case the
167  // date specifier lacks some critical fields
168  if ( ( gettimeofday( &tstamp, nullptr ) ) || ( gmtime_r( &tstamp.tv_sec, &broken_down_time ) == nullptr ) )
169  {
170  SVR_FERR( 1, "Error getting system time!" );
171  throw px::generic_err();
172  }
173  tstamp.tv_usec = 0;
174 
175  while ( nbytes < ( ( size_t )( _size - 1 ) ) )
176  {
177  assert( i < nregs );
178 
179  if ( _spec->double_specifier( nbytes ) )
180  {
181  // This register contains a full 16-bit integer value
182  // which is a component of the date/time
183 
184  _spec->translate( nbytes,
185  val[i],
186  broken_down_time,
187  tstamp );
188  }
189  else
190  {
191  // Each byte of this register contains a different
192  // 8-bit integer value which is a component of the
193  // date/time
194 
195  _spec->translate( nbytes,
196  ( val[i] >> 8 ) & 0x00FF,
197  broken_down_time,
198  tstamp );
199  _spec->translate( nbytes + 1,
200  val[i] & 0x00FF,
201  broken_down_time,
202  tstamp );
203  }
204 
205  nbytes += 2;
206  i++;
207  }
208 
209  if ( nbytes != _size )
210  {
211  // We end with a half-register value
212 
213  assert( nbytes == ( size_t )( _size - 1 ) );
214  assert( i < nregs );
215 
216  _spec->translate( nbytes,
217  ( val[i] >> 8 ) & 0x00FF,
218  broken_down_time,
219  tstamp );
220 
221  i++;
222  }
223 
224  assert( i == nregs );
225 
226  // broken_down_time contains the translated date/time, minus the
227  // sub-second fields (which have been used to fill in tstamp.tv_usec)
228  tstamp.tv_sec = mktime( &broken_down_time );
229 
230  return ( px::CValue( tstamp ) );
231 }
