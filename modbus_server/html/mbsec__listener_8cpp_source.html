104 void CMbSecListener::handle_ssl_accept( const boost::system::error_code &err,
105  const std::shared_ptr<CMbSecConnection> &pconn )
106 {
107  PUSH_FUNC;
108 
109  assert( pconn );
110  assert( pconn->_sock );
111  //
112  // Check whether the server was stopped by a signal before this completion
113  // handler had a chance to run.
114  if ( !_acceptor || ( _acceptor && !_acceptor->is_open() ) )
115  {
116  return;
117  }
118 
119  SYSLOG_DEBUG( 2, "Accepted new connection from '%s'",

121 
122  if ( err )
123  {
124  SYSLOG_FERR( 1, "%s", err.message().c_str() );
125  }
126  else
127  {
128  // Success - add this connection to the I/O manager, and
129  // start handling requests
130  _owner.add( pconn );
131  pconn->start_handshake();
132  }
133 
134  // Now prepare for the next incoming connection
