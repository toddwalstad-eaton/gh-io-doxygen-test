36 CGlobalEDS::CGlobalEDS( const px::string &peds_json_file )
37  : _is_initialized( false )
38 {
39  PUSH_FUNC;
40 
41  if ( !CModbusUtil::does_file_exist( peds_json_file ) )
42  {
43  std::cerr << "Error: JSON EDS file " << peds_json_file << " does not exist!" << std::endl;
44  return;
45  }
46 
47  px::string content = px::CUtil::read_file_to_string( peds_json_file );
48 
49  if ( content.length() <= 0 )
50  {
51  std::cerr << "Error: Nothing read from json file!" << std::endl;
52  return;
53  }
54 
55  // A top level value contains either array or Object.
56  json_spirit::mValue val;
57 
58  SVR_DEBUG( 4, "Read PEDS json file %s", peds_json_file.c_str() );
59 
60  if ( json_spirit::read_string( content, val ) )
61  {
62  //=================================================
63  // The format of Modbus TCP JSON:
64  // { "Peds" : {
65  // "version": 1,
66  // "canonical_model_name": "PXG Modbus TCP register map",
67  // "map_name": "Device EDS file name"
68  // "description": "PXG Modbus TCP register mapping file",
69  // "entries": [
70  // { "name": "DisplayTemplate",
71  // "address": 1000,
72  // "size": 64,
73  // "value_type": "STRING"
74  // },
75  // } }
76  //=================================================
77 
78  auto top_obj = val.get_obj()[PED_TOP_OBJ].get_obj();
79 
80  _version = read_string_value( top_obj, px::string( PED_VERSION ) );
81  _canonical_model_name = read_string_value( top_obj, px::string( PED_CAN_MODEL_NAME ) );
82  _description = read_string_value( top_obj, px::string( PED_DESCRIPTION ) );
83  _eds_file = CModbusUtil::get_file_name_from_path( peds_json_file );
84  _peds_file = CModbusUtil::replace_file_extension( _eds_file, FILE_EXT_PEDS );
85 
86  SVR_DEBUG( 4, " version %s, canonical_model_name %s, eds_file %s, description %s",
87  _version.c_str(), _canonical_model_name.c_str(), _eds_file.c_str(), _description.c_str() );
88 
89  // process entry
90  process_entries( top_obj[PED_ENTRIES].get_array() );
91 
92  _is_initialized = true;
93  }
94 }

146 peds_ptr_t CGlobalEDS::create_peds_object()
147 {
148  PUSH_FUNC;
149 
150  peds_ptr_t peds( new px::proto::Peds() );
151 
152  peds->set_protocol( PED_PROTOCOL_TYPE_MODBUS );
153  peds->set_version( _version.c_str() );
154  peds->set_description( _description.c_str() );
155  peds->add_canonical_model_name( _canonical_model_name.c_str() );
156  peds->set_type( PEDS_TYPE_GLOBAL );
157  peds->set_name( _canonical_model_name.c_str() );
158  peds->set_eds_file( _eds_file.c_str() );
159 
160  px::proto::PedsEntry *entry = nullptr;
161 
162  for ( auto &ch_entry : _entries )
163  {
164  entry = peds->add_entries();
165 
166  if ( ch_entry.name.size() )
167  {
168  entry->set_name( ch_entry.name.c_str() );
169  }
170  if ( ch_entry.alternate_name.size() )
171  {
172  entry->set_alternate_name( ch_entry.alternate_name.c_str() );
173  }
174  if ( ch_entry.value_type.size() )
175  {
176  entry->set_value_type( ch_entry.value_type.c_str() );
177  }
178  if ( ch_entry.date_spec.size() )
179  {
180  entry->set_date_spec( ch_entry.date_spec.c_str() );
181  }
182  if ( ch_entry.address )
183  {
184  entry->set_address( ch_entry.address.get() );
185  }
186  if ( ch_entry.size )
187  {
188  entry->set_size( ch_entry.size.get() );
189  }
190  if ( ch_entry.discrete )
191  {
192  entry->set_discrete( ch_entry.discrete.get() );
193  }
194  if ( ch_entry.device )
195  {
196  entry->set_device( ch_entry.device.get() );
197  }
198  if ( ch_entry.writable )
199  {
200  entry->set_writable( ch_entry.writable.get() );
201  }
202  if ( ch_entry.command )
203  {
204  entry->set_command( ch_entry.command.get() );
205  }
206 
207  // Make sure there is a date_spec field if data type is TIME
208  if ( entry->has_value_type() )
209  {
210  if ( MB_DATETIME_TYPE == entry->value_type() )
211  {
212  if ( !entry->has_date_spec() )
213  {
214  entry->set_date_spec( MB_DEFAULT_DATETIME_SPEC );
215  }
216  }
217  }
218 
219  // Special case where we need to figure out if this is a device command
220  // and set 'command' attribute.
221  if ( ch_entry.writable && ch_entry.name.size() )
222  {
223  if ( ch_entry.writable.get() && ( DEV_CMD_PREFIX == ch_entry.name.at( 0 ) ) )
224  {
225  entry->set_command( true );
226  }
227  }
228  }
229 
230  SVR_DEBUG( 4, "Added %d total entries to peds", peds->entries_size() );
231 
232  // Do some error checking here to make sure we have the minimum requirements for a peds file
233  if ( peds->entries_size() == 0 )
234  {
235  peds.reset();
236  }
237 
238  return ( peds );
239 }

244 void CGlobalEDS::process_entries( json_spirit::mArray &entries )
245 {
246  PUSH_FUNC;
247 
248  // "entries": [
249  // { "name": "DisplayTemplate",
250  // "address": 1000,
251  // "size": 64,
252  // "value_type": "STRING"
253  // },
254  // ...
255  // ]
256 
257  for ( json_spirit::mValue entry : entries )
258  {
259  auto entry_obj = entry.get_obj();
260  CEntry ch_entry;
261  bool is_discrete = false;
262 
263  for ( size_t i = 0; i < PED_PARAM_COUNT; i++ )
264  {
265  ::json_spirit::mObject::const_iterator mpos = entry_obj.find( param_fields[i].tag );
266  if ( mpos != entry_obj.end() )
267  {
268  ::json_spirit::mValue mval = mpos->second;
269  if ( mval.type() == param_fields[i].jtype )
270  {
271  SVR_DEBUG( 8, " Parsing Ped field %s", mpos->first.c_str() );
272  switch ( i )
273  {
274  case PED_PARAM_NAME:
275  ch_entry.name = mval.get_str();
276  break;
277  case PED_PARAM_ALT_NAME:
278  ch_entry.alternate_name = mval.get_str();
279  break;
280  case PED_PARAM_VALTYPE:
281  ch_entry.value_type = mval.get_str();
282  break;
283  case PED_PARAM_DATE_SPEC:
284  ch_entry.date_spec = mval.get_str();
285  break;
286  case PED_PARAM_ADDRESS:
287  ch_entry.address = mval.get_int();
288  break;
289  case PED_PARAM_SIZE:
290  ch_entry.size = mval.get_int();
291  break;
292  case PED_PARAM_DISCRETE:
293  ch_entry.discrete = mval.get_bool();
294  if ( ch_entry.discrete )
295  {
296  is_discrete = true;
297  }
298  break;
299  case PED_PARAM_DEVICE:
300  ch_entry.device = mval.get_bool();
301  break;
302  case PED_PARAM_WRITABLE:
303  ch_entry.writable = mval.get_bool();
304  break;
305  default:
306  break;
307  }
308  }
309  }
310  }
311 
312  // Don't add entry unless we have minimum required data
313  if ( is_discrete && ch_entry.name.size() && ch_entry.address )
314  {
315  _entries.push_back( ch_entry );
316  }
317  else if ( ch_entry.name.size() && ch_entry.address && ch_entry.size && ch_entry.value_type.size() )
318  {
319  _entries.push_back( ch_entry );
320  }
321  }
322 
323  SVR_DEBUG( 4, "Entry count %zu", _entries.size() );
324 }
