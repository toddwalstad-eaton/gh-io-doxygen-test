55 void CModbusReadDiscretes::execute()
56 {
57  PUSH_FUNC;
58 
59  // First, we parse out the command arguments
60 
61  if ( _pdu.data.size() != MB_PDU_BYTES_EXPECTED )
62  {
63  SVR_FDEBUG( 2, "Bad PDU payload size: %zu bytes", _pdu.data.size() );
64  _prot.err( MB_ILLEGAL_FUNCTION );
65  return;
66  }
67 
68  px::asio::iobuf_const_iter_t ibuf = _pdu.data.begin();
69 
70  _addr = px::asio::CUtil::read_u16( ibuf );
71  _ninputs = px::asio::CUtil::read_u16( ibuf );
72 
73  if ( ( MB_MAX_DISCRETE_ADDR - _addr ) < _ninputs )
74  {
75  SVR_FDEBUG( 2, "Discrete range too large (%u + %u)", ( unsigned )_addr, ( unsigned )_ninputs );
76  _prot.err( MB_ILLEGAL_DATA_ADDRESS );
77  return;
78  }
79 
80  if ( ( _ninputs == 0 ) || ( _ninputs > MB_MAX_DISCRETE_REQ ) )
81  {
82  SVR_FDEBUG( 2, "Invalid request length (%u)", ( unsigned )_ninputs );
83  _prot.err( MB_ILLEGAL_DATA_VALUE );
84  return;
85  }
86 
87  // Next we create a maplist with the relevant parameter mappings
88  fetch_fields();
89 }

94 bool CModbusReadDiscretes::post( response_ptr_t resp )
95 {
96  PUSH_FUNC;
97 
98  if ( _timeout )
99  {
100  return ( true );
101  }
102 
103  if ( resp->status != px::cds::RequestStatus::SUCCEEDED )
104  {
105  // Unknown error encountered in CDS thread; since the request
106  // can't be fulfilled, we'll just send back an error
107  _prot.err( MB_SLAVE_DEVICE_FAILURE );
108  return ( true );
109  }
110 
111  std::shared_ptr<cds::getv_field_t> cds_resp =
112  std::dynamic_pointer_cast<cds::getv_field_t>( resp );
113 
114  // The response contains filled-in values for all leaves we fetched,
115  // including both data values and [potentially] dynamic configuration.
116  assert( _maplist );
117 
118 
119  try
120  {
121  auto vals = _maplist->translate_vals( cds_resp->fields );
122  assert( vals.size() == _ninputs );
123 
124  _pdu.data.clear();
125  _pdu.data.push_back( ( _ninputs + 7 ) / 8 );
126 
127  size_t count = 0;
128  while ( count < _ninputs )
129  {
130  _pdu.data.push_back( 0 );
131  for ( size_t i = 0; ( i < 8 ) && ( count < _ninputs ); i++, count++ )
132  {
133  _pdu.data.back() |= ( vals[count] << i );
134  }
135  }
136  }
137  catch ( const px::generic_err & )
138  {
139  _prot.err( MB_ILLEGAL_DATA_ADDRESS );
140  return ( true );
141  }
142 
143  // Command successful - return the response data
144  _prot.respond();
145  return ( true );
146 }
