274 void command_setup()
275 {
276  PUSH_FUNC;
277 
278  if ( g_snmpget_path.empty() )
279  {
280  struct stat st;
281  if ( stat( "/bin/snmpget", &st ) == 0 )
282  {
283  g_snmpget_path = "/bin/snmpget";
284  }
285  else if ( stat( "/usr/bin/snmpget", &st ) == 0 )
286  {
287  g_snmpget_path = "/usr/bin/snmpget";
288  }
289  else
290  {
291  SYSLOG_ERR( 0, "Cannot find 'snmpget' program" );
292  return;
293  }
294  }
295  g_snmpget_cmd.clear();
296  if ( g_snmp_user_settings.has_enable_v1() && g_snmp_user_settings.enable_v1() )
297  {
298  px::string comm_str;
299  if ( g_snmp_user_settings.has_ro_community() &&
300  ( !g_snmp_user_settings.ro_community().empty() ) )
301  {
302  comm_str = px::to_pxstr( g_snmp_user_settings.ro_community() );
303  }
304  else if ( g_snmp_user_settings.has_rw_community() &&
305  ( !g_snmp_user_settings.rw_community().empty() ) )
306  {
307  comm_str = px::to_pxstr( g_snmp_user_settings.rw_community() );
308  }
309  if ( !comm_str.empty() )
310  {
311  SYSLOG_DEBUG( 2, "Polling via SNMPv1 community string %s",
312  comm_str.c_str() );
313  g_snmpget_cmd = g_snmpget_path + " -r 3 -t 5 -M \"\" -m \"\" -Ot ";
314  g_snmpget_cmd += "-v 1 -c '" + comm_str + "' ";
315  }
316  }
317  if ( g_snmpget_cmd.empty() && g_snmp_user_settings.has_enable_v3() &&
318  g_snmp_user_settings.enable_v3() )
319  {
320  for ( auto u : g_snmp_user_settings.v3_users() )
321  {
322  px::string levArg;
323  px::string username;
324  if ( !u.has_username() )
325  {
326  continue;
327  }
328  username = px::to_pxstr( u.username() );
329  if ( username.empty() )
330  {
331  continue;
332  }
333  if ( u.has_use_priv() && u.use_priv() )
334  {
335  levArg = "-l authPriv ";
336  }
337  else if ( u.has_use_auth() && u.use_auth() )
338  {
339  levArg = "-l authNoPriv ";
340  }
341  else
342  {
343  g_snmpget_cmd = g_snmpget_path + " -r 3 -t 5 -M \"\" -m \"\" -Ot -v 3 -u '" + username + "' ";
344  break; // no need to look at other users
345  }
346  if ( !levArg.empty() )
347  {
348  if ( g_usmusers.count( username ) == 0 )
349  {
350  SYSLOG_DEBUG( 2, "Skipping user %s because no auth/priv info available", username.c_str() );
351  continue;
352  }
353  std::vector<px::string> toks = px::tokenize( g_usmusers[username] );
354  if ( toks.size() < 11 )
355  {
356  SYSLOG_DEBUG( 2, "Skipping user %s because can't figure out its usmUser line", username.c_str() );
357  continue;
358  }
359  g_snmpget_cmd = g_snmpget_path + " -r 3 -t 5 -M \"\" -m \"\" -Ot -v 3 -u '";
360  g_snmpget_cmd += username + "' " + levArg + "-e " + toks[3];
361  g_snmpget_cmd += " -a SHA -3k " + toks[8] + " ";
362  if ( levArg == "-l authPriv " )
363  {
364  g_snmpget_cmd += "-x AES -3K " + toks[10] + " ";
365  }
366  break;
367  }
368  } // end loop over v3 users
369  }
370  if ( !g_snmpget_cmd.empty() )
371  {
372  px::string localHostIp = "127.0.0.1";
373  px::string sysUpTimeOid = ".1.3.6.1.2.1.1.3.0";
374 
375  if ( !g_acc_ctrl_snmp.has_port() )
376  {
377  // No port in ACCESS CTRL OOB, not good but let's assume the default
378  g_snmpget_cmd += localHostIp + ":161 ";
379  g_snmpget_cmd += sysUpTimeOid;
380  }
381  else
382  {
383  // Use the port in the snmpget
384  uint32_t snmpdPort = g_acc_ctrl_snmp.port();
385  std::stringstream snmpdPortStream;
386  snmpdPortStream << snmpdPort;
387  px::string snmpdPortString( px::to_pxstr( snmpdPortStream.str() ) );
388 
389  g_snmpget_cmd += localHostIp + ":" + snmpdPortString + " " + sysUpTimeOid;
390  }
391  }
392  SYSLOG_DEBUG( 2, "Set request command to: [%s]", g_snmpget_cmd.c_str() );
393 }

403  g_close_watchdog = 0;
404 
405  // Parse the command-line options
406  try
407  {
408  g_options.reset( new COptions( argc, argv ) );
409  }

412  // Command-line parsing failed; usage was already shown
413  return ( rv );
414  }
415 
416  int sig_setup_rv = signal_handler_setup();
417  if ( sig_setup_rv )
418  {
419  // Can't send yet SYSLOG_ERR(0, "Signal handler Failure %d", sig_setup_rv );
420  fprintf( stderr, "Signal handler Failure %d", sig_setup_rv );
421  rv = sig_setup_rv;
422  return ( rv );
423  }
424 
425  // daemonize and set uid and gid for "snmp"
426  config_process();
427  init_logging();
428  // Sorry, but this process ALWAYS daemonizes, so you won't see this debug output
429  SYSLOG_INFO( 1, "Starting snmpd_wd" );
430 
431  // Check for sudden shutdown (real life example: configsubscriber restarts snmpd twice)
432  if ( g_close_watchdog )
433  {
434  SYSLOG_NOTICE( 1, "Stopping snmpd_wd before watchdog registration." );
435  return ( 0 ); // Return = OK, because someone killed us gracefully
436  }
437 
438  try
439  {

441 
442  if ( !get_cds_data( cds_hndlr ) )
443  {
444  SYSLOG_ERR( 0, "Failed to find necessary CDS data" );
445  return ( rv );
446  }
447 
448  if ( !is_snmpd_enabled_for_wd_monitoring() )
449  {
450  // snmpd is disabled, no need to continue
451  return ( 0 );
452  }
453 
454  if ( g_close_watchdog )
455  {
456  SYSLOG_NOTICE( 1, "Stopping snmpd_wd before watchdog registration." );
457  return ( 0 ); // Return = OK, because someone killed us gracefully
458  }
459  // else
460  SYSLOG_DEBUG( 5, "Got the OOB data; sysObjectID = %s", g_snmp_data_settings.sys_object_id().c_str() );
461 
462  // Set up command for queries
463  command_setup();
464  if ( g_snmpget_cmd.empty() )
465  {
466  SYSLOG_WARN( 0, "Cannot figure out how to query SNMP agent; this watchdog won't hunt!" );
467  return ( 1 );
468  }
469 
470  // Pre-Check one last time
471  if ( g_close_watchdog )
472  {
473  SYSLOG_NOTICE( 1, "Stopping snmpd_wd before watchdog registration." );
474  return ( 0 ); // Return = OK, because someone killed us gracefully
475  }
476  // register watchdog, ping every (watchdog_timeout / 3) seconds,
477  // and de-register when process is terminated by a signal.
478  watchdog_setup();
479  while ( !g_close_watchdog )
480  {
481  if ( is_snmpd_alive() )
482  {
483  ping_watchdog();
484  }
485  sleep( g_options->watchdog_timeout / 3 );
486  }
487 
488  close_watchdog();
489  // all good for our return
490  rv = 0;
491  }
492  catch ( const px::generic_err &ex )
493  {
494  if ( !g_close_watchdog )
495  {
496  const char *what_it_was = ex.what();
497  // Common problem in testing: shutting down suddently while still starting up
498  // The signature of this is no message in ex
499  if ( ( what_it_was == nullptr ) || ( strlen( what_it_was ) == 0 ) )
500  {
501  SYSLOG_NOTICE( 1, "Exiting snmpd_wd; CDS state indeterminate." );
502  }
503  else
504  {
505  SYSLOG_ERR( 0, "CDS Failure: %s", ex.what() );
506  }
507  }
508  else
509  {
510  SYSLOG_NOTICE( 1, "Exit CDS: %s", ex.what() );
511  }
512  }
513 
514  return ( rv );
515 }
