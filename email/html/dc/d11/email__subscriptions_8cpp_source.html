215 {
216  PUSH_FUNC;
217

219  {
220  SVR_DEBUG( 8, "Subscription callback ID: %s", id.c_str() );
221  }
222 
223  std::shared_ptr<px::cds::CSubscriberNotify> sub_notify;
224  sub_notify = std::dynamic_pointer_cast<px::cds::CSubscriberNotify>( response );
225 
226  if ( !sub_notify->alarms.empty() )
227  {
228  SVR_INFO( 3, "Received %zu alarm update(s)", sub_notify->alarms.size() );
229 
230  for ( auto &alarm: sub_notify->alarms )
231  {
232  // update the alarm history
233  std::shared_ptr<CEmailAlarmStatus> alarm_status = _work_mgr->update_alarm_history( &alarm );
234 
235  // create a new work item
236  std::shared_ptr<CEmailWorkItem> item = std::make_shared<CEmailWorkItem>( &alarm, alarm_status );
237  SVR_DEBUG( 4, "Alarm Update: %s", item->alarm_update_str().c_str() ); // log the details of the alarm update
238 
239  std::shared_ptr<CEmailChannelStatus> channel_status = _work_mgr->update_channel_history( &alarm );
240 
241  // check spam threshold for new alarms
242  if ( item->is_new_alarm() )
243  {
244  uint32_t threshold = _cfg->get_config_int_value( px::email::CONFIG_MINIMUM_SEND_INTERVAL, px::email::DEFAULT_MINIMUM_SEND_INTERVAL );
245 
246  px::CWallTime now = px::CWallTime::now();
247  timeval tm_now = now.to_timeval();
248 
249  px::CWallTime *prev = channel_status->last_new_status();
250  timeval tm_prev = prev->to_timeval();
251 
252  if ( ( tm_prev.tv_sec != 0 ) && ( ( tm_now.tv_sec - tm_prev.tv_sec ) < static_cast<signed>( threshold ) ) )
253  {
254  SVR_DEBUG( 4, "Alarm Update ignored - minimum time threshold between sending a NEW alarm email has not been reached" );
255  continue;
256  }
257  channel_status->set_last_new_status();
258  }
259 
260  // check spam threshold for cleared alarms
261  if ( item->is_cleared_alarm() )
262  {
263  uint32_t threshold = _cfg->get_config_int_value( px::email::CONFIG_MINIMUM_SEND_INTERVAL, px::email::DEFAULT_MINIMUM_SEND_INTERVAL );
264 
265  px::CWallTime now = px::CWallTime::now();
266  timeval tm_now = now.to_timeval();
267 
268  px::CWallTime *prev = channel_status->last_clear_status();
269  timeval tm_prev = prev->to_timeval();
270 
271  if ( ( tm_prev.tv_sec != 0 ) && ( ( tm_now.tv_sec - tm_prev.tv_sec ) < static_cast<signed>( threshold ) ) )
272  {
273  SVR_DEBUG( 4, "Alarm Update ignored - minimum time threshold between sending a CLEARED alarm email has not been reached" );
274  continue;
275  }
276  channel_status->set_last_clear_status();
277  }
278 
279  // queue the update for processing
280  _work_mgr->queue_pending_work_item( item );
281  }
