60 {
61  PUSH_FUNC;
62 
63  // make sure we have an absolute path
64  if ( ( _config_file.size() > 0 ) && ( _config_file.at( 0 ) != '/' ) )
65  {
66  px::string err = "Specified configuration file is not an absolute path";
67  SYSLOG_FERR( 1, err.c_str() );
68  throw px::generic_err( err.c_str() );
69  }
70 
71  // check for the existence of the configuration file
72  if ( !check_config_file_exists() )
73  {
74  px::string err = "Configuration file does not exist";
75  SYSLOG_FERR( 1, err.c_str() );
76  throw px::generic_err( err.c_str() );
77  }
78 
79  // parse the config file
80  xml_document<> doc;
81  xml_node<> *root_node;
82 
83  // read the xml file into a vector
84  ifstream xml_file( _config_file.c_str() );
85  vector<char> buffer( ( istreambuf_iterator<char>( xml_file ) ), istreambuf_iterator<char>() );
86  buffer.push_back( '\0' );
87 
88  // parse the buffer
89  doc.parse<0>( &buffer[0] );
90 
91  // find our root node
92  root_node = doc.first_node( px::email::CONFIG_TAG_ROOT.c_str() );
93 
94  // iterate over the nodes
95  for ( xml_node<> *node = root_node->first_node( px::email::CONFIG_TAG_NODE.c_str() ); node; node = node->next_sibling() )
96  {
97  px::string key = node->first_attribute( px::email::CONFIG_TAG_NAME.c_str() )->value();
98  px::string value = node->first_attribute( px::email::CONFIG_TAG_VALUE.c_str() )->value();
99 
100  // update list of configuration values
101  if ( !key.empty() && !value.empty() )
102  {
103  _config_values[key] = value;
104  }
105  }
106 
107  // check to make sure we have configuration values
108  if ( !verify_config_file() )
109  {
110  px::string err = "Invalid/empty configuration file";
111  SYSLOG_FERR( 1, err.c_str() );
112  throw px::generic_err( err.c_str() );
113  }
114 }
