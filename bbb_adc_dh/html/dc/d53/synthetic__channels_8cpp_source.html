244 // Called once per sample interval to update all the channel data.
245 void CSyntheticChannels::update_channel_values( float adc_slider_val, uint8_t poll_counter )
246 {
247  PUSH_FUNC;
248 
249  // First, compute the cycle_value for this time, based on current seconds
250  timeval tnow = px::CMonoTime::now().to_timeval();
251 
252  // Get the modulo factor, 0 to 1843 secs
253  unsigned int cycle_factor = tnow.tv_sec % _cycle_period_secs;
254 
255  // Compute the cycle_value as how far it is from 1/2 the modulus (absolute value), scaled to 1
256  int delta = cycle_factor - ( _cycle_period_secs / 2 );
257  cycle_value = ( ( float ) std::abs( delta ) ) / ( _cycle_period_secs / 2 );
258  // SVR_LOG_INFO(SVR_LOG_PARAMETERS, 5, "cycle_value = %f for %u secs", cycle_value, cycle_factor );
259 
260  for ( size_t idx = 0; idx < DEFAULT_NUM_CHANNELS; idx++ )
261  {
262  synth_channel_t *ch_ptr = &channel_array[idx];
263  if ( ch_ptr->params.poll_slot != poll_counter )
264  {
265  continue;
266  }
267 
268  float rnval, pf_range;
269  switch ( ch_ptr->params.ch_type )
270  {
271  case sampled_float:
272  // Simply store the given value
273  ch_ptr->value = adc_slider_val;
274  break;
275  case power_factor:
276  /* Roughly determine as minimum + cycle_value * range + random value (from range_random)
277  * The range is funky for PF; calculate as max + min + 1 to get the right value
278  * And to get the right negative PF values (from 1 to -0.99, -0.98, -0.97, etc)
279  * we start by adding to the maximum, and if the number is > 1, then subtract 2 from it,
280  * so 1.01 becomes -0.99 */
281  rnval = _random_gen.uniform( 0.0, ch_ptr->params.range_random );
282  pf_range = ch_ptr->params.range_max + ch_ptr->params.range_min + 1.0;
283  ch_ptr->value = ch_ptr->params.range_max + ( cycle_value * pf_range ) + rnval;
284  if ( ch_ptr->value.as_float() > 1.0 )
285  {
286  float tmp = ch_ptr->value.as_float();
287  ch_ptr->value = tmp - 2.0;
288  }
289  break;
290  case random_float:
291  // Determine as minimum + cycle_value * range + random value (from range_random)
292  rnval = _random_gen.uniform( 0.0, ch_ptr->params.range_random );
293  ch_ptr->value = ch_ptr->params.range_min + ( cycle_value * ( ch_ptr->params.range_max - ch_ptr->params.range_min ) ) + rnval;
294  break;
295  case computed_watts:
296  // Use idx-2 for Volts and idx-1 for Current, multiply together with PF from idx=1
297  // This will be a signed value, depending on the PF sign
298  {
299  float tmp_v = channel_array[idx - 6].value.as_float();
300  float tmp_i = channel_array[idx - 3].value.as_float();
301  float tmp_pf = channel_array[DEFAULT_PF_IDX].value.as_float();
302  ch_ptr->value = tmp_v * tmp_i * tmp_pf;
303  }
304  break;
305  case computed_wh:
306  /* Should accumulate watts * sample interval, but will goose this up to accumulate faster
307  * by taking 1sec as the time interval. (1sec = 1/3600 hours)
308  * Wattage is from idx-1. Will only accumulate if wattage is positive.
309  * Have to test for the NaN (initial value) or INFINITY, and zero it out before summing. */
310  {
311  double tmp_wh = ch_ptr->value.as_double();
312  if ( !std::isfinite( tmp_wh ) )
313  {
314  ch_ptr->value = 0;
315  }
316  double tmp_w = channel_array[idx - 3].value.as_double();
317  if ( tmp_w > 0 )
318  {
319  ch_ptr->value = tmp_wh + ( tmp_w * ( 1.0 / 3600.0 ) );
320  }
321  }
322  break;
323  case static_int:
324  /* Just output a value of 1 for now */
325  ch_ptr->value = 1;
326  break;
327  default:
328  break;
329  // Should be no more cases!
330  }
331  }
332 
333  // Now update the values in the CDS for all channels
334  set_channel_values( poll_counter );
335 }
