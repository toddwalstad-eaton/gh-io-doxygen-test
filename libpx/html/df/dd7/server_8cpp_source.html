325 {
326  unsigned int nbytes;
327 
328  // If no callback is set don't bother sending, client won't see it
329  if ( unlikely( response->callback_handle == CALLBACK_NONE ) )
330  {
331  return ( true );
332  }
333 
334  // Get the appropriate server connection
335  conn_list_lock.lock();
336  auto i = connections.find( response->client_id );
337  if ( i == connections.end() )
338  {
339  conn_list_lock.unlock();
340  return ( true );// silently discarding
341  }
342  std::shared_ptr<CServerConnection> conn = i->second;
343  // At this point even if the connection gets closed, the queue won't
344  // get deleted until 'conn' goes out of scope
345  conn_list_lock.unlock();
346 
347  // If the response is too large to fit in the queue req_pack() will
348  // throw too_much_data and we will propagate that
349  try
350  {
351  // Allow only one thread at a time to send on the queue
352  conn->response_queue_lock.lock();
353  nbytes = req_pack( response.get(), // Can throw px::generic_err or px::too_much_data
354  ( unsigned char * )( conn->response_queue->get() ),
355  conn->response_queue->chunk_size() );
356  // Signal the other end that we have put the message in place.
357  // If this throws generic_err we will propagate that
358  conn->response_queue->write( nbytes ); // Can throw toolkit::ipc::no_data or toolkit::ipc::generic_err
359  }
360  catch ( px::too_much_data )
361  {
362  // Response is too big to ever fit in the queue. Tell the client
363  // if possible.
364  response->clean();
365  response->status = px::cds::RequestStatus::RESPONSE_TOO_BIG;
366  try
367  {
368  nbytes = req_pack( response.get(), // Can throw px::generic_err or px::too_much_data
369  ( unsigned char * )( conn->response_queue->get() ),
370  conn->response_queue->chunk_size() );
371  conn->response_queue->write( nbytes ); // Can throw toolkit::ipc::no_data or toolkit::ipc::generic_err
372  }
373  catch ( toolkit::ipc::no_data )
374  {
375  // Queue full, little to do but fail
376  conn->response_queue_lock.unlock();
377  return ( false );
378  }
379  catch ( px::too_much_data )
380  {
381  // No reason for a message to ever be too big AFTER clean!

383  conn->response_queue_lock.unlock();
384  return ( false );
385  }
386  catch ( toolkit::ipc::generic_err )
387  {
388  // Queue data issue, follow same action as px::generic error
389  conn->response_queue_lock.unlock();

396  }
397  }

401  response->clean();

403  try

409  }
410  catch ( toolkit::ipc::no_data )

415  }
416  catch ( px::too_much_data )

419  LIBPX_ERR( 0, "Message too big after clean() (shouldn't happen)" );
420  conn->response_queue_lock.unlock();

422  }

425  // Queue data issue, follow same action as px::generic error
426  conn->response_queue_lock.unlock();
427  throw px::generic_err( "Caught response queue write exception ipc::generic_err inside ipc::no_data exception" );
428  }
429  catch ( px::generic_err )

681 int CServer::update_uid( ClientID id )
682 {
683  std::lock_guard<std::mutex> lock( conn_list_lock );
684  auto got = connections.find( id );
685 
686  if ( unlikely( ( got == connections.end() ) || ( !got->second ) ) )
687  {
688  return ( -1 );
689  }
690  struct ucred creds;
691  socklen_t len = sizeof( creds );
692 
693  // If the UID hasn't been set yet, set it from socket creds, else just return it.
694  if ( ( got->second->uid == -1 ) || ( got->second->uid == 0 ) )
695  {
696  if ( getsockopt( got->second->socket_fd, SOL_SOCKET, SO_PEERCRED, &creds, &len ) == 0 )
697  {
698  got->second->uid = creds.uid;
699  got->second->permissions_valid = false;
700  if ( got->second->gids.empty() )
701  {

703  }
704  }
705  }

709 void CServer::serve_client_conn_req( ClientID client_id, int client_socket )
710 {
711  struct iovec iov;
712  struct ucred *credptr;
713  struct ucred creds;
714  int cred_data;
715  struct msghdr msg;
716  struct cmsghdr *cmsg;
717  ssize_t nrecv;
718  char control[CMSG_SPACE( sizeof( struct ucred ) )];
719 
720  iov.iov_base = &cred_data;
721  iov.iov_len = sizeof( int );
722 
723  memset( &msg, 0, sizeof( msg ) );
724  msg.msg_iov = &iov;
725  msg.msg_iovlen = 1;
726  msg.msg_name = NULL;
727  msg.msg_namelen = 0;
728  msg.msg_control = control;
729  msg.msg_controllen = sizeof( control );
730  nrecv = ::recvmsg( client_socket, &msg, MSG_DONTWAIT );
731  if ( nrecv == -1 )
732  {
733  // No data on this one, move on
734  // Happens in busy systems due to races, have not seen any data
735  // lost yet due to it
736  LIBPX_WARN( 2, "No data on socket %d", client_socket );
737  return;
738  }
739  cmsg = CMSG_FIRSTHDR( &msg );
740  if ( ( cmsg == NULL ) ||
741  ( cmsg->cmsg_level != SOL_SOCKET ) ||
742  ( cmsg->cmsg_type != SCM_CREDENTIALS ) )
743  {
744  // bad message sent; restart loop
745  LIBPX_ERR( 0, "Bad message received on socket %d", client_socket );
746  return;
747  }
748  credptr = reinterpret_cast<struct ucred *>( CMSG_DATA( cmsg ) );
749  memcpy( ( void * )&creds, ( void * )credptr, sizeof( struct ucred ) );
750  if ( cmsg->cmsg_len != CMSG_LEN( sizeof( struct ucred ) ) )
751  {
752  // bad credentials sent; restart loop
753  LIBPX_ERR( 0, "Bad credentials received on socket %d", client_socket );
754  return;
755  }
756 
757  // Determine the Linux UID of the credentials passed
758  update_uid( client_id, credptr->uid );
759 }
