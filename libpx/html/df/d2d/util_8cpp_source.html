85  if ( src.empty() )
86  {
87  return ( retval );
88  }
89

92  newpos = src.find( delimiter, pos );
93  if ( newpos != src.npos )
94  {
95  retval.push_back( src.substr( pos, newpos - pos ) );

102  }
103  while ( newpos != src.npos );
104 
105  return ( retval );

112 {
113  pid_t process;
114  int retval;
115 
116  process = fork();
117  switch ( process )
118  {
119  case 0:
120  // Child process
121  setsid();
122  process = fork();
123  switch ( process )
124  {
125  case 0:
126  // Grandchild process
127  if ( chdir( "/" ) )
128  {
129  throw px::generic_err();
130  }
131 
132  // Close stdout, stderr, and stdin
133  close( 0 );
134  close( 1 );
135  close( 2 );
136 
137  // Send all the usual I/O to /dev/null
138  // Redirect stdin
139  if ( open( "/dev/null", O_RDWR ) == -1 )
140  {
141  throw px::generic_err();
142  }
143 
144  // coverity[RESOURCE_LEAK]
145  // Redirect stdout/stderr. Coverity considers these 2 calls leaks, but the magic comment above should tell it to suppress it's complaints
146  if ( ( dup( 0 ) == -1 ) || ( dup( 0 ) == -1 ) )
147  {
148  throw px::generic_err();
149  }
150 
151  // Set umask appropriately
152  umask( 022 );
153  // Fall to bottom...
154  break;
155  case -1:
156  // Error in fork()
157  exit( 1 );
158  break;
159  default:
160  // Child process
161  exit( 0 );
162  break;
163  }
164  break;
165  case -1:
166  throw px::generic_err();
167  break;
168  default:
169  wait( &retval );
170  if ( !WIFEXITED( retval ) || WEXITSTATUS( retval ) )
171  {
172  throw px::generic_err();
173  }
174  exit( 0 );
175  break;
176  }

271 //
272 // Set directory and children writable, a directory at a time, recursively, ignoring errors.
273 //
274 static void set_dir_and_children_writable( const char *path )
275 {
276  struct stat st;
277  char entry_path[PATH_MAX];
278 
279  DIR *dir = opendir( path );
280 
281  if ( dir == nullptr )
282  {
283  return;
284  }
285  struct dirent prev_entry;
286  struct dirent *entry;
287  memset( &prev_entry, 0, sizeof( prev_entry ) );
288  while ( readdir_r( dir, &prev_entry, &entry ) == 0 )
289  {
290  if ( entry == nullptr )
291  {
292  break;
293  }
294  if ( strcmp( entry->d_name, ".." ) == 0 )
295  {
296  continue;
297  }
298  snprintf( entry_path, PATH_MAX, "%s/%s", path, entry->d_name );
299  entry_path[PATH_MAX - 1] = '\0';
300  LIBPX_DEBUG( 7, "Checking file / directory %s for making writable", entry_path );
301  // open so we can use fstat and fchmod to avoid TOCTOU problems
302  int fd = open( entry_path, O_RDONLY );
303  if ( fd < 0 )
304  {
305  // Debug, not error, because we want to be resilient when not in development
306  LIBPX_DEBUG( 2, "File or directory %s could not be opened, error %d", entry_path, errno );
307  continue;
308  }
309  if ( fstat( fd, &st ) != 0 )
310  {
311  LIBPX_DEBUG( 2, "File or directory %s could not be stat-ed, error %d", entry_path, errno );
312  ( void ) close( fd );
313  continue;
314  }
315  if ( S_ISDIR( st.st_mode ) )
316  {
317  if ( strcmp( entry->d_name, "." ) != 0 )
318  {
319  // Recurse here
320  set_dir_and_children_writable( entry_path );
321  }
322  }
323  if ( ( st.st_mode & 0222 ) == 0222 )
324  {
325  // already world-writable
326  ( void ) close( fd );
327  LIBPX_DEBUG( 9, "File or directory %s already world-writable, skipping", entry_path );
328  continue;
329  }
330  int rc = fchmod( fd, st.st_mode | 0222 );
331  if ( rc != 0 )
332  {
333  LIBPX_DEBUG( 2, "Could not set file or directory %s world-writable, error %d.", entry_path, errno );
334  }
335  ( void ) close( fd );
336  }
337  ( void ) closedir( dir );
338 }
339 
340 #endif

342 //
343 // If changing from root to something else, and there are files in the coverage
344 // directory, make them world-writable so our destination user can create / write to
345 // them.
346 //
347 static void coverage_compensate( uid_t dest_uid )
348 {
349 #ifdef CONFIG_COVERAGE
350  if ( ( geteuid() != 0 ) || ( dest_uid == 0 ) )
351  {
352  return; // nothing to do
353  }
354  // If the environment variable is not set, gcov instrumentation is trying to save in
355  // /home/something/something anyway, which will not work, so no need to go farther
356  char *coverage_dir_path = getenv( "GCOV_PREFIX" );
357  if ( coverage_dir_path == nullptr )
358  {
359  LIBPX_DEBUG( 2, "GCOV_PREFIX not set in environment, not trying to save the files" );
360  return;
361  }
362  DIR *dir = opendir( coverage_dir_path );
363  if ( dir == nullptr )
364  {
365  LIBPX_DEBUG( 2, "Coverage directory %s could not be opened (error %d), not compensating for setuid in code coverage", coverage_dir_path, errno );
366  return;
367  }
368  struct dirent prev_entry;
369  struct dirent *entry;
370  memset( &prev_entry, 0, sizeof( prev_entry ) );
371  bool found_something = false;
372  while ( readdir_r( dir, &prev_entry, &entry ) == 0 )
373  {
374  if ( entry == nullptr )
375  {
376  break;
377  }
378  if ( ( strcmp( entry->d_name, "." ) != 0 ) && ( strcmp( entry->d_name, ".." ) != 0 ) )
379  {
380  found_something = true;
381  break;
382  }
383  }
384  ( void ) closedir( dir );
385  if ( found_something )
386  {
387  set_dir_and_children_writable( coverage_dir_path );
388  }
389  else
390  {
391  LIBPX_DEBUG( 2, "Coverage directory %s is empty, no need to compensate for setuid", coverage_dir_path );
392  }
393 #endif
394 }

621 void px::CUtil::drop_root_keep_bind( const px::string &user,
622  const px::string &group )
623 {
624  // Step 1: Drop all capabilities (permitted and effective) except for
625  // low port binding, changing UID, changing GID
626  struct __user_cap_header_struct hdr;
627  struct __user_cap_data_struct data;
628 
629  memset( &data, 0, sizeof( data ) );
630  hdr.pid = 0;// ourselves
631  hdr.version = _LINUX_CAPABILITY_VERSION_1;
632  if ( ::syscall( SYS_capget, &hdr, &data ) != 0 )
633  {
634  throw px::generic_err( "Could not capget, as root" );
635  }
636  data.effective &= ( ( 1 << CAP_SETUID ) | ( 1 << CAP_SETGID ) | ( 1 << CAP_NET_BIND_SERVICE ) | ( 1 << CAP_NET_RAW ) );
637  data.permitted &= ( ( 1 << CAP_SETUID ) | ( 1 << CAP_SETGID ) | ( 1 << CAP_NET_BIND_SERVICE ) | ( 1 << CAP_NET_RAW ) );
638  data.inheritable &= ( ( 1 << CAP_SETUID ) | ( 1 << CAP_SETGID ) | ( 1 << CAP_NET_BIND_SERVICE | ( 1 << CAP_NET_RAW ) ) );
639  if ( ::syscall( SYS_capset, &hdr, &data ) != 0 )
640  {
641  throw px::generic_err( "Could not capset, as root" );
642  }
643 
644  // Step 2: Set a flag that allows us to keep our Linux "capabilities"
645  // across setuid() and similar calls
646  int rv = ::prctl( PR_SET_KEEPCAPS, 1, 0, 0, 0 );
647  if ( rv )
648  {
649  throw px::generic_err( "Could not PR_SET_KEEPCAPS" );
650  }
651 
652  // Step 3: Remove any supplementary group memberships
653  if ( ::setgroups( 0, NULL ) != 0 )
654  {
655  throw px::generic_err( "Failed to remove any supplementary group memberships" );
656  }
657 
658  // Step 4: Assume the unprivileged user/group IDs.
659  // These may throw, we will propagate the exceptions.
660  px::CUtil::set_gid( group );
661  px::CUtil::set_uid( user );
662 
663  // Step 5: Drop the capabilities we may still have for setuid/setgid
664  if ( ::syscall( SYS_capget, &hdr, &data ) != 0 )
665  {
666  throw px::generic_err( "Could not capget, as user" );
667  }
668  data.effective &= ( 1 << CAP_NET_BIND_SERVICE | ( 1 << CAP_NET_RAW ) );
669  data.permitted &= ( 1 << CAP_NET_BIND_SERVICE | ( 1 << CAP_NET_RAW ) );
670  data.inheritable &= ( 1 << CAP_NET_BIND_SERVICE | ( 1 << CAP_NET_RAW ) );
671  // If we still have bind permission, activate it
672  if ( ( data.permitted ) != 0 )
673  {
674  data.effective = data.permitted;
675  }
676  if ( ::syscall( SYS_capset, &hdr, &data ) != 0 )
677  {
678  throw px::generic_err( "Could not capset, as user" );
679  }
680  // Done.
681 }

683 // format_uuid_v3or5 -- make a UUID from a (pseudo)random 128-bit number
684 void format_uuid_v3or5( _uuid_t *uuid, unsigned char hash[16], int v )
685 {
686  // Convert UUID to local byte order
687  memcpy( uuid, hash, sizeof *uuid );
688  uuid->time_low = ntohl( uuid->time_low );
689  uuid->time_mid = ntohs( uuid->time_mid );

695  uuid->clock_seq_hi_and_reserved &= 0x3F;
696  uuid->clock_seq_hi_and_reserved |= 0x80;

925 bool px::CUtil::get_process_gids( int pid, std::unordered_set<int> &gids )
926 {
927  PUSH_FUNC;
928  bool success = false;
929 
930  LIBPX_DEBUG( 7, "Getting GIDs for process %d", pid );
931  std::string stat_path = "/proc/" + std::to_string( pid ) + "/status";
932  FILE *fp = fopen( stat_path.c_str(), "r" );
933  char *line_buf;
934  size_t line_size;
935  if ( fp )
936  {
937  line_buf = nullptr;
938  while ( getline( &line_buf, &line_size, fp ) != -1 )
939  {
940  if ( unlikely( line_buf == nullptr ) )
941  {
942  continue;
943  }
944  if ( memcmp( line_buf, "Groups:\t", 8 ) != 0 )
945  {
946  if ( memcmp( line_buf, "Gid:\t", 5 ) != 0 )
947  {
948  free( line_buf );
949  line_buf = nullptr;
950  continue;
951  }
952  }
953  if ( memcmp( line_buf, "Gid:\t", 5 ) == 0 )
954  {
955  errno = 0;
956  char *this_start = line_buf + 5;
957  char *this_end = this_start;
958  int this_num = strtol( this_start, &this_end, 10 );
959  if ( ( errno == 0 ) && ( this_end != this_start ) )
960  {
961  LIBPX_DEBUG( 7, "Primary GID %d", this_num );
962  gids.insert( this_num );
963  }
964  free( line_buf );
965  line_buf = nullptr;
966  continue;
967  }
968  success = true;
969  char *this_start = line_buf + 8;
970  char *this_end = this_start;
971  while ( ( *this_end != '\0' ) && ( *this_end != '\n' ) )
972  {
973  errno = 0;
974  int this_num = strtol( this_start, &this_end, 10 );
975  if ( errno == 0 )
976  {
977  if ( this_end == this_start )
978  {
979  break; // at end of line
980  }
981  LIBPX_DEBUG( 7, "Found GID %d", this_num );
982  gids.insert( this_num );
983  if ( *this_end == ' ' )
984  {
985  this_start = this_end + 1;
986  }
987  else
988  {
989  break;
990  }
991  }
992  else
993  {
994  success = false;
995  break;
996  }
997  }
998  free( line_buf );
999  break;
1000  }
1001  fclose( fp );
1002  }
1003  return ( success );
1004 }

1008 //
1009 void px::CUtil::drop_root_keep_net_admin( const px::string &user,
1010  const px::string &group )
1011 {
1012  // Drop all Linux capabilities except for
1013  // CAP_NET_ADMIN - To perform various network-related operations.
1014  struct __user_cap_header_struct hdr;
1015  struct __user_cap_data_struct data;
1016 
1017  memset( &data, 0, sizeof( data ) );
1018  hdr.pid = 0;
1019  hdr.version = _LINUX_CAPABILITY_VERSION_3;
1020 
1021  if ( ::syscall( SYS_capget, &hdr, &data ) != 0 )
1022  {
1023  throw px::generic_err( "Could not get Linux capabilities, as a root" );
1024  }
1025 
1026  data.effective &= ( ( 1 << CAP_SETUID ) | ( 1 << CAP_SETGID ) | ( 1 << CAP_NET_ADMIN ) );
1027  data.permitted &= ( ( 1 << CAP_SETUID ) | ( 1 << CAP_SETGID ) | ( 1 << CAP_NET_ADMIN ) );
1028  data.inheritable &= ( ( 1 << CAP_SETUID ) | ( 1 << CAP_SETGID ) | ( 1 << CAP_NET_ADMIN ) );
1029 
1030  if ( ::syscall( SYS_capset, &hdr, &data ) != 0 )
1031  {
1032  throw px::generic_err( "Could not set Linux capabilites, as a root" );
1033  }
1034 
1035  // Set a flag that allows us to keep our Linux "capabilities"
1036  // across setuid() and similar calls
1037  int rv = ::prctl( PR_SET_KEEPCAPS, 1, 0, 0, 0 );
1038  if ( rv )
1039  {
1040  throw px::generic_err( "Could not PR_SET_KEEPCAPS" );
1041  }
1042 
1043  // Remove any supplementary group memberships
1044  if ( ::setgroups( 0, NULL ) != 0 )
1045  {
1046  throw px::generic_err( "Failed to remove any supplementary group memberships" );
1047  }
1048 
1049  // Switch to Linux group
1050  px::CUtil::set_gid( group );
1051 
1052  // Switch to Linux user
1053  px::CUtil::set_uid( user );
1054 
1055  // We may still have setuid/setgid capabilities, hence drop them also.
1056  if ( ::syscall( SYS_capget, &hdr, &data ) != 0 )
1057  {
1058  px::string msg = "Could not get Linux capabilities, as user " + user;
1059  throw px::generic_err( msg );
1060  }
1061 
1062  data.effective &= ( 1 << CAP_NET_ADMIN );
1063  data.permitted &= ( 1 << CAP_NET_ADMIN );
1064  data.inheritable &= ( 1 << CAP_NET_ADMIN );
1065 
1066  // If we still have CAP_NET_ADMIN permission, activate it.
1067  if ( ( data.permitted ) != 0 )
1068  {
1069  data.effective = data.permitted;
1070  }
1071 
1072  if ( ::syscall( SYS_capset, &hdr, &data ) != 0 )
1073  {
1074  px::string msg = "Could not set Linux capabilities, as user " + user;
1075  throw px::generic_err( msg );
1076  }
1077 }
