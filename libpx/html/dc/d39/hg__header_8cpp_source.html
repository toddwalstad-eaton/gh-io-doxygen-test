55 // Parse and validate that this is a correct Mercury Header.
56 //
57 // Mercury Msg Header V2 (8 bytes total):
58 // 1. header_magic_byte : 1 byte
59 // 2. header_version : 1 byte
60 // 3. header_size : 1 byte
61 // 4. msg_size : 4 bytes
62 // 5. msg_type : 1 bytes
63 //
64 // Mercury Msg Header V3 (25 bytes total):
65 // 1. header_magic_byte : 1 byte
66 // 2. header_version : 1 byte
67 // 3. header_size : 1 byte
68 // 4. msg_size : 4 bytes
69 // 5. msg_type : 1 bytes
70 // -----------------------------
71 // 6. msg_api : 1 byte
72 // 7. msg_publisher_uuid: 16 bytes (if msg_src_ipv6_flag is 1)
73 px::asio::parse_result_t CMercuryHeader::parse( px::asio::iobuf_const_iter_t start,
74  px::asio::iobuf_const_iter_t end )
75 {
76  PUSH_FUNC;
77 
78  assert( end >= start );
79 
80  _error_message.clear();
81 
82  px::asio::parse_result_t hdr_rv = parse_header_only( start, end );
83  if ( !hdr_rv.first )
84  {
85  return ( hdr_rv ); // header parse failed
86  }
87 
88  uint32_t recd_message_size = ( size_t )( end - hdr_rv.second );
89  if ( _message_size > recd_message_size )
90  {
91  // _error_message = err_text; -- no, don't declare a problem
92  LIBPX_DEBUG( 5, "Parsing Header complete, but msg incomplete: %u of %u",
93  _message_size, recd_message_size );
94 
95  hdr_rv.first = boost::tribool( boost::logic::indeterminate );
96  // Reset the start pointer, too
97  hdr_rv.second = start;
98  }
99  return ( hdr_rv );
100 }

102 px::asio::parse_result_t CMercuryHeader::parse_header_only( px::asio::iobuf_const_iter_t start,
103  px::asio::iobuf_const_iter_t end )
104 {
105  PUSH_FUNC;
106 
107  assert( end >= start );
108 
109  bool failed = false;
110  bool incomplete = false; // ie, not enough header bytes received yet
111  size_t buflen = ( size_t ) ( end - start );
112  px::asio::iobuf_const_iter_t ibuf = start;
113  char err_text[60] = { 0 };
114  _error_message.clear();
115 
116  LIBPX_DEBUG( 8, "Parsing Mercury header from buffer segment of size %zu", buflen );
117 
118  size_t exp_hdr_len = HEADER_VERSION_2_SIZE; // has too be at least this long
119  // Eg: we expect only V3 headers for ctc_agent via websockets
120  if ( _min_header_version == HEADER_VERSION_3 )
121  {
122  exp_hdr_len = HEADER_VERSION_3_SIZE;
123  }
124 
125  if ( buflen < exp_hdr_len )
126  {
127  snprintf( err_text, 60, "Received message too short, %zu < %zu header size", buflen, exp_hdr_len );
128  failed = incomplete = true;
129  }
130  else
131  {
132  // Get the initial values first
133  uint8_t header_magic_byte = px::asio::CUtil::read_u8( ibuf );
134  _header_version = px::asio::CUtil::read_u8( ibuf );
135  _header_size = px::asio::CUtil::read_u8( ibuf );
136  _message_size = px::asio::CUtil::read_u32( ibuf );
137  _message_type = ( mercury_msg_format_t ) px::asio::CUtil::read_u8( ibuf );
138 
139  if ( header_magic_byte != HEADER_MAGIC_BYTE )
140  {
141  snprintf( err_text, 60, "First (\"magic\") Mercury header byte incorrect: %hhu",
142  header_magic_byte );
143  failed = true;
144  }
145  else if ( ( _header_version < _min_header_version ) || ( _header_version > _max_header_version ) )
146  {
147  snprintf( err_text, 60, "Wrong Mercury Header version: %hhu not in range %hhu:%hhu",
148  _header_version, _min_header_version, _max_header_version );
149  failed = true;
150  }
151 
152  if ( !failed )
153  {
154  if ( _header_version == HEADER_VERSION_2 )
155  {
156  exp_hdr_len = HEADER_VERSION_2_SIZE;
157  }
158  else
159  {
160  exp_hdr_len = HEADER_VERSION_3_SIZE;
161  }
162 
163  if ( _header_size != exp_hdr_len )
164  {
165  snprintf( err_text, 60, "Wrong Mercury Header size: %hhu instead of %zu",
166  _header_size, exp_hdr_len );
167  failed = true;
168  }
169  else if ( buflen < exp_hdr_len )
170  {
171  failed = incomplete = true;
172  }
173  else if ( ( _required_message_type != HG_MSG_NOT_SUPPORT ) &&
174  ( _required_message_type != _message_type ) )
175  {
176  snprintf( err_text, 60, "Wrong Mercury message type: %hhu instead of %hhu",
177  ( uint8_t ) _message_type, ( uint8_t ) _required_message_type );
178  failed = true;
179  }
180  }
181 
182  if ( !failed && ( _header_version == HEADER_VERSION_3 ) )
183  {
184  /* We don't need to validate the message ID. */
185  _message_api = ( mercury_msg_api_t ) px::asio::CUtil::read_u8( ibuf );
186 
187  // If we need to match the UUID, then check byte-by-byte
188  if ( _require_uuid_match )
189  {
190  for ( size_t i = 0; i < 16; i++ )
191  {
192  uint8_t ubyte = px::asio::CUtil::read_u8( ibuf );
193  if ( ubyte != _uuid[i] )
194  {
195  snprintf( err_text, 60, "Wrong UUID at %zu: %hhu instead of %hhu",
196  ( i + 1 ), ubyte, _uuid[i] );
197  failed = true;
198  break;
199  }
200  }
201  }
202  // Else just read it into our _uuid vector
203  // Will pay more attention later when we need to reroute to external devices via UUIDs
204  else
205  {
206  _uuid.clear();
207  for ( int i = 0; i < 16; i++ )
208  {
209  _uuid.push_back( px::asio::CUtil::read_u8( ibuf ) );
210  }
211  }
212  }
213  }
214 
215  if ( incomplete )
216  {
217  LIBPX_INFO( 6, "Mercury header parsing incomplete at %zu bytes", buflen );
218  }
219  else if ( failed )
220  {
221  LIBPX_NOTICE( 4, "Mercury header parsing failed: %s", err_text );
222  LIBPX_DBGDUMP( 8, static_cast<const void *>( &( *start ) ), buflen, " Bad Header bytes: " );
223  _error_message = err_text;
224  }
225  // else
226  else if ( _header_version == HEADER_VERSION_2 )
227  {
228  LIBPX_DEBUG( 8,
229  "Parsed v2 Mercury Header: Header Size=%hhu, Message Size=%u, Message Type=%hhu",
230  _header_size, _message_size, ( uint8_t ) _message_type );
231  }
232  else
233  {
234  LIBPX_DEBUG( 8,
235  "Parsed v3 Mercury Header: Header Size=%hhu, Message Size=%u, Message Type=%hhu, API ID=%hhu",
236  _header_size, _message_size, ( uint8_t ) _message_type, ( uint8_t ) _message_api );
237  LIBPX_DBGDUMP( 8, _uuid.data(), _uuid.size(), " UUID: " );
238  }
239 
240  if ( incomplete )
241  {
242  boost::tribool tristate( boost::logic::indeterminate );
243  // Return second pointing to the start of the buffer, still
244  return ( std::make_pair( tristate, start ) );
245  }
246  else
247  {
248  // ibuf is pointing to the first byte after the header (at least on success)
249  return ( std::make_pair( !failed, ibuf ) );
250  }
251 }
