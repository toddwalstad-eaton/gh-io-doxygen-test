88  int signum )
89 {
90  PUSH_FUNC;
91 
92  // Call other handlers first, in case we're quitting
93  for ( auto h : _additional_handlers )
94  {
95  if ( likely( h ) )
96  {
97  h( err, signum );
98  }
99  }
100  if ( err )
101  {
102  LIBPX_FERR( 1, "%s", err.message().c_str() );
103  if ( !_owner.is_quit_signal_received() )
104  {
105  _owner.quit();
106  }
107  }
108  else

116  case SIGHUP:

118  case SIGINT:
119  case SIGTERM:
120  case SIGSEGV:

122  {

124  SYSLOG_NOTICE( 1, "Caught signal %d, exiting...", signum );
125  _owner.quit();

128 
129  default:
130  // Nothing to do, this signal is probably for one of the additional_handlers
