147 std::vector<px::string> px::COptionsBase::break_line( const px::string &str,
148  size_t nchars,
149  const px::string &sep )
150 {
151  assert( !str.empty() );
152  assert( !sep.empty() );
153 
154  std::vector<px::string> retval;
155 
156  size_t pos = 0, eolpos;
157  while ( pos != str.npos )
158  {
159  // Is there a newline char within the next nchars characters?
160  // If so, put a break there!
161  eolpos = str.find( "\n", pos );
162  if ( ( eolpos != str.npos ) && ( ( eolpos - pos ) < nchars ) )
163  {
164  // Skip the newline char for this string so we don't interfere with indenting
165  retval.push_back( px::string( str, pos, ( eolpos - pos ) ) );
166  ++eolpos; // now skip the newline
167  }
168  // Is this the last line? If so, push it all as one string (unless contains linefeeds)
169  else if ( pos + nchars >= str.size() )
170  {
171  retval.push_back( px::string( str, pos ) );
172  eolpos = str.npos;
173  }
174  else
175  {
176  // Look for the last separator
177  eolpos = str.find_last_of( sep.c_str(), pos + nchars - 1 );
178  // If no separator, or before this position,
179  if ( ( eolpos == str.npos ) ||
180  ( eolpos < pos ) ||
181  ( eolpos - pos < ( nchars / 2 ) ) )
182  {
183  // Hard break
184  eolpos = pos + nchars;
185  retval.push_back( px::string( str, pos, nchars ) );
186  }
187  else
188  {
189  // Normal (good) break
190  retval.push_back( px::string( str, pos,
191  eolpos - pos ) );
192  }
193  }
194 
195  pos = eolpos;
196 
197  while ( ( pos != str.npos ) &&
198  ( pos < str.size() ) &&
199  ( isspace( str[pos] ) ) )
200  {
201  pos++;
202  }
203 
204  if ( pos >= str.size() )
205  {
206  pos = str.npos;
207  }
208  }
209 
210  return ( retval );
211 }
