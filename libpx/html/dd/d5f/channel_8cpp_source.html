197 void px::CChannel::to( px::proto::MclEntry &arg ) const
198 {
199  if ( channel_settings != nullptr )
200  {
201  auto cs = arg.mutable_channel_configuration();
202  if ( unlikely( cs == nullptr ) )
203  {
204  throw px::generic_err();
205  }
206  channel_settings->to( *cs );
207  // Set trend fields, if there is no trend config then keep empty
208  if ( channel_settings->flags.get( px::channel_flag::trending_enabled ) ||
209  channel_settings->flags.get( px::channel_flag::has_min ) ||
210  channel_settings->flags.get( px::channel_flag::has_max ) ||
211  channel_settings->flags.get( px::channel_flag::has_avg ) ||
212  channel_settings->flags.get( px::channel_flag::has_actual ) )
213  {
214  px::proto::TrendConfiguration *tc = arg.mutable_trend_configuration();
215  if ( unlikely( tc == nullptr ) )
216  {
217  throw px::generic_err();
218  }
219  tc->set_trending_enabled( channel_settings->flags.get( px::channel_flag::trending_enabled ) );
220  tc->set_has_min( channel_settings->flags.get( px::channel_flag::has_min ) );
221  tc->set_has_max( channel_settings->flags.get( px::channel_flag::has_max ) );
222  tc->set_has_avg( channel_settings->flags.get( px::channel_flag::has_avg ) );
223  tc->set_has_actual( channel_settings->flags.get( px::channel_flag::has_actual ) );
224  }
225  else
226  {
227  arg.clear_trend_configuration();
228  }
229  }
230 
231  // Possible lossiness: Fields in AlarmConfiguration proto are only
232  // included once in the proto, included once for each CAlarmTrigger.
233  // Use the first and hope for the best.
234  if ( alarm_settings.size() > 0 )
235  {
236  auto as = arg.mutable_alarm_configuration();
237  if ( unlikely( as == nullptr ) )
238  {
239  throw px::generic_err();
240  }
241  alarm_settings.begin()->to( *as );
242  for ( auto &i : alarm_settings )
243  {
244  auto at = as->add_triggers();
245  if ( unlikely( at == nullptr ) )
246  {
247  throw px::generic_err();
248  }
249  i.to( *at );
250  }
251  }
252  if ( ( channel_settings != nullptr ) && ( channel_settings->parent != PX_ID_UNSET ) )
253  {
254  arg.set_parent_id( channel_settings->parent );
255  }
256 }

305 bool px::CChannel::get_field_by_name( const px::string &field_string, px::CValue &field_value ) const
306 {
307  bool rv = false;
308 
309  // Check for empty field string first
310  if ( likely( field_string.size() > 0 ) )
311  {
312  std::vector<px::string> field_names = px::CUtil::split_string( field_string, '.' );
313  // Check for special cases and convert (punt for now for many)
314  if ( field_names[0] == "alarm_settings" )
315  {}
316  else if ( field_names[0] == "state" )
317  {}
318  else if ( field_names[0] == "value" )
319  {
320  if ( value.type() != px::datatype::invalid )
321  {
322  field_value = value;
323  rv = true;
324  }
325  }
326  else if ( field_names[0] == "timestamp" )
327  {}
328  else if ( field_names[0] == "trend_accumulator" )
329  {}
330  else if ( field_names[0] == "channel_settings" )
331  {
332  // Let channel_settings class handle it
333  if ( likely( channel_settings ) && ( field_names.size() > 1 ) )
334  {
335  px::string sub_field_string = field_string.substr( field_names[0].size() + 1 );
336  // Special case - pass "value" down to channel_settings so we can
337  // return the correct mapped value.
338  if ( ( field_names.size() == 2 ) && ( field_names[1] == "multistate_map" ) )
339  {
340  sub_field_string += ".";
341  sub_field_string += value.as_string();
342  }
343  return ( channel_settings->get_field_by_name( sub_field_string, field_value ) );
344  }
345  }
346  else
347  {
348  // Convert from MclEntry to googlePB::Message
349  // can use libpx GetFieldValue to get the value.
350  px::proto::MclEntry protoMclEntry;
351  to( protoMclEntry );
352 
353  try
354  {
355  if ( !field_string.empty() )
356  {
357  px::CValue new_value = px::CProtoOPS::GetFieldValue( field_string,
358  protoMclEntry );
359  if ( new_value.type() != px::datatype::invalid )
360  {
361  field_value = new_value;
362  rv = true;
363  }
364  }
365  else
366  {
367  LIBPX_NOTICE( 3,
368  "CChannel get_field_by_name() failed due to empty field" );
369  }
370  }
371  catch ( const px::generic_err &gx )
372  {
373  LIBPX_NOTICE( 3,
374  "CChannel get_field_by_name() failed with error: %s",
375  gx.what() );
376  }
377  }
378  }
379  return ( rv );
380 }
