29  /* Determine the end-of-interval time for the given start time.
30  * We will use the convenient px::db::get_normalized_interval_timestamp(), though
31  * that assumes an offset of 0, so we then add the offset to it, then
32  * convert those epoch minutes to minutes in this hour. */
33  uint32_t end_of_interval_epoch_mins = 0;
34  bool on_interval = false;
35 
36  // Get the interval end time (in minutes UTC since 1970) given this start timestamp,
37  // but first remove the offset minutes (or we may jump ahead), and then add them back to the result
38  px::CWallTime adjusted_start = start - px::CTimestamp( 60 * starting_offset_minute );
39  px::db::get_normalized_interval_timestamp( adjusted_start, interval_minutes, end_of_interval_epoch_mins, on_interval );
40 
41  if ( unlikely( end_of_interval_epoch_mins == 0 ) )
42  {
43  // bad timestamp; just return False
44  return ( false );
45  }
46  end_of_interval_epoch_mins += starting_offset_minute;
47  if ( on_interval )
48  {
49  // If start time was right at the end of its interval, then we want to bump up by one to the next interval
50  end_of_interval_epoch_mins += interval_minutes;
51  }
52  timeval tv {0, 0 };
53  tv.tv_sec = end_of_interval_epoch_mins * 60;
54  // To prevent retriggering in this minute (if we check at both 00:01:00 and 00:01:10), force the check
55  // to be 1 sec after the top of the minute
56  tv.tv_sec++;
57 
58  // Convert to a WallTime

60 
61  /* If end time ("now") is past the end of interval, return True

63  * but we were triggering at both 12:16:00 and 12:16:10.
64  * So just make sure we are past the EOI. */
