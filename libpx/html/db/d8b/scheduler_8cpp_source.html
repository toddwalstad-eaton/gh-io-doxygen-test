191 void CScheduler::timer_cb( const boost::system::error_code &err )
192 {
193  PUSH_FUNC;
194 
195  std::vector<std::shared_ptr<ITask> > runnable_tasks;
196 
197  if ( err.value() == boost::system::errc::success )
198  {
199  { // lock scope
200  std::lock_guard<std::mutex> lock{_mutex};
201  px::CWallTime current_time = px::CWallTime::now();
202 
203  // ToDo: this could be optimized, if necessary
204  for ( const auto &trigger : _trigger_map )
205  {
206  if ( trigger.second->in_range( _prev_time, current_time ) )
207  {
208  auto task_pair = _scheduled_tasks.find( trigger.first );
209  if ( task_pair != _scheduled_tasks.end() )
210  {
211  runnable_tasks.push_back( task_pair->second );
212  }
213  }
214  }
215  _prev_time = current_time;
216  }
217 
218  for ( const auto &task : runnable_tasks )
219  {
220  run_task_async( task );
221  }
222  }
223  else
224  {
225  LIBPX_FERR( 0, "Failure in timer: %s", boost::diagnostic_information( err ).c_str() );
226  }
227 
228  if ( _work.get() )
229  {
230  reset_timer();
231  }
232 }
