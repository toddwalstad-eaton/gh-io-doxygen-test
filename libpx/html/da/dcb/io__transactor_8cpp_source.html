92 {
93  PUSH_FUNC;
94 
95  if ( _prot->chunk_size <= 0 )
96  {
97  _prot->chunk_size = 128;
98  }
99 
100  if ( end == _rbuf.begin() + _offset )
101  {
102  // Common case of a single, complete request: clear the
103  // buffer after parsing
104  _rbuf.resize( _prot->chunk_size );
105  _offset = 0;
106  }
107  else if ( ( end < _rbuf.begin() ) || ( end > _rbuf.begin() + _offset ) )
108  {
109  // this should not happen.
110  LIBPX_ERR( 1, "The read buf might be corrupted by I/O interleaving or other"
111  "logic errors. The offset is beyond the buffer size." );
112  // try to fix it by clearing the rbuf.
113  _rbuf.resize( _prot->chunk_size );
114  _offset = 0;
115  }
116  else
117  {
118  // the offset is valid.
119  // Additional data in buffer may be for the next request;
120  // we leave it and shift the bytes down.
121  size_t cmd_len = end - _rbuf.begin();
122  _rbuf.erase( _rbuf.begin(), _rbuf.begin() + cmd_len );
123  // with the check above, this assert should never fail.
124  // assert(cmd_len<_offset);
125  _offset -= cmd_len;

213  // try to fix it by marking all data in the buffer consumed.
214  _offset = _rbuf.size();

216  // parse and read more if needed.
