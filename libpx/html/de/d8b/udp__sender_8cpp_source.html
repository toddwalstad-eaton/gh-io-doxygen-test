44  const CProtocolHandler &prot,

153 void CUdpSender::write_more( const std::shared_ptr<io_buf_t> &pbuf )
154 {
155  PUSH_FUNC;
156 
157  if ( ( pbuf != nullptr ) && pbuf.get() && _udp_socket->is_open() )
158  {
159  LIBPX_DEBUG( 3, "Begin async_write to '%s'",
160  _remote_endpoint->address().to_string().c_str() );
161  _udp_socket->async_send_to( boost::asio::buffer( *pbuf ), *_remote_endpoint,
162  boost::bind( &CUdpSender::on_write, this,
163  boost::asio::placeholders::error, boost::asio::placeholders::bytes_transferred ) );
164  }
165  else
166  {
167  if ( !_udp_socket->is_open() )
168  {
169  LIBPX_WARN( 2, "Sender closed - suppress async_write to '%s'",
170  _remote_endpoint->address().to_string().c_str() );
171  }
172  else
173  {
174  std::string local = ( ( _remote_endpoint != nullptr )
175  ? _remote_endpoint->address().to_string()
176  : "Unknown" );
177  LIBPX_DEBUG( 3, "Sender message to '%s'", local.c_str() );
178  }
179  // Need to write nothing to keep machine moving
180  pbuf->resize( 0 );
181  _udp_socket->async_send_to( boost::asio::null_buffers(), *_remote_endpoint,
182  boost::bind( &CUdpSender::on_write, this,
183  boost::asio::placeholders::error, boost::asio::placeholders::bytes_transferred ) );
184  }
185 }

190 void CUdpSender::on_write( const boost::system::error_code &err, size_t bytes_transferred )
191 {
192  PUSH_FUNC;
193 
194  if ( !err )
195  {
196  handle_write( err, bytes_transferred );
197  }
198  else
199  {
200  // Don't pass error to handle_read() unless you want it to remove actor
201  if ( !_shutdown )
202  {
203  reset_socket();
204  boost::system::error_code no_error;
205  handle_write( no_error, 0 );
206  }
207  else
208  {
209  handle_write( err, 0 );
210  }
211  }
