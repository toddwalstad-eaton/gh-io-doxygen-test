315 static px::string parse_one_message( const ::json_spirit::Object &object,
316  const google::protobuf::Descriptor *desc,
317  google::protobuf::Message *msg )
318 {
319  unsigned int i;
320 
321  px::string rv = "";
322 
323  // Iterate over the fields of the object
324  for ( i = 0; i < object.size(); i++ )
325  {
326  px::string name = px::to_pxstr( object[i].name_ );
327  ::json_spirit::Value val = object[i].value_;
328  ::json_spirit::Value_type type = val.type();
329  const google::protobuf::Reflection *r = msg->GetReflection();
330  const google::protobuf::FieldDescriptor *fd =
331  desc->FindFieldByName( px::to_basicstr( name ) );
332 
333  // If value is null, nothing to do
334  if ( val.is_null() )
335  {
336  continue;
337  }
338  if ( unlikely( r == nullptr ) )
339  {
340  return ( "Error: Reflection error" );
341  }
342  if ( fd == nullptr )
343  {
344  // No existing field with that name.
345  // Nothing useful we can do.
346  std::ostringstream s;
347  s << "Warning: Field '" << name << "' not found in protobuf";
348  rv.append( px::to_pxstr( s.str() ) );
349  continue;
350  }
351  // Is the value an array? If so, iterate over the values and set
352  if ( type == ::json_spirit::array_type )
353  {
354  if ( fd->label() != google::protobuf::FieldDescriptor::LABEL_REPEATED )
355  {
356  rv.append( "Error: Array value given for non-repeated protobuf field" );
357  }
358  else
359  {
360  ::json_spirit::Array a = val.get_array();
361  unsigned int j;
362  r->ClearField( msg, fd );
363  for ( j = 0; j < a.size(); j++ )
364  {
365  rv.append( set_single_value( a[j], msg, r, fd ) );
366  }
367  }
368  }
369  else
370  {
371  // If field is repeated, we should have gotten an array
372  if ( fd->label() == google::protobuf::FieldDescriptor::LABEL_REPEATED )
373  {
374  rv.append( "Error: Non-array value given for repeated protobuf field" );
375  }
376  else
377  {
378  rv.append( set_single_value( val, msg, r, fd ) );
379  }
380  }
381  } // End loop over fields of the object
382  return ( rv );
383 }

432 static px::string json_to_protobuf_internal( bool is_qualified,
433  const px::string &json,
434  const px::string &protobuf_type,
435  std::vector<std::shared_ptr<google::protobuf::Message> > &msgs )
436 {
437  assert( !json.empty() );
438 
439  ::json_spirit::Value top;
440  ::json_spirit::Object top_obj;
441  ::json_spirit::Array list;
442  unsigned int i;
443  px::string rv = "";
444 
445  // e.g. [ { "Channel":{...} }, { "Device":{...} } ]
446  bool is_individual_array = false;
447  // One message only, e.g. { "Channel":{...} }
448  bool is_single_message = false;
449 
450  // is_qualified: e.g. [ {...}, {...} ] (type given in separate param)
451  // Else array of same type e.g. { "Channel": [ {...},{...} ] }
452 
453  try
454  {
455  std::string std_json( json.c_str() );
456  read_string_or_throw( std_json, top );
457  }
458  catch ( ::json_spirit::Error_position &e )
459  {
460  std::ostringstream s;
461  s << "JSON parse Error: " << e.reason_ << " at line " << e.line_ << ", column " << e.column_;
462  rv = px::to_pxstr( s.str() );
463  return ( rv );
464  }
465  catch ( std::exception &ex )
466  {
467  rv = "JSON parse exception: ";
468  rv += ex.what();
469  return ( rv );
470  }
471 
472  if ( top.type() == ::json_spirit::array_type )
473  {
474  if ( !is_qualified )
475  {
476  is_individual_array = true;
477  }
478  }
479  else if ( top.type() == ::json_spirit::obj_type )
480  {
481  top_obj = top.get_obj();
482  if ( is_qualified ||
483  ( ( top_obj.size() == 1 ) &&
484  ( top_obj[0].value_.type() == ::json_spirit::obj_type ) ) )
485  {
486  is_single_message = true;
487  }
488  else if ( ( top_obj.size() != 1 ) ||
489  ( top_obj[0].value_.type() != ::json_spirit::array_type ) )
490  {
491  return ( "JSON must be a single object, array of objects, or object containing one array value" );
492  }
493  }
494  else
495  {
496  return ( "JSON must be a single object, array of objects, or object containing one array value" );
497  }
498 
499  if ( is_single_message )
500  {
501  google::protobuf::Message *m = nullptr;
502  const google::protobuf::Descriptor *desc = nullptr;
503  px::string result;
504  if ( is_qualified )
505  {
506  result = new_message_given_name( protobuf_type, &m, &desc );
507  }
508  else
509  {
510  result = new_message_given_name( px::to_pxstr( top_obj[0].name_ ), &m, &desc );
511  }
512  if ( result != "" )
513  {
514  return ( result );
515  }
516  if ( is_qualified )
517  {
518  result = parse_one_message( top_obj, desc, m );
519  }
520  else
521  {
522  result = parse_one_message( top_obj[0].value_.get_obj(), desc, m );
523  }
524  if ( result == "" )
525  {
526  msgs.clear();
527  msgs.push_back( std::shared_ptr<google::protobuf::Message>( m ) );
528  return ( "" );
529  }
530  else
531  {
532  std::size_t found = result.find( "Error: " );
533  if ( found != std::string::npos )
534  {
535  delete m;
536  return ( result );
537  }
538  else
539  {
540  // Proceed with warning if there is no error but some warnings.
541  msgs.clear();
542  msgs.push_back( std::shared_ptr<google::protobuf::Message>( m ) );
543  return ( result );
544  }
545  }
546  }
547 
548  // Past this point we have an array, with zero, one or N message-type
549  // specifiers.
550 
551  px::string proto_name;
552  if ( is_qualified )
553  {
554  list = top.get_array();
555  proto_name = protobuf_type;
556  }
557  else if ( is_individual_array )
558  {
559  list = top.get_array();
560  }
561  else
562  {
563  list = top_obj[0].value_.get_array();
564  proto_name = px::to_pxstr( top_obj[0].name_ );
565  }
566  msgs.clear();
567  for ( i = 0; i < list.size(); i++ )
568  {
569  if ( list[i].type() != ::json_spirit::obj_type )
570  {
571  std::ostringstream s;
572  s << "Array item " << i << " is not a JSON object\n";
573  rv.append( px::to_pxstr( s.str() ) );
574  continue;
575  }
576  google::protobuf::Message *m = nullptr;
577  const google::protobuf::Descriptor *desc = nullptr;
578  ::json_spirit::Object this_obj = list[i].get_obj();
579  if ( is_individual_array )
580  {
581  if ( this_obj.size() != 1 )
582  {
583  std::ostringstream s;
584  s << "Array item " << i << ": JSON object has too many parts\n";
585  rv.append( px::to_pxstr( s.str() ) );
586  continue;
587  }
588  proto_name = px::to_pxstr( this_obj[0].name_ );
589  if ( this_obj[0].value_.type() != ::json_spirit::obj_type )
590  {
591  std::ostringstream s;
592  s << "Array item " << i << " is not a JSON object\n";
593  rv.append( px::to_pxstr( s.str() ) );
594  continue;
595  }
596  this_obj = this_obj[0].value_.get_obj();
597  }
598  px::string result = new_message_given_name( proto_name, &m, &desc );
599  if ( result != "" )
600  {
601  rv.append( result );
602  continue;
603  }
604  result = parse_one_message( this_obj, desc, m );
605  if ( result == "" )
606  {
607  msgs.push_back( std::shared_ptr<google::protobuf::Message>( m ) );
608  }
609  else
610  {
611  std::size_t found = result.find( "Error: " );
612  if ( found == std::string::npos )
613  {
614  // Only proceed if there is no error
615  msgs.push_back( std::shared_ptr<google::protobuf::Message>( m ) );
616  rv.append( result );
617  }
618  else
619  {
620  // Proceed if there are warnings
621  std::ostringstream s;
622  delete m;
623  s << "Array item " << i << ": " << result << "\n";
624  rv.append( px::to_pxstr( s.str() ) );
625  }
626  }
627  }
628  return ( rv );
629 }

724 static px::string json_string_escape( const px::string &in )
725 {
726  px::string rv;
727 
728  rv.reserve( in.size() );
729  // JSON spec requires escaping \, " and \x00-\x1f.
730  // If bytes >=0x80 are found, they are assumed to be the UTF-8 encoding
731  // of a single character. If they don't form valid UTF-8, a Unicode
732  // replacement character is substituted.
733  // If the string is using single characters >=0x80 that are not UTF-8,
734  // the "binary" value type should be used to encode them as base64
735  // text instead.
736  const unsigned char *pc = ( const unsigned char * )( in.c_str() );
737  size_t rem = in.size();
738  while ( rem > 0 )
739  {
740  uint32_t codepoint = px::pxstr_first_utf8_codepoint( pc, rem );
741  if ( likely( ( codepoint > 0x1f ) && ( codepoint < 0x80 ) ) )
742  {
743  if ( codepoint == '\\' )
744  {
745  rv.append( "\\\\" );// two backslashes
746  }
747  else if ( codepoint == '"' )
748  {
749  rv.append( "\\\"" );
750  }
751  else
752  {
753  rv.push_back( codepoint & 0x7f );
754  }
755  }
756  else
757  {
758  rv.append( "\\u" );
759  if ( unlikely( codepoint > 0xFFFF ) )
760  {
761  // Need to represent this as a UTF-16 surrogate pair
762  codepoint -= 0x10000;
763  uint16_t high = 0xD800 | ( codepoint >> 10 );
764  uint16_t low = 0xDC00 | ( codepoint & 0x03FF );
765  rv.push_back( hex_char( ( high & 0xF000 ) >> 12 ) );
766  rv.push_back( hex_char( ( high & 0xF00 ) >> 8 ) );
767  rv.push_back( hex_char( ( high & 0xF0 ) >> 4 ) );
768  rv.push_back( hex_char( high & 0xF ) );
769  rv.append( "\\u" );
770  codepoint = low;
771  }
772  rv.push_back( hex_char( ( codepoint & 0xF000 ) >> 12 ) );
773  rv.push_back( hex_char( ( codepoint & 0xF00 ) >> 8 ) );
774  rv.push_back( hex_char( ( codepoint & 0xF0 ) >> 4 ) );
775  rv.push_back( hex_char( codepoint & 0xF ) );
776  }
777  } // end loop over each code-point
778  return ( rv );
779 }
