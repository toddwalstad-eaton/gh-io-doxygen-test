186 void CUdsDgramListener::write_more( const std::shared_ptr<io_buf_t> &pbuf )
187 {
188  PUSH_FUNC;
189 
190  // We will not have a valid _sender_endpoint address if the socket was reset, eg due to errors
191  if ( ( pbuf != nullptr ) && pbuf.get() && ( pbuf->size() > 0 ) &&
192  _socket->is_open() && !_sender_endpoint->path().empty() )
193  {
194  LIBPX_DEBUG( 3, "Begin async_write to '%s'", _sender_endpoint->path().c_str() );
195  _socket->async_send_to( boost::asio::buffer( *pbuf ), *_sender_endpoint,
196  boost::bind( &CUdsDgramListener::on_write,
197  this,
198  boost::asio::placeholders::error,
199  boost::asio::placeholders::bytes_transferred ) );
200  }
201  else
202  {
203  if ( !_socket->is_open() )
204  {
205  LIBPX_DEBUG( 3, "Listener closed - suppress async_write to '%s'",
206  _sender_endpoint->path().c_str() );
207  }
208  else if ( _sender_endpoint->path().empty() )
