32  // types except binary. We want to just ignore binary values.
33  // CValue converts them to 0.
34  return;
35  }

37  // calculate sum, count, min, max, and last
38  if ( std::isnan( v ) )
39  {
40  // ignore NAN

42  }
43  ++_count;
44  if ( likely( ( _sum < ( DBL_MAX / 2 ) ) && ( v < ( DBL_MAX / 2 ) ) ) )
45  {

50  }
51  else if ( _shift < 31 )

54  }

58  }
59  }
60  else

65  // This will not lose precision unless _shift > (1023-52).
66  _shift++;
