684  LIBPX_ERR( 0, " generic error exception caught when request queue failed:%s ", e.what() );
685  }
686  catch ( ... )
687  {
688  LIBPX_ERR( 0, "Unknown Exception caught during submitting request..." );
689  logged_once = true;
690  }
691 
692  if ( unlikely( xid == 0 ) )
693  {
694  LIBPX_WARN( 1, "submit() in submit_wait() returned 0 xid, retrying" );
695  size_t retry_count = 1; // Since already tried once.
696  while ( ( xid == 0 ) && ( retry_count++ <= 3 ) )
697  {
698  try
699  {
700  xid = submit( req, cbh, true );
701  }
702  catch ( const px::no_connection &e )
703  {
704  if ( !logged_once )
705  {
706  LIBPX_ERR( 0, " No connection exception caught during submitting request:%s ", e.what() );
707  logged_once = true;
708  }
709  }
710  catch ( ... )
711  {
712  if ( !logged_once )
713  {
714  LIBPX_ERR( 0, "Unknown exception caught during submitting request" );
715  logged_once = true;
716  }
717  }
718  usleep( 100000 );
719  }
720  if ( xid != 0 )
721  {
722  LIBPX_WARN( 1, "submit() retries succeeded" );
723  }
724  else
725  {
726  waiting_mutex.unlock();
727  LIBPX_ERR( 0, "submit() failed after retries in submit_wait()" );
728  rv.reset( new CRequestResponse() );
729  rv->status = RequestStatus::NOT_ABLE_TO_EXECUTE;
730  rv->transaction_id = req.transaction_id;
731  return ( rv );
732  }
733  }
734  // Intentionally lock a second time, this will make us sleep until
735  // the callback releases the mutex. If the callback has somehow already
736  // been called, no problem, this won't sleep.
737  waiting_mutex.lock();
738  // At this point the callback has been called.
739  unregister_callback( cbh );
740  rv = cb->response;
741  waiting_mutex.unlock(); // we locked twice after all
742 
743  // Provide some useful debug info!
744  if ( rv && ( rv->status != px::cds::RequestStatus::SUCCEEDED ) )
745  {
746  LIBPX_DEBUG( 5, "CDS Response: %s", rv->get_status_text() );
747  }
748  if ( unlikely( !rv ) )
749  {
750  LIBPX_ERR( 0, "submit_wait() returning null response" );
751  }
752  return ( rv );
753 }

838 void CClient::bg_thread_func()
839 {
840  struct pollfd topoll[2];
841  int rc;
842  unsigned char *shared_buf;
843  unsigned int bytes_to_read, remaining_len;
844  std::shared_ptr<CRequestResponse> resp;
845  std::pair<size_t, bool> read_rc;
846  /* Instantiate a Log pointer for this thread.
847  * If this is not set, messages from the background thread, including
848  * callback execution, will not appear or be logged.
849  */
850  boost::thread_specific_ptr<toolkit::CSysLog> _thr_local_syslog;
851 
852  _thr_local_syslog.reset( new toolkit::CSysLog );
853  /* Set the verbosity level to that for the CDS Thread level's.
854  * This is the verbosity level that our callbacks (invoked on this thread) will have.
855  */
856  if ( likely( _thr_local_syslog.get() ) )
857  {
858  _thr_local_syslog.get()->verbosity = _libpx_verb_array[LIBPX_LOG_CDS_THREAD];
859  }
860  /* Kick off syslogging with a message.
861  * This, it turns out, is necessary to do early; otherwise, if the first
862  * logged message happens in a later callback, it will segfault
863  * (for poorly understood reasons).
864  */
865  LIBPX_INFO( 2, "Inside startup of CDS Client background thread." );
866 
867  px::CUtil::set_thread_name( "CDSClient RX" );
868  memset( topoll, 0, sizeof( topoll ) );
869  topoll[0].fd = cds_response_queue->fd();
870  topoll[0].events = POLLIN;
871  topoll[1].fd = bg_thread_stop_fd;
872  topoll[1].events = POLLIN;
873 
874  while ( 1 )
875  {
876  // Wait for something interesting to happen
877  rc = ::poll( topoll, 2, -1 );
878  if ( rc < 0 )
879  {
880  // Syscall can be interrupted, try again
881  if ( errno == EINTR )
882  {
883  continue;
884  }
885  // Error, terminate thread
886  LIBPX_ERR( 2, "Terminating CDS Client background thread due to error." );
887  break;
888  }
889  if ( topoll[1].revents & POLLIN )
890  {
891  // Main thread requested us to exit, do so
892  LIBPX_INFO( 3, "Terminating CDS Client background thread as requested." );
893  break;
894  }
895  if ( !( topoll[0].revents & POLLIN ) )
896  {
897  // Nothing to do, go back to waiting
898  continue;
899  }
900  do // read() may need to be called twice
901  {
902  shared_buf = ( unsigned char * )cds_response_queue->get();
903  try
904  {
905  read_rc = cds_response_queue->read(); // Can throw toolkit::ipc::no_data or toolkit::ipc::generic_err
906  }
907  catch ( toolkit::ipc::no_data )
908  {
909  // FD said we have data, but we don't.
910  // Crude but better options are unavailable
911  usleep( 100000 );
912  break;
913  }
914  catch ( toolkit::ipc::generic_err )
915  {
916  // FD said we have a data buffer problem
917  usleep( 100000 );
918  break;
919  }
920 
921  bytes_to_read = read_rc.first;
922  while ( bytes_to_read > 0 )
923  { // we may get more than one at once
924  resp = req_unpack( shared_buf, bytes_to_read, &remaining_len );
925  if ( !resp )
926  {
927  // Unpack failed, give up on this read
928  break;
929  }
930  shared_buf += ( bytes_to_read - remaining_len );
931  bytes_to_read = remaining_len;
932  if ( unlikely( resp->callback_handle == CALLBACK_NONE ) )
933  {
934  // Don't know why we got this response, ignore it
935  continue;
936  }
937  else if ( resp->is_subscription() || resp->is_unsolicited() )
938  {
939  const char *mtype = ( resp->is_subscription() ? "Subscribed" : "Unsolicited" );
940  // Message came either from a subscription, or on the CDS's own initiative
941  LIBPX_DEBUG( 9, "handle %s, with callback handle=%d\ttoken=%s", mtype, resp->callback_handle, resp->session_token.c_str() );
942  std::shared_ptr<CCallback> ucb;
943  callbacks_mutex.lock();
944  // We use the same callback for subscribed and truly unsolicited updates
945  // (somewhat misnamed, since subscribed updates are its main use)
946  if ( unsol_cb )
947  {
948  ucb = unsol_cb;
949  }
950  callbacks_mutex.unlock();
951  if ( ucb )
952  {
953  LIBPX_DEBUG( 8, "Trying %s callback, with callback handle=%d\ttoken=%s", mtype, resp->callback_handle, resp->session_token.c_str() );
954  // Protect ourselves against exceptions in the callback code
955  try
956  {
957  ucb->cds_response_callback( resp );
958  }
959  catch ( const std::exception &e )
960  {
961  // Eg, std::errc::resource_deadlock_would_occur
962  LIBPX_ERR( 2, "Callback for %s threw an exception inside the background thread: %s", mtype, e.what() );
963  // But we survived that one, and keep on ticking ...
964  }
965  }
966  else
967  {
968  LIBPX_DEBUG( 9, "Queue %s message, with callback handle=%d\ttoken=%s", mtype, resp->callback_handle, resp->session_token.c_str() );
969  response_queue_mutex.lock();
970  response_queue.push_back( resp );
971  response_queue_mutex.unlock();
972  }
973  }
974  else if ( resp->callback_handle == CALLBACK_ENQUEUE )
975  {
976  // The client hasn't registered a callback handler for submit() responses,
977  // so just queue up this response
978  response_queue_mutex.lock();
979  response_queue.push_back( resp );
980  response_queue_mutex.unlock();
981  }
982  else
983  {
984  // This is the response to a submit() request; find the right callback via its handle
985  std::shared_ptr<CCallback> cb;
986  callbacks_mutex.lock();
987  auto cbi = callbacks.find( resp->callback_handle );
988  if ( cbi != callbacks.end() )
989  {
990  cb = cbi->second;
991  }
992  callbacks_mutex.unlock();
993  if ( likely( cb ) )
994  {
995  LIBPX_DEBUG( 9, "Trying response callback, with callback handle=%d TransID=%u token=%s",
996  resp->callback_handle, resp->transaction_id, resp->session_token.c_str() );
997  // Protect ourselves against exceptions in the callback code
998  try
999  {
1000  cb->cds_response_callback( resp );
1001  }
1002  catch ( const std::exception &e )
1003  {
1004  // Eg, std::errc::resource_deadlock_would_occur
1005  LIBPX_ERR( 2, "Response callback threw an exception inside the background thread: %s", e.what() );
1006  // But we survived that one, and keep on ticking ...
1007  }
1008  }
1009  }
1010  } // end loop over this receive batch
1011  }
1012  while ( read_rc.second == true ); // read() again if queue asks us to
1013  } // end loop forever, ending thread execution
1014 }
