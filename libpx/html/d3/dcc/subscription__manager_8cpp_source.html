116  // from _subscribers, _subscriber_map and _item_map
117  size_t idx = 0;
118  // _subscriber_map
119  pthread_rwlock_wrlock( &_rwlock );
120  auto isub = _subscriber_map.find( id );
121  if ( unlikely( isub == _subscriber_map.end() ) )
122  {
123  // didn't exist, do nothing
124  pthread_rwlock_unlock( &_rwlock );
125  return;
126  }
127  idx = isub->second;
128  _subscriber_map.erase( isub );
129  CSubscriberInfo *subscriber = nullptr;

132  subscriber = &_subscribers[idx];

135  subscriber = nullptr;
136  }
137  }
138  assert( subscriber != nullptr ); // should not happen
139  if ( subscriber != nullptr )

141  // _item_map

143  // Check the map holding partial OOB tags if appropriate
144  for ( const auto &item : subscriber->get_items() )
145  {

153  if ( imatch != sublist.end() )

167  else

170  }
171  }
172  // remove this item if no other subscriber is interested in.
173  if ( sublist.empty() )
174  {
175  _item_map.erase( imap );
176  }
177  }
178  }
179  else
180  {
181  auto imap = _partial_tag_items_map.find( item );
182  if ( imap != _partial_tag_items_map.end() )
183  {
184  auto &sublist = imap->second;
185  auto imatch = std::find( sublist.begin(), sublist.end(), idx );
186  if ( imatch != sublist.end() )
187  {
188  // remove sub_index from the _item_map
189  // Ordering isn't important, so do this for efficiency
190  std::swap( *imatch, sublist.back() );
191  sublist.pop_back();
192  }
193  // remove this item if no other subscriber is interested in.
194  if ( sublist.empty() )
195  {
196  _partial_tag_items_map.erase( imap );
197  }
198  }
199  }
200  }
201  // _subscribers
202  // Remove this subscriber itself from the list, by swapping in the last
203  // subscriber in the list (if necessary)
204  size_t last_idx = _subscribers.size() - 1;
205  if ( idx != last_idx )
206  {
207  _subscribers[idx] = std::move( _subscribers.back() );

209  if ( igoodsub != _subscriber_map.end() )
210  {
211  assert( igoodsub->second == last_idx );
212  igoodsub->second = idx;
213  // Clean up per-item references to this index
214  for ( const auto &item : _subscribers[idx].get_items() )
215  {
216  // If not found in the standard items map, check the map holding partial OOB tags
217  auto imap = _item_map.find( item );
218  if ( imap != _item_map.end() )
219  {
220  auto &sublist = imap->second;
221  auto imatch = std::find( sublist.begin(), sublist.end(),
222  last_idx );
223  if ( imatch != sublist.end() )
224  {
225  // Point at the new index instead
226  *imatch = idx;
227  }
228  }
229  else
230  {
231  auto imap = _partial_tag_items_map.find( item );
232  if ( imap != _partial_tag_items_map.end() )
233  {
234  auto &sublist = imap->second;
235  auto imatch = std::find( sublist.begin(), sublist.end(),
236  last_idx );
237  if ( imatch != sublist.end() )

239  // Point at the new index instead

241  }
242  }
243  }
244  }
245  }
246  }
