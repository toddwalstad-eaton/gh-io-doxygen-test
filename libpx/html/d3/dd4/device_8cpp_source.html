573 void px::CDevice::from( const px::proto::Device &device, bool &bump_change_signature )
574 {
575  bump_change_signature = false;
576  if ( device.has_id() )
577  {
578  auto new_id = ( px::id_t )device.id();
579  if ( _id != new_id )
580  {
581  bump_change_signature = true;
582  }
583  _id = new_id;
584  }
585  if ( device.has_parent_id() )
586  {
587  if ( device.parent_id() != 0 )
588  {
589  auto new_parent_id = ( px::id_t )device.parent_id();
590  if ( !parent || ( *parent != new_parent_id ) )
591  {
592  bump_change_signature = true;
593  }
594  parent.reset( new_parent_id );
595  }
596  }
597  if ( device.has_change_signature() )
598  {
599  // do not bump change signature here.
600  change_signature = ( uint64_t )device.change_signature();
601  }
602 
603  for ( int d = 0; d < device.devices_size(); d++ )
604  {
605  proto::Device sdev = device.devices( d );
606  if ( sdev.has_id() )
607  {
608  const auto new_id = sdev.id();
609  if ( sub_devices.count( new_id ) == 0 )
610  {
611  bump_change_signature = true;
612  sub_devices.insert( new_id );
613  }
614  }
615  else
616  {
617  throw px::generic_err( "No device ID" );
618  }
619  }
620  for ( int d = 0; d < device.channels_size(); d++ )
621  {
622  proto::Channel sch = device.channels( d );
623  if ( sch.has_id() )
624  {
625  const auto new_id = sch.id();
626  if ( channels.count( new_id ) == 0 )
627  {
628  bump_change_signature = true;
629  // Add the ID to our set of channels
630  channels.insert( new_id );
631  // And if it has a name, put it in our map
632  // of names to IDs
633  if ( sch.has_canonical_name() )
634  {
635  channel_names[sch.canonical_name().c_str()] = new_id;
636  }
637  }
638  }
639  else
640  {
641  throw px::generic_err( "No channel ID" );
642  }
643  }
644 
645  if ( device.has_name() )
646  {
647  const auto &new_name = px::to_pxstr( device.name() );
648  if ( name != new_name )
649  {
650  bump_change_signature = true;
651  }
652  name = new_name;
653  }
654  if ( device.has_canonical_model_name() )
655  {
656  const auto &new_canonical_model_name = px::to_pxstr( device.canonical_model_name() );
657  if ( canonical_model_name != new_canonical_model_name )
658  {
659  bump_change_signature = true;
660  }
661  canonical_model_name = new_canonical_model_name;
662  }
663  if ( device.has_network_address() )
664  {
665  const auto &new_network_address = px::to_pxstr( device.network_address() );
666  if ( network_address != new_network_address )
667  {
