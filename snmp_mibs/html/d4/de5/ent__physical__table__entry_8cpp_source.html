65 // Uses the px::CDevice data for this instance to update this class' members
66 void CEntPhysicalTableEntry::update_table_members( bool is_update )
67 {
68  // Fill out the other members here, first those for all instances
69  entPhysicalDescr = _px_device.vendor;
70  if ( entPhysicalDescr.length() > 0 )
71  {
72  entPhysicalDescr += " ";
73  }
74  entPhysicalDescr += _px_device.model;
75 
76  if ( _px_device.role == px::device_role::port )
77  {
78  if ( entPhysicalDescr.length() < 2 )
79  {
80  entPhysicalDescr = _px_device.name; // a better alternative for port devices
81  entPhysicalDescr += " Port";
82  }
83 
84  entPhysicalClass = 10; // port(10)
85  entPhysicalIsFRU = 2; // false(2)
86  }
87  else if ( _px_device.role == px::device_role::subdevice )
88  {
89  entPhysicalIsFRU = 2; // false(2)
90  }
91 
92  SYSLOG_INFO( 5, "%s entPhysicalTable entry for device %" PRIu64 ": %s",
93  ( is_update ? "Updating" : "Adding" ),
94  _px_device.id(), entPhysicalDescr.c_str() );
95 
96  // Best effort to get a Parent position (serial address, normally)
97  entPhysicalParentRelPos = std::strtol( _px_device.network_address.c_str(), NULL, 0 );
98 
99  // Publisher (root device) only stuff:
100  if ( _px_device.id() == PX_ID_ROOT )
101  {
102  entPhysicalClass = 5; // container(5)
103  entPhysicalParentRelPos = 0;
104 
105  // Get the linux software version
106  struct utsname buf;
107  int res = uname( &buf );
108  if ( res == 0 )
109  {
110  entPhysicalSoftwareRev = buf.release;
111  }
112  }
113 }
