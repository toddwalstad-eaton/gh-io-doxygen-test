114 void CActiveAlarmsTableEntry::update( const px::CAlarmUpdate &upd )
115 {
116  if ( seq_index > upd.index )
117  {
118  return; // Already updated from a later sequence
119  }
120  seq_index = upd.index;
121  time = upd.timestamp;
122  if ( time.msec() < 0 )
123  {
124  px::CWallTime zero_time( 0, 0 );
125  time = zero_time;
126  }
127  if ( upd.closed )
128  {
129  level = alarmLevel::closed;
130  }
131  else if ( upd.ack_level >= upd.max_reached_trigger_level )
132  {
133  level = alarmLevel::acknowledged;
134  acknowledge_trap_sent = false;
135  }
136  else
137  {
138  level = alarmLevel::cautionary;
139  clear_trap_sent = false;
140  }
141  bool desc_set = false;
142  if ( likely( upd.trigger ) )
143  {
144  if ( !upd.trigger->custom_message.empty() )
145  {
146  description = upd.trigger->custom_message;
147  desc_set = true;
148  }
149  if ( ( upd.trigger->priority < 5000 ) && ( level == alarmLevel::cautionary ) )
150  {
151  level = alarmLevel::critical;
152  }
153  }
154  if ( !desc_set )
155  {
156  description = _source;
157  if ( upd.closed )
158  {
159  description += "closed";
160  }
161  else if ( upd.ack_level >= upd.max_reached_trigger_level )
162  {
163  description += "acknowledged";
164  }
165  else if ( seq_index > 0 )
166  {
167  description += "updated";
168  }
169  else
170  {
171  description += "added";
172  }
173  }
174  value = _source + upd.value.as_string();
175 }

177 uint8_t CActiveAlarmsTableEntry::get_ent_alarm_status_level( bool cleared )
178 {
179  uint8_t retval = ENTITYALARMSTATUS_UNKNOWN_FLAG;
180 
181  SYSLOG_DEBUG( 8, "Alarm Table entry status level %d", ( int ) level );
182  switch ( level )
183  {
184  case alarmLevel::critical:
185  if ( cleared )
186  {
187  retval = ENTITYALARMSTATUS_UNDERREPAIR_FLAG;
188  }
189  else
190  {
191  retval = ENTITYALARMSTATUS_CRITICAL_FLAG;
192  }
193  break;
194  case alarmLevel::cautionary:
195  if ( cleared )
196  {
197  retval = ENTITYALARMSTATUS_UNDERREPAIR_FLAG;
198  }
199  else
200  {
201  retval = ENTITYALARMSTATUS_WARNING_FLAG;
202  }
203  break;
204  // the thought here is an ACK'd or cleared alarm has some level of repair
205  // (either by the user or by the device)
206  case alarmLevel::acknowledged:
207  case alarmLevel::cleared:
208  retval = ENTITYALARMSTATUS_UNDERREPAIR_FLAG;
209  break;
210  case alarmLevel::closed:
211  retval = ENTITYALARMSTATUS_NONE;
212  break;
213  // active and unknown shouldn't ever happen
214  case alarmLevel::active:
215  case alarmLevel::unknown:
216  default:
217  retval = ENTITYALARMSTATUS_UNKNOWN_FLAG;
218  break;
219  }
220 
221  return ( retval );
222 }
