44 bool CPwrMeterRtPhaseMeasuresTableEntry::check_add_table_entry_one_phase( std::shared_ptr<CPwrMeterMib> pwr_meter_mib_instance,
45  const px::CDevice &px_device,
46  const std::vector<px::CChannelSettings> &settings,
47  uint8_t phase_num )
48 {
49  PUSH_FUNC;
50 
51  if ( unlikely( !pwr_meter_mib_instance ) )
52  {
53  SYSLOG_FWARN( 1, "uninitialized mib instance class" );
54  return ( false );
55  }
56  uint32_t id = ( uint32_t ) px_device.id();
57  bool had_existing_row = false;
58  uint64_t keyval = CMmedsNodeTableEntry::compose_row_key( id, phase_num );
59  {
60  std::unique_lock<std::mutex> _lock{pwr_meter_mib_instance->rt_phase_measures_table_map_lock};
61  auto i = pwr_meter_mib_instance->rt_phase_measures_table_map.find( keyval );
62  if ( pwr_meter_mib_instance->rt_phase_measures_table_map.find( keyval ) !=
63  pwr_meter_mib_instance->rt_phase_measures_table_map.end() )
64  {
65  SYSLOG_DEBUG( 6, "Found existing pwrMeterRtPhaseMeasuresTable entry for device %u phase %u", id, phase_num );
66  had_existing_row = true;
67  if ( px_device.change_signature == i->second.dev_change_sig )
68  {
69  SYSLOG_DEBUG( 6, "Channel membership unchanged, nothing to do" );
70  return ( false );
71  }
72  }
73  else
74  {
75  SYSLOG_DEBUG( 6, "New (to pwrMeterRtPhaseMeasuresTable) device %u phase %u", id, phase_num );
76  }
77  } // end lock scope
78 
79  CPwrMeterRtPhaseMeasuresTableEntry this_row( id, settings, phase_num );
80  this_row.dev_change_sig = px_device.change_signature;
81  // set the Phase enum
82  px::proto::IndexTypeEnums idx_type = px::proto::INDEX_PH_1;
83  if ( phase_num == 2 )
84  {
85  idx_type = px::proto::INDEX_PH_2;
86  }
87  else if ( phase_num == 3 )
88  {
89  idx_type = px::proto::INDEX_PH_3;
90  }
91 
92  bool success;
93  const char *base_oid;
94  const ::px::proto::MibTable *mib_table;
95  std::tie( success, base_oid, mib_table ) =
96  CMmedsNode::lookup_mmeds_table( pwr_meter_mib_instance->mmeds,
97  this_row.table_name() );
98  if ( success ) // There's a mmeds for this table
99  {
100  std::string table_oid( base_oid ); // get the base OID first
101  table_oid += mib_table->text_oid();
102  SYSLOG_DEBUG( 6, "pwrMeterRtPhaseMeasuresTableEntry Processing table %s nodes for Device %u phase %u, oid: %s", mib_table->name().c_str(),
103  id, phase_num, table_oid.c_str() );
104 
105  for ( int i = 0; i < mib_table->mib_columns_size(); i++ )
106  {
107  const ::px::proto::MibObject &column_data = mib_table->mib_columns( i );
108  /* Screen out unimplemented nodes, like inaccessible Indices.
109  * Do this by skipping nodes that don't have an ASN.1 type.
110  */
111  if ( !column_data.has_asn1_type() )
112  {
113  continue;
114  }
115 
116  /* Screen out the other phases */
117  if ( column_data.secondary_index() != idx_type )
118  {
119  continue;
120  }
121 
122  SYSLOG_DEBUG( 9, "Checking node %s", column_data.name().c_str() );
123 
124  std::shared_ptr<CMmedsNode> node_ptr =
125  CMmedsNode::make_mmeds_node( pwr_meter_mib_instance->cds_handler, px_device,
126  column_data, NULL );
127  if ( node_ptr && ( node_ptr->_column_oid > 0 ) )
128  {
129  // Then this one is valid; add it to our map
130  // For this MIB, we are also interested in rescaling for units; apply that here
131  for ( const px::CChannelSettings &chsttngs : settings )
132  {
133  if ( chsttngs.id() == node_ptr->_channel_id )
134  {
135  node_ptr->set_units_scalar( chsttngs.get_units_scaling() );
136  break;
137  }
138  }
139  // Move the CMmedsNode data out of the shared_ptr into the map
140  long column_oid = node_ptr->_column_oid;
141  bool success;
142  std::tie( std::ignore, success ) = this_row._row_map.insert(
143  std::make_pair( column_oid, std::move( *node_ptr.get() ) ) );
144  if ( success )
145  {
146  SYSLOG_DEBUG( 6, "Successfully added %s as column %lu to our row map of %zu nodes",
147  column_data.name().c_str(), column_oid, this_row._row_map.size() );
148  }
149  }
150  } // end loop over columns
151  } // end if (have a mmeds)
152 
153  {
154  std::unique_lock<std::mutex> _lock{pwr_meter_mib_instance->rt_phase_measures_table_map_lock};
155  if ( had_existing_row )
156  {
157  pwr_meter_mib_instance->rt_phase_measures_table_map.erase( keyval );
158  }
159  if ( this_row._row_map.empty() )
160  {
161  SYSLOG_DEBUG( 6, "No columns, so no pwrMeterRtPhaseMeasuresTable entry for Device %u ph %u", id, phase_num );
162  return ( false );
163  }
164  else
165  {
166  // Move the row into our table
167  bool emplace_result;
168  std::tie( std::ignore, emplace_result ) = pwr_meter_mib_instance->rt_phase_measures_table_map.emplace(
169  std::make_pair( keyval, std::move( this_row ) ) );
170  SYSLOG_DEBUG( 8, "Inserted pwrMeterRtPhaseMeasuresTable entry at key value %" PRIu64 ", dev %u ph %u, %s", keyval, id, phase_num,
171  emplace_result ? "succeeded" : "failed" );
172  return ( emplace_result );
173  }
174  } // end lock scope
