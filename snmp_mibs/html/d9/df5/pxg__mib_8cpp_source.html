74 void CPxgMib::init_services( std::shared_ptr<CSnmpCdsHandler> cds_handler )
75 {
76  PUSH_FUNC;
77 
78  _cds_handler = cds_handler;
79  _doing_discovery = true;
80  if ( likely( _cds_handler ) )
81  {
82  _alarm_cb_hdl = _cds_handler->add_alarm_callback( std::bind( &CPxgMib::process_alarm_update, this, std::placeholders::_1 ) );
83  // Can't use do_alarm_discovery - we only want open ones
84  px::cds::CGetAlarms ga;
85  ga.active_only = true;
86  ga.only_latest_update = true;
87  SYSLOG_DEBUG( 8, "Fetching open alarms" );
88  std::shared_ptr<px::cds::CGetAlarms> rv =
89  std::dynamic_pointer_cast<px::cds::CGetAlarms>( _cds_handler->submit_wait( ga ) );
90  if ( likely( rv && ( rv->status == px::cds::RequestStatus::SUCCEEDED ) ) )
91  {
92  SYSLOG_DEBUG( 8, "Processing open alarms" );
93  for ( auto &u : rv->alarms )
94  {
95  process_alarm_update( u );
96  }
97  }
98  else
99  {
100  SYSLOG_ERR( 0, "Open alarm discovery failed, PXG MIB alarm table may have missing entries" );
101  }
102  }
103  else
104  {
105  SYSLOG_FERR( 0, "Init without CDS handler pointer, PXG MIB won't work" );
106  }
107  _doing_discovery = false;
108  _closed_alarms.clear();
109 }

174 void CPxgMib::process_alarm_updates()
175 {
176  PUSH_FUNC;
177  SYSLOG_DEBUG( 8, "Processing queued alarm updates" );
178 
179  while ( 1 )
180  {
181  px::CAlarmUpdate update;
182  {
183  std::unique_lock<std::mutex> _list_lock{_pending_alarm_updates_mutex};
184  if ( _pending_alarm_updates.empty() )
185  {
186  return;
187  }
188  update = _pending_alarm_updates.front();
189  _pending_alarm_updates.pop_front();
190  } // end lock on pending updates
191 
192  // If startup discovery is going on, we might see a closing update
193  // via subscription BEFORE seeing a prior update. Ignore if that
194  // has happened.
195  if ( unlikely( _doing_discovery && ( _closed_alarms.count( update.id ) != 0 ) ) )
196  {
197  SYSLOG_DEBUG( 8, "Ignoring post-closure update on alarm %" PRIu64,
198  update.id );
199  continue;
200  }
201  px::string device_name_field;
202  {
203  std::unique_lock<std::mutex> _lock{_open_alarms_mutex};
204  auto i = _open_alarms.find( update.id );
205  if ( i == _open_alarms.end() )
206  {
207  SYSLOG_DEBUG( 8, "Update: new (to us) alarm ID %" PRIu64,
208  update.id );
209  // Get device and channel names
210  px::string dev_name, chan_name;
211  if ( update.device_id != PX_ID_UNSET )
212  {
213  try
214  {
215  px::CDevice dev = _cds_handler->get_device( update.device_id );
216  if ( !dev.name.empty() )
217  {
218  dev_name = dev.name;
219  device_name_field = dev.name;
220  }
221  else if ( !dev.model.empty() )
222  {
223  dev_name = dev.model;
224  }
225  else if ( !dev.canonical_model_name.empty() )
226  {
227  dev_name = dev.canonical_model_name;
228  }
229  if ( !dev.network_address.empty() )
230  {
231  dev_name += "@" + dev.network_address;
232  }
233  }
234  catch ( px::generic_err msg )
235  {
236  SYSLOG_DEBUG( 5,
237  "Alarm ID %" PRIu64 " on unknown device %" PRIu64 ", ignoring",
238  update.id, update.device_id );
239  continue; // go back for more, then
240  }
241  }
242  if ( update.channel_id != PX_ID_UNSET )
243  {
244  px::cds::CGetChannelSettings gcs;
245  std::shared_ptr<px::cds::CGetChannelSettings> rv;
246  gcs.ids.push_back( update.channel_id );
247  rv = std::dynamic_pointer_cast<px::cds::CGetChannelSettings>( _cds_handler->submit_wait( gcs ) );
248  if ( rv && ( !rv->channels.empty() ) )
249  {
250  // Get the best name for this channel. No length limit.
251  chan_name = rv->channels[0].get_best_name( 0 );
252  }
253  }
254  SYSLOG_DEBUG( 6, "Alarm update for Device name %s, channel name %s", dev_name.c_str(),
255  chan_name.c_str() );
256  CActiveAlarmsTableEntry ent( update, dev_name, chan_name );
257  if ( !device_name_field.empty() )
258  {
259  ent.dev_name = device_name_field; // for traps
260  }
261  ent.dev_id = update.device_id;
262 
263  // update the alarm table cleared state if necessary
264  if ( update.trigger_condition_cleared )
265  {
266  if ( ( ent.level == alarmLevel::critical ) ||
267  ( ent.level == alarmLevel::cautionary ) )
268  {
269  ent.level = alarmLevel::cleared;
270  }
271  }
272  // add alarm to dev/alarm table
273  _open_alarms.insert( std::make_pair( update.id, ent ) );
274  send_trap( update.id, ent );
275  }
276  else
277  {
278  SYSLOG_DEBUG( 8, "Update on known alarm ID %" PRIu64, update.id );
279  i->second.update( update );
280 
281  // update the alarm table cleared state if necessary
282  if ( update.trigger_condition_cleared )
283  {
284  if ( ( i->second.level == alarmLevel::critical ) ||
285  ( i->second.level == alarmLevel::cautionary ) )
286  {
287  i->second.level = alarmLevel::cleared;
288  }
289  }
290  send_trap( update.id, i->second );
291  }
292  if ( update.closed )
293  {
294  SYSLOG_DEBUG( 8, "Update: alarm %" PRIu64 " closed", update.id );
295  // remove alarm in dev/alarm table
296  _open_alarms.erase( update.id );
297  if ( unlikely( _doing_discovery ) )
298  {
299  _closed_alarms.insert( update.id );
300  }
301  }
302  } // end _open_alarms lock scope
303  } // end loop
304 }

306 void CPxgMib::send_trap( px::id_t alarm_id, CActiveAlarmsTableEntry ent )
307 {
308  oid trap_oid[] =
309  { 1, 3, 6, 1, 4, 1, 534, 8, 1, 0, 7 }; // powerChainCriticalAlarm
310  oid alarmID_oid[] =
311  { 1, 3, 6, 1, 4, 1, 534, 8, 1, 1, 2, 2, 1, 1, 0 /*index*/};
312  oid alarmSequenceIndex_oid[] =
313  { 1, 3, 6, 1, 4, 1, 534, 8, 1, 1, 2, 2, 1, 2,
314  0 /* insert index here */ };
315  oid alarmDescription_oid[] =
316  { 1, 3, 6, 1, 4, 1, 534, 8, 1, 1, 2, 2, 1, 3,
317  0 /* insert index here */ };
318  oid alarmValue_oid[] =
319  { 1, 3, 6, 1, 2, 1, 16, 3, 1, 1, 5, 0 /* insert index here */ };
320  oid entPhysicalName_oid[] =
321  { 1, 3, 6, 1, 2, 1, 47, 1, 1, 1, 1, 7, 0 /* insert index here */ };
322 
323  if ( _doing_discovery )
324  {
325  SYSLOG_DEBUG( 8, "Skipping trap send since we're processing initial alarm discovery" );
326  return;
327  }
328  auto i = _open_alarms.find( alarm_id );
329 
330  switch ( ent.level )
331  {
332  case alarmLevel::critical:
333  SYSLOG_DEBUG( 5, "Constructing powerChainCriticalAlarm trap" );
334  trap_oid[10] = 7;
335  break;
336  case alarmLevel::cautionary:
337  SYSLOG_DEBUG( 5, "Constructing powerChainCautionaryAlarm trap" );
338  trap_oid[10] = 8;
339  break;
340  case alarmLevel::acknowledged:
341  SYSLOG_DEBUG( 5, "Constructing powerChainAlarmAcknowledged trap" );
342  trap_oid[10] = 9;
343  if ( i != _open_alarms.end() )
344  {
345  i->second.acknowledge_trap_sent = true;
346  }
347  break;
348  case alarmLevel::cleared:
349  SYSLOG_DEBUG( 5, "Constructing powerChainAlarmCleared trap" );
350  trap_oid[10] = 10;
351  if ( i != _open_alarms.end() )
352  {
353  i->second.clear_trap_sent = true;
354  }
355  break;
356  case alarmLevel::closed:
357  // When an alarm is closed, we need to send either the cleared or
358  // acknowledged trap just before the closed trap. BE SURE TO NOT
359  // LEAVE ent.level = alarmLevel::closed BEFORE THE RECURSIVE CALL
360  // OR THERE WILL BE INFINITE RECURSION!!!
361  if ( ent.clear_trap_sent )
362  {
363  ent.level = alarmLevel::acknowledged;
364  }
365  else
366  {
367  ent.level = alarmLevel::cleared;
368  }
369  send_trap( alarm_id, ent );
370  ent.level = alarmLevel::closed;
371  SYSLOG_DEBUG( 5, "Constructing powerChainAlarmClosed trap" );
372  trap_oid[10] = 11;
373  break;
374  default:
375  SYSLOG_DEBUG( 1, "Ignoring trap for un-handled alarmLevel" );
376  return;
377  }
378  netsnmp_variable_list *var_list = NULL;
379  snmp_varlist_add_variable( &var_list,
380  snmptrap_oid, snmptrap_oid_len,
381  ASN_OBJECT_ID,
382  trap_oid, sizeof( trap_oid ) );
383 
384  // Add trap objects
385  alarmID_oid[14] = alarm_id;
386  alarmSequenceIndex_oid[14] = alarm_id;
387  alarmDescription_oid[14] = alarm_id;
388  alarmValue_oid[11] = alarm_id;
389  entPhysicalName_oid[12] = ent.dev_id;
390 
391  snmp_varlist_add_variable( &var_list,
392  alarmID_oid, OID_LENGTH( alarmID_oid ),
393  ASN_INTEGER, &alarm_id, sizeof( alarm_id ) );
394 
395  snmp_varlist_add_variable( &var_list,
396  alarmSequenceIndex_oid,
397  OID_LENGTH( alarmSequenceIndex_oid ),
398  ASN_INTEGER, &ent.seq_index,
399  sizeof( ent.seq_index ) );
400 
401  snmp_varlist_add_variable( &var_list,
402  alarmDescription_oid,
403  OID_LENGTH( alarmDescription_oid ),
404  ASN_OCTET_STR, ent.description.c_str(),
405  ent.description.size() );
406 
407  snmp_varlist_add_variable( &var_list,
408  alarmValue_oid,
409  OID_LENGTH( alarmValue_oid ),
410  ASN_OCTET_STR, ent.value.c_str(),
411  ent.value.size() );
412 
413  snmp_varlist_add_variable( &var_list,
414  entPhysicalName_oid,
415  OID_LENGTH( entPhysicalName_oid ),
416  ASN_OCTET_STR, ent.dev_name.c_str(),
417  ent.dev_name.size() );
418 
419  send_v2trap( var_list );
420  snmp_free_varbind( var_list );
421 }
