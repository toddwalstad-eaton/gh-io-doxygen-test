42 
43  if ( unlikely( !pwr_meter_mib_instance ) )

52  auto i = pwr_meter_mib_instance->energy_measures_table_map.find( id );
53  if ( pwr_meter_mib_instance->energy_measures_table_map.find( id ) !=
54  pwr_meter_mib_instance->energy_measures_table_map.end() )
55  {
56  SYSLOG_DEBUG( 6, "Found existing pwrMeterEnergyMeasuresTable entry for device %u", id );
57  had_existing_row = true;
58  if ( px_device.change_signature == i->second.dev_change_sig )
59  {
60  SYSLOG_DEBUG( 6, "Channel membership unchanged, nothing to do" );
61  return ( false );
62  }
63  }
64  else
65  {
66  SYSLOG_DEBUG( 6, "New (to pwrMeterEnergyMeasuresTable) device %u", id );
67  }
68  } // end lock scope
69 
70  CPwrMeterEnergyMeasuresTableEntry this_row( id, settings );
71  this_row.dev_change_sig = px_device.change_signature;
72 
73  bool success;
74  const char *base_oid;
75  const ::px::proto::MibTable *mib_table;
76  std::tie( success, base_oid, mib_table ) =
77  CMmedsNode::lookup_mmeds_table( pwr_meter_mib_instance->mmeds,
78  this_row.table_name() );
79  if ( success ) // There's a mmeds for this table
80  {
81  std::string table_oid( base_oid ); // get the base OID first
82  table_oid += mib_table->text_oid();
83  SYSLOG_DEBUG( 6, "pwrMeterEnergyMeasuresTableEntry Processing table %s nodes for Device %u, oid: %s", mib_table->name().c_str(),
84  id, table_oid.c_str() );
85 
86  for ( int i = 0; i < mib_table->mib_columns_size(); i++ )
87  {
88  const ::px::proto::MibObject &column_data = mib_table->mib_columns( i );
89  /* Screen out unimplemented nodes, like inaccessible Indices.
90  * Do this by skipping nodes that don't have an ASN.1 type.
91  */
92  if ( !column_data.has_asn1_type() )
93  {
94  continue;
95  }
96  SYSLOG_DEBUG( 9, "Checking node %s", column_data.name().c_str() );
97 
98  std::shared_ptr<CMmedsNode> node_ptr =
99  CMmedsNode::make_mmeds_node( pwr_meter_mib_instance->cds_handler, px_device,
100  column_data, nullptr );
101  if ( node_ptr && ( node_ptr->_column_oid > 0 ) )
102  {
103  // Then this one is valid; add it to our map
104  // For this MIB, we are also interested in rescaling for units; apply that here
105  for ( const px::CChannelSettings &chsttngs : settings )
106  {
107  if ( chsttngs.id() == node_ptr->_channel_id )
108  {
109  node_ptr->set_units_scalar( chsttngs.get_units_scaling() );
110  break;
111  }
112  }
113  // Move the CMmedsNode data out of the shared_ptr into the map
114  long column_oid = node_ptr->_column_oid;
115  bool success;
116  std::tie( std::ignore, success ) = this_row._row_map.insert(
117  std::make_pair( column_oid, std::move( *node_ptr.get() ) ) );
118  if ( success )
119  {
120  SYSLOG_DEBUG( 6, "Successfully added %s as column %lu to our row map of %zu nodes",
121  column_data.name().c_str(), column_oid, this_row._row_map.size() );
122  }
123  }
124  } // end loop over columns
125  } // end if (have a mmeds)
126 
127  {
128  std::unique_lock<std::mutex> _lock{pwr_meter_mib_instance->energy_measures_table_map_lock};
129  if ( had_existing_row )
130  {
131  pwr_meter_mib_instance->energy_measures_table_map.erase( id );
132  }
133  if ( this_row._row_map.empty() )
134  {
135  SYSLOG_DEBUG( 6, "No columns, so no pwrMeterEnergyMeasuresTable entry for Device %u", id );
136  return ( false );
137  }
138  else
139  {
140  // Move the row into our table
141  // But if you enable the ctor/xtor debug messages, you'll see two copy constructors called here,
142  // so we're failing to do a good clean move.
143  // And the settings are still carried through, for now (during initialization)
144  bool emplace_result;
145  std::tie( std::ignore, emplace_result ) = pwr_meter_mib_instance->energy_measures_table_map.emplace(
146  std::make_pair( id, std::move( this_row ) ) );
147  return ( emplace_result );
148  }
149  } // end lock scope
