20 /* Factory function which checks to see if we need to add any pcdSensorTable entries
21  * for the given Device. */
22 bool CPcdSensorTableEntry::check_add_table_entry( std::shared_ptr<CPcdMib> pcd_mib_instance,
23  const px::CDevice &px_device,
24  const std::vector<px::CChannelSettings> &settings )
25 {
26  PUSH_FUNC;
27 
28  uint32_t idev_id = ( uint32_t ) px_device.id();
29 
30  bool success = true;
31  bool return_result = false;
32 
33  // For the pcdSensorTable:
34  const char *base_oid;
35  const ::px::proto::MibTable *mib_table;
36  std::tie( success, base_oid, mib_table ) =
37  CMmedsNode::lookup_mmeds_table( pcd_mib_instance->_mmeds, CPcdSensorTableEntry::stable_name() );
38 
39  if ( likely( success ) )
40  {
41  /* The pcdSensorTable is different: it has one set of nodes for all possible entries.
42  * But determining which are true entries is based on the mcl_names of the first (index) node.
43  * But! Some of them are wild-carded, so any matching Channel up to the wildcard is a match.
44  */
45  const ::px::proto::MibObject &index_col_data = mib_table->mib_columns( 0 );
46  assert( index_col_data.secondary_index() == px::proto::IndexTypeEnums::INDEX_NEXT );
47  assert( index_col_data.mcl_names().size() > 2 ); // not empty
48 
49  std::vector<px::CChannelSettings> matches =
50  CSnmpCdsHandler::find_matching_channels( settings, index_col_data.mcl_names() );
51 
52  if ( matches.empty() )
53  {
54  SYSLOG_DEBUG( 6, "No pcdSensorTable entries for Device %u", idev_id );
55  }
56  // Check to see if we have already processed this device's Sensors - and if they are unchanged
57  else
58  {
59  std::unique_lock<std::mutex> _lock{pcd_mib_instance->_pcd_sensor_table_map_lock};
60  bool had_existing_row = false;
61  size_t same_cs_count = 0;
62  // Make Coverity happy, limit to uint8_t range:
63  uint8_t count_matches = ( uint8_t ) std::min( matches.size(), ( size_t ) 255 );
64  for ( uint8_t i = 1; i <= count_matches; i++ )
65  {
66  uint64_t keyval = CMmedsNodeTableEntry::compose_row_key( idev_id, i );
67  auto pos = pcd_mib_instance->_pcd_sensor_table_map.find( keyval );
68  if ( pos != pcd_mib_instance->_pcd_sensor_table_map.end() )
69  {
70  had_existing_row = true;
71  if ( pos->second._dev_change_sig == px_device.change_signature )
72  {
73  same_cs_count++;
74  }
75  }
76  }
77  if ( had_existing_row )
78  {
79  if ( same_cs_count == matches.size() )
80  {
81  SYSLOG_DEBUG( 6, "Found %zu existing pcdSensorTable entries for Device %u", same_cs_count, idev_id );
82  return ( success ); // This is the other exit point
83  }
84  else
85  {
86  // Something has changed; we need to remove all the existing nodes for this Device
87  // and redo them all
88  SYSLOG_DEBUG( 5, "Erasing existing pcdSensorTable entries for Device %u and redoing them", idev_id );
89  /* Coverity pointed out that we could get an infinite loop if matches.size() > 255;
90  * we don't believe that can happen in practice, but this keeps the code correct: */
91  uint8_t matches_size = std::min<uint8_t>( matches.size(), 255 );
92  for ( uint8_t i = 1; i <= matches_size; i++ )
93  {
94  uint64_t keyval = CMmedsNodeTableEntry::compose_row_key( idev_id, i );
95  pcd_mib_instance->_pcd_sensor_table_map.erase( keyval );
96  }
97  }
98  }
99  }
100 
101  /* The running index/counter which holds the value to use for the "next" pcdSensor row to be added.
102  * Starting at 1 (for each Device), will be incremented when a pcdSensor row is added.
103  */
104  uint8_t next_sensors_index = 1;
105 
106  // Loop through the set of matching Channels and create a row for each
107  for ( const px::CChannelSettings &match_ch : matches )
108  {
109  uint64_t keyval = CMmedsNodeTableEntry::compose_row_key( idev_id, next_sensors_index );
110  CPcdSensorTableEntry this_entry( px_device.id(), next_sensors_index, match_ch );
111  this_entry._dev_change_sig = px_device.change_signature;
112 
113  std::string table_oid( base_oid ); // get the base OID first
114  table_oid += mib_table->text_oid();
115  SYSLOG_DEBUG( 6, "Processing table %s nodes for Device %u, sec index %u, oid: %s",
116  mib_table->name().c_str(), idev_id, next_sensors_index, table_oid.c_str() );
117 
118  /* This table does need a "special" node validator, so bind the call to our member function.
119  * Note that "this_entry" is passed by reference to make this work (without '&' this_entry
120  * argument is copy-constructed into the bind() but is then unrelated to _this_ entry).
121  */
122  using std::placeholders::_1;
123  using std::placeholders::_2;
124  CMmedsNode::validate_special_node_t f_validate_special =
125  std::bind( &CPcdSensorTableEntry::validate_special_nodes, &this_entry, _1, _2 );
126 
127  // Now skip the first (ie, index) node and process each of the others
128  for ( int i = 1; i < mib_table->mib_columns_size(); i++ )
129  {
130  const ::px::proto::MibObject &column_data = mib_table->mib_columns( i );
131 
132  std::shared_ptr<CMmedsNode> node_ptr =
133  CMmedsNode::make_mmeds_node( pcd_mib_instance->_cds_handler, px_device,
134  column_data, f_validate_special, next_sensors_index );
135  if ( node_ptr && ( node_ptr->_column_oid > 0 ) )
136  {
137  // Then this one is valid; add it to our map
138  // Move the CMmedsNode data out of the shared_ptr into the map
139  long column_oid = node_ptr->_column_oid;
140  bool success;
141  std::tie( std::ignore, success ) = this_entry._row_map.insert(
142  std::make_pair( column_oid, std::move( *node_ptr.get() ) ) );
143 
144  if ( success )
145  {
146  SYSLOG_DEBUG( 6, "Successfully added %s as column %lu to our row map of %zu nodes",
147  column_data.name().c_str(), column_oid, this_entry._row_map.size() );
148  }
149  }
150  }
151 
152  // Move the row into our table
153  bool result;
154  {
155  std::unique_lock<std::mutex> _lock{pcd_mib_instance->_pcd_sensor_table_map_lock};
156  std::tie( std::ignore, result ) = pcd_mib_instance->_pcd_sensor_table_map.emplace(
157  std::make_pair( keyval, std::move( this_entry ) ) );
158  SYSLOG_DEBUG( 6, "Insert pcdSensorTable entry at key value %" PRIu64 ", dev %u index %u, %s",
159  keyval, idev_id, next_sensors_index, result ? "succeeded" : "failed" );
160  }
161  if ( result )
162  {
163  return_result = true;
164  }
165 
166  // Increment the Index for the "next" possible entry
167  next_sensors_index++;
168  }
169  }
170 
171  return ( return_result );
172 }

174 // Table-specific function to validate a "special" node, when this is its Translation Type.
175 bool CPcdSensorTableEntry::validate_special_nodes( CMmedsNode &node, const px::proto::MibObject &mmeds )
176 {
177  PUSH_FUNC;
178  bool success = true;
179 
180  /* We will use the stored Channel Settings for the Channel that this row is based on.
181  * All columns here will pass / are valid.
182  */
183  switch ( node._column_oid )
184  {
185  case COLUMN_PCDSENSORNAME:
186  node._const_text = _ch_setting.name;
187  // Set the _translation_type now to "constant value"
188  // so we don't later try to re-evaluate it as a "special"
189  node._translation_type = px::proto::XLT_CONSTANT;
190  break;
191  case COLUMN_PCDSENSORREADING:
192  // No default value; retrieve value at run-time from its Channel ID
193  node._channel_id = _ch_setting.id();
194  // Set the _translation_type now to "none" so we get it directly and
195  // so we don't later try to re-evaluate it as a "special"
196  node._translation_type = px::proto::XLT_NONE;
197  break;
198  case COLUMN_PCDSENSORUNITS:
199  if ( !_ch_setting.units.empty() )
200  {
201  node._const_text = _ch_setting.units;
202  }
203  else if ( mmeds.failover_action() == px::proto::FAIL_DFLT_STR )
204  {
205  node._const_text = mmeds.failover_default().c_str();
206  }
207  // Else (unlikely): if not set by anything, is just an empty string
208  node._translation_type = px::proto::XLT_CONSTANT;
209  break;
210  default:
211  success = false;
212  break;
213  }
214 
215  return ( success );
216 }
