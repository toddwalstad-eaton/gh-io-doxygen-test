75 // Updates our map whenever CCdsHandler updates the state data for a device in its map.
76 void CEntityStateMib::device_state( const px::asio::CCdsDevice &container )
77 {
78  PUSH_FUNC;
79 
80  px::id_t dev_id = container._px_device.id();
81  char text[120];
82  px::CBitset<px::realtime_state> last_state, rt_state;
83  bool last_flag, rt_flag;
84 
85  rt_state = container._px_device.get_state();
86  rt_flag = rt_state.get( px::realtime_state::disabled ) || rt_state.get( px::realtime_state::not_communicating );
87 
88  // Just to be sure that we've gotten the change, copy the container over to our MIB table
89  auto map_pos = _ent_state_table_map.find( dev_id );
90  if ( map_pos != _ent_state_table_map.end() )
91  {
92  last_state = map_pos->second.last_state;
93  last_flag = last_state.get( px::realtime_state::disabled ) || last_state.get( px::realtime_state::not_communicating );
94 
95  map_pos->second.update_px_device( container._px_device );
96 
97  snprintf( text, 120, "device_state_callback: New state %02X for device %s with ID = %" PRIu64 " \n",
98  rt_state.as_uint32(), container._px_device.name.c_str(), dev_id );
99  text[119] = 0;
100  SYSLOG_INFO( 5, "%s", text );
101  }
102  else
103  {
104  // set flag to trigger trap
105  last_flag = !rt_flag;
106  // Re-Create the map entry for this device (same as Add). Give it the pointer to "this".
107  _ent_state_table_map.emplace( std::make_pair( dev_id, CEntStateTableEntry( container ) ) );
108 
109  snprintf( text, 120, "device_state_callback: state %02X for ADDED missing device %s with ID = %" PRIu64 " \n",
110  rt_state.as_uint32(), container._px_device.name.c_str(), dev_id );
111  text[119] = 0;
112  SYSLOG_NOTICE( 2, "%s", text );
113  }
114 
115  DEBUGMSGTL( ( "entityStateMIB", "%s", text ) );
116 
117  //if (rt_state.get(px::realtime_state::disabled) != cur_state.get(px::realtime_state::disabled)
118  // || rt_state.get(px::realtime_state::not_communicating) != cur_state.get(px::realtime_state::not_communicating) )
119  if ( rt_flag != last_flag )
120  {
121  if ( rt_flag )
122  {
123  send_entStateOperDisabled_trap( dev_id );
124  }
125  else
126  {
127  send_entStateOperEnabled_trap( dev_id );
128  }
129  }

144 void CEntityStateMib::got_alarm_update( const px::CAlarmUpdate &upd )
145 {
146  PUSH_FUNC;
147 
148  SYSLOG_DEBUG( 8, "Ent State MIB got alarm update, global count %" PRIu64,
149  upd.global_count );
150  if ( upd.closed )
151  {
152  device_alarm_state( upd.device_id, upd.id, ENTITYALARMSTATUS_NONE );
153  return;
154  }
155  if ( upd.ack_level >= upd.max_reached_trigger_level )
156  {
157  device_alarm_state( upd.device_id, upd.id, ENTITYALARMSTATUS_UNDERREPAIR_FLAG );
158  return;
159  }
160  if ( upd.trigger_condition_cleared )
161  {
162  device_alarm_state( upd.device_id, upd.id, ENTITYALARMSTATUS_UNDERREPAIR_FLAG );
163  return;
164  }
165  if ( upd.trigger && ( upd.trigger->priority < 5000 ) )
166  {
167  device_alarm_state( upd.device_id, upd.id, ENTITYALARMSTATUS_CRITICAL_FLAG );
168  return;
169  }
170  device_alarm_state( upd.device_id, upd.id, ENTITYALARMSTATUS_WARNING_FLAG );
171 }
