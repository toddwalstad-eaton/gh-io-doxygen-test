24 /* Factory function which checks to see if we need to add a pcdDigitalOutputsTable entries
25  * for the given Device.
26  * If we already have entries for this Device, does nothing.
27  * Else will process the possible pcdDigitalOutputsTable columns ("nodes") for this Device
28  * to see if we have any valid columns. If yes, will add this entry to the table.
29  */
30 bool CPcdDigitalOutputsTableEntry::check_add_table_entry( std::shared_ptr<CPcdMib> pcd_mib_instance,
31  const px::CDevice &px_device,
32  const std::vector<px::CChannelSettings> &settings )
33 {
34  PUSH_FUNC;
35 
36  uint32_t idev_id = ( uint32_t ) px_device.id();
37 
38  uint8_t output_index = 0;
39 
40  bool success = true;
41  bool return_result = false;
42  // Limit output_index so we don't loop forever checking
43  while ( success && ( output_index <= CPcdDigitalOutputsTableEntry::MAX_DIGITAL_OUTPUT_INDEX ) )
44  {
45  uint64_t keyval = CMmedsNodeTableEntry::compose_row_key( idev_id, output_index );
46  {
47  bool had_existing_row = false;
48  bool same_sig = false;
49 
50  std::unique_lock<std::mutex> _lock{pcd_mib_instance->_pcd_digital_outputs_table_map_lock};
51  auto pos = pcd_mib_instance->_pcd_digital_outputs_table_map.find( keyval );
52  if ( pos != pcd_mib_instance->_pcd_digital_outputs_table_map.end() )
53  {
54  had_existing_row = true;
55  if ( pos->second._dev_change_sig == px_device.change_signature )
56  {
57  same_sig = true;
58  }
59  }
60  SYSLOG_DEBUG( 6, "Found existing pcdDigitalOutputsTable entry for Device %u, row %u",
61  idev_id, output_index );
62 
63  if ( had_existing_row )
64  {
65  if ( same_sig )
66  {
67  SYSLOG_DEBUG( 6, "Found existing pcdDigitalOutputsTable entry for Device %u", idev_id );
68  return ( success ); // This is the other exit point
69  }
70  else
71  {
72  // Something has changed; we need to remove all the existing nodes for this Device
73  // and redo them all
74  SYSLOG_DEBUG( 5, "Erasing existing pcdDigitalOutputsTable entry for Device %u and "
75  "reconstructing it", idev_id );
76  pcd_mib_instance->_pcd_digital_outputs_table_map.erase( keyval );
77  }
78  }
79  }
80 
81  CPcdDigitalOutputsTableEntry this_entry( px_device.id(), output_index,
82  settings );
83  this_entry._dev_change_sig = px_device.change_signature;
84 
85  // For the pcdDigitalOutputsTable:
86  const char *base_oid;
87  const ::px::proto::MibTable *mib_table;
88  std::tie( success, base_oid, mib_table ) =
89  CMmedsNode::lookup_mmeds_table( pcd_mib_instance->_mmeds,
90  this_entry.table_name() );
91 
92  if ( success )
93  {
94  // Then we succeeded in the lookup
95  std::string table_oid( base_oid ); // get the base OID first
96  table_oid += mib_table->text_oid();
97  SYSLOG_DEBUG( 6, "Processing table %s nodes for Device %u, sec index %u, oid: %s",
98  mib_table->name().c_str(),
99  idev_id, output_index, table_oid.c_str() );
100 
101  for ( int i = 0; i < mib_table->mib_columns_size(); i++ )
102  {
103  const ::px::proto::MibObject &column_data = mib_table->mib_columns( i );
104  SYSLOG_DEBUG( 9, "Checking node %s", column_data.name().c_str() );
105 
106  std::shared_ptr<CMmedsNode> node_ptr =
107  CMmedsNode::make_mmeds_node( pcd_mib_instance->_cds_handler, px_device,
108  column_data, nullptr, output_index );
109  if ( node_ptr && ( node_ptr->_column_oid > 0 ) )
110  {
111  // Then this one is valid; add it to our map
112  // Move the CMmedsNode data out of the shared_ptr into the map
113  long column_oid = node_ptr->_column_oid;
114  // Set columns with static values
115  switch ( column_oid )
116  {
117  case COLUMN_PCDDIGITALOUTPUTINDEX:
118  node_ptr->_const_int = static_cast<int32_t>( output_index );
119  break;
120  case COLUMN_PCDDIGITALOUTPUTNAME:
121  // Find the settings of the Channel ID matched on
122  for ( auto ch_setting : settings )
123  {
124  if ( node_ptr->_channel_id == ch_setting.id() )
125  {
126  node_ptr->_const_text = ch_setting.canonical_name;
127  }
128  }
129  break;
130  case COLUMN_PCDDIGITALOUTPUTPOSSIBLEVALUES:
131  // Find the settings of the Channel ID matched on
132  for ( auto ch_setting : settings )
133  {
134  if ( node_ptr->_channel_id == ch_setting.id() )
135  {
136  node_ptr->_const_text = ch_setting.multistate_map.semi_separated_labels();
137  }
138  // No multistate map labels - default to 0;1
139  if ( node_ptr->_const_text == "" )
140  {
141  node_ptr->_const_text = "0;1";
142  }
143  }
144  break;
145  case COLUMN_PCDDIGITALOUTPUTPRESENTVALUE:
146  case COLUMN_PCDDIGITALOUTPUTLASTCHANGE:
147  node_ptr->_const_text = "";
148  default:
149  // No defaults; retrieve value at run-time
150  break;
151  }
152 
153  bool success;
154  std::tie( std::ignore, success ) = this_entry._row_map.insert(
155  std::make_pair( column_oid, std::move( *node_ptr.get() ) ) );
156 
157  if ( success )
158  {
159  SYSLOG_DEBUG( 6, "Successfully added %s as column %lu to our row map of %zu nodes",
160  column_data.name().c_str(), column_oid, this_entry._row_map.size() );
161  }
162  }
163  else if ( column_data.failover_action() == ::px::proto::FailoverActionEnums::FAIL_NO_ROW )
164  {
165  // No more columns to check
166  break;
167  }
168  }
169  }
170 
171  if ( this_entry._row_map.empty() )
172  {
173  SYSLOG_DEBUG( 6, "No columns, so no pcdDigitalOutputsTable entry for Device %u, row %u",
174  idev_id, output_index );
175  }
176  else
177  {
178  // Move the row into our table
179  bool result;
180  {
181  std::unique_lock<std::mutex> _lock{pcd_mib_instance->_pcd_digital_outputs_table_map_lock};
182  std::tie( std::ignore, result ) = pcd_mib_instance->_pcd_digital_outputs_table_map.emplace(
183  std::make_pair( keyval, std::move( this_entry ) ) );
184  SYSLOG_DEBUG( 6, "Insert pcdDigitalOutputsTable entry at key value %" PRIu64 ", dev %u index %u, %s",
185  keyval, idev_id, output_index, result ? "succeeded" : "failed" );
186  }
187  if ( result )
188  {
189  return_result = true;
190  }
191  }
192  ++output_index;
193  }
194 
195  return ( return_result );
196 }

211 int CPcdDigitalOutputsTableEntry::map_present_value( px::CChannel ch_data,
212  px::CValue ch_val, netsnmp_request_info *request ) const
213 {
214  PUSH_FUNC;
215 
216  int err_val = SNMP_ERR_NOERROR;
217  px::CMultistateMap ms_map = ch_data.channel_settings->multistate_map;
218 
219  px::string mapped_label = ms_map.get_mapped_label( ch_val );
220  if ( mapped_label != "" )
221  {
222  snmp_set_var_typed_value( request->requestvb, ASN_OCTET_STR,
223  mapped_label.c_str(), mapped_label.size() );
224  }
225  else
226  {
227  // No multistate map labels: use default 0 for false, 1 for true
228  if ( ch_val == false )
229  {
230  snmp_set_var_typed_value( request->requestvb, ASN_OCTET_STR,
231  "0", 1 );
232  }
233  else
234  {
235  snmp_set_var_typed_value( request->requestvb, ASN_OCTET_STR,
236  "1", 1 );
237  }
238  }
239 
240  return ( err_val );
241 }

263  ret_val = true;
264  }
265  // Else, leave invalid
266  }
267 
268  return ( ret_val );
