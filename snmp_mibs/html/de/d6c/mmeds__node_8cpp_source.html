44 // Try to map from the MIB Mapping EDS data for this scalar or column to CDS or constant data.
45 bool CMmedsNode::map_from_mmeds( std::shared_ptr<CSnmpCdsHandler> cds_handler,
46  const px::CDevice &px_device, const px::proto::MibObject &mib_obj,
47  const validate_special_node_t &validate_special_node_func,
48  const uint8_t row_index )
49 {
50  PUSH_FUNC;
51 
52  bool success = false;
53  px::id_t dev_id = px_device.id();
54 
55  /* Set the common elements before we do special processing or the others
56  * For the Column OID, take a field like "text_oid": "1.1.1.4" and
57  * extract just the last position */
58  oid oids[12]; // 12 ought to be plenty
59  size_t oid_len = 12; // array size
60  if ( snmp_parse_oid( mib_obj.text_oid().c_str(), oids, &oid_len ) && ( oid_len > 0 ) )
61  {
62  // Success; we got an oid
63  _dev_id = dev_id;
64  _column_oid = oids[oid_len - 1];
65  _asn1_type = ( u_char ) mib_obj.asn1_type();
66  _translation_type = mib_obj.translation();
67  _is_writable = mib_obj.writable();
68  }
69  else
70  {
71  SYSLOG_NOTICE( 2, "Failed in parsing the oid_text (%s) for %s of Device %u; cannot continue",
72  mib_obj.text_oid().c_str(), mib_obj.name().c_str(), ( uint32_t ) dev_id );
73  return ( success );
74  }
75 
76  // Process by the translation type and see if we have support for this instance
77  switch ( _translation_type )
78  {
79  case px::proto::XLT_NONE:
80  case px::proto::XLT_MULTIPLY: // treated alike in this area
81  case px::proto::XLT_DIVIDE:
82  // See if we have one of the listed MCL channels
83  _channel_id = cds_handler->lookup_mcl_channel( dev_id, mib_obj.mcl_names() );
84  if ( _channel_id != 0 )
85  {
86  _translation_scalar = mib_obj.translation_scalar();
87  success = true;
88  }
89  break;
90 
91  case px::proto::XLT_NUM_ROWS:
92  // See if we have one of the listed MCL channels with wild carded
93  // row number
94  _channel_id = cds_handler->lookup_mcl_channel_by_row( dev_id,
95  row_index, mib_obj.mcl_names() );
96  if ( _channel_id != 0 )
97  {
98  success = true;
99  }
100  break;
101 
102  case px::proto::XLT_SPECIAL:
103  if ( validate_special_node_func )
104  {
105  success = validate_special_node_func( *this, mib_obj );
106  if ( success && ( _channel_id == 0 ) )
107  {
108  // If that didn't provide a channel ID, do so now
109  _channel_id = cds_handler->lookup_mcl_channel( dev_id, mib_obj.mcl_names() );
110  }
111  }
112  else
113  {
114  SYSLOG_NOTICE( 1, "Missing handler for \"special\" node %s", mib_obj.name().c_str() );
115  }
116  break;
117 
118  default:
119  SYSLOG_NOTICE( 1, "No support yet for Translation Type %s",
120  px::proto::TranslationTypeEnums_Name( mib_obj.translation() ).c_str() );
121  break;
122  }
123 
124  // ToDo: look at failover_action here if failed
125 
126  if ( success )
127  {
128  SYSLOG_DEBUG( 5, "Success in creating %s at column %lu from Device %u:%u", mib_obj.name().c_str(),
129  _column_oid, ( uint32_t ) dev_id, ( uint32_t ) _channel_id );
130  }
131  else
132  {
133  // Failure is very common
134  SYSLOG_DEBUG( 8, "Failed to create %s at column %lu for Device %u", mib_obj.name().c_str(),
135  _column_oid, ( uint32_t ) dev_id );
136  }
137 
138  return ( success );
139 }

141 /* Static function to lookup the given table (by its name) in the MIB Mapping EDS data,
142  * and also return its base OID. */
143 std::tuple<bool, const char *, const ::px::proto::MibTable *> CMmedsNode::lookup_mmeds_table(
144  std::shared_ptr<px::proto::MibMapEds> mmeds, const std::string &table_name )
145 {
146  PUSH_FUNC;
147 
148  for ( int i = 0; i < mmeds->mib_groups_size(); i++ )
149  {
150  if ( mmeds->mib_groups( i ).has_table() )
151  {
152  // Pointers are mutables in protobuf structures, even though we'd prefer to keep them const
153  ::px::proto::MibTable *mib_table = mmeds->mutable_mib_groups( i )->mutable_table();
154  if ( mib_table->name() == table_name )
155  {
156  SYSLOG_DEBUG( 6, "Found MIB Mapping for Table %s", table_name.c_str() );
157  /* Make and return a tuple with 3 elements:
158  * 0) A boolean which is true if the lookup was successful (found the table)
159  * 1) A pointer to the base OID (not including the table OID elements)
160  * 2) A pointer to the EDS data for this table */
161  return ( std::make_tuple( true, mmeds->eds_info().root_object_oid().c_str(), mib_table ) );
162  }
163  }
164  }
165 
166  // But if we get here, it's bad news. Return an empty-ish tuple with false first.
167  SYSLOG_WARN( 1, "Failed to find MIB Mapping for Table %s", table_name.c_str() );
168  return ( std::make_tuple( false, nullptr, nullptr ) );
169 }

171 // Provide the appropriate value in the request MIB var structure, using this node's data.
172 int CMmedsNode::fill_mibvar_value( std::shared_ptr<CSnmpCdsHandler> cds_handler,
173  netsnmp_request_info *request, const fill_special_value_t &fill_special_value_func ) const
174 {
175  PUSH_FUNC;
176 
177  int err_val = SNMP_ERR_NOERROR;
178  // If XLT_CONSTANT but type is not string, _translation_scalar is
179  // the source of the constant value we will use, so use it for simple initialization
180  px::CValue cds_value( _translation_scalar );
181 
182  if ( !cds_handler->is_communicating_device( _dev_id ) )
183  {
184  err_val = SNMP_NOSUCHINSTANCE; // No value for an invalid device
185  }
186  // If the Translation Type is "special", call the provided functor
187  else if ( ( _translation_type == px::proto::XLT_SPECIAL ) && fill_special_value_func )
188  {
189  err_val = fill_special_value_func( *this, cds_value );
190  }
191  // For the normal CDS Channel Value case:
192  else if ( _channel_id != 0 )
193  {
194  if ( cds_handler->get_channel_value( _channel_id, cds_value ) &&
195  cds_value.is_valid() )
196  {
197  // Screen out NaN and Infinite values
198  if ( ( ( cds_value.type() == px::datatype::float32 ) || ( cds_value.type() == px::datatype::float64 ) ) &&
199  !std::isfinite( cds_value.as_double() ) )
200  {
201  err_val = SNMP_NOSUCHINSTANCE;
202  }
203  // That worked; now see if we have to scale this. SNMP only works with Ints, not Floats
204  else if ( _translation_type == px::proto::XLT_MULTIPLY )
205  {
206  cds_value = ( cds_value.as_float() * _translation_scalar );
207  }
208  else if ( ( _translation_type == px::proto::XLT_DIVIDE ) && ( _translation_scalar != 0 ) )
209  {
210  cds_value = ( cds_value.as_float() / _translation_scalar );
211  }
212  }
213  else
214  {
215  err_val = SNMP_NOSUCHINSTANCE; // Couldn't get value, or is unset
216  }
217  }
218  // Else is constant data
219  else if ( _translation_type == px::proto::XLT_CONSTANT )
220  {
221  if ( _asn1_type == ASN_OCTET_STR )
222  {
223  cds_value = _const_text;
224  }
225  // Else if integer type, was already set in cds_value from _translation_scalar
226  }
227  else
228  {
229  SYSLOG_WARN( 2, "No means provided to fill node; using %d", _translation_scalar );
230  }
231 
232  if ( err_val == SNMP_ERR_NOERROR )
233  {
234  switch ( _asn1_type )
235  {
236  case ASN_INTEGER:
237  // If we need to scale this for its units (if we have been provided with a _units_scalar), do so
238  if ( _units_scalar )
239  {
240  // Force the cds_value to double to maintain precision for integers of more than 23 bits
241  cds_value = ( cds_value.as_double() * _units_scalar.get() );
242  }
243  // If this is a float value, bump it by 0.5 before converting to improve the round-off accuracy
244  if ( ( cds_value.type() == px::datatype::float32 ) || ( cds_value.type() == px::datatype::float64 ) )
245  {
246  double dval = cds_value.as_double();
247  if ( dval >= 0 )
248  {
249  dval += 0.5;
250  }
251  else
252  {
253  dval -= 0.5;// subtract 0.5 for negative values
254  }
255  cds_value = dval;
256  }
257  snmp_set_var_typed_integer( request->requestvb, ASN_INTEGER, cds_value.as_int32() );
258  break;
259  case ASN_OCTET_STR:
260  snmp_set_var_typed_value( request->requestvb, ASN_OCTET_STR,
261  cds_value.as_string().c_str(), cds_value.as_string().size() );
262  break;
263  default:
264  SYSLOG_ERR( 1, "Need to add support this ASN.1 type %u", _asn1_type );
265  err_val = SNMP_NOSUCHOBJECT;
266  break;
267  }
268  }
269 
270  if ( err_val == SNMP_ERR_NOERROR )
271  {
272  SYSLOG_DEBUG( 8, "Providing CDS value %s for MIB Table column %lu", cds_value.as_string().c_str(), _column_oid );
273  }
274 
275  return ( err_val );
276 }
