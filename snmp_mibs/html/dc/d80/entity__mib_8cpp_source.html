74 void snmp_now_running_callback( unsigned int /* clientreg */, void * /* clientarg */ )
75 {
76  try
77  {
78  // Re-initialize the logging
79  // If snmpd daemonized, we are now in a new thread
80  // and the old _thr_local_syslog pointer will be invalid
81  init_logging();
82 
83  // Disconnect and reconnect to the CDS to ensure a valid connection
84  assert( _this_mib_instance );
85  assert( g_cds_handler );
86  g_cds_handler->reconnect_to_cds();
87 
88  // Set up subscription to alarm updates
89  g_cds_handler->subscribe_alarm_callback();
90 
91  /* While the Entity, Entity State, and PXG (Alarms) MIBs don't care about channel data,
92  * the other MIBs (PCD, Power Meter, UPS, etc) do want channel data; so let's turn this on.
93  */
94  bool want_channels = true;
95  // Bind the calls to our member functions, to use in the call to CdsHandler
96  using std::placeholders::_1;
97  using std::placeholders::_2;
98 
99  // Register our Entity MIB callbacks first. Then use
100  // init_device_map() to bind to the internal muxing callbacks in
101  // CSnmpCdsHandler.
102 
103  g_cds_handler->add_add_device_callback( std::bind( &CEntityMib::add_device_callback,
104  _this_mib_instance, _1 ) );
105  g_cds_handler->add_del_device_callback( std::bind( &CEntityMib::delete_device_callback,
106  _this_mib_instance, _1 ) );
107  g_cds_handler->add_device_state_callback( std::bind( &CEntityMib::device_state_callback,
108  _this_mib_instance, _1 ) );
109 
110 
111  px::asio::CCdsHandler::cds_device_callback_t f_add_device =
112  std::bind( &CSnmpCdsHandler::got_add_device, g_cds_handler, _1 );
113  px::asio::CCdsHandler::cds_device_callback_t f_update_device =
114  std::bind( &CEntityMib::update_device_callback, _this_mib_instance, _1 );
115  px::asio::CCdsHandler::device_delete_callback_t f_delete_device =
116  std::bind( &CSnmpCdsHandler::got_del_device, g_cds_handler, _1 );
117  px::asio::CCdsHandler::cds_device_callback_t f_device_state =
118  std::bind( &CSnmpCdsHandler::got_dev_state, g_cds_handler, _1 );
119  px::asio::CCdsHandler::cds_ch_settings_callback_t f_ch_settings =
120  px::asio::CCdsHandler::cds_ch_settings_callback_t(); // default to empty functor
121  if ( want_channels )
122  {
123  f_ch_settings = std::bind( &CSnmpCdsHandler::channel_settings_callback,
124  g_cds_handler, _1, _2 );
125  }
126  // Start discovering and adding our devices to the MIBs
127  g_cds_handler->init_device_map( f_add_device, f_update_device,
128  f_delete_device, f_device_state, want_channels, f_ch_settings );
129  // Won't return until it has done the scanning of the device tree
130  // Now enter the CDS subscriptions for device changes
131  g_cds_handler->subscribe_device_callbacks();
132  // Look again, to see if any new popped up between our initial
133  // discovery and the subscription. Will automatically skip
134  // any branches of the tree that have not changed
135  g_cds_handler->do_static_discovery( PX_ID_ROOT );
136 
137  // touch /var/run/[mypid].ready
138  size_t LEN = 32;
139  char cmd[LEN];
140  snprintf( cmd, LEN, "touch /var/run/ready/%d", getpid() );
141  cmd[LEN - 1] = '\0';
142  system( cmd );
143  }
144  catch ( px::generic_err &ex )
145  {
146  DEBUGMSGTL( ( "entityMIB", "Failed to reconnect to CDS for updates \n" ) );
147  SYSLOG_WARN( 1, "Failed to reconnect to CDS for updates" );
148  }
149 }

157 DLLEXPORT void init_etn_mibs_base( void )
158 {
159  // bool is_first_time = true;
160  if ( !_this_mib_instance )
161  {
162  DEBUGMSGTL( ( "entityMIB", "Initializing C++ handling \n" ) );
163 
164  // Initialize logging for these initialization steps
165  init_logging();
166 
167  // Register our own signal handlers, esp for stack tracing
168  // This may be dicey, and compete with net-snmp agent's doing the same.
169  // Might need to call "if (verbosity > 1)" or somesuch
170  pcd_register_handlers();
171 
172  g_cds_handler.reset( new CSnmpCdsHandler() ); // We are OK with the default arg values
173 
174  /* Instantiate and open our main class */
175  _this_mib_instance.reset( new CEntityMib() );
176  }
177  else
178  {
179  SYSLOG_INFO( 1, "Reinitializing Entity MIB submodule" );
180  // is_first_time = false;
181  }
182 
183  /* Register and initialize the one scalar in the Entity MIB: */
184  init_entityMIB_scalar( _this_mib_instance );
185  /* Register our table */
186  init_entPhysicalTable( _this_mib_instance );
187 
188  /* Now start the Device tree discovery, unless we already have devices. */
189  if ( _this_mib_instance->_physical_table_map.empty() )
190  {
191  _this_mib_instance->init_services();
192  }
193 
194  // if ( is_first_time )
195  {
196  /* Initialize the class that manages the MIB-II sys group. */
197  _this_mib_instance->_eaton_sys_manager.init_services( g_cds_handler );
198  }
199 
200  /* Now register the callback to take the next steps after snmpd has daemonized
201  * and started normal running. We want a one-shot (non-repeating) alarm; we have
202  * no client data to provide (clientarg*) and don't care about keeping the handle returned.
203  * Set kind of arbitrarily for 2s. */
204  unsigned int when = 2; // 2sec
205  unsigned int flags = 0;
206  snmp_alarm_register( when, flags, &snmp_now_running_callback, nullptr );
207 }

296 void CEntityMib::add_device_callback( const px::asio::CCdsDevice &container )
297 {
298  PUSH_FUNC;
299 
300  /* If the Syslog Debug level is down at default but we want higher, raise it.
301  * This happens because this function is really running on the CDS Client's background thread.
302  */
303  if ( ( SYSLOG_PTR != nullptr ) && ( SYSLOG_PTR->verbosity < ( unsigned int ) eaton_debug_level ) )
304  {
305  SYSLOG_PTR->verbosity = eaton_debug_level;
306  }
307 
308  uint32_t dev_id = container._px_device.id();
309  char text[120];
310  snprintf( text, 120, "add_device_callback: Adding device %s with ID = %u \n", container._px_device.name.c_str(), dev_id );
311  text[119] = 0;
312  SYSLOG_INFO( 5, "%s", text );
313  DEBUGMSGTL( ( "entityMIB", "%s", text ) );
314  // Create the map entry for this device. Give it the pointer to "this".
315  _physical_table_map.emplace( std::make_pair( dev_id, CEntPhysicalTableEntry( container, _this_mib_instance ) ) );
316  // Do likewise for the entityState table map
317  _ent_state_manager->add_device( container );
318 
319  // Update the entLastChangeTime to "now" since this is a table change
320  update_entLastChangeTime();
321

330 //cds_device_callback_t& update_device_callback for device_settings changes
331 void CEntityMib::update_device_callback( const px::asio::CCdsDevice &container )
332 {
333  PUSH_FUNC;
334 
335  /* If the Syslog Debug level is down at default but we want higher, raise it.
336  * This happens because this function is really running on the CDS Client's background thread.
337  */
338  if ( ( SYSLOG_PTR != nullptr ) && ( SYSLOG_PTR->verbosity < ( unsigned int ) eaton_debug_level ) )
339  {
340  SYSLOG_PTR->verbosity = eaton_debug_level;
341  }
342 
343  uint32_t idev_id = ( uint32_t ) container._px_device.id();
344  char text[120];
345  DEBUGMSGTL( ( "entityMIB", "%s", text ) );
346  // Just to be sure that we've gotten the change, copy the container over to our MIB table
347  auto map_pos = _physical_table_map.find( idev_id );
348  if ( map_pos != _physical_table_map.end() )
349  {
350  map_pos->second.update_px_device( container._px_device );
351  snprintf( text, 120, "update_device_callback: Updated device %s with ID = %u \n",
352  container._px_device.name.c_str(), idev_id );
353  text[119] = 0;
354  SYSLOG_INFO( 5, "%s", text );
355  }
356  else
357  {
358  // Re-Create the map entry for this device (same as Add). Give it the pointer to "this".
359  _physical_table_map.emplace( std::make_pair( idev_id, CEntPhysicalTableEntry( container, _this_mib_instance ) ) );
360  snprintf( text, 120, "update_device_callback: ADDED missing device %s with ID = %u \n",
361  container._px_device.name.c_str(), idev_id );
362  text[119] = 0;
363  SYSLOG_NOTICE( 2, "%s", text );
364  }
365  // Update the entLastChangeTime to "now" since this is a table change
366  update_entLastChangeTime();
367 }
