21 /* Factory function which checks to see if we need to add a pcdMeasuresTable entry
22  * for the given Device.
23  * If we already have an entry for this Device, does nothing.
24  * Else will process the possible pcdMeasuresTable columns ("nodes") for this Device
25  * to see if we have any valid columns. If yes, will add this entry to the table.
26  */
27 bool CPcdMeasuresTableEntry::check_add_table_entry( std::shared_ptr<CPcdMib> pcd_mib_instance,
28  const px::CDevice &px_device,
29  const std::vector<px::CChannelSettings> &settings )
30 {
31  PUSH_FUNC;
32 
33  uint32_t idev_id = ( uint32_t ) px_device.id();
34  if ( pcd_mib_instance->_pcd_measures_table_map.find( idev_id ) !=
35  pcd_mib_instance->_pcd_measures_table_map.end() )
36  {
37  SYSLOG_DEBUG( 6, "Found existing pcdMeasuresTable entry for Device %u", idev_id );
38  return ( false );
39  }
40 
41  CPcdMeasuresTableEntry this_row( px_device.id(), settings );
42 
43  // For the pcdMeasuresTable:
44  bool success;
45  const char *base_oid;
46  const ::px::proto::MibTable *mib_table;
47  // std::tuple<bool, const std::string&, const ::px::proto::MibTable&> tuple_res =
48  std::tie( success, base_oid, mib_table ) =
49  CMmedsNode::lookup_mmeds_table( pcd_mib_instance->_mmeds,
50  this_row.table_name() );
51  if ( success )
52  {
53  // Then we succeeded in the lookup
54  std::string table_oid( base_oid ); // get the base OID first
55  table_oid += mib_table->text_oid();
56  SYSLOG_DEBUG( 6, "Processing table %s nodes for Device %u, oid: %s", mib_table->name().c_str(),
57  idev_id, table_oid.c_str() );
58 
59  /* This table does need a "special" node handler, so bind the call to our member function.
60  * Note that "this_row" is passed by reference to make this work (without '&' this_row
61  * argument is copy-constructed into the bind() but is then unrelated to _this_ row).
62  */
63  using std::placeholders::_1;
64  using std::placeholders::_2;
65  CMmedsNode::validate_special_node_t f_validate_special =
66  std::bind( &CPcdMeasuresTableEntry::validate_special_nodes, &this_row, _1, _2 );
67 
68  for ( int i = 0; i < mib_table->mib_columns_size(); i++ )
69  {
70  const ::px::proto::MibObject &column_data = mib_table->mib_columns( i );
71  /* Screen out unimplemented nodes, like inaccessible Indices.
72  * Do this by skipping nodes that don't have an ASN.1 type.
73  */
74  if ( !column_data.has_asn1_type() )
75  {
76  continue;
77  }
78  SYSLOG_DEBUG( 9, "Checking node %s", column_data.name().c_str() );
79 
80  std::shared_ptr<CMmedsNode> node_ptr =
81  CMmedsNode::make_mmeds_node( pcd_mib_instance->_cds_handler, px_device,
82  column_data, f_validate_special );
83  if ( node_ptr && ( node_ptr->_column_oid > 0 ) )
84  {
85  // Then this one is valid; add it to our map
86  // For this MIB, we are also interested in rescaling for units; apply that here
87  for ( const px::CChannelSettings &chsttngs : settings )
88  {
89  if ( chsttngs.id() == node_ptr->_channel_id )
90  {
91  node_ptr->set_units_scalar( chsttngs.get_units_scaling() );
92  break;
93  }
94  }
95  // Move the CMmedsNode data out of the shared_ptr into the map
96  long column_oid = node_ptr->_column_oid;
97  bool success;
98  std::tie( std::ignore, success ) = this_row._row_map.insert(
99  std::make_pair( column_oid, std::move( *node_ptr.get() ) ) );
100  if ( success )
101  {
102  SYSLOG_DEBUG( 6, "Successfully added %s as column %lu to our row map of %zu nodes",
103  column_data.name().c_str(), column_oid, this_row._row_map.size() );
104  }
105  }
106  }
107  }
108 
109  if ( this_row._row_map.empty() )
110  {
111  SYSLOG_DEBUG( 6, "No columns, so no pcdMeasuresTable entry for Device %u", idev_id );
112  return ( false );
113  }
114  else
115  {
116  // Move the row into our table
117  bool result;
118  std::tie( std::ignore, result ) = pcd_mib_instance->_pcd_measures_table_map.emplace(
119  std::make_pair( idev_id, std::move( this_row ) ) );
120  return ( result ); // hopefully true!
121  }
122 }

124 // Table-specific function to validate a "special" node, when this is its Translation Type.
125 bool CPcdMeasuresTableEntry::validate_special_nodes( CMmedsNode &node, const px::proto::MibObject &mmeds )
126 {
127  PUSH_FUNC;
128  bool success = false;
129 
130  /* First question is "did the column before us get added to this row?"
131  * because if not then this column doesn't exist either.
132  */
133  long previous_column = node._column_oid - 1;
134  mmeds_node_map_itr_t parent_pos = _row_map.find( previous_column );
135  if ( parent_pos != _row_map.end() )
136  {
137  // Now find the Channel Settings for the parent node, checking by ID
138  for ( const px::CChannelSettings &ch_setts : _ch_settings )
139  {
140  if ( ch_setts.id() != parent_pos->second._channel_id )
141  {
142  continue;
143  }
144 
145  // Now, if this Parent Channel has Units, use them
146  px::string volts_tag;
147  if ( ch_setts.units.size() > 2 )
148  {
149  volts_tag = ch_setts.units;
150  }
151  // Otherwise, see if this node has a default string
152  else if ( mmeds.failover_action() == px::proto::FAIL_DFLT_STR )
153  {
154  volts_tag = mmeds.failover_default().c_str();
155  }
156  // See if we succeeded; if so, preserve the tag as a constant value
157  // so we don't later try to re-evaluate the Value as a "special"
158  if ( volts_tag.size() > 0 )
159  {
160  success = true;
161  node._const_text = volts_tag;
162  node._translation_type = px::proto::XLT_CONSTANT;
163  SYSLOG_DEBUG( 6, "Found special tag for %s = %s", mmeds.name().c_str(), volts_tag.c_str() );
164  }
165  break; // Done processing
166  }
167  if ( !success )
168  {
169  SYSLOG_DEBUG( 9, "No success getting special tag for %s", mmeds.name().c_str() );
170  }
171  }
172  else
173  {
174  SYSLOG_DEBUG( 9, "No previous column %lu node (of %zu nodes) for %s", previous_column,
175  _row_map.size(), mmeds.name().c_str() );
176  }
177 
178  return ( success );
179 }
