18 /* Local C-code callback handler for the sys group objects that we manage for gets and sets.
19  * De-references the pointer to our handler instance and calls it.
20  * This conforms to the function template for Netsnmp_Node_Handler.
21  */
22 static int _local_handle_sysVar( netsnmp_mib_handler *handler, netsnmp_handler_registration *reginfo,
23  netsnmp_agent_request_info *reqinfo, netsnmp_request_info *request )
24 {
25  int retval = SNMP_ERR_GENERR;
26 
27  if ( reginfo && reginfo->my_reg_void )
28  {
29  std::function<Netsnmp_Node_Handler> *my_fptr_instance =
30  reinterpret_cast<std::function<Netsnmp_Node_Handler> *>( reginfo->my_reg_void );
31  if ( *my_fptr_instance != nullptr )
32  {
33  retval = ( *my_fptr_instance )( handler, reginfo, reqinfo, request );
34  }
35  }
36  return ( retval );
37 }

39 // Function to initialize sys objects via the CdsHandler.
40 void CEatonSystemMib::init_services( std::shared_ptr<CSnmpCdsHandler> cds_handler )
41 {
42  PUSH_FUNC;
43 
44  using namespace std::placeholders; // for _1, _2, etc
45 
46  SYSLOG_DEBUG( 4, "Initializing CEatonSystemMib." );
47 
48  _cds_handler = cds_handler;
49 
50  px::CDevice root_device = cds_handler->get_device( PX_ID_ROOT );
51  if ( unlikely( root_device.id() != PX_ID_ROOT ) )
52  {
53  SYSLOG_FERR( 1, "Could not access root device; cannot proceed." );
54  // Best to just do nothing further, but carry on with the rest.
55  return;
56  }
57  if ( root_device.contact.empty() )
58  {
59  px::cds::CSetFieldValue sfv_req;
60  sfv_req.fields.emplace( px::cds::field_message_type::device, "contact", _default_contact,
61  px::cds::persistence::save_always, PX_ID_ROOT, "" );
62  SYSLOG_INFO( 5, "Initializing root Device.contact to %s", _default_contact.c_str() );
63  // We'll just do a fire-and-forget - we expect this will always work
64  _cds_handler->submit_forget( sfv_req );
65  }
66  if ( root_device.location.empty() )
67  {
68  px::cds::CSetFieldValue sfv_req;
69  sfv_req.fields.emplace( px::cds::field_message_type::device, "location", _default_location,
70  px::cds::persistence::save_always, PX_ID_ROOT, "" );
71  SYSLOG_INFO( 5, "Initializing root Device.location to %s", _default_location.c_str() );
72  _cds_handler->submit_forget( sfv_req );
73  }
74 
75  /* Now:
76  * Use netsnmp_inject_handler() to add our access handler to the existing registrations
77  * for sysContact and sysLocation objects (recreate or fetch the existing registration, as we can)
78  *
79  * Write an access handler (like Toolkit 2.x's handle_tkScalar() in toolkitHanlders.c)
80  * which handles the gets and sets, and does not allow the next (original watcher) handler
81  * to do any work.
82  */
83 
84  /* First, for sysContact. Look into agent/mibgroup/mibII/system_mib.c to see how
85  * sysContact was registered, then mimic that after unregistering the original handler. */
86  oid sys_object_oid[] = { SYS_GROUP_OID, SYS_CONTACT_IDX };
87  int retval = unregister_mib( sys_object_oid, OID_LENGTH( sys_object_oid ) );
88  if ( retval != MIB_UNREGISTERED_OK )
89  {
90  SYSLOG_FERR( 1, "Could not unregister sysContact; cannot proceed." );
91  return;
92  }
93  /* Define our std::function pointer here.
94  * We store the definition in this class so it persists.
95  * Need to bind while declaring 4 placeholders for this handler.
96  */
97  _fptr_sysVar = std::bind( &CEatonSystemMib::handle_sysVar, this, _1, _2, _3, _4 );
98  _sysContact_reg =
99  netsnmp_create_handler_registration( "mibII/sysContact", &_local_handle_sysVar,
100  sys_object_oid, OID_LENGTH( sys_object_oid ),
101  HANDLER_CAN_RWRITE );
102  if ( _sysContact_reg == NULL )
103  {
104  SYSLOG_FERR( 1, "Could not get handler for sysContact; cannot proceed." );
105  return;
106  }
107  /* After attempting various games with std::function::target() and such, and failing to
108  * get useful results, decided that the easiest route was just to use the available pointer
109  * to store a pointer to the instance of our handler, and dereference when called back.
110  */
111  _sysContact_reg->my_reg_void = ( void * ) &_fptr_sysVar;
112  retval = netsnmp_register_scalar( _sysContact_reg );
113  if ( retval != MIB_REGISTERED_OK )
114  {
115  SYSLOG_FERR( 1, "Could not register sysContact; cannot proceed." );
116  netsnmp_unregister_handler( _sysContact_reg );
117  _sysContact_reg = nullptr;
118  return;
119  }
120 
121  /* Then same for sysLocation. If sysContact succeeded, then this is *really* likely to,
122  * as well, so not getting torqued out about error recovery. */
123  sys_object_oid[SYS_OBJECT_IDX_POS] = SYS_LOCATION_IDX;
124  retval = unregister_mib( sys_object_oid, OID_LENGTH( sys_object_oid ) );
125  if ( retval != MIB_UNREGISTERED_OK )
126  {
127  SYSLOG_FERR( 1, "Could not unregister sysLocation; cannot proceed." );
128  return;
129  }
130  _sysLocation_reg =
131  netsnmp_create_handler_registration( "mibII/sysLocation", &_local_handle_sysVar,
132  sys_object_oid, OID_LENGTH( sys_object_oid ),
133  HANDLER_CAN_RWRITE );
134  if ( _sysLocation_reg == NULL )
135  {
136  SYSLOG_FERR( 1, "Could not get handler for sysLocation; cannot proceed." );
137  return;
138  }
139  _sysLocation_reg->my_reg_void = ( void * ) &_fptr_sysVar;
140  retval = netsnmp_register_scalar( _sysLocation_reg );
141  if ( retval != MIB_REGISTERED_OK )
142  {
143  SYSLOG_FERR( 1, "Could not register sysLocation; cannot proceed." );
144  netsnmp_unregister_handler( _sysLocation_reg );
145  _sysLocation_reg = nullptr;
146  return;
147  }
148 }

169 // Callback handler for the sys group objects that we manage for gets and sets.
170 int CEatonSystemMib::handle_sysVar( netsnmp_mib_handler * /* handler */, netsnmp_handler_registration *reginfo,
171  netsnmp_agent_request_info *reqinfo, netsnmp_request_info *request )
172 {
173  PUSH_FUNC;
174  /*
175  * We are never called for a GETNEXT if it's registered as an
176  * "instance", as it's "magically" handled for us.
177  */
178 
179  /*
180  * an instance handler also only hands us one request at a time, so
181  * we don't need to loop over a list of requests; we'll only get one.
182  */
183 
184  int retval = SNMP_ERR_GENERR;
185  netsnmp_assert( reginfo->handlerName != NULL );
186  if ( reginfo->handlerName == NULL )
187  {
188  return ( retval );
189  }
190 
191  px::asio::cds_device_map_citer_t map_pos;
192  if ( !_cds_handler->get_map_device( PX_ID_ROOT, map_pos ) )
193  {
194  SYSLOG_FERR( 1, "Could not access root device; cannot proceed." );
195  return ( retval );
196  }
197  const px::CDevice &root_device = map_pos->second._px_device;
198 
199  // Get pointer to the info of the specific varbind requested here
200  netsnmp_variable_list *requestvb = request->requestvb;
201  oid object_idx = requestvb->name[SYS_OBJECT_IDX_POS];
202  const char *obj_text;
203  px::string field_name;
204  char text[80];
205 
206  switch ( object_idx )
207  {
208  case SYS_CONTACT_IDX:
209  obj_text = root_device.contact.c_str();
210  field_name = "contact";
211  break;
212  case SYS_LOCATION_IDX:
213  obj_text = root_device.location.c_str();
214  field_name = "location";
215  break;
216  default:
217  snprint_objid( text, 80, requestvb->name, requestvb->name_length );
218  SYSLOG_FERR( 1, "Unhandled OID %s request; cannot proceed.", text );
219  return ( retval );
220  }
221 
222  switch ( reqinfo->mode )
223  {
224  case MODE_GET:
225  snmp_set_var_typed_value( requestvb, ASN_OCTET_STR, obj_text, strlen( obj_text ) );
226  retval = SNMP_ERR_NOERROR;
227  break;
228 
229  /*
230  * Write-support (modes for Set commands)
231  */
232  case MODE_SET_RESERVE1:
233  /* Check write authorization privilege for snmpd (user "snmp")
234  * We do not expect this failure to actually occur, since snmpd has these
235  * permissions always.
236  * The purpose of this code is to guard against a future inadvertent revision
237  * of those permissions; if the failure is legitimate, then it should be audit logged.
238  */
239  if ( !_cds_handler->check_permission( px::permissions::data_write ) &&
240  !_cds_handler->check_permission( px::permissions::data_admin ) )
241  {
242  retval = SNMP_ERR_READONLY; // No write for you!!
243  SYSLOG_NOTICE( 1, "Bad permissions for Set of %s", field_name.c_str() );
244  SYSLOG_DEBUG( 1, "FIXME: this Bad Permissions failure should be audit logged." );
245  break;
246  }
247 
248  // Check that the new contents are a string, of length <= 255
249  if ( requestvb->type != ASN_OCTET_STR )
250  {
251  retval = SNMP_ERR_WRONGTYPE; /* Or v1 SNMP_ERR_BADVALUE */
252  }
253  else if ( requestvb->val_len >= SYS_STRING_LEN )
254  {
255  retval = SNMP_ERR_TOOBIG;
256  }
257  else
258  {
259  retval = SNMP_ERR_NOERROR;
260  }
261  break;
262 
263  case MODE_SET_RESERVE2:
264  // Decided that we don't need to manage a conditional backup,
265  // since nothing will happen until we call SetField, and then
266  // the CdsHandler will update out local Device copy
267  // // Make a backup copy of the current string contents
268  // // Will use the ->data area provided for us
269  // requestvb->data = netsnmp_strdup( obj_text.c_str() );
270  // requestvb->dataFreeHook = netsnmp_free;
271  retval = SNMP_ERR_NOERROR;
272  break;
273 
274  case MODE_SET_FREE:
275  /*
276  * Free resources allocated in RESERVE1 and/or RESERVE2.
277  * Something failed somewhere, and the states below won't be called.
278  * The backup we might have created will be freed by snmp_free_var_internals()
279  */
280  retval = SNMP_ERR_NOERROR;
281  break;
282 
283  case MODE_SET_ACTION:
284  // // Set the contents of the new string into our local device copy
285  // obj_text.assign( requestvb->val, requestvb->val_len );
286  retval = SNMP_ERR_NOERROR;
287  break;
288 
289  case MODE_SET_UNDO:
290  // // Restore the original text from our backup copy
291  // obj_text = requestvb->data;
292  retval = SNMP_ERR_NOERROR;
293  break;
294 
295  case MODE_SET_COMMIT:
296  {
297  // Now send the new Device field to the CDS to make it persistent
298  // Decided to do this instead of CUpdateDevice since we're mucking with the root device here
299  px::cds::CSetFieldValue sfv_req;
300  px::string new_text;
301  new_text.assign( ( const char * ) requestvb->val.string, requestvb->val_len );
302  sfv_req.fields.emplace( px::cds::field_message_type::device, field_name, new_text,
303  px::cds::persistence::save_always, PX_ID_ROOT, "" );
304  SYSLOG_INFO( 5, "Updated root Device.%s from %s to %s", field_name.c_str(), obj_text, new_text.c_str() );
305  // We'll just do a fire-and-forget - we expect this will always work
306  _cds_handler->submit_forget( sfv_req );
307 
308  // Setup the info for Audit logging.
309  // Using the Configuration log for this Device field change.
310  char msg_text[140];
311  snprintf( msg_text, 140, "Set main Device.%s to %s", field_name.c_str(), new_text.c_str() );
312  audit_log_msg( AUDIT_LOG_CONFIG, LOG_INFO, CEE_ACT_write, CEE_STS_success, reqinfo, msg_text );
313 
314  retval = SNMP_ERR_NOERROR;
315  }
316  break;
317 
318  default:
319  /*
320  * we should never get here, so this is a really bad error
321  */
322  snmp_log( LOG_ERR, "unknown mode (%d) in handle_sysVar for %s \n",
323  reqinfo->mode, reginfo->handlerName );
324  break;
325  }
326 
327  if ( retval != SNMP_ERR_NOERROR )
328  {
329  // Report the error but make a "normal" return otherwise
330  netsnmp_request_set_error( request, retval );
331  retval = SNMP_ERR_NOERROR;
332  }
333 
334  return ( retval );
335 }
