79 // Uses the px::CDevice data for this instance to update this class' members
80 void CEntStateTableEntry::update_table_members( bool is_update )
81 {
82  // Fill out the other members here, first those for all instances
83  last_state = _px_device.get_state();
84 
85  // Just one of these two for entStateAdmin, and mostly same for entStateOper:
86  if ( last_state.get( px::realtime_state::disabled ) )
87  {
88  entStateAdmin = ENTITYADMINSTATE_LOCKED;
89  entStateOper = ENTITYOPERSTATE_DISABLED;
90  }
91  else
92  {
93  entStateAdmin = ENTITYADMINSTATE_UNLOCKED;
94  // entStateOper is 'enabled' too, unless non-comm or pass-through
95  if ( last_state.get( px::realtime_state::not_communicating ) )
96  {
97  entStateOper = ENTITYOPERSTATE_DISABLED;
98  }
99  else if ( last_state.get( px::realtime_state::pass_through ) )
100  {
101  entStateOper = ENTITYOPERSTATE_TESTING;
102  }
103  else
104  {
105  entStateOper = ENTITYOPERSTATE_ENABLED;
106  }
107  }
108 
109  // entStateUsage is always ENTITYUSAGESTATE_BUSY for us
110 
111  // entStateAlarm is updated via update_entStateAlarm
112 
113  // entStateStandby is always ENTITYSTANDBYSTATUS_PROVIDINGSERVICE
114 
115  // ToDo: ignore uninteresting changes (particularly waveform_available)
116  // for entStateLastChanged
117  //
118 
119  // For now, if we're here, we update this field
120  // Convert the latest timestamp (ignore deci-seconds) to a DateAndTime "string" and copy it in
121  // Using the library function date_n_time() to convert this, which takes timeval (UTC) input
122  size_t max_strlen = 11;
123  time_t tsecs = px::CWallTime::now().tv_sec;
124  char *entStateLastChanged_ptr = ( char * ) date_n_time( &tsecs, &max_strlen );
125  entStateLastChanged_len = max_strlen;
126  for ( size_t i = 0; i < entStateLastChanged_len; ++i )
127  {
128  entStateLastChanged[i] = entStateLastChanged_ptr[i];
129  }
130 
131  SYSLOG_INFO( 5, "%s entStateTable entry for device %" PRIu64 ": %s",
132  ( is_update ? "Updating" : "Adding" ),
133  _px_device.id(), _px_device.name.c_str() );
134 }

136 // update this table entry's entStateAlarm
137 void CEntStateTableEntry::update_entStateAlarm( const px::id_t alarm_id, const uint8_t alm_level )
138 {
139  PUSH_FUNC;
140 
141  auto map_pos = _alarm_statuses.find( alarm_id );
142  if ( map_pos != _alarm_statuses.end() )
143  {
144  if ( alm_level == ENTITYALARMSTATUS_NONE )
145  {
146  SYSLOG_DEBUG( 5, "Removing alarm %" PRIu64 " ", alarm_id );
147  _alarm_statuses.erase( map_pos );
148  }
149  else
150  {
151  map_pos->second = alm_level;
152  }
153  }
154  else
155  {
156  // Don't add the alarm if the status is none
157  if ( alm_level != ENTITYALARMSTATUS_NONE )
158  {
159  // Add a new alarm
160  _alarm_statuses.insert(
161  std::make_pair( alarm_id, alm_level ) );
162  }
163  }
164 
165  // re-calculate the entStateAlarm
166  uint8_t tmp_entStateAlarm = ENTITYALARMSTATUS_NONE;
167 
168  for ( auto &alarm_statuses_itr : _alarm_statuses )
169  {
170  tmp_entStateAlarm |= alarm_statuses_itr.second;
171  }
172 
173  if ( tmp_entStateAlarm & ENTITYALARMSTATUS_CRITICAL_FLAG )
174  {
175  tmp_entStateAlarm = ENTITYALARMSTATUS_CRITICAL_FLAG;
176  }
177  else if ( tmp_entStateAlarm & ENTITYALARMSTATUS_MAJOR_FLAG )
178  {
179  tmp_entStateAlarm = ENTITYALARMSTATUS_MAJOR_FLAG;
180  }
181  else if ( tmp_entStateAlarm & ENTITYALARMSTATUS_MINOR_FLAG )
182  {
183  tmp_entStateAlarm = ENTITYALARMSTATUS_MINOR_FLAG;
184  }
185  else if ( tmp_entStateAlarm & ENTITYALARMSTATUS_WARNING_FLAG )
186  {
187  tmp_entStateAlarm = ENTITYALARMSTATUS_WARNING_FLAG;
188  }
189  else if ( tmp_entStateAlarm & ENTITYALARMSTATUS_UNDERREPAIR_FLAG )
190  {
191  tmp_entStateAlarm = ENTITYALARMSTATUS_UNDERREPAIR_FLAG;
192  }
193  else if ( tmp_entStateAlarm & ENTITYALARMSTATUS_INDETERMINATE_FLAG )
194  {
195  tmp_entStateAlarm = ENTITYALARMSTATUS_INDETERMINATE_FLAG;
196  }
197  else if ( tmp_entStateAlarm & ENTITYALARMSTATUS_UNKNOWN_FLAG )
198  {
199  tmp_entStateAlarm = ENTITYALARMSTATUS_UNKNOWN_FLAG;
200  }
201  else
202  {
203  // ENTITYALARMSTATUS_NONE, leave current state
204  }
205 
206  SYSLOG_DEBUG( 8, "Setting entStateAlarm to %u", tmp_entStateAlarm );
207  entStateAlarm = tmp_entStateAlarm;
208 }
