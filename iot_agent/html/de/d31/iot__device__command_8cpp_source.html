345 // Utility function to unpack the command params and put them into our _cmd_params map.
346 bool IotDeviceCommand::unpack_cmd_params( const char *commandParams )
347 {
348  PUSH_FUNC;
349 
350  bool retval = true; // normally expect that all is well.
351  if ( !commandParams )
352  {
353  SVR_DEBUG( 7, "No commandParams to process." );
354  return ( retval );
355  }
356 
357  // Clear any old contents
358  _cmd_params.clear();
359  // The following is usually redundant with the "unpacked" output below, so limit its length
360  SVR_DEBUG( 7, "Processing DeviceCommand params: '%.120s'", commandParams );
361  // A top level value contains either array or Object; we expect an Object.
362  px::string content( commandParams );
363  json_spirit::mValue top; // Get this as a map Value, not a vector (Value)
364  json_spirit::mObject top_obj;
365 
366  // JSON Spirit docs say read_string_or_throw() is 3x slower than read_string()
367  // But it does report any JSON errors it finds, so probably worth it here
368  try
369  {
370  json_spirit::read_string_or_throw( content, top );
371  if ( top.type() != ::json_spirit::obj_type )
372  {
373  SVR_ERR( 1, "Failure processing JSON for command parameters - wrong format" );
374  retval = false;
375  }
376  else
377  {
378  top_obj = top.get_obj(); // Get the map object
379  SVR_DEBUG( 9, "Size of top is %zu", top_obj.size() );
380  std::string param_name;
381  px::string param_text;
382  json_spirit::mValue param_val; // even inside, still a mapped Value
383  for ( auto mvpair : top_obj )
384  {
385  std::tie( param_name, param_val ) = mvpair;
386  // Insert each entry into our map member, once we decode the type
387  // We only expect a few types; others are failures, for now
388  switch ( param_val.type() )
389  {
390  case json_spirit::Value_type::str_type:
391  param_text = param_val.get_str().c_str();
392  if ( !param_text.empty() && px::pxstr_to_printable( param_text ) )
393  {
394  SVR_DEBUG( 2, "Had to remove unprintable chars from %s", param_name.c_str() );
395  }
396  _cmd_params[param_name] = px::CValue( param_text );
397  break;
398  case json_spirit::Value_type::bool_type:
399  _cmd_params[param_name] = px::CValue( param_val.get_bool() );
400  break;
401  case json_spirit::Value_type::int_type: // assuming not 64b (u)ints
402  _cmd_params[param_name] = px::CValue( param_val.get_int() );
403  break;
404  default:
405  SVR_ERR( 1, "Failure processing command parameter \'%s\' - wrong type", param_name.c_str() );
406  retval = false;
407  break;
408  }
409 
410  if ( retval )
411  {
412  // Limit the length of the string, eg for 5500 Base64-encoded bytes
413  SVR_DEBUG( 5, "Unpacked command parameter \'%s\' = %.80s", param_name.c_str(),
414  _cmd_params[param_name].as_string().c_str() );
415  }
416  }
417  }
418  }
419  catch ( json_spirit::Error_position &ep )
420  {
421  SVR_NOTICE( 1, "Error in processing commandParams \"%s\" at Line:Pos %d:%d - %s", commandParams, ep.line_, ep.column_, ep.reason_.c_str() );
422  retval = false;
423  }
424  catch ( const boost::exception & )
425  {
426  // The base exception covers a lot of evils.
427  // Wanted to catch boost::exception_detail::clone_impl, eg under param_val.get_bool()
428  SVR_NOTICE( 1, "Failure processing commandParams \'%s\'", commandParams );
429  retval = false;
430  }
431  catch ( const std::runtime_error &ex )
432  {
433  SVR_NOTICE( 1, "Failure processing commandParams \'%s\' - %s", commandParams, ex.what() );
434  retval = false;
435  }
436 
437  return ( retval );
438 }

1122  {

1124  set_command_resp_msg( commandResponse, "Address + Size points beyond end of receive buffer." );
1125  return ( bOK );
1126  }
1127 
1128  // Now we're ready to decode the string
1129  std::vector<unsigned char> data_bin( px::CBase64Converter::decode( data_encoded ) );
1130  // Make sure it wasn't more than expected; should be <= chunk size
1131  if ( data_bin.size() > chunk_size )
1132  {
1133  SVR_NOTICE( 1, "Decoded size %zu > expected chunk size %zu.", data_bin.size(), chunk_size );
1134  commandResponse->status = HTTP_STATUS_BAD_REQUEST;
1135  set_command_resp_msg( commandResponse, "Decoded size > expected chunk size." );
1136  return ( bOK );
1137  }
1138 
1139  // Copy the bytes into our buffer. We already checked against buffer overrun.
1140  for ( uint8_t one_byte : data_bin )
1141  {
1142  _recv_buffer[offset++] = one_byte;
1143  }
1144  // Save the last pointer, as long as it is greater
1145  // (So we provide some accommodation for out-of-order data messages)
1146  if ( offset > _buffer_end_idx )
1147  {
1148  _buffer_end_idx = offset;
1149  }
1150

1152  if ( last_chunk )
1153  {

1156  _image_filename = "/tmp/image_XXXXXX.tar";
1157  int suffix_length = strlen( ".tar" );
1158  SVR_INFO( 5, "Opening temp file, template = %s, suffix length = %d", ( char * )_image_filename.data(), suffix_length );
1159  ssize_t bytes_written = 0;
1160  int fd = mkstemps( ( char * )_image_filename.data(), suffix_length );
1161  if ( fd != -1 )
1162  {
1163  bytes_written = write( fd, _recv_buffer, _buffer_end_idx );
1164  close( fd );
1165  if ( ( uint32_t ) bytes_written == _buffer_end_idx )
1166  {
1167  SVR_INFO( 1, "Finished receiving image, and wrote %u bytes to temp file %s",
1168  ( uint32_t ) bytes_written, _image_filename.c_str() );
1169  _iot_core->_fw_up_manager->request_fw_up_state( FirmwareUpdateManager::update_states::START_IMAGE_UPDATE );
1170  bOK = true; // Success!
1171  }
1172  else if ( bytes_written == -1 )
1173  {
1174  SVR_ERR( 0, "Failed to write received Image to temp file: %s", strerror( errno ) );
1175  }
1176  else
1177  {
1178  // Wrote less than the full size - must have run out of room
1179  SVR_ERR( 0, "Failed to write full received Image to temp file: only %u / %u bytes written.",
1180  ( uint32_t ) bytes_written, _buffer_end_idx );
1181  unlink( _image_filename.c_str() ); // delete whatever is there in the file
1182  }
1183  }
1184  else
1185  {
1186  SVR_ERR( 0, "Failed to open temp file for received Image: %s", strerror( errno ) );
1187  }
1188  // Now we can free our buffer, whether we succeeded or not
1189  free( _recv_buffer );
1190  _recv_buffer = nullptr;
1191  _expected_recv_size = _buffer_end_idx = 0;
1192  // If we failed to write the recv_buffer out, end this FirmwareUpdate process
1193  if ( !bOK )
1194  {
1195  complete_firmware_update( CEE_STS_failure, "Failed to write received Image to temp file" );
1196  }
1197  // Else, subscribe to updates in the ImageUpdate OOB for this command UUID
1198  else
1199  {
1200  // First, set our callback functor
1201  // Bind the calls to our member functions, to use in the call to CdsHandler
1202  using std::placeholders::_1;
1203  CIotAgentCdsHandler::OOB_Callback_t f_handle_img_up_status =
1204  std::bind( &IotDeviceCommand::handle_img_up_status, this, _1 );
1205  assert( f_handle_img_up_status ); // The bind worked, right?
1206  _iot_core->_cds_handler->assign_oob_callback( f_handle_img_up_status );
1207  _iot_core->_cds_handler->subscribe_image_updates( _cmd_uuid );
1208  }
1209  }
1210  else
1211  {
1212  _iot_core->_fw_up_manager->request_fw_up_state( FirmwareUpdateManager::update_states::RECEIVING_IMAGE );
1213  bOK = true; // partial success
1214  }
1215  // Now setup the response
1216  if ( bOK )
1217  {
1218  commandResponse->status = HTTP_STATUS_OK;
1219  set_command_resp_msg( commandResponse, "OK" );
1220  }
1221  else
1222  {
1223  commandResponse->status = HTTP_STATUS_INTERNAL_SERVER_ERROR;
1224  set_command_resp_msg( commandResponse, "Failed to save the image." );
1225  }
1226 
1227  return ( bOK );
1228 }

1807 bool IotDeviceCommand::handle_generic_cmds( const IOT_DEVICE_COMMAND *command, const std::unordered_map<std::string, px::CValue> cmd_params,
1808  std::string &cmd_line, IOT_DEVICE_COMMAND_RESPONSE *commandResponse )
1809 {
1810  bool success = false;
1811 
1812  // if str_val is not provided, param 'command' should be present
1813  if ( cmd_line.empty() )
1814  {
1815  if ( cmd_params.find( COMMAND ) == cmd_params.end() )
1816  {
1817  SVR_ERR( 2, "Could not find param '%s' in %s", COMMAND, command->commandParams );
1818  set_command_resp_msg( commandResponse, "Command missing param 'command'." );
1819  return ( success );
1820  }
1821  }
1822 
1823  bool leave_newline = true;
1824  int cmd_status = -1;
1825  std::vector<std::string> outlines;
1826  std::string res_msg;
1827 
1828  // Form the command line string from the params received
1829  if ( !cmd_params.empty() )
1830  {
1831  for ( auto it = cmd_params.begin(); it != cmd_params.end(); ++it )
1832  {
1833  if ( !it->second.as_string().empty() )
1834  {
1835  cmd_line += " ";
1836  cmd_line += px::to_basicstr( it->second.as_string() );
1837  cmd_line += " ";
1838  }
1839  }
1840  }
1841 
1842  // Verify that the command line is not empty
1843  if ( cmd_line.empty() )
1844  {
1845  set_command_resp_msg( commandResponse, "No command received for execution." );
1846  commandResponse->status = HTTP_STATUS_BAD_REQUEST;
1847  return ( success );
1848  }
1849  SVR_DEBUG( 5, "handle_generic_cmds the command to execute is : %s", cmd_line.c_str() );

1852  // If present, block the command to avoid any unwanted commands being executed.
1853  if ( ( cmd_line.find( ';' ) != std::string::npos ) || ( cmd_line.find( "&&" ) != std::string::npos ) ||
1854  ( cmd_line.find( "||" ) != std::string::npos ) )
1855  {
1856  SVR_ERR( 2, "Invalid Command. Use of concatenation characters is restricted in Device Commands." );
1857  set_command_resp_msg( commandResponse, "Invalid command. Use of concatenation characters is restricted in Device Commands." );
1858  commandResponse->status = HTTP_STATUS_BAD_REQUEST;
1859  return ( success );
1860  }

1862  // First, pass in the verbosity level for this source file
1863  etn::CLinuxUtil::set_verbosity_level( _svr_verb_array[SVR_LOG_IOT] );
1864  success = etn::CLinuxUtil::shell_cmd( "Executing command", cmd_line.c_str(), BUILT_IN_COMMANDS_TIMEOUT,
1865  leave_newline, cmd_status, outlines );
1866 
1867  for ( size_t i = 0; i < outlines.size(); ++i )
1868  {
1869  res_msg += outlines[i];
1870  }
1871 
1872  // Escape all double quotes in the response message as the
1873  // Eaton SDK serializer will fail to parse it.
1874  // Also, replace the % sign by adding escape character as double %
1875  res_msg = px::add_esc_char( res_msg.c_str() );
1876 
1877  // Strip off the final \n, if any
1878  SVR_DEBUG( 5, "handle_generic_cmds the command res_msg is : %s", res_msg.c_str() );
1879  auto idx = res_msg.find_last_of( '\n' );
1880  if ( ( idx != std::string::npos ) && ( idx + 2 >= res_msg.size() ) )
1881  {
1882  res_msg.erase( idx );
1883  }
1884 
1885  if ( success )
1886  {
1887  // Yes - the program gave us a thumbs up!
1888  commandResponse->status = HTTP_STATUS_OK;
1889  set_command_resp_msg( commandResponse, res_msg.c_str() );
1890  }
1891  else if ( cmd_status == -1 )
1892  {
1893  /* Error reported by pclose() */
1894  commandResponse->status = HTTP_STATUS_INTERNAL_SERVER_ERROR;
1895  set_command_resp_msg( commandResponse, res_msg.c_str() );
1896  }
1897  else
1898  {

1901  if ( WIFEXITED( cmd_status ) )
1902  {
1903  // We already know that the exit code wasn't 0 since retval was False
1904  // uint8_t exit_code = WEXITSTATUS( cmd_status );
1905  // if ( exit_code != 0 ) {

1910  set_command_resp_msg( commandResponse, res_msg.c_str() );
1911  // Abnormal termination, eg from a signal. Treat as failure.
1912  }
1913  else

1915  //Check if the termination was due to a signal

1931  set_command_resp_msg( commandResponse, res_msg.c_str() );

1933  }
