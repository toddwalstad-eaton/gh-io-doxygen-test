263 const IOT_DATA_CHANNEL_REALTIMES_ITEM *CIotChannel::get_iot_channel_realtime_struct( const px::CWallTime &last_realtime,
264  bool ignore_last_realtime )
265 {
266  PUSH_FUNC;
267 
268  // Ask the CDS for our Value
269  px::CValue ch_value;
270  px::CWallTime val_realtime( 0, 0 );
271  if ( !ignore_last_realtime )
272  {
273  val_realtime = last_realtime;
274  }
275 
276  bool success;
277  try
278  {
279  success = _iot_core->_cds_handler->get_channel_value( _ch_id, ch_value, val_realtime );
280  }
281  catch ( const px::generic_err &ex )
282  {
283  SVR_NOTICE( 1, "Failed to get channel realtimes for channel %u: %s", ( uint32_t ) _ch_id, ex.what() );
284  success = false;
285  }
286 
287  // We will get the value's timestamp on a valid return
288  if ( !success )
289  {
290  if ( ignore_last_realtime )
291  {
292  SVR_NOTICE( 2, "Failed to get Value update for Channel %" PRIu64, _ch_id );
293  }
294  else
295  {
296  SVR_DEBUG( 9, "Channel %" PRIu64 " is older than last_realtime timestamp, not updating", _ch_id );
297  }
298  return ( nullptr );
299  }
300  SVR_DEBUG( 8, "Updating Channel %" PRIu64 "%s", _ch_id, ch_name.c_str() );
301 
302  _iot_ch_rt_struct.time = val_realtime.tv_sec;
303  _iot_ch_rt_struct.milliseconds = val_realtime.tv_nsec / 1000000; // nsec -> msec
304  _present_value = ch_value.as_string();
305  _iot_ch_rt_struct.value = _present_value.c_str();
306  SVR_DEBUG( 9, "Got Value %s for Channel %" PRIu64 "", _iot_ch_rt_struct.value, _ch_id );
307 
308  return ( &_iot_ch_rt_struct );
309 }

311 /* Alternate Getter for a pointer to the IOT_DATA_CHANNEL_REALTIMES_ITEM member of this class
312  * after it has been unconditionally updated with the latest channel present value,
313  * and its timestamp updated (optionally "rounded" - with the milliseconds parameter zeroed) */
314 const void *CIotChannel::get_iot_channel_item_for_realtime_value( bool for_trend )
315 {
316  const void *ret_val = nullptr;
317 
318  px::CWallTime tnow = px::CWallTime::now();
319 
320  if ( for_trend )
321  {
322  px::CValue ch_value;
323  px::CWallTime val_realtime( 0, 0 );
324  bool success;
325  time_t trend_cadence = CONFIG_IOT_REALTIME_CADENCE_SECS;
326  if ( trend_cadence < 300 )
327  {
328  trend_cadence = 300; // Set to 5 minutes if realtime cadence is less, which is the default.
329  }
330  try
331  {
332  success = _iot_core->_cds_handler->get_channel_value( _ch_id, ch_value, val_realtime );
333  }
334  catch ( const px::generic_err &ex )
335  {
336  SVR_NOTICE( 1, "Failed to get channel value for channel %u: %s", ( uint32_t ) _ch_id, ex.what() );
337  success = false;
338  }
339 
340  // We will get the value's timestamp on a valid return (time it last changed)
341  if ( success )
342  {
343  /* After trying the raw timestamps, and being disappointed that static or unchanging data
344  * might be, eg, 8 days old in val_realtime.tv_sec, we have decided now to use a rounded timestamp,
345  * set to the trend interval just ended.
346  * Revised to now use the configured trend cadence (not always 5 minutes). */
347  long modulo_count = tnow.tv_sec % trend_cadence;
348  _iot_ch_trend_struct.time = tnow.tv_sec - modulo_count;
349  _trend_actValue = ch_value.as_string();
350  _iot_ch_trend_struct.actValue = _trend_actValue.c_str();
351  _iot_ch_trend_struct.avgValue = nullptr;
352  _iot_ch_trend_struct.minValue = nullptr;
353  _iot_ch_trend_struct.maxValue = nullptr;
354  ret_val = ( const void * ) &_iot_ch_trend_struct;
355  }
356  // else, leave ret_val as nullptr
357  }
358  else
359  {
360  // Get the latest const IOT_DATA_CHANNEL_REALTIMES_ITEM* in the normal way
361  ret_val = ( const void * ) get_iot_channel_realtime_struct( tnow, true );
362  }
363  return ( ret_val );
364 }
