113  std::map<px::string, px::proto::IotBlobUpload> upload_local_map;
114  std::unique_lock<std::mutex> lock( pending_upload_list_lock );
115  SVR_DEBUG( 8, "Getting the pending blob uploads list to initiate the upload process." );
116  upload_local_map = std::move( _blob_upload_pending_list );
117  _blob_upload_pending_list.clear();
118  lock.unlock();

120  DEVICELOG_INFO( "Upload Blob", "Starting blob upload for %zu files", upload_local_map.size() );
121  std::map<px::string, px::proto::IotBlobUpload>::iterator item;
122  for ( item = upload_local_map.begin(); item != upload_local_map.end(); item++ )
123  {
124  if ( !want_thread_stop && _iot_core->_iot_client->is_connected() )
125  {
126  std::string result_msg;
127  enum px::proto::IotBlobUploadErrorEnums thread_result = launch_blob_upload_thread( item->first, item->second );
128  if ( thread_result == px::proto::IotBlobUploadErrorEnums::IOT_BLOB_UPLOAD_INITIALIZED )
129  {
130  SVR_INFO( 4, "Successfully launched thread for blob upload for %s.", item->first.c_str() );
131  }
132  else
133  {
134  DEVICELOG_ERR( "Upload Blob", "Failed to launch thread for blob upload task for %s; result code = %u", item->first.c_str(), thread_result );
135  }
136  if ( is_blob_upload_thread_launched() )
137  {
138  std::tie( thread_result, result_msg ) = get_results();
139  if ( thread_result == px::proto::IotBlobUploadErrorEnums::IOT_BLOB_UPLOAD_DONE_SUCCESS )
140  {
141  DEVICELOG_INFO( "Upload Blob", "Successfully completed thread for blob upload task for %s", item->first.c_str() );
142  }
143  else if ( thread_result == px::proto::IotBlobUploadErrorEnums::IOT_BLOB_UPLOAD_INTERRUPTED )
144  {
145  if ( item->second.has_blob_up_response() && ( ( item->second.blob_up_response().retry_count() + 1 ) < BLOB_UPLOAD_MAX_RETRY_COUNT ) )
146  {
147  std::unique_lock<std::mutex> lock( pending_upload_list_lock );
148  _blob_upload_pending_list.emplace( item->first, item->second );
149  DEVICELOG_ERR( "Upload Blob", "Blob upload for %s interrupted due to connection loss.", item->first.c_str() );
150  }
151  }
152  else
153  {
154  DEVICELOG_ERR( "Upload Blob", "Failed to complete thread for blob upload task for %s; result code = %u: %s", item->first.c_str(), thread_result, result_msg.c_str() );
155  }
156  }
157  handle_blob_upload_resp( item->first, thread_result, result_msg );
158  }
159  else

161  SVR_DEBUG( 8, "Adding to pending list else case" );
162  std::unique_lock<std::mutex> lock( pending_upload_list_lock );
163  _blob_upload_pending_list.emplace( item->first, item->second );

165  SVR_NOTICE( 2, "%s", err_text.c_str() );

236 enum px::proto::IotBlobUploadErrorEnums BlobUploadManager::launch_blob_upload_thread( const px::string &upload_oob_tag, const px::proto::IotBlobUpload &blob_upld_req )
237 {
238  PUSH_FUNC;
239 
240  enum px::proto::IotBlobUploadErrorEnums launch_result;
241  want_thread_stop = false;
242 
243  try
244  {
245  /* Launch the blob upload task in a separate thread using std::async and std::future
246  * for the results that we will wait on.
247  * Uncrustify really messes up the appearance of this lambda:
248  * [this] { return ( blob_upload_task() ); } */
249  blob_upload_future_result = std::async( std::launch::async, [&upload_oob_tag, &blob_upld_req, this] {
250  return ( blob_upload_task( upload_oob_tag, blob_upld_req ) );
251  } );
252  SVR_INFO( 5, "Launched the blob upload thread for %s", upload_oob_tag.c_str() );
253  launch_result = px::proto::IotBlobUploadErrorEnums::IOT_BLOB_UPLOAD_INITIALIZED;
254  blob_upload_thread_is_launched = true;
255  }
256  catch ( const std::exception &ex )
257  {
258  // Caught either std::system_error or std::bad_alloc; equally bad to us.
259  SVR_ERR( 0, "Failed to launch blob upload thread for %s: %s", upload_oob_tag.c_str(), ex.what() );
260  launch_result = px::proto::IotBlobUploadErrorEnums::IOT_BLOB_UPLOAD_DONE_ERROR;
261  }
262 
263  return ( launch_result );

266 // Utility to check to see if the blob upload task (thread) is done.
267 bool BlobUploadManager::check_is_done()
268 {
269  PUSH_FUNC;
270 
271  // This function should not be called unless we have launched the blob upload task thread,
272  // and are waiting on this future.
273  if ( !blob_upload_future_result.valid() )
274  {
275  // But if already done, will just provide the warning message and signal "Done".
276  SVR_WARN( 1, "Found that the blob upload thread was already Done." );
277  return ( true );
278  }
279 
280  SVR_DEBUG( 9, "Checking to see if the blob upload thread is done..." );
281  // We use wait_for( 1s ) here to test to see if the future has been
282  // set by the completion of the thread, with that very short timeout
283  std::future_status status = blob_upload_future_result.wait_for( std::chrono::seconds( 1 ) );
284  if ( status == std::future_status::ready )
285  {
286  SVR_DEBUG( 5, "Found that the blob upload thread is done." );
287  return ( true );
288  }
289  else
290  {
291  SVR_DEBUG( 9, "Found that the blob upload thread is Not Done yet..." );
292  return ( false );
293  }
