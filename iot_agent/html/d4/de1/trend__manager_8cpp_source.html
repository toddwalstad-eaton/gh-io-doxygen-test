148  }
149  // Add additional delay during catchup, to avoid hitting the per-minute message limit
150  if ( !request_latest )
151  {
152  ThreadAPI_Sleep( catchup_cadence_secs * 1000 );

158  }
159

161  {
162  SVR_DEBUG( 5, "Published Trend updates at %lu for devices %s", interval_time_secs, good_devs.c_str() );

177  }
178  else
179  {
180  SVR_INFO( 5, "No trend data found for catchup time %lu; done with this interval.", requested_interval_time );
181  // The catchup task will manage removing this entry from our list of catchup intervals
182  }
183  }
184 
185  return ( bOverall );

275 bool TrendManager::check_is_done()
276 {
277  PUSH_FUNC;
278 
279  // This function should not be called unless we have launched the catchup task thread,
280  // and are waiting on this future.
281  if ( !catchup_future_result.valid() )
282  {
283  // But if already done, will just provide the warning message and signal "Done".
284  SVR_WARN( 1, "Found that the catchup thread was already Done." );
285  return ( true );
286  }
287 
288  SVR_DEBUG( 9, "Checking to see if the catchup thread is done..." );
289  // We use wait_for( 1s ) here to test to see if the future has been
290  // set by the completion of the thread, with that very short timeout
291  std::future_status status = catchup_future_result.wait_for( std::chrono::seconds( 1 ) );
292  if ( status == std::future_status::ready )
293  {
294  SVR_DEBUG( 5, "Found that the catchup thread is done." );
295  return ( true );
296  }
297  else
298  {
299  SVR_DEBUG( 9, "Found that the catchup thread is Not Done yet..." );
300  return ( false );

348 // This is the task that does the trend catchup.
349 TrendManager::catchup_result_t TrendManager::catchup_task()
350 {
351  catchup_result_t rv( INTERRUPTED, "" );
352 
353  // We need to re-establish the syslog pointer, since we are in a new thread
354  if ( _thr_local_syslog.get() == nullptr )
355  {
356  _thr_local_syslog.reset( new toolkit::CSysLog );
357  /* Set the verbosity level to that for CDS operations */
358  if ( likely( _thr_local_syslog.get() ) )
359  {
360  _thr_local_syslog.get()->verbosity = _svr_verb_array[SVR_LOG_CDS_HANDLE];
361  }
362  /* Kick off syslogging with a message... below */
363  }
364 
365  // Test for the gotta-haves
366  if ( !_iot_core || !_iot_core->_oob_handler || !_iot_core->_iot_client || !_iot_core->_iot_client->is_connected() )
367  {
368  rv = catchup_result_t( DONE_ERROR, "Bad start or misconfigured." );
369  return ( rv );
370  }
371 
372  SVR_INFO( 2, "Beginning Trend Catchup Task." );
373 
374  time_t request_interval_time = _iot_core->_oob_handler->catchup_oob_handler.get_next_trend_catchup_interval( trend_cadence_secs );
375  while ( ( request_interval_time != 0 ) && !want_thread_stop && _iot_core->_iot_client->is_connected() )
376  {
377  // If we aren't already waiting on this interval time:
378  if ( request_interval_time != catchup_ack_timestamp )
379  {
380  /* Publish this interval's data:
381  * The IoT SDK takes care of locking for us, to avoid thread contention. */
382  if ( publish_channel_trend_values( trend_cadence_secs, request_interval_time ) )
383  {
384  // We sent some message(s); Now sleep this thread for so many milliseconds
385  ThreadAPI_Sleep( catchup_cadence_secs * 1000 );
386  // If the thread is being killed now, loop back and exit
387  if ( want_thread_stop )
388  {
389  continue;
390  }
391  }
392  }
393  // If there are still unacknowledged trend responses, just wait and try again
394  if ( !catchup_msg_ids.empty() )
395  {
396  ThreadAPI_Sleep( catchup_cadence_secs * 1000 );
397  continue;
398  }
399  else if ( !want_thread_stop )
400  {
401  // Done with this interval; advance to the next. Don't try to use the CDS if we are being stopped, though.
402  _iot_core->_oob_handler->catchup_oob_handler.remove_trend_catchup_interval( request_interval_time, trend_cadence_secs );
403  request_interval_time = _iot_core->_oob_handler->catchup_oob_handler.get_next_trend_catchup_interval( trend_cadence_secs );
404  }
405  }
406 
407  if ( request_interval_time == 0 )
408  {
409  rv = catchup_result_t( DONE_SUCCESS, "Done: All trend data caught up." );
410  }
411  else if ( want_thread_stop || !_iot_core->_iot_client->is_connected() )
412  {
413  rv = catchup_result_t( INTERRUPTED, "Catchup processing was Interrupted; not complete." );
414  }
415  // Else, rv should have been set in the while loop at some error condition
416 
417  return ( rv );
418 }
