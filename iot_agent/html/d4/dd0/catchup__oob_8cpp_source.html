26 // Reads in the IotCatchup OOB contents, and stores it.
27 void CatchupOob::read_catchup_oob()
28 {
29  PUSH_FUNC;
30 
31  // Get the IotCatchup OOB:
32  px::proto::toolkitStruct tk_catchup_oob;
33  bool bOK = _cds_handler->get_oob( IOT_CATCHUP_TAG, tk_catchup_oob );
34  if ( bOK && tk_catchup_oob.has_iot_catch_up() )
35  {
36  const px::proto::IotHistoricCatchup &iot_catchup_data = tk_catchup_oob.iot_catch_up();
37  last_trend_published_time = ( time_t ) iot_catchup_data.last_trend_published_time();
38  trend_catchup_spans.clear();
39  for ( int i = 0; i < iot_catchup_data.trend_catchup_spans_size(); i++ )
40  {
41  const px::proto::IotTimeSpan &span = iot_catchup_data.trend_catchup_spans( i );
42  time_t begin_span = ( time_t ) span.begin_span();
43  time_t end_span = ( time_t ) span.end_span();
44  // Reality check these
45  if ( ( begin_span == 0 ) && ( end_span == 0 ) )
46  {
47  SVR_WARN( 1, "Bad span value(s): %ld to %ld", begin_span, end_span );
48  continue;
49  }
50  auto one_span = std::make_pair( begin_span, end_span );
51  trend_catchup_spans.push_back( one_span );
52  SVR_INFO( 2, "Found trend catchup span: %ld to %ld", begin_span, end_span );
53  }
54  }
55  else
56  {
57  SVR_WARN( 1, "Did not find %s OOB", IOT_CATCHUP_TAG );
58  }
59  if ( last_trend_published_time == 0 )
60  {
61  SVR_INFO( 2, "No initial value provided for last_trend_published_time" );
62  }
63  else
64  {
65  SVR_INFO( 2, "Read last_trend_published_time = %ld from %s", last_trend_published_time, IOT_CATCHUP_TAG );
66  }
67 
68  if ( trend_catchup_spans.empty() )
69  {
70  SVR_INFO( 2, "No initial trend catchup spans found in %s", IOT_CATCHUP_TAG );
71  }
72 }

75 void CatchupOob::write_catchup_oob()
76 {
77  PUSH_FUNC;
78 
79  // Read in the IotCatchup OOB first:
80  px::proto::toolkitStruct tk_catchup_oob;
81  if ( !_cds_handler->get_oob( IOT_CATCHUP_TAG, tk_catchup_oob ) )
82  {
83  // This is unlikely, but if it happens that there is no such OOB, then fill out some basic data for it.
84  tk_catchup_oob.set_log_changes( false );
85  tk_catchup_oob.set_description( IOT_CATCHUP_TAG );
86  tk_catchup_oob.set_label( IOT_CATCHUP_TAG );
87  }
88 
89  // Fill out the data
90  px::proto::IotHistoricCatchup *iot_catchup_data = tk_catchup_oob.mutable_iot_catch_up();
91  iot_catchup_data->set_last_trend_published_time( ( uint64_t ) last_trend_published_time );
92  iot_catchup_data->clear_trend_catchup_spans();
93 
94  if ( !trend_catchup_spans.empty() )
95  {
96  for ( auto &span : trend_catchup_spans )
97  {
98  px::proto::IotTimeSpan *new_span = iot_catchup_data->add_trend_catchup_spans();
99  new_span->set_begin_span( ( uint64_t ) span.first );
100  new_span->set_end_span( ( uint64_t ) span.second );
101  }
102  }
103 
104  bool persist = true;
105  _cds_handler->set_oob( IOT_CATCHUP_TAG, tk_catchup_oob, persist );
106  // That was a fire-and-forget, so we are done
107 }

183 /* Remove the indicated interval from the trend_catchup_spans, presumably because it has been published
184  * and successfully acknowledged. */
185 void CatchupOob::remove_trend_catchup_interval( time_t interval_time, time_t cadence_secs )
186 {
187  PUSH_FUNC;
188 
189  if ( trend_catchup_spans.empty() )
190  {
191  SVR_WARN( 1, "No catchup intervals exist; can't remove %ld", interval_time );
192  return;
193  }
194  if ( cadence_secs < 300 )
195  {
196  cadence_secs = 300; // ie, 5 minute trend cadence
197  }
198 
199  auto test_span = trend_catchup_spans.begin();
200  size_t on_iter_count = 0;
201  bool found = false;
202  while ( test_span != trend_catchup_spans.end() && !found )
203  {
204  on_iter_count++;
205  if ( test_span->first == interval_time )
206  {
207  // Found it; see if it is also the end interval
208  found = true;
209  if ( test_span->second == interval_time )
210  {
211  SVR_INFO( 2, "Removing span's last interval: %ld", interval_time );
212  trend_catchup_spans.erase( test_span );
213  if ( trend_catchup_spans.empty() )
214  {
215  SVR_INFO( 1, "Removed last; no more catchup intervals exist." );
216  }
217  }
218  else
219  {
220  // Need to advance the first (begin_secs) to the next interval
221  test_span->first += cadence_secs;
222  }
223 
224  break;
225  }
226  test_span++;
227  }
228  if ( found )
229  {
230  write_catchup_oob();
231  if ( on_iter_count != 1 )
232  {
233  SVR_WARN( 1, "Found interval to remove, but not in the first span." );
234  }
235  }
236 }

238 /* Look through the trend_catchup_spans, and see if any entries
239  * are older than the allowed catchup interval, and remove any that are too old. */
240 void CatchupOob::purge_old_spans( time_t cadence_secs )
241 {
242  PUSH_FUNC;
243 
244  // Determine the latest timestamp that is allowed; OK to use non-normalized time here
245  time_t tnow = px::CWallTime::now().tv_sec;
246  time_t oldest_time = tnow - ( 60 * 60 * allowed_catchup_window );
247 
248  auto test_span = trend_catchup_spans.begin();
249  bool need_write = false;
250  while ( test_span != trend_catchup_spans.end() )
251  {
252  // We have seen nonsense data, with a span beginning later than "now", so remove that span, too.
253  if ( ( test_span->second < oldest_time ) || ( test_span->first > tnow ) )
254  {
255  // The end of the span is too old; we can remove this span entirely
256  test_span = trend_catchup_spans.erase( test_span );
257  need_write = true;
258  continue; // continue since we already advanced the iterator here
259  }
260  if ( test_span->first < oldest_time )
261  {
262  // The beginning of the span is not in range, but the end is; remove any too-old intervals.
263  while ( test_span->first < oldest_time )
264  {
265  // Need to advance the first (begin_secs) to the next interval
266  test_span->first += cadence_secs;
267  // Reality check
268  if ( test_span->first > test_span->second )
269  {
270  test_span->first = test_span->second;
271  break;
272  }
273  }
274  need_write = true;
275  }
276  test_span++;
277  }
278 
279  if ( need_write )
280  {
281  SVR_INFO( 2, "Removed some intervals that were older than the allowed catchup window of %u hours.",
282  allowed_catchup_window );
283  write_catchup_oob();
284  }
