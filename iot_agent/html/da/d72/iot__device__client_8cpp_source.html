492  }

496  }

780 // This is the main switchyard for IoT Agent activities, once we are configured and running
781 // Called by the core's update timer to manage next state / action.
782 // If we don't alter the next_msec_timeout, it will stay at the default of _UPDATE_TIMER_MSECS
783 // that was set by the core's calling timer function.
784 void CIotDeviceClient::handle_timer_update( time_t tnow_secs, unsigned int &next_msec_timeout )
785 {
786  PUSH_FUNC;
787 
788  px::proto::IotConnectionStatusEnums status;
789  // Switch for the non-connected states
790  switch ( _client_state )
791  {
792  case CONNECTION_FAILED:
793  // We have already updated the Status OOB. We don't want to send any messages yet.
794  // This was not a fatal failure, so we should close the connection, wait 1 min, and try again
795  next_msec_timeout = _RETRY_TIMER_MSECS;
796  close();
797  _client_state = CONNECTING;
798  SVR_NOTICE( 2, "Will sleep %d min and try again.", ( _RETRY_TIMER_MSECS / 60000 ) );
799  return;
800 
801  case DO_EXIT:
802  // Now we can throw to CIotAgentCore::execute() since we are out of our callbacks and back under ASIO control
803  throw px::generic_err( "Fatal connection issue encountered." );
804 
805  case DO_RECONNECT:
806  SVR_NOTICE( 2, "Will close connection and try to reconnect again." );
807  // Force a close(), which will null out the _client_handle, wait 2s, and try again to connect
808  close();
809  _client_state = CONNECTING;
810  // We won't update the OOB status in this case; still shows the problem that got us here
811  return;
812 
813  // Normally done as part of startup; if done later, we are re-connecting:
814  case CONNECTING:
815  {
816  // Validate the value of cadence set in the config file
817  static_assert( CONFIG_IOT_REALTIME_CADENCE_SECS != 0, "Error: Invalid value of CONFIG_IOT_REALTIME_CADENCE_SECS variable" );
818 #if CONFIG_IOT_REALTIME_CADENCE_SECS > 0 && CONFIG_IOT_REALTIME_CADENCE_SECS < 300
819  // Note: though we would like values here that multiply up evenly into 300s, there isn't a
820  // fundamental reason why any value <300s couldn't work.
821  static_assert( CONFIG_IOT_REALTIME_CADENCE_SECS == 1 || CONFIG_IOT_REALTIME_CADENCE_SECS == 5 ||
822  CONFIG_IOT_REALTIME_CADENCE_SECS == 15 || CONFIG_IOT_REALTIME_CADENCE_SECS == 30 ||
823  CONFIG_IOT_REALTIME_CADENCE_SECS == 60,
824  "Error: Invalid value of CONFIG_IOT_REALTIME_CADENCE_SECS variable" );
825 #else
826  // Limited by the choices available in CDS CTrendStore::QueryTrends, in minutes
827  static_assert( CONFIG_IOT_REALTIME_CADENCE_SECS / 60 == 5 || CONFIG_IOT_REALTIME_CADENCE_SECS / 60 == 30 ||
828  CONFIG_IOT_REALTIME_CADENCE_SECS / 60 == 60 || CONFIG_IOT_REALTIME_CADENCE_SECS / 60 == 120 ||
829  CONFIG_IOT_REALTIME_CADENCE_SECS / 60 == 240 || CONFIG_IOT_REALTIME_CADENCE_SECS / 60 == 480 ||
830  CONFIG_IOT_REALTIME_CADENCE_SECS / 60 == 720 || CONFIG_IOT_REALTIME_CADENCE_SECS / 60 == 1440,
831  "Error: Invalid value of CONFIG_IOT_REALTIME_CADENCE_SECS variable" );
832 #endif
833  SVR_NOTICE( 2, "IoT device client state: CONNECTING" );
834  const char *error_reason = nullptr;
835  px::proto::IotAgentErrorStateEnums connect_error_code;
836  std::tie( status, connect_error_code, error_reason ) = connect();
837  // Leave the client state unchanged
838  if ( connect_error_code != px::proto::IotAgentErrorStateEnums::IOT_NO_ERROR )
839  {
840  SVR_NOTICE( 2, "Connect failed: %s", error_reason );
841  }
842  else
843  {
844  SVR_NOTICE( 2, "Connect succeeded." );
845  }
846  }
847  return;
848 
849  case IN_NET_DIAGS:
850  // See if the Net Diagnostic test thread is done yet
851  if ( !_iot_core->_diag->check_is_done() )
852  {
853  SVR_DEBUG( 8, "Skipping timer pop, waiting for network diags task to complete" );
854  }
855  else
856  {
857  handle_net_diagnostic_result();
858  }
859  return;
860 
861  default:
862  // OK to proceed
863  break;
864  }
865 
866  if ( _client_handle == nullptr )
867  {
868  // Then we closed the last connection and it's time now to reopen. Set up states to go to CONNECTING
869  status = px::proto::IotConnectionStatusEnums::IOT_CONNECTING;
870  _client_state = CONNECTING;
871  }
872  else
873  {
874  // Get the latest status
875  status = check_connection_status( _client_handle );
876  }
877 
878  // Before checking on publishing regular data, check the Firmware Update status
879  if ( status == px::proto::IotConnectionStatusEnums::IOT_OPEN )
880  {
881  if ( !_iot_core->_fw_up_manager->handle_update_state() )
882  {
883  // The Firmware Update won't let us continue to publish data here, so exit
884  return;
885  }
886  }
887  else if ( status == px::proto::IotConnectionStatusEnums::IOT_CLOSING )
888  {
889  // Doing a quick exit to shutdown; don't proceed further here
890  return;
891  }
892 
893  if ( ( status == px::proto::IotConnectionStatusEnums::IOT_OPEN ) ||
894  ( status == px::proto::IotConnectionStatusEnums::IOT_CONNECTING ) ||
895  ( status == px::proto::IotConnectionStatusEnums::IOT_CS_OTHER_ERROR ) )
896  {
897  px::CWallTime new_realtime;
898  switch ( _client_state )
899  {
900  case SEND_ALL:
901  SVR_INFO( 5, "IoT device client state: SEND_ALL" );
902  if ( build_send_device_tree() )
903  {
904  _client_state = SEND_REALTIMES;
905  }
906  break;
907  case SEND_DEV_TREE:
908  SVR_INFO( 5, "IoT device client state: SEND_DEV_TREE" );
909  if ( build_send_device_tree() )
910  {
911  // Only advance if that succeeded, else retry next time.
912  publish_device_realtime_state();// Send a Real-Time state update as well
913  _client_state = SEND_CH_RT_VALUES;
914  }
915  break;
916  case SEND_REALTIMES:
917  // Send the Device States now (just for the Publisher device), then next send RTs and Trends
918  SVR_INFO( 5, "IoT device client state: SEND_REALTIMES" );
919  publish_device_realtime_state();
920  _client_state = SEND_CH_RT_TRENDS;
921  break;
922  case SEND_DEV_RT_STATES:
923  SVR_INFO( 5, "IoT device client state: SEND_DEV_RT_STATES" );
924  publish_device_realtime_state();
925  _client_state = SEND_CH_RT_VALUES;
926  break;
927  case SEND_CH_RT_TRENDS:
928  {
929  SVR_INFO( 5, "IoT device client state: SEND_CH_RT_TRENDS" );
930  // Publish all R-T channel values
931  /* The time "now" before publishing, to use as _last_realtime next time
932  * There is a small possible overlap in published data, but this is
933  * better than missing data that falls in the time between "right now" and
934  * "now after publishing". */
935  new_realtime = px::CWallTime::now();
936  publish_channel_realtime_values( true, _last_realtime );
937  _last_realtime = new_realtime;
938 
939  // Publish the current Trend Values from each device (if they are available)
940  trend_manager.publish_channel_trend_values( realtime_cadence );
941 
942  // Adding a defensive measure; if we get stuck in an uncompleted FirmwareUpdate,
943  // detect that condition and clear it here so another can be run sometime later
944  if ( strlen( _iot_core->_cmd_handler->get_cmd_uuid() ) > 2 )
945  {
946  SVR_NOTICE( 2, "Clearing leftover Device Command (%s) metadata.", _iot_core->_cmd_handler->get_cmd_uuid() );
947  _iot_core->_cmd_handler->clear_cmd_params();
948  }
949  // Leave the timeout value alone
950  _client_state = SEND_CH_RT_VALUES;
951 
952  // And update our count of Messages Confirmed (write out the OOB)
953  update_msg_count();
954  }
955  break;
956 
957  case SEND_CH_RT_VALUES:
958  default:
959  {
960  //SVR_NOTICE( 2, "IoT device client state: Default/SEND_CH_RT_VALUES" );
961 
962  // Before we send, check that we have been getting confirmation responses.
963  // Otherwise, we have a failed connection attempt and it will have launched net_diagnostics.
964  if ( !check_confirmed_msg_count() )
965  {
966  SVR_NOTICE( 2, "Will sleep %d min and try again.", ( _RETRY_TIMER_MSECS / 60000 ) );
967  next_msec_timeout = _RETRY_TIMER_MSECS;
968  break; // We are closed; don't send any updates this time
969  }
970 
971  bool do_all_realtimes, do_changed_realtimes, do_trends, do_blob_upload = false;
972  std::tie( do_all_realtimes, do_changed_realtimes, do_trends, do_blob_upload ) = check_for_publish_times( tnow_secs );
973 #ifndef DONT_USE_UPLOADTOBLOB
974  // If _last_blob_upload_check_time is invalid, say at the start of IoT Agent, set it to current time.
975  if ( !_last_blob_upload_check_time.valid() )
976  {
977  _last_blob_upload_check_time = px::CWallTime::now();
978  }
979 #endif
980 
981  // Publish the realtime channel values once in this interval
982  if ( do_all_realtimes )
983  {
984  // Publish all R-T channel values
985 #if CONFIG_RT_AS_TREND_ACTUALS == y
986  publish_channel_realtimes_as_trend_values();
987  sent_all_realtimes_this_interval = true;
988 #else
989  // @ToDo: Do we want to send all RTs every 5 minutes, but not if CONFIG_RT_AS_TREND_ACTUALS?
990  // Does this make sense? Has everyone agreed to this?
991  // We want to publish realtime channel values once, since the specified trend interval has been reached.
992  publish_channel_realtime_values( true, _last_realtime );
993  _last_realtime = px::CWallTime::now();
994  sent_all_realtimes_this_interval = true;
995 #endif
996 
997 #ifdef HAVE_ROUTED_CH_LIST
998  // If the feature is enabled, in addition maybe send the routed trend messages
999  publish_channel_realtimes_as_trend_values( true );
1000 #endif
1001  }
1002 
1003  // Publish only if the cadence configured is reached
1004  // Always Publish just R-T channel values that have been updated since the _last_realtime timestamp
1005  if ( do_changed_realtimes )
1006  {
1007  publish_channel_realtime_values( false, _last_realtime );
1008  _last_realtime = px::CWallTime::now();
1009  }
1010 
1011  // Publish the Trend updates once every 5 minutes if the cadence is less than 5 mins
1012  // else sync with the cadence interval
1013  if ( do_trends )
1014  {
1015  // Take this opportunity to see, if we were running a trend catchup thread, whether it has completed.
1016  if ( trend_manager.is_catchup_thread_launched() && trend_manager.check_is_done() )
1017  {
1018  enum TrendManager::catchup_results result_code;
1019  const char *result_msg;
1020  std::tie( result_code, result_msg ) = trend_manager.get_results();
1021  if ( result_code == TrendManager::DONE_SUCCESS )
1022  {
1023  SVR_INFO( 2, "Successfully completed thread for trend catchup task: %s", result_msg );
1024  }
1025  else
1026  {
1027  SVR_ERR( 0, "Failed to complete thread for trend catchup task; result code = %u: %s", result_code, result_msg );
1028  }
1029  }
1030  // As a heartbeat (especially important if no RTs or Trends are available to publish),
1031  // send the Device States (for just the publisher). Send this before trends for batching reasons.
1032  publish_device_realtime_state();
1033  // Publish the latest Channel Trend values
1034  // It has been noted that the trend data may be incomplete if cadence is less than 5 mins
1035  // this soon after the trend interval is supposed to have closed, since it depends on data updates.
1036  trend_manager.publish_channel_trend_values( realtime_cadence );
1037  sent_trends_this_interval = true;
1038  // And update our count of Messages Confirmed (write out the OOB)
1039  update_msg_count();
1040  }
1041 #ifndef DONT_USE_UPLOADTOBLOB
1042  if ( do_blob_upload )

1599  * callback execution, will not appear or be logged.

1701  error_reason = "Error while sending to IoT Hub";
1702  // Close this connection and do a quick retry to connect
1703  _client_state = DO_RECONNECT;
1704  break;
1705  // case IOTHUB_CLIENT_CONNECTION_OK: - already handled
1706  case IOTHUB_CLIENT_CONNECTION_NO_NETWORK:
1707  default:
1708  // Since we don't have a specific error, run the net_diagnostic tests
1709  // to see if we can determine a specific problem
1710  if ( reasonCode == IOTHUB_CLIENT_CONNECTION_NO_NETWORK )
1711  {
1712  SVR_INFO( 2, "Starting net_diag_tests() due to IOTHUB_CLIENT_CONNECTION_NO_NETWORK." );
1713  }
1714  else
1715  {
1716  SVR_NOTICE( 2, "Starting Net Diagnostic tests due to non-specific IoT Error reasonCode %u.", reasonCode );
1717  }
1718  _client_state = _iot_core->_diag->launch_net_diagnostics();
1719  break;
1720  }
1721  // Set the status we will update to.
1722  // Don't allow a fatal result IFF we have successfully connected - must be some recoverable error in that case.
1723  if ( fatal && !_iot_core->_oob_handler->_ever_connected )
1724  {
1725  status = px::proto::IotConnectionStatusEnums::IOT_CONNECTION_FAILURE;
1726  // We can't throw in this callback, but set the client state that will do so when back in ASIO control
1727  _client_state = DO_EXIT;
1728  }
1729  else
1730  {
1731  // Check if firmware update is in progress and cancel if yes.
1732  if ( strlen( _iot_core->_cmd_handler->get_cmd_uuid() ) > 2 )
1733  {
1734  std::string text( error_reason );
1735  text = "Cancelling firmware update due to " + text;
1736  SVR_NOTICE( 2, "%s", text.c_str() );
1737  _iot_core->_cmd_handler->complete_firmware_update( CEE_STS_cancel, text.c_str() );
1738  _iot_core->_cmd_handler->clear_cmd_params();
1739  }
1740  status = px::proto::IotConnectionStatusEnums::IOT_CLOSED;
1741  }
1742 
1743  // Update the IoT Status OOB
1744  _iot_core->_oob_handler->set_oob_status( status, error_state, error_reason );
1745 }

1886 // Determines the current connection status via the Eaton Device SDK.
1887 px::proto::IotConnectionStatusEnums CIotDeviceClient::check_connection_status( void *device_handle )
1888 {
1889  PUSH_FUNC;
1890 
1891  px::proto::IotConnectionStatusEnums status = px::proto::IotConnectionStatusEnums::IOT_CS_UNKNOWN;
1892  px::proto::IotAgentErrorStateEnums error_code = px::proto::IotAgentErrorStateEnums::IOT_UNKNOWN;
1893  px::proto::IotConnectionStatusEnums last_agent_status = _iot_core->_oob_handler->get_agent_status();
1894  px::proto::IotAgentErrorStateEnums last_error_state = _iot_core->_oob_handler->get_connection_error_state();
1895  const char *error_reason = "Connection status Unknown";
1896  IOT_CONNECTION_STATUS conn_stat_struct; // this is a struct
1897  bool valid_status = false;
1898  bool update_traffic_count = false;
1899  // First, make sure we don't need to shutdown now, before proceeding
1900  if ( _iot_core->need_to_stop() )
1901  {
1902  return ( px::proto::IotConnectionStatusEnums::IOT_CLOSING );
1903  }
1904  if ( device_handle != nullptr )
1905  {
1906  valid_status = iot_getStatus( ( IOT_DEVICE_HANDLE ) device_handle, &conn_stat_struct );
1907  }
1908  if ( valid_status )
1909  {
1910  // conn_stat_struct is valid; map to our status type
1911  error_code = sdk_to_protobuf_err_code( conn_stat_struct.lastErrorCode, &error_reason );
1912 
1913  switch ( conn_stat_struct.connectionStatus )
1914  {
1915  case IOT_OPENED:
1916  // Just because the IoT SDK tells us we're "open" doesn't mean it's so,
1917  // until we get at least one confirmed message
1918  if ( last_agent_status == px::proto::IotConnectionStatusEnums::IOT_CONNECTING )
1919  {
1920  // When we call the SDK's initializeConnection(), before iot_open(), it zeroes out msgConfirmed
1921  if ( conn_stat_struct.msgConfirmed == 0 )
1922  {
1923  // No change, still connecting
1924  status = px::proto::IotConnectionStatusEnums::IOT_CONNECTING;
1925  update_traffic_count = true;
1926  }
1927  else
1928  {
1929  // Show the "happy" result
1930  is_now_connected( ( IOT_DEVICE_HANDLE ) device_handle );
1931  // Set up the fields for the OOB update
1932  status = px::proto::IotConnectionStatusEnums::IOT_OPEN;
1933  error_reason = "";
1934  update_traffic_count = true;
1935  }
1936  }
1937  else if ( conn_stat_struct.msgConfirmed > 0 )
1938  {
1939  status = px::proto::IotConnectionStatusEnums::IOT_OPEN;
1940  update_traffic_count = true;
1941  }
1942  // else no change; nothing's really happened yet.
1943  break;
1944  case IOT_CLOSING:
1945  status = px::proto::IotConnectionStatusEnums::IOT_CLOSING;
1946  break;
1947  case IOT_CLOSED:
1948  status = px::proto::IotConnectionStatusEnums::IOT_CLOSED;
1949  break;
1950  case IOT_CONNECTING:
1951  status = px::proto::IotConnectionStatusEnums::IOT_CONNECTING;
1952  break;
1953  case IOT_OTHER:
1954  status = px::proto::IotConnectionStatusEnums::IOT_CS_OTHER_ERROR;
1955  break;
1956  case IOT_UNKNOWN:
1957  default:
1958  status = px::proto::IotConnectionStatusEnums::IOT_CS_UNKNOWN;
1959  break;
1960  }
1961  }
1962  // Now see if we need to update
1963  if ( update_traffic_count )
1964  {
1965  _iot_core->_oob_handler->update_traffic_count( ( uint32_t ) conn_stat_struct.msgConfirmed );
1966  }
1967  if ( ( status != last_agent_status ) || ( error_code != last_error_state ) )
1968  {
1969  _iot_core->_oob_handler->set_oob_status( status, error_code, error_reason );
1970  }
1971  return ( status );
1972 }

1974 void CIotDeviceClient::is_now_connected( IOT_DEVICE_HANDLE device_handle )
1975 {
1976  _is_connected = true;
1977  time_t tnow = px::CWallTime::now().tv_sec; // time in UTC, ignore tv_nsec portion
1978  // Setup the info for Audit logging. Provide no username or IP address; set the appname, however
1979  toolkit::CCeeLogMsg cee_info( CEE_ACT_connect, CEE_STS_success, "", "", "IoT Agent" );
1980  cee_info._hostname = _iot_core->_oob_handler->get_hosted_uri();
1981  SVR_INFO( 2, "Success: Connected to %s", _iot_core->_oob_handler->get_hosted_uri().c_str() );
1982  AUDIT_LOG_MSG( AUDIT_LOG_SESSION, LOG_INFO, cee_info, "%s at [%ld] epoch secs", CONNECTION_OPEN_MSG, tnow );
1983 
1984  // Enable batching, and use the configured DoWork interval as the batching interval (for AMQP)
1985  // Note that this means that we had no batching before this point, which is the desired operation,
1986  // so we can see the initial message confirmations as soon as possible
1987  // (which is how we get to this spot in the first place).
1988  iot_set_enable_batching( device_handle, true, DO_WORK_INTERVAL_SECS );
1989 
1990 #ifndef DONT_USE_UPLOADTOBLOB
1991  // After connecting, first check the blob upload catch up
1992  SVR_DEBUG( 8, "Check if catch up of blob uploads is required." );
1993  blob_upload_manager.init();
1994 #endif
1995 
1996  uint32_t trend_cadence = realtime_cadence;
1997  if ( realtime_cadence < 300 )
1998  {
1999  trend_cadence = 300; // 5 minute trends, our default
2000  }
2001 
2002  /* We want to handle the following cases for determing whether to add a catch-up span
2003  * (defined as a pair of "Begin" and "End" times).
2004  * We also have to handle the normal sequence where on first connect, we have used the SEND_CH_RT_TRENDS state
2005  * and published an initial trend update, which can't count in this determination of prior span; in that case,
2006  * we have to rely on the "previous" published trend interval time.
2007  *
2008  * Sequence: Powerup, empty OOB, last_trend_time read as 0, ...
2009  * Signature: last_trend_time = 0
2010  * Action: No catchup span added
2011  *
2012  * Sequence: Powerup (with previous time == 0), last_trend_time read as XX, have published new trend data (awaiting time = YY)
2013  * Signature: previous time = 0, last_trend_time = XX, awaiting time = YY
2014  * Action: Begin = (last_trend_time = XX) + cadence, End = (awaiting time = YY) - cadence
2015  * Note: If was a short power outage or process restart, XX may == YY and Begin may be > End and thus no span will be added.
2016  *
2017  * Sequence: Powerup, have published new trend data at YY and been acknowledged, previous time read as XX, last_trend_time read as YY, not awaiting time
2018  * Signature: previous time = XX, last_trend_time = YY = now (normalized to prior interval close time), not awaiting
2019  * Action: Begin = (previous time = XX) + cadence, End = (last_trend_time = normalized time = YY) - cadence
2020  * Note: If XX + cadence == YY, then Begin will be > End and thus no span will be added, since there is no gap.
2021  *
2022  * Sequence: Have been running, previous time == XX - cadence, last_trend_time = XX, long outage (> cadence),
2023  * reconnected, have published new trend data (awaiting time = YY)
2024  * Signature: previous time = XX - cadence, last_trend_time = XX, awaiting time = YY
2025  * Action: Begin = (last_trend_time = XX) + cadence, End = (awaiting time = YY) - cadence
2026  *
2027  * Sequence: Have been running, previous time = (XX < normalized time - cadence), last_trend_time = YY, long outage (> cadence),
2028  * reconnected, have published new trend data at normalized time = YY and been acknowledged already (not awaiting)
2029  * Signature: previous time = XX, last_trend_time = YY, not awaiting
2030  * Action: Begin = (previous time = XX) + cadence, End = (last_trend_time = normalized time = YY) - cadence
2031  *
2032  * Sequence: Have been running (ie, previous time == YY - cadence), last_trend_time = YY, short outage, reconnected, no new trend data (not awaiting)
2033  * Signature: previous time = YY - cadence, last_trend_time = YY, not awaiting trends, now (normalized to prior interval close time) == YY
2034  * Action: No catchup span added
2035  * Note: This is the same result as Begin = XX + cadence, End = now (normalized to prior interval close time) - cadence
2036  *
2037  * Reality check: if Begin > End, or Begin > normalized time, do nothing (no new catchup span)
2038  *
2039  * Solution:
2040  * 1) If last_trend_time == 0 : No catchup (no history)
2041  * 2) If awaiting : Set End = awaiting time - cadence; Else : End = normalized time - cadence
2042  * 3) If previous time = 0 : Begin = last_trend_time + cadence;
2043  * 3b) Else if previous time + cadence == last_trend_time : Begin = last_trend_time + cadence; Else : Begin = previous time + cadence
2044  * 4) If Begin > End : do nothing (no new catchup span)
2045  */
2046 
2047  time_t previous_trend_time = _iot_core->_oob_handler->catchup_oob_handler.get_previous_trend_published_time();
2048  time_t last_trend_time = _iot_core->_oob_handler->catchup_oob_handler.get_last_trend_published_time();
2049  time_t normalized_trend_time = ( tnow / trend_cadence ) * trend_cadence; // when we would have just published trend data
2050 
2051  if ( last_trend_time == 0 )
2052  {
2053  SVR_NOTICE( 2, "No history, so no trend catchup." );
2054  }
2055  else
2056  {
2057  time_t end_time;
2058  if ( !trend_manager.check_all_trend_msgs_acknowledged() )
2059  {
2060  end_time = trend_manager.get_awaiting_ack_timestamp() - trend_cadence;
2061  }
2062  else
2063  {
2064  end_time = normalized_trend_time - trend_cadence;
2065  }
2066  time_t begin_time = last_trend_time + trend_cadence;
2067  if ( ( previous_trend_time != 0 ) && ( ( previous_trend_time + ( time_t ) trend_cadence ) < last_trend_time ) )
2068  {
2069  // This is the case where last_trend_time is probably == normalized_trend_time and we have published upon reconnect
2070  begin_time = previous_trend_time + trend_cadence;
2071  }
2072 
2073  if ( begin_time > end_time )
2074  {
2075  SVR_INFO( 2, "No gap in history, so no trend catchup." );
2076  }
2077  else
2078  {
2079  _iot_core->_oob_handler->catchup_oob_handler.add_trend_catchup_span( begin_time, end_time );
2080  }
2081  }
2082 
2083  // Now determine if we need to launch the trend catchup thread / task:
2084  // if we haven't done so already, and if there is at least one interval needing catchup
2085  if ( !trend_manager.is_catchup_thread_launched() &&
