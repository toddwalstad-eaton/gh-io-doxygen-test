122 void CIotAgentCdsHandler::handle_response( cds::response_ptr_t resp )
123 {
124  PUSH_FUNC;
125 
126  // Something bad happened if we can't get to our core
127  assert( _iot_core );
128 
129  SVR_DEBUG( 9, "-------------- CIotAgentCdsHandle::handle_response begin" );
130 
131  if ( ( !resp->is_subscription() ) && ( !resp->is_unsolicited() ) )
132  {
133  SVR_NOTICE( 6, "Unexpected CDS response published to us" );
134  //SVR_DEBUG(6, "Handle CDS request response end");
135  }
136  else if ( resp->is_unsolicited() )
137  {
138  // SVR_DEBUG(6, "Handle CDS unsolicited response begin");
139  SVR_NOTICE( 6, "Unexpected CDS unsolicited message published to us" );
140  }
141  /* Else was something that was subscribed to.
142  * Maybe it was something in the base CdsHandler, but that we are not directly interested in
143  * But probably it is our ImageUpdate OOB. As long as our functor has been set.
144  */
145  else if ( _fn_oob_update )
146  {
147  std::shared_ptr<px::cds::CSubscriberNotify> sub_notify =
148  std::dynamic_pointer_cast<px::cds::CSubscriberNotify>( resp );
149  // If that worked, and was our subscription ID, and as long as the event type is
150  if ( sub_notify && ( sub_notify->event == px::sub::subscription_type::oob_data ) )
151  {
152  if ( sub_notify->subscriber_id == _oob_sub_id )
153  {
154  // Now see if the expected OOB is in what was given to us as changed.
155  // (Thus we ignore if it was in the cleared list.)
156  if ( sub_notify->oobs.find( _image_update_tag ) != sub_notify->oobs.end() )
157  {
158  px::proto::toolkitStruct tk_struc = sub_notify->oobs[_image_update_tag];
159  // Try to extract our OOB structure
160  if ( tk_struc.has_image_update_results() )
161  {
162  // Finally! As long as that succeeded, call the callback
163  SVR_DEBUG( 7, "Calling handler for %s OOB", _image_update_tag.c_str() );
164  _fn_oob_update( tk_struc );
165  }
166  else
167  {
168  SVR_NOTICE( 2, "Got OOB update for %s, but not image_update_results!!", _image_update_tag.c_str() );
169  }
170  }
171  else
172  {
173  SVR_NOTICE( 2, "Subscribed OOB update, but not for %s!!", _image_update_tag.c_str() );
174  }
175  }
176 #ifndef DONT_USE_UPLOADTOBLOB
177  else if ( sub_notify->subscriber_id == _blob_upld_oob_sub_id )
178  {
179  // Now see if the expected OOB is in what was given to us as changed.
180  // (Thus we ignore if it was in the cleared list.)
181  for ( auto it = sub_notify->oobs.begin(); it != sub_notify->oobs.end(); ++it )
182  {
183  if ( it->first.find( BLOB_UPLOAD_TAG_BASE ) != std::string::npos )
184  {
185  _blob_upload_tag = it->first;
186  break;
187  }
188  }
189  if ( !_blob_upload_tag.empty() )
190  {
191  px::proto::toolkitStruct tk_struc = sub_notify->oobs[_blob_upload_tag];
192  // Try to extract our OOB structure
193  if ( tk_struc.has_iot_blob_upload() )
194  {
195  // Finally! As long as that succeeded, call the callback
196  SVR_DEBUG( 7, "Calling handler for %s OOB", _blob_upload_tag.c_str() );
197  _fn_oob_update( tk_struc );
198  }
199  else
200  {
201  SVR_NOTICE( 2, "Got OOB update for %s, but not iot blob upload!!", _image_update_tag.c_str() );
202  }
203  }
204  else
205  {
206  SVR_NOTICE( 2, "Subscribed OOB update, but %s not found!!", _blob_upload_tag.c_str() );
207  }
208  }
209 #endif
210  else
211  {
212  SVR_INFO( 9, "Subscription update received with id %u but not for us.", ( uint32_t ) sub_notify->subscriber_id );
213  }
214  }
215  else
216  {
217  SVR_INFO( 9, "Subscription notification but not for OOB data." );
218  }
219  }
220  else
221  {
222  /* Here either because this is "other" data (especially the frequent Channel Value updates),
223  * or else because there is No handler for OOB updates. */
224  }
225 
226  SVR_DEBUG( 9, "-------------- CIotAgentCdsHandle::handle_response end" );
227 }
