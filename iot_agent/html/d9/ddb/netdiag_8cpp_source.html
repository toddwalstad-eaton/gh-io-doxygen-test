96 {
97  try
98  {
99  SVR_DEBUG( 4, "Trying TCP connection to %s port %hu", ip_addr.c_str(), port );
100  // Try to create a socket connection to the target address
101  struct sockaddr_in target_addr;
102  memset( &target_addr, 0, sizeof( target_addr ) );
103  target_addr.sin_family = AF_INET;
104  target_addr.sin_port = htons( port );
105  target_addr.sin_addr.s_addr = inet_addr( ip_addr.c_str() );
106  if ( unlikely( INADDR_NONE == target_addr.sin_addr.s_addr ) )
107  {
108  SVR_WARN( 2, "Bad IP address in TCP test: %s", ip_addr.c_str() );
109  return ( false );
110  }
111 
112  int sockfd;
113  /* Create the socket "manually" */
114  sockfd = socket( AF_INET, SOCK_STREAM, 0 );
115  if ( unlikely( sockfd == -1 ) )
116  {
117  SVR_WARN( 2, "Error creating listening socket for TCP test." );
118  return ( false );
119  }
120 
121  // To avoid long timeouts, make this non-blocking, and use select()'s timeout
122  if ( fcntl( sockfd, F_SETFL, O_NONBLOCK ) == -1 )
123  {
124  int errnum = errno;
125  close( sockfd );
126  SVR_WARN( 2, "Socket initialization failed, error: %s", strerror( errnum ) );
127  return ( false );
128  }
129 
130  fd_set read_set;
131  FD_ZERO( &read_set );
132  FD_SET( sockfd, &read_set );
133 
134  if ( connect( sockfd, ( struct sockaddr * ) &target_addr, sizeof( target_addr ) ) == -1 )
135  {
136  int errnum = errno;
137  // Since we are now non-blocking, the normal case here is to have gotten EINPROGRESS
138  if ( errnum == EINPROGRESS )
139  {
140  struct timeval timeout;
141  timeout.tv_sec = CONNECT_TIMEOUT_SECS;
142  timeout.tv_usec = 0;
143  errnum = 0;
144  int selection = select( sockfd + 1, &read_set, NULL, NULL, &timeout );
145  if ( selection == -1 )
146  {
147  errnum = errno;
148  }
149  else if ( selection == 0 )
150  {
151  // Then the select operation timed out
152  errnum = ETIMEDOUT;
153  }
154  }
155  if ( errnum != 0 )
156  {
157  close( sockfd );
158  SVR_WARN( 2, "TCP Connection failed, error: %s", strerror( errnum ) );
159  return ( false );
160  }
161  }
162 
163  // else
164  SVR_DEBUG( 4, "TCP Connection succeeded, closing" );
165  close( sockfd );
166  return ( true );
167  }
168  catch ( const std::exception &e )
169  {
170  SVR_ERR( 2, "Exception trying TCP connection: %s", e.what() );
171  return ( false );
172  }

653  return ( std::make_tuple( px::proto::IotAgentErrorStateEnums::IOT_HOST_DOWN, "IoT Cloud Service unreachable", false ) );
654  }

657  }

675  }
