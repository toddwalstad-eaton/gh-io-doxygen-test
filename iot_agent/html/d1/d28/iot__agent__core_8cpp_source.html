245 void CIotAgentCore::execute()
246 {
247  PUSH_FUNC;
248  assert( _options ); // we've had problems long before now if this isn't valid!
249  assert( _oob_handler.get() );
250 
251  px::proto::IotAgentErrorStateEnums closing_error_code =
252  px::proto::IotAgentErrorStateEnums::IOT_NO_ERROR;
253  px::string closing_reason;
254  const char *error_reason = nullptr;
255 
256  // First, see if we should even proceed:
257  if ( !_oob_handler->_agent_enabled )
258  {
259  return;
260  }
261 
262  bool bOK = false;
263 
264  try
265  {
266  _diag.reset( new CNetDiag( getptr() ) );
267 
268  /* Instantiate the Eaton IoT Device SDK Client: */
269  _iot_client.reset( new CIotDeviceClient( *( manager.get() ), getptr() ) );
270  assert( _iot_client.get() );
271 
272  // Initialize the endpoint
273  closing_error_code = _iot_client->initialize( &error_reason );
274  if ( closing_error_code != px::proto::IotAgentErrorStateEnums::IOT_NO_ERROR )
275  {
276  closing_reason = error_reason;
277  }
278  else
279  {
280  bOK = true;
281  }
282  }
283  catch ( const boost::system::system_error &e )
284  {
285  SVR_ERR( 0, "System error encountered in setting up I/O manager: %s", e.what() );
286  closing_error_code = px::proto::IotAgentErrorStateEnums::IOT_SYSTEM_ERROR;
287  closing_reason = e.what();
288  bOK = false; // ensure that we close out below
289  }
290 
291  if ( !bOK )
292  {
293  // Our attempt to open the network port(s) failed;
294  // Update status with the reason we are exiting
295  _oob_handler->set_oob_status( px::proto::IotConnectionStatusEnums::IOT_CLOSED,
296  closing_error_code, closing_reason.c_str() );
297  // Free up our manager and exit below (CDS disconnected once we reset the manager)
298  manager->quit();
299  manager.reset();
300  }
301 
302  if ( manager.get() && !manager->is_stopped() ) // if we created a valid IO Manager, and it hasn't been killed yet
303  {
304  // Create our update timer
305  _realtime_timer = px::asio::CMsecTimer::create( *( manager.get() ), this, _UPDATE_TIMER_MSECS );
306 
307  bool update_status_oob = true; // Usually we will have a new status to update with
308  bool got_started = false; // So we can tell initial failures from later ones.
309  px::proto::IotConnectionStatusEnums connection_status = px::proto::IotConnectionStatusEnums::IOT_CS_UNKNOWN;
310 
311  if ( bOK )
312  {
313  // Create the DeviceCommand and Firmware Update handlers
314  _cmd_handler.reset( new IotDeviceCommand( getptr() ) );
315  _fw_up_manager.reset( new FirmwareUpdateManager( getptr() ) );
316 
317  try
318  {
319  // We will be attempting to connect, if we get here:
320  _oob_handler->set_oob_status( px::proto::IotConnectionStatusEnums::IOT_CONNECTING,
321  px::proto::IotAgentErrorStateEnums::IOT_NO_ERROR, "Initializing..." );
322 
323  // We are ready to run and connect, (we believe) if we get here:
324  SVR_DEBUG( 2, "Entering the run() loop" );
325  // Update our count of Messages Confirmed so far
326  _iot_client->update_msg_count();
327  // Kick off our update timer
328  _realtime_timer->update_timeout( _UPDATE_TIMER_MSECS );
329 
330  // Signal to any waiting proceeses that the IoT Agent is ready for requests
331  ready();
332  got_started = true;
333  // Make sure we have the right client state to call for connect,
334  // since the device discovery makes a lot of attempts to set it to something else
335  _iot_client->request_client_state( CIotDeviceClient::client_states::CONNECTING );
336 
337  // This will block indefinitely, and do all the real work
338  manager->run();
339 
340  // If we get here, give this connection status and reason
341  connection_status = px::proto::IotConnectionStatusEnums::IOT_CLOSED;
342  closing_reason = "Exiting iot_agent now.";
343  }
344  catch ( const boost::system::system_error &e )
345  {
346  SVR_ERR( 0, "System error encountered in I/O manager: %s", e.what() );
347  if ( got_started )
348  {
349  connection_status = px::proto::IotConnectionStatusEnums::IOT_CLOSED;
350  }
351  else
352  {
353  connection_status = px::proto::IotConnectionStatusEnums::IOT_CONNECTION_FAILURE;
354  }
355  closing_error_code = px::proto::IotAgentErrorStateEnums::IOT_SYSTEM_ERROR;
356  closing_reason = e.what();
357  }
358  catch ( const px::generic_err &ex )
359  {
360  // Probably a connection error
361  SVR_ERR( 0, "IoT Connection failed: %s", ex.what() );
362  closing_reason = ex.what();
363  // In this case, we have already set the Status OOB the way we want it
364  update_status_oob = false;
365  }
366  catch ( const std::exception &e )
367  {
368  SVR_ERR( 0, "System error encountered: %s", e.what() );
369  if ( got_started )
370  {
371  connection_status = px::proto::IotConnectionStatusEnums::IOT_CLOSED;
372  }
373  else
374  {
375  connection_status = px::proto::IotConnectionStatusEnums::IOT_CONNECTION_FAILURE;
376  }
377  closing_error_code = px::proto::IotAgentErrorStateEnums::IOT_SYSTEM_ERROR;
378  closing_reason = e.what();
379  }
380  }
381 
382  SVR_NOTICE( 1, "Exiting the %s run() loop: %s", COMMON_APP_NAME, closing_reason.c_str() );
383  // Cancel our timer
384  _realtime_timer->update_timeout( 0 );
385  sleep( 1 ); // let it take effect
386 
387  if ( update_status_oob )
388  {
389  // Update status with the reason we are exiting - before we lose our CDS login
390  _oob_handler->set_oob_status( connection_status, closing_error_code, closing_reason.c_str() );
391  }
392 
393  // Closeout, if we haven't already, and free our resources.
394  shutdown();
395  // When closing, the manager will also send a message to clear the watchdog
396  manager->quit();
397  manager.reset();
398  // Now we can close this, since all message to the CDS have been sent.
399  _cds_handler.reset();
400  }

515 // _Very_ similar to main's sdk_custom_logger, but without using va_list as an argument
516 void iot_custom_logger( enum IOT_LOG_LEVELS level, const char *funcName, const char *msg_format, ... )
517 {
518  #define MAX_LOG_MSG_LEN 512
519  char msg_text[MAX_LOG_MSG_LEN];
520  va_list vargs;
521  va_start( vargs, msg_format );
522  vsnprintf( msg_text, MAX_LOG_MSG_LEN, msg_format, vargs );
523  va_end( vargs );
524 
525  // Sometimes called from threads that are spawned just to do one call,
526  // so the _thr_local_syslog has to be re-established for that transient thread
527  if ( _thr_local_syslog.get() == nullptr )
528  {
529  _thr_local_syslog.reset( new toolkit::CSysLog() );
530  /* Set the verbosity level to that for the IoT SDK */
531  if ( likely( _thr_local_syslog.get() ) )
532  {
533  _thr_local_syslog.get()->verbosity = _svr_verb_array[SVR_LOG_IOT];
534  }
535  /* Kick off syslogging with a message... below */
536  }
537 
538  // Get the level that is certain to be logged
539  unsigned int certain_log_level = _svr_verb_array[SVR_LOG_IOT];
540 
541  if ( likely( SYSLOG_PTR ) )
542  {
543  switch ( level )
544  {
545  case IOT_LOG_ERROR:
546  // Treat like SVR_LOG_FERR(0, ...) and always log
547  SYSLOG_PTR->err( 0, funcName, "%s", msg_text );
548  break;
549 
550  case IOT_LOG_INFO:
551  // Treat like SVR_LOG_FINFO(2, ...)
552  /* Was going to check against _svr_verb_array[SVR_LOG_IOT],
553  * but the fact that we get this message means that we
554  * specifically asked for INFO level messages, so always log it. */
555  SYSLOG_PTR->info( certain_log_level, funcName, "%s", msg_text );
556  break;
557 
558  case IOT_LOG_DEBUG:
559  default: // IOT_LOG_NONE pretty unlikely...
560  // Treat like SVR_LOG_FDEBUG(6, ...)
561  SYSLOG_PTR->debug( certain_log_level, funcName, "%s", msg_text );
