47 bool FirmwareUpdateManager::handle_update_state()
48 {
49  PUSH_FUNC;
50  bool ok_to_publish = true;
51 
52  switch ( current_state )
53  {
54  case SEND_TRANSFER_IMAGE:
55  SVR_INFO( 5, "IoT FirmwareUpdateManager state: SEND_TRANSFER_IMAGE" );
56  iot_core->_cmd_handler->send_device_to_cloud_command( TRANSFER_IMAGE );
57  // The IotDeviceCommand instance remembers what it needs to receive the file transfer
58  current_state = RECEIVING_IMAGE; // should start receiving, or else time out
59  timeout_count = 0;
60  {
61  // Quick fix: to keep from timing out (in the RECEIVE_TIMEOUT state) because of the pauses for batching,
62  // disable batching while in the firmware update process.
63  IOT_DEVICE_HANDLE device_handle = iot_core->_iot_client->_client_handle;
64  iot_set_enable_batching( device_handle, false, iot_core->_iot_client->DO_WORK_INTERVAL_SECS );
65  }
66  break;
67  case RECEIVING_IMAGE:
68  // Clear the _recvtimeout_count but set the state for next time to RECEIVE_TIMEOUT
69  // When we do receive an ImageTransfer message, it will reset the state to this one,
70  // thus preventing the timeout from occurring - normally.
71  timeout_count = 0;
72  current_state = RECEIVE_TIMEOUT;
73  break;
74  case RECEIVE_TIMEOUT:
75  {
76  if ( ++timeout_count >= MAX_RECV_TIMEOUT_COUNT )
77  {
78  SVR_NOTICE( 0, "Canceling FirmwareUpdate process of %s due to timeout while receiving ImageTransfers.",
79  iot_core->_cmd_handler->get_cmd_params_str( "name" ).c_str() );
80  iot_core->_cmd_handler->complete_firmware_update( CEE_STS_cancel,
81  "Timed out while receiving ImageTransfers." );
82  current_state = IDLE;
83  }
84  // else, just stay in this state.
85  }
86  break;
87  case START_IMAGE_UPDATE:
88  start_firmware_update(); // will set the next current_state to go to
89  timeout_count = 0;
90  // next_msec_timeout = _VERY_SHORT_TIMER_MSECS;
91  break;
92  case DOING_IMAGE_UPDATE:
93  {
94  // Here we are replicating the fgets() portion of shell_cmd(), for one line
95  if ( read_shell_line() )
96  {
97  // Leave current_state set to DOING_IMAGE_UPDATE to come back for more
98  }
99  else
100  {
101  // Reached EOF, so move to the next stage
102  current_state = FINISH_IMAGE_UPDATE;
103  }
104  // next_msec_timeout = _VERY_SHORT_TIMER_MSECS;
105  }
106  break;
107  case FINISH_IMAGE_UPDATE:
108  // Let our function handle it
109  finish_cmd_session();
110  break;
111  case IDLE:
112  default:
113  // Nothing to do
114  break;
115  }
116 
117  // Within this timer-driven function, see if our CdsHandler needs any Device or Channel data update requests
118  // to be sent
119  bool have_more_to_do = iot_core->_cds_handler->check_request_queue();
120 
121  /* Finally, if there are any devices that we need to create the ImageVersion OOB for,
122  * do that now for one device, and remove it from the set.
123  * But also now waiting until device change traffic has slowed down, else the version data
124  * may be incomplete. */
125  if ( ok_to_publish && !have_more_to_do && !request_image_version_list.empty() )
126  {
127  px::id_t dev_id = request_image_version_list.front(); // Copy the first Device ID
128  // Now remove that entry from the set
129  request_image_version_list.pop_front();
130  SVR_DEBUG( 6, "Processing (perhaps) one of the queued devices (%u) for ImageVersion", ( uint32_t ) dev_id );
131  // SVR_DEBUG(1, "%s: Client state = %u", px::CWallTime::now().localtime_str().c_str(), current_state );
132  std::shared_ptr<CIotDataDevice> iot_dev_ptr = iot_core->_iot_client->lookup_iot_data_device( dev_id );
133  if ( iot_dev_ptr )
134  {
135  SVR_INFO( 5, "Creating/Updating ImageVersion OOB for device %u", ( uint32_t ) dev_id );
136  bool success = iot_dev_ptr->update_image_version_oob();
137  if ( !success )
138  {
139  // Well that failed; requeue this device to try again sometime later
140  queue_request_image_version_update( dev_id );
141  }
142  }
143  // Else, since this device isn't our our IoT Device Map, we'll just drop it from this request queue
144  }
145 
146  return ( ok_to_publish );
147 }

282 // Function to read one line of input from the shell that is executing
283 bool FirmwareUpdateManager::read_shell_line()
284 {
285  PUSH_FUNC;
286 
287  bool got_one = false;
288  // Here we are replicating the fgets() portion of shell_cmd(), for one line
289  try
290  {
291  char cmd_output[CMD_OUTPUT_MAX];
292  memset( cmd_output, 0, CMD_OUTPUT_MAX );
293  // See if fgets() gets a line or nothing (ie, EOF)
294  if ( ( shell_pipe != nullptr ) && ( fgets( cmd_output, CMD_OUTPUT_MAX, shell_pipe ) != NULL ) )
295  {
296  // Strip any trailing newline - this is the recommended, "never fails" approach
297  cmd_output[strcspn( cmd_output, "\n" )] = 0;
298  SVR_INFO( 5, "Shell cmd output line: %s", cmd_output );
299  std::string out( cmd_output );
300  if ( !out.empty() )
301  {
302  shell_outlines.push_back( out );
303  }
304  got_one = true; // even if was an empty line
305  }
306  else
307  {
308  // Reached EOF
309  }
310  }
311  catch ( ... )
312  {
313  SVR_ERR( 0, "Caught exception while reading shell output." );
314  shell_outlines.push_back( std::string( "Caught exception while reading shell output." ) );
315  }
316  return ( got_one );
317 }

319 // Function to close out a Firmware Update shell session and clean up.
320 void FirmwareUpdateManager::finish_cmd_session()
321 {
322  PUSH_FUNC;
323 
324  int cmd_status = -1; // default is failure.
325  px::string image_type = iot_core->_cmd_handler->get_cmd_params_str( IMAGE_TYPE_TAG );
326  // Get the Device ID of the current device
327  px::id_t dev_id = iot_core->_cmd_handler->get_cmd_device();
328  // Finish off like shell_cmd()
329  if ( shell_pipe != nullptr )
330  {
331  // First, finish reading any input
332  while ( read_shell_line() )
333  { }
334  cmd_status = pclose( shell_pipe );
335  shell_pipe = nullptr;
336  if ( cmd_status == -1 )
337  {
338  /* Error reported by pclose(), or previous state */
339  SVR_NOTICE( 0, "Failure while ending Firmware Update." );
340  }
341  }
342  if ( cmd_status != -1 )
343  {
344  /* Use macros described under wait() to inspect `status' in order
345  to determine success/failure of command executed by popen() */
346  if ( WIFEXITED( cmd_status ) )
347  {
348  uint8_t exit_code = WEXITSTATUS( cmd_status );
349  if ( exit_code == 0 )
350  {
351  // Yes - the program gave us a thumbs up! (Success)
352  px::string result_msg( "Completed update of " );
353  result_msg += image_type;
354  result_msg += " for Device ";
355  result_msg += std::to_string( dev_id ).c_str();
356  SVR_INFO( 0, "%s", result_msg.c_str() );
357  // iot_core->_cmd_handler->complete_firmware_update( CEE_STS_success, result_msg.c_str() );
358  // complete_firmware_update() should have been already called
359 
360  // So now advance to the next device, if any, else close out
361  dev_id = iot_core->_cmd_handler->get_cmd_device( true ); // advance_to_next = true
362  if ( dev_id > 0 )
363  {
364  SVR_INFO( 5, "Advancing to update the next device: %u", ( uint32_t ) dev_id );
365  current_state = START_IMAGE_UPDATE;
366  }
367  else
368  {
369  SVR_INFO( 5, "Done updating all devices." );
370  iot_core->_cmd_handler->clear_cmd_params();
371  current_state = IDLE; // back to normal now
372  }
373  return;
374  }
375  else
376  {
377  // Some failure in doing update, but done with that failure
378  SVR_WARN( 1, "Firmware Update app returned exit code %u", exit_code );
379  // Continue below now...
380  }
381  }
382  }
383 
384  px::string result_msg( "Failed update of " );
385  result_msg += image_type;
386  result_msg += " for Device ";
387  result_msg += std::to_string( dev_id ).c_str();
388  if ( !shell_outlines.empty() )
389  {
390  std::string last_line;
391  result_msg += ": ";
392  for ( const std::string &line : shell_outlines )
393  {
394  SVR_NOTICE( 2, "FwUp Cmd out: %s", line.c_str() );
395  last_line = line;
396  result_msg += last_line.c_str();
397  }
398  }
399  SVR_NOTICE( 0, "Final FwUp result: %s", result_msg.c_str() );
400  if ( cmd_status != -1 )
401  {
402  // Then we had a controlled update failure, but may have more devices to process
403  // Have already called complete_firmware_update() from handle_img_up_status()
404  // So now advance to the next device, if any, else close out
405  dev_id = iot_core->_cmd_handler->get_cmd_device( true ); // advance_to_next = true
406  if ( dev_id > 0 )
407  {
408  SVR_INFO( 5, "Advancing to update the next device: %u", ( uint32_t ) dev_id );
409  current_state = START_IMAGE_UPDATE;
410  }
411  else
412  {
413  SVR_INFO( 5, "Done updating all devices." );
414  iot_core->_cmd_handler->clear_cmd_params();
415  current_state = IDLE; // back to normal now
416  }
417  }
418  else
419  {
420  // We're here because we failed for some system reason, so don't try to do more
421  iot_core->_cmd_handler->clear_cmd_params();
422  iot_core->_cmd_handler->complete_firmware_update( CEE_STS_failure, result_msg.c_str() );
423  // That set the next current_state, so we can exit now
424  }
425 }
