711 // Reads in the IotDeviceSettings OOB and maps its data into the given settings_map.
712 bool CIotOobHandler::read_iot_device_settings_oob( gdse_device_settings_map_t &settings_map )
713 {
714  PUSH_FUNC;
715 
716  settings_map.clear();
717  // Get the IotDeviceSettings OOB:
718  px::proto::toolkitStruct dev_settings_oob;
719  bool bOK = _cds_handler->get_oob( IOT_DEV_SETTINGS_TAG, dev_settings_oob );
720  if ( bOK && ( dev_settings_oob.generic_dev_settings_table_size() > 0 ) )
721  {
722  for ( const px::proto::GenericDeviceSettingsEntry &dev_settings : dev_settings_oob.generic_dev_settings_table() )
723  {
724  px::id_t dev_id = dev_settings.cds_id();
725  gdse_entries_map_t entries_map;
726  entries_map.clear();
727  for ( const px::proto::GenericTableEntry &setting_entry : dev_settings.settings() )
728  {
729  if ( setting_entry.has_label() && setting_entry.has_value() )
730  {
731  // Must have a label and a value; might have a description
732  std::string details( "" );
733  if ( setting_entry.has_description() )
734  {
735  details = setting_entry.description();
736  }
737  px::CValue settings_val( setting_entry.value() );
738  // Now insert it in the map
739  SVR_INFO( 5, "Reading Device %u setting %s = %s - %s", ( uint32_t ) dev_id, setting_entry.label().c_str(),
740  settings_val.as_string().c_str(), details.c_str() );
741  entries_map[setting_entry.label()] = std::make_pair( settings_val, details );
742  }
743  }
744  if ( !entries_map.empty() )
745  {
746  settings_map[dev_id] = entries_map;
747  }
748  }
749  }

758 // Write the _iot_settings_map out to the IotDeviceSettings OOB.
759 bool CIotOobHandler::update_iot_device_settings_oob( bool force_update )
760 {
761  PUSH_FUNC;
762 
763  bool bOK = true;
764  if ( !force_update )
765  {
766  gdse_device_settings_map_t old_settings_map;
767  bOK = read_iot_device_settings_oob( old_settings_map );
768  if ( bOK && ( old_settings_map == _iot_settings_map ) )
769  {
770  SVR_INFO( 5, "IotDeviceSettings have not changed; no update." );
771  return ( bOK );
772  }
773  }
774 
775  if ( _iot_settings_map.empty() )
776  {
777  _cds_handler->clear_oob( IOT_DEV_SETTINGS_TAG );
778  }
779  else
780  {
781  px::proto::toolkitStruct dev_settings_oob;
782  for ( const auto &dev_pair : _iot_settings_map )
783  {
784  px::id_t dev_id;
785  gdse_entries_map_t entries_map;
786  std::tie( dev_id, entries_map ) = dev_pair;
787  if ( entries_map.empty() )
788  {
789  continue; // Skip; how did this happen?
790  }
791  // Get one mutable new entry:
792  px::proto::GenericDeviceSettingsEntry *dev_settings =
793  dev_settings_oob.add_generic_dev_settings_table();
794  dev_settings->set_cds_id( dev_id );
795  for ( const auto &set_pair : entries_map )
796  {
797  std::string label;
798  px::CValue value;
799  std::string details;
800  gdse_setting_pair_t val_pair;
801  std::tie( label, val_pair ) = set_pair;
802  std::tie( value, details ) = val_pair;
803  px::proto::GenericTableEntry *setting_entry = dev_settings->add_settings();
804  SVR_DEBUG( 6, "Writing device %u setting %s = %s - %s", ( uint32_t ) dev_id, label.c_str(),
805  value.as_string().c_str(), details.c_str() );
806  setting_entry->set_label( label );
807  px::proto::DataValue *proto_val = setting_entry->mutable_value();
808  value.to( *proto_val ); // convert from px::CValue to px::proto::Datavalue
809  if ( !details.empty() )
810  {
811  setting_entry->set_description( details );
812  }
813  }
814  }
815  // At the end of all that, write out the OOB (fire-and-forget, minimal error checking)
816  _cds_handler->set_oob( IOT_DEV_SETTINGS_TAG, dev_settings_oob, true ); // set persistently
817  }
818  return ( bOK );
819 }

821 /* Add a per-device IoT setting value to our internal map and optionally write out the map to
822  * the IotDeviceSettings OOB. */
823 bool CIotOobHandler::add_device_setting( px::id_t dev_id, const std::string &label, const px::CValue &value,
824  const std::string &details, bool dont_write_yet )
825 {
826  PUSH_FUNC;
827 
828  bool bOK = true;
829 
830  if ( ( dev_id == 0 ) || label.empty() )
831  {
832  // bad data; actually return a False in this case
833  return ( false );
834  }
835  // Prepare the inner-most pair
836  gdse_setting_pair_t spair = std::make_pair( value, details );
837  // See if we already have this device in the map
838  auto dpos = _iot_settings_map.find( dev_id );
839  if ( dpos != _iot_settings_map.end() )
840  {
841  // Then see if we also have this label as an entry

843  auto epos = entry_map.find( label );
844  if ( epos != entry_map.end() )
845  {
846  // Put in the new pair
847  epos->second = spair;
848  }
849  else
850  {
851  entry_map[label] = spair;
852  }
853  SVR_INFO( 5, "Added/updated IoT Setting %s = %s for device %u", label.c_str(),
854  value.as_string().c_str(), ( uint32_t ) dev_id );
855  }
856  else
857  {
858  // All new
859  gdse_entries_map_t entry_map;
860  entry_map[label] = spair;
861  _iot_settings_map[dev_id] = entry_map;
862  SVR_INFO( 5, "Added new IoT Setting %s = %s for device %u", label.c_str(),
863  value.as_string().c_str(), ( uint32_t ) dev_id );
864  }
865 
866  if ( !dont_write_yet )
867  {
868  bOK = update_iot_device_settings_oob( false ); // Don't force the update
869  }
870  return ( bOK );
