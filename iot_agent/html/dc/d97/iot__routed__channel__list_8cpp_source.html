49  {
50  try

52  std::string content( reqd_oob.generic().stringdata() );

54  {

56  return;

63  // Use read_string_or_throw() instead of read_string() to increase reliability.

65  // but it seems to be more reliable in reading JSON content.
66  json_spirit::read_string_or_throw( content, val );
67  const auto &enabled_obj = get_object_item( val, ROUTED_CH_LIST_ENABLE );
68  if ( enabled_obj.type() == ::json_spirit::bool_type )
69  {
70  enabled = enabled_obj.get_bool();
71  }
72  if ( !enabled )
73  {
74  SVR_INFO( 2, "Routed Channel List is not enabled." );
75  return;
76  }
77  SVR_INFO( 2, "Routed Channel List IS enabled." );
78 
79  const auto &target_obj = get_object_item( val, ROUTED_CH_LIST_TARGET ); // Will this throw if not found?
80  routed_target = target_obj.get_str();
81  if ( !routed_target.empty() )
82  {
83  SVR_INFO( 2, "Will use routed_target of '%s' for:", routed_target.c_str() );
84  }
85  else
86  {

89  }
90 
91  // Get array of channel names
92  const auto &ch_array_obj = get_object_item( val, ROUTED_CH_LIST_ARRAY );
93  if ( ch_array_obj.type() == ::json_spirit::array_type )
94  {
95  // Parse out canonical channel names
96  for ( auto &channel : ch_array_obj.get_array() )
97  {
98  // auto &ch_obj = channel.get_obj();
99  std::string channel_name = channel.get_str();
100  if ( !channel_name.empty() )
101  {
102  SVR_DEBUG( 8, " selective routed channel [%s]", channel_name.c_str() );
103  channel_list.insert( channel_name );

105  }

107  }
108  }
