54 // The purpose of this class is to create and update a number of
55 // "synthetic" channels of real-timish data.
56 //
57 // The data will appear like real data, with a slowly time-varying
58 // cycle and per-sample random component on top of a nominal value.
59 //
60 // The channels will be chosen to represent the various types of normal data:
61 // floats, negative numbers, accumulating values, and power factor.
62 //
63 class CSyntheticChannels
64 {
65 public:
66 
67  // Default Ctor.
68  // Will initialize channel data from MQTTDeviceEDS protobuf data.
69  //
70  CSyntheticChannels( std::shared_ptr<px::proto::MQTTDeviceEDS> &eds_data, px::string &root_topic_path );
71 
72  // Default Xtor
73  virtual ~CSyntheticChannels();
74 
75  // Called once per sample interval to update all the channel data.
76  //
77  void update_channel_values();
78 
79  // Array of synthetic channels built from the MQTT Device EDS
80  //
81  synth_vec_t channels;
82 
83 private:
84  // The prefix path for the channel sub_path values in the MQTT Device EDS. Should have trailing '/'.
85  // Example: "GO/Dev_ID/"
86  //
87  px::string _root_topic_path;
88 
89  // The time-varying component of the synthetic data.
90  // Updated each sample interval to the next value.
91  // Varies like a sine wave from 0 to 1, centered around 0.5.
92  //
93  double _cycle_value;
94 
95  // Define our time-varying cycle period, in seconds.
96  // Want something longer than the trend cycle (5min),
97  // but also that doesn't cycle too obviously in the trend results,
98  // so use 30 min * 60s + 44s.
99  //
100  const unsigned int _cycle_period_secs = 1844;
101 
102  // A random generator for our synthesized channels.
103  // Using the default "seed", given by the system. */
104  px::CRandom _random_gen;
105 };
