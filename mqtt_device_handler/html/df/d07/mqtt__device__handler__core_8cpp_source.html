116 void CMQTTCore::execute()
117 {
118  PUSH_FUNC;
119 
120  assert( _options ); // we've had problems long before now if this isn't valid!
121 
122  // Read and parse the MQTT Product EDS file
123  subscriber_client_t cfg;
124  cfg.eds_filename = _options->device_eds;
125  if ( CMQTTUtil::read_device_eds( cfg ) )
126  {
127  SYSLOG_ERR( 0, "read_device_eds %s returned an error. Exiting.", cfg.eds_filename.c_str() );
128  return;
129  }
130 
131  // Instantiate the MQTT Subscriber Client
132  std::shared_ptr<CMQTTClient> mqtt_client;
133  try
134  {
135  mqtt_client.reset( new CMQTTClient( getptr(), cfg, _options->port, _options->id ) );
136  if ( !mqtt_client.get() )
137  {
138  SYSLOG_ERR( 0, "Problem encountered when creating mqtt_client" );
139  return;
140  }
141 
142  // Register the client CDS subscriber callback
143  px::cds::CClient::get()->register_unsolicited_callback( mqtt_client );
144 
145  // Start the client on its way
146  mqtt_client->start();
147  }
148  catch ( const px::generic_err &e )
149  {
150  SYSLOG_ERR( 0, "Problem encountered while starting mqtt_client instance: %s", e.what() );
151  return;
152  }
153 
154  ready();
155 
156  uint32_t count = 0;
157  uint32_t ping_timeout = _options->watchdog_timeout / 3;
158  while ( !_want_to_stop )
159  {
160  count++;
161  sleep( 1 );
162  if ( count >= ping_timeout )
163  {
164  SYSLOG_FDEBUG( 9, "ping watchdog" );
165  count = 0;
166  _cds_handler->ping_watchdog_s();
167  }
168  } // end loop forever
169  _cds_handler->clear_watchdog();
170 
171  px::cds::CClient::get()->unregister_unsolicited_callback();
172  mqtt_client.reset();
173 }
