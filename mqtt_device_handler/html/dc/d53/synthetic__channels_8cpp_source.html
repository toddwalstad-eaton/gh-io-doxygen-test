168 // Called once per sample interval to update all the channel data.
169 void CSyntheticChannels::update_channel_values()
170 {
171  // First, compute the _cycle_value for this time, based on current seconds
172  timeval tnow;
173 
174  tnow = px::CMonoTime::now().to_timeval();
175  if ( -1 == tnow.tv_sec )
176  {
177  SYSLOG_DEBUG( 0, "px::CMonoTime::now().to_timeval() returned invalid time" );
178  return;
179  }
180 
181  // Get the modulo factor, 0 to 1843 secs
182  unsigned int cycle_factor = tnow.tv_sec % _cycle_period_secs;
183 
184  // Compute the _cycle_value as how far it is from 1/2 the modulus (absolute value), scaled to 1
185  int delta = cycle_factor - ( _cycle_period_secs / 2 );
186  _cycle_value = ( ( double ) std::abs( delta ) ) / ( _cycle_period_secs / 2 );
187 
188  double rnval;
189  for ( uint32_t i = 0; i < channels.size(); i++ )
190  {
191  if ( channels[i].params.is_dynamic )
192  {
193  // Determine as minimum + _cycle_value * range + random value (from d_rand)
194  rnval = _random_gen.uniform( 0.0, channels[i].params.d_rand );
195  *channels[i].message = ( double )( channels[i].params.d_min + ( _cycle_value * ( channels[i].params.d_max - channels[i].params.d_min ) ) + rnval );
196  SYSLOG_DEBUG( 6, "Update %s\t\t [%s]", channels[i].params.topic.c_str(), channels[i].message->as_string().c_str() );
197  }
198  else if ( channels[i].params.is_accum )
199  {
200  double tmp = channels[i].message->as_double();
201  rnval = _random_gen.uniform( 0.0, channels[i].params.d_rand );
202  if ( !std::isfinite( tmp ) )
203  {
204  *channels[i].message = ( double )0.0;
205  }
206  else
207  {
208  *channels[i].message = ( double )( tmp + rnval );
209  }
210  SYSLOG_DEBUG( 6, "Update %s\t\t [%s]", channels[i].params.topic.c_str(), channels[i].message->as_string().c_str() );
211  }
212  }
213 }
