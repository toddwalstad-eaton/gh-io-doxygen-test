182 void my_connect_callback( struct mosquitto *mosq, void * /* obj */, int result )
183 {
184  if ( result != MOSQ_ERR_SUCCESS )
185  {
186  switch ( result )
187  {
188  case MOSQ_ERR_INVAL:
189  SYSLOG_ERR( 0, "Invalid input. Does your topic contain '+' or '#'?" );
190  break;
191  case MOSQ_ERR_NOMEM:
192  SYSLOG_ERR( 0, "Out of memory when trying to publish message." );
193  break;
194  case MOSQ_ERR_NO_CONN:
195  SYSLOG_ERR( 0, "Client not connected when trying to publish." );
196  break;
197  case MOSQ_ERR_PROTOCOL:
198  SYSLOG_ERR( 0, "Protocol error when communicating with broker." );
199  break;
200  case MOSQ_ERR_PAYLOAD_SIZE:
201  SYSLOG_ERR( 0, "Message payload is too large." );
202  break;
203  default:
204  SYSLOG_ERR( 0, "Unknown connect error %d", result );
205  break;
206  }
207  mosquitto_disconnect( mosq );
208  }
209  else
210  {
211  status = STATUS_CONNACK_RECVD;
212  if ( fw_up_mode && !request_topic_root.empty() )
213  {
214  px::string subscribe_path = request_topic_root;
215  subscribe_path += dev_id;
216  subscribe_path += "/";
217  subscribe_path += request_topic_suffix;
218  request_topic_path = subscribe_path;
219  subscribe_path += "/#";
220  // GO/<device id>/Updates/SW/<name> e.g. GO/0001/Updates/SW/AIC
221  // GO/+/Updates/SW/#
222 
223  mosquitto_subscribe( mosq, NULL, subscribe_path.c_str(), 0 );
224  SYSLOG_DEBUG( 0, "Subscribing to %s - %s, request_topic_path %s",
225  subscribe_path.c_str(), mosquitto_connack_string( result ), request_topic_path.c_str() );
226  }
227  }
228 }

268 void my_message_callback( struct mosquitto *mosq, void * /* obj */, const struct mosquitto_message *message )
269 {
270  if ( ( message->payload != NULL ) && message->payloadlen )
271  {
272  SYSLOG_DEBUG( 1, "my_message_callback: %s, %d bytes", message->topic, message->payloadlen );
273  SYSLOG_DBGDUMP( 4, message->payload, message->payloadlen );
274 
275  if ( strstr( ( const char * )message->topic, request_topic_path.c_str() ) != NULL )
276  {
277  // Check if the file is open, or if this is the first SRecord message
278  if ( fileptr ||
279  ( ( ( size_t )message->payloadlen >= first_line_start.size() ) &&
280  ( strncmp( first_line_start.c_str(), ( const char * )message->payload, first_line_start.size() ) == 0 ) ) )
281  {
282  px::string publish_path = response_topic_root;
283  publish_path += dev_id;
284  publish_path += "/";
285  publish_path += response_topic_suffix; // TODO include any image specific strings that were at the end of the request topic
286 
287  // e.g. GO/1/Updates/Status
288  // Currently using a topic path for the device being updated
289  // Note: GO/Ser_Num/AICInfo/UpdInfo/UpdStats is the topic used for the IoT cloud communication
290  int rc;
291  px::string filename = "/tmp/fw_";
292  if ( fileptr == NULL )
293  {
294  current_update_publish_path = publish_path;
295  filename += dev_id;
296  filename += ".srec";
297  fileptr = fopen( filename.c_str(), "w" );
298  SYSLOG_DEBUG( 3, " fopen %s for topic %s returned %s", filename.c_str(), publish_path.c_str(), ( fileptr == NULL ) ? "NULL" : "Success" );
299  }
300  else if ( current_update_publish_path != publish_path )
301  {
302  // Ignore the new request
303  SYSLOG_WARN( 0, "Ignoring new firmware update request for since one is already in progress" );
304  return;
305  }
306 
307  // Save the SRecord(s) to a file. The message->payload can contain SRecords as:
308  // 1) A single line at a time.
309  // 2) Multiple lines at a time (Ephesus).
310  //
311  // The SRecord line endings may also not be consistent, so we will make sure
312  // that each line terminates in a line-feed (Unix file format).
313 
314  if ( fileptr != NULL )
315  {
316  char *start = ( char * )message->payload;
317  char *end = NULL;
318  bool done = false;
319  size_t srec_len = 0;
320 
321  // Append each SRecord as a separate line in the file.
322  do
323  {
324  // Get the start of the next SRecord
325  start = strchr( start, SREC_TAG );
326  if ( ( start == NULL ) || ( start >= ( ( char * )message->payload + message->payloadlen ) ) )
327  {
328  break;
329  }
330 
331  // Get the end of the next SRecord
332  end = strchr( start + 1, SREC_TAG );
333  if ( end == NULL )
334  {
335  end = ( char * )message->payload + message->payloadlen - 1;
336  done = true;
337  }
338 
339  // Handle the various line ending cases so that the result is a 'LF'.
340  if ( *( end - 1 ) == LF_TAG )
341  {
342  end--;
343  }
344  if ( *( end - 1 ) == CR_TAG )
345  {
346  end--;
347  *end = LF_TAG;
348  }
349  if ( *end == CR_TAG )
350  {
351  *end = LF_TAG;
352  }
353  srec_len = ( size_t )( end + 1 - start );
354 
355  SYSLOG_DEBUG( 5, " srec_len %zu, remaining %zu, end 0x%02X", srec_len, ( size_t )( ( char * )message->payload + message->payloadlen - start ), *end );
356 
357  fwrite( start, srec_len, 1, fileptr );
358  SYSLOG_DEBUG( 3, " fwrite %d", srec_len );
359 
360  // If there are no line ending characters between SRecords, terminate this SRecord with a line-feed.
361  if ( *end != LF_TAG )
362  {
363  *start = LF_TAG;
364  fwrite( start, 1, 1, fileptr );
365  }
366 
367  start = end;
368  }
369  while ( !done );
370  }
371  else
372  {
373  SYSLOG_ERR( 0, "Failed to open file %s", filename.c_str() );
374  }
375 
376  // See if the string contains the last_line_start characters (anywhere)
377  // Safe to test since the 'S' character can't appear in hex data bytes
378  SYSLOG_DEBUG( 1, "checking if last message..." );
379  last_ptr = strstr( ( const char * )message->payload, last_line_start.c_str() );
380  if ( last_ptr == NULL )
381  {
382  // If that didn't work, try one hard-coded alternative for ending (this IS a test program!)
383  last_ptr = strstr( ( const char * )message->payload, "S7" );
384  }
385 
386  if ( last_ptr != NULL )
387  {
388  SYSLOG_DEBUG( 1, "last message: last line marker: %.3s", last_ptr );
389  // If last SRecord, close the file
390  if ( fileptr != NULL )
391  {
392  fclose( fileptr );
393  fileptr = NULL;
394  SYSLOG_DEBUG( 3, " fclose done" );
395  }
396  // Publish the reception of the SRecord and completion of update
397  pub_return_code[0] = ( char )UPDATE_ERR_COMPLETE;
398  }
399  else
400  {
401  // Publish the reception of the last SRecord
402  pub_return_code[0] = ( char )UPDATE_ACK_OK;
403  }
404 
405  rc = mosquitto_publish( mosq, &mid_sent, publish_path.c_str(), ( size_t )1, &pub_return_code[0], 1, false );
406 
407  // Check if we succeeded
408  if ( rc != MOSQ_ERR_SUCCESS )
409  {
410  SYSLOG_ERR( 0, "Failed to publish to %s! Error = %d", publish_path.c_str(), rc );
411  }
412  }
413  else
414  {
415  SYSLOG_DEBUG( 1, " my_message_callback not interested in message content %d", message->payloadlen );
416  }
417  }
418  else
419  {
420  SYSLOG_DEBUG( 1, " my_message_callback not interested in topic %s", message->topic );
421  }
422  }
423  else
424  {
425  SYSLOG_DEBUG( 1, " my_message_callback: %s (null)", message->topic );
426  }
427 }

579 // Check if there is an endian mismatch between the local system and the MQTT device.
580 //
581 bool init_endian_mismatch( std::shared_ptr<px::proto::MQTTDeviceEDS> &eds_data )
582 {
583  bool endian_mismatch = false;
584  bool big_endian = is_big_endian();
585 
586  if ( eds_data->eds_info().has_endian() )
587  {
588  switch ( eds_data->eds_info().endian() )
589  {
590  case px::proto::MQTTEndianType::ENDIAN_BIG:
591  if ( !big_endian )
592  {
593  endian_mismatch = true;
594  }
595  break;
596  case px::proto::MQTTEndianType::ENDIAN_LITTLE:
597  if ( big_endian )
598  {
599  endian_mismatch = true;
600  }
601  break;
602  }
603  SYSLOG_DEBUG( 2, "Device EDS endian type [%s], system is big_endian [%s], endian_mismatch [%s]",
604  px::proto::MQTTEndianType_Name( eds_data->eds_info().endian() ).c_str(), ( big_endian ? "Yes" : "NO" ), ( endian_mismatch ? "Yes" : "NO" ) );
605  }
606  else
607  {
608  SYSLOG_DEBUG( 2, "No endian specified in Device EDS. System is_big_endian [%s]",
609  ( is_big_endian() ? "Yes" : "NO" ) );
610  }
611  return ( endian_mismatch );
612 }

614 // Check if the MQTT device EDS has an LWT entry, and intialize the mosq_config parameters.
615 //
616 void check_eds_lwt( std::shared_ptr<px::proto::MQTTDeviceEDS> &eds_data, px::string &root_topic_path, bool endian_mismatch )
617 {
618  // Check for 'lwt' test field entry and required parameters
619  if ( !eds_data->eds_info().has_test() ||
620  !eds_data->eds_info().test().has_lwt() ||
621  !eds_data->eds_info().test().lwt().has_topic() ||
622  !eds_data->eds_info().test().lwt().has_type() ||
623  !eds_data->eds_info().test().lwt().has_payload() )
624  {
625  return;
626  }
627 
628  // Reset any existing LWT data
629  if ( cfg.will_topic )
630  {
631  free( cfg.will_topic );
632  cfg.will_topic = NULL;
633  }
634  if ( cfg.will_payload )
635  {
636  free( cfg.will_payload );
637  cfg.will_payload = NULL;
638  }
639 
640  std::string will_topic = root_topic_path.c_str();
641  will_topic += eds_data->eds_info().test().lwt().topic();
642  cfg.will_topic = ( char * )malloc( will_topic.size() + 4 );
643  snprintf( cfg.will_topic, will_topic.size() + 4, "%s", will_topic.c_str() );
644 
645  px::proto::MQTTDataType will_type = eds_data->eds_info().test().lwt().type();
646  std::string will_payload = eds_data->eds_info().test().lwt().payload();
647 
648  // Start out with the most likely MQTT LWT data types
649  uint8_t lwt_uint8 = 0;
650  uint16_t lwt_uint16 = 0;
651  uint32_t lwt_uint32 = 0;
652 
653  cfg.will_payloadlen = 0;
654  switch ( will_type )
655  {
656  case px::proto::MQTTDataType::MQTT_BOOL:
657  case px::proto::MQTTDataType::MQTT_UINT8:
658  cfg.will_payloadlen = 1;
659  cfg.will_payload = ( char * )calloc( cfg.will_payloadlen, sizeof( uint8_t ) );
660  lwt_uint8 = ( uint8_t )( std::stoul( will_payload ) );
661  memcpy( cfg.will_payload, &lwt_uint8, cfg.will_payloadlen );
662  break;
663  case px::proto::MQTTDataType::MQTT_UINT16:
664  cfg.will_payloadlen = 2;
665  cfg.will_payload = ( char * )calloc( cfg.will_payloadlen, sizeof( uint16_t ) );
666  lwt_uint16 = ( uint16_t )( std::stoul( will_payload ) );
667  if ( endian_mismatch )
668  {
669  lwt_uint16 = bswap_16( lwt_uint16 );
670  }
671  memcpy( cfg.will_payload, &lwt_uint16, cfg.will_payloadlen );
672  break;
673  case px::proto::MQTTDataType::MQTT_UINT32:
674  cfg.will_payloadlen = 4;
675  cfg.will_payload = ( char * )calloc( cfg.will_payloadlen, sizeof( uint32_t ) );
676  lwt_uint32 = std::stoul( will_payload );
677  lwt_uint32 = std::stoul( will_payload );
678  if ( endian_mismatch )
679  {
680  lwt_uint32 = bswap_32( lwt_uint32 );
681  }
682  memcpy( cfg.will_payload, &lwt_uint32, cfg.will_payloadlen );
683  break;
684  default:
685  break;
686  }
687 
688  cfg.will_qos = 0;
689  if ( eds_data->eds_info().test().lwt().has_qos() )
690  {
691  switch ( eds_data->eds_info().test().lwt().qos() )
692  {
693  case px::proto::MQTTQoSType::QOS_1:
694  cfg.will_qos = 1;
695  break;
696  case px::proto::MQTTQoSType::QOS_2:
697  cfg.will_qos = 2;
698  break;
699  default:
700  break;
701  }
702  }
703  SYSLOG_DEBUG( 2, "LWT from EDS: topic [%s], payload [%s], payloadlen %lu, qos %d",
704  cfg.will_topic, eds_data->eds_info().test().lwt().payload().c_str(), cfg.will_payloadlen, cfg.will_qos );
705 }

1090  break;

1101  snprintf( tmp_message, sizeof( tmp_message ), "%s", synth->channels[i].message->as_string().c_str() );

1124  break;
