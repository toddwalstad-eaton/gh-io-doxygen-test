5 /*
6  * sha1.c
7  *
8  * Description:
9  * This file implements the Secure Hashing Algorithm 1 as
10  * defined in FIPS PUB 180-1 published April 17, 1995.
11  *
12  * The SHA-1, produces a 160-bit message digest for a given
13  * data stream. It should take about 2**n steps to find a
14  * message with the same digest as a given message and
15  * 2**(n/2) to find any two messages with the same digest,
16  * when n is the digest size in bits. Therefore, this
17  * algorithm can serve as a means of providing a
18  * "fingerprint" for a message.
19  *
20  * Portability Issues:
21  * SHA-1 is defined in terms of 32-bit "words". This code
22  * uses <stdint.h> (included via "sha1.h" to define 32 and 8
23  * bit unsigned integer types. If your C compiler does not
24  * support 32 bit unsigned integers, this code is not
25  * appropriate.
26  *
27  * Caveats:
28  * SHA-1 is designed to work with messages less than 2^64 bits
29  * long. Although SHA-1 allows a message digest to be generated
30  * for messages of any number of bits less than 2^64, this
31  * implementation only works with messages with a length that is
32  * a multiple of the size of an 8-bit character.
33  *
34  */

53  * SHA1Reset
54  *
55  * Description:
56  * This function will initialize the SHA1Context in preparation

73

90 /*
91  * SHA1Result
92  *
93  * Description:
94  * This function will return the 160-bit message digest into the
95  * Message_Digest array provided by the caller.
96  * NOTE: The first octet of hash is stored in the 0th element,
97  * the last octet of hash in the 19th element.
98  *
99  * Parameters:
100  * context: [in/out]
101  * The context to use to calculate the SHA-1 hash.
102  * Message_Digest: [out]
103  * Where the digest is returned.
104  *
105  * Returns:
106  * sha Error Code.
107  *
108  */
109 int SHA1Result( SHA1Context *context,
110  uint8_t Message_Digest[], unsigned length )
111 {
112  unsigned int i;
113 
114  if ( !context || !Message_Digest )
115  {
116  return ( shaNull );
117  }
118 
119  if ( context->Corrupted )
120  {
121  return ( context->Corrupted );
122  }
123 
124  if ( !context->Computed )
125  {
126  SHA1PadMessage( context );
127  for ( i = 0; i < 64; ++i )
128  {
129  /* message may be sensitive, clear it out */
130  context->Message_Block[i] = 0;
131  }
132  context->Length_Low = 0;/* and clear length */
133  context->Length_High = 0;
134  context->Computed = 1;
135  }
136 
137  // MODIFIED
138  //for (i = 0; i < SHA1HashSize; ++i)
139  if ( length > SHA1HashSize )
140  {
141  length = SHA1HashSize;
142  }
143  for ( i = 0; i < length; ++i )
144  {
145  Message_Digest[i] = ( uint8_t )( context->Intermediate_Hash[i >> 2] >> 8 * ( 3 - ( i & 0x03 ) ) );
146  }
147 
148  return ( shaSuccess );
149 }

151 /*
152  * SHA1Input
153  *
154  * Description:
155  * This function accepts an array of octets as the next portion
156  * of the message.
157  *
158  * Parameters:
159  * context: [in/out]
160  * The SHA context to update
161  * message_array: [in]
162  * An array of characters representing the next portion of
163  * the message.
164  * length: [in]
165  * The length of the message in message_array
166  *
167  * Returns:
168  * sha Error Code.
169  *
170  */
171 int SHA1Input( SHA1Context *context,
172  const uint8_t *message_array,
173  unsigned length )
174 {
175  if ( !length )
176  {
177  return ( shaSuccess );
178  }
179 
180  if ( !context || !message_array )
181  {
182  return ( shaNull );
183  }
184 
185  if ( context->Computed )
186  {
187  context->Corrupted = shaStateError;
188  return ( shaStateError );
189  }
190 
191  if ( context->Corrupted )
192  {
193  return ( context->Corrupted );
194  }
195  while ( length-- && !context->Corrupted )
196  {
197  context->Message_Block[context->Message_Block_Index++] =
198  ( *message_array & 0xFF );
199 
200  context->Length_Low += 8;
201  if ( context->Length_Low == 0 )
202  {
203  context->Length_High++;
204  if ( context->Length_High == 0 )
205  {
206  /* Message is too long */
207  context->Corrupted = 1;
208  }
209  }
210 
211  if ( context->Message_Block_Index == 64 )
212  {
213  SHA1ProcessMessageBlock( context );
214  }
215 
216  message_array++;
217  }
218 
219  return ( shaSuccess );
220 }

222 /*
223  * SHA1ProcessMessageBlock
224  *
225  * Description:
226  * This function will process the next 512 bits of the message
227  * stored in the Message_Block array.
228  *
229  * Parameters:
230  * None.
231  *
232  * Returns:
233  * Nothing.
234  *
235  * Comments:
236  * Many of the variable names in this code, especially the
237  * single character names, were used because those were the
238  * names used in the publication.
239  *
240  *
241  */
242 void SHA1ProcessMessageBlock( SHA1Context *context )
243 {
244  const uint32_t K[] = { /* Constants defined in SHA-1 */
245  0x5A827999,
246  0x6ED9EBA1,
247  0x8F1BBCDC,
248  0xCA62C1D6
249  };
250  int t; /* Loop counter */
251  uint32_t temp; /* Temporary word value */
252  uint32_t W[80]; /* Word sequence */
253  uint32_t A, B, C, D, E; /* Word buffers */
254 
255  /*
256  * Initialize the first 16 words in the array W
257  */
258  for ( t = 0; t < 16; t++ )
259  {
260  W[t] = context->Message_Block[t * 4] << 24;
261  W[t] |= context->Message_Block[t * 4 + 1] << 16;
262  W[t] |= context->Message_Block[t * 4 + 2] << 8;
263  W[t] |= context->Message_Block[t * 4 + 3];
264  }
265 
266  for ( t = 16; t < 80; t++ )
267  {
268  W[t] = SHA1CircularShift( 1, W[t - 3] ^ W[t - 8] ^ W[t - 14] ^ W[t - 16] );
269  }
270 
271  A = context->Intermediate_Hash[0];
272  B = context->Intermediate_Hash[1];
273  C = context->Intermediate_Hash[2];
274  D = context->Intermediate_Hash[3];
275  E = context->Intermediate_Hash[4];
276 
277  for ( t = 0; t < 20; t++ )
278  {
279  temp = SHA1CircularShift( 5, A ) +
280  ( ( B & C ) | ( ( ~B ) & D ) ) + E + W[t] + K[0];
281  E = D;
282  D = C;
283  C = SHA1CircularShift( 30, B );
284  B = A;
285  A = temp;
286  }
287 
288  for ( t = 20; t < 40; t++ )
289  {
290  temp = SHA1CircularShift( 5, A ) + ( B ^ C ^ D ) + E + W[t] + K[1];
291  E = D;
292  D = C;
293  C = SHA1CircularShift( 30, B );
294  B = A;
295  A = temp;
296  }
297 
298  for ( t = 40; t < 60; t++ )
299  {
300  temp = SHA1CircularShift( 5, A ) +
301  ( ( B & C ) | ( B & D ) | ( C & D ) ) + E + W[t] + K[2];
302  E = D;
303  D = C;
304  C = SHA1CircularShift( 30, B );
305  B = A;
306  A = temp;
307  }
308 
309  for ( t = 60; t < 80; t++ )
310  {
311  temp = SHA1CircularShift( 5, A ) + ( B ^ C ^ D ) + E + W[t] + K[3];
312  E = D;
313  D = C;
314  C = SHA1CircularShift( 30, B );
315  B = A;
316  A = temp;
317  }
318 
319  context->Intermediate_Hash[0] += A;
320  context->Intermediate_Hash[1] += B;
321  context->Intermediate_Hash[2] += C;
322  context->Intermediate_Hash[3] += D;
323  context->Intermediate_Hash[4] += E;
324 
325  context->Message_Block_Index = 0;
326 }

328 /*
329  * SHA1PadMessage
330  *
331  * Description:
332  * According to the standard, the message must be padded to an even
333  * 512 bits. The first padding bit must be a '1'. The last 64
334  * bits represent the length of the original message. All bits in
335  * between should be 0. This function will pad the message
336  * according to those rules by filling the Message_Block array
337  * accordingly. It will also call the ProcessMessageBlock function
338  * provided appropriately. When it returns, it can be assumed that
339  * the message digest has been computed.
340  *
341  * Parameters:
342  * context: [in/out]
343  * The context to pad
344  * ProcessMessageBlock: [in]
345  * The appropriate SHA*ProcessMessageBlock function
346  * Returns:
347  * Nothing.
348  *
349  */
350 
351 void SHA1PadMessage( SHA1Context *context )
352 {
353  /*
354  * Check to see if the current message block is too small to hold
355  * the initial padding bits and length. If so, we will pad the
356  * block, process it, and then continue padding into a second
357  * block.
358  */
359  if ( context->Message_Block_Index > 55 )
360  {
361  context->Message_Block[context->Message_Block_Index++] = 0x80;
362  while ( context->Message_Block_Index < 64 )
363  {
364  context->Message_Block[context->Message_Block_Index++] = 0;
365  }
366 
367  SHA1ProcessMessageBlock( context );
368 
369  while ( context->Message_Block_Index < 56 )
370  {
371  context->Message_Block[context->Message_Block_Index++] = 0;
372  }
373  }
374  else
375  {
376  context->Message_Block[context->Message_Block_Index++] = 0x80;
377  while ( context->Message_Block_Index < 56 )
378  {
379  context->Message_Block[context->Message_Block_Index++] = 0;
380  }
381  }
382 
383  /*
384  * Store the message length as the last 8 octets
385  */
386  context->Message_Block[56] = ( uint8_t )( context->Length_High >> 24 );
387  context->Message_Block[57] = ( uint8_t )( context->Length_High >> 16 );
388  context->Message_Block[58] = ( uint8_t )( context->Length_High >> 8 );
389  context->Message_Block[59] = ( uint8_t )( context->Length_High );
390  context->Message_Block[60] = ( uint8_t )( context->Length_Low >> 24 );
391  context->Message_Block[61] = ( uint8_t )( context->Length_Low >> 16 );
392  context->Message_Block[62] = ( uint8_t )( context->Length_Low >> 8 );
393  context->Message_Block[63] = ( uint8_t )( context->Length_Low );
394 
395  SHA1ProcessMessageBlock( context );
396 }
