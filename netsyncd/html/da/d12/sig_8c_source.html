103 static void sig_handler( int arg )
104 {
105  char fake = 0;
106 
107  switch ( arg )
108  {
109  case SIGSEGV:
110  fprintf( stderr, "Segmentation Fault\n" );
111  syslog( LOG_CRIT, "Segmentation Fault" );
112  trace();
113  case SIGILL:
114  fprintf( stderr, "Illegal Instruction\n" );
115  syslog( LOG_CRIT, "Illegal Instruction" );
116  trace();
117  case SIGFPE:
118  fprintf( stderr, "FP Exception\n" );
119  syslog( LOG_CRIT, "FP Exception" );
120  trace();
121  case SIGHUP:
122  case SIGINT:
123  case SIGQUIT:
124  case SIGTERM:
125  fprintf( stderr, "Exit signal\n" );
126  syslog( LOG_CRIT, "Exit signal" );
127  /* Tell the process to exit */
128  __signal_exit = 1;
129  write( __signal_pipepair[WRITE_PIPE], &fake, 1 );
130  break;
131  case SIGPIPE:
132  default:
133  break;
134  }
135 }
