241  }

377 unsigned char CDeviceIoStats::get_max_errors( const CDevice *owner_device, const err_t error )
378 {
379  PUSH_FUNC;
380  unsigned char max_errors;
381 
382  // Determine the current I/O state.
383  switch ( error )
384  {
385  case err_t::SEND_TIMEOUT:
386  case err_t::RECEIVE_TIMEOUT:
387  case err_t::COMMUNICATION_IO_ERROR:
388  max_errors = _max_rx_timeout_errors;
389  break;
390 
391  case err_t::BROKEN_PIPE_ERROR:
392  max_errors = 1;
393  break;
394 
395  // Modbus
396  case err_t::ILLEGAL_FUNCTION:
397  case err_t::ILLEGAL_DATA_ADDRESS:
398  case err_t::ILLEGAL_DATA_VALUE:
399  case err_t::SLAVE_DEVICE_FAILURE:
400  case err_t::ACKNOWLEDGE:
401  case err_t::SLAVE_DEVICE_BUSY:
402  case err_t::NEGATIVE_ACKNOWLEDGE:
403  case err_t::MEMORY_PARITY_ERROR:
404  case err_t::UNKNOWN_MODBUS_EXCEPTION_ERROR:
405  case err_t::GATEWAY_PATH_UNAVAILABLE:
406  case err_t::TARGET_DEVICE_FAILED_TO_RESPOND:
407  // Gateway errors
408  case err_t::GATEWAY_UNAVAILABLE:
409  case err_t::GATEWAY_UNRESPONSIVE:
410  case err_t::UNDETERMINED_RESULT:
411  if ( ( ( error == err_t::ILLEGAL_DATA_ADDRESS ) ||
412  ( error == err_t::SLAVE_DEVICE_FAILURE ) ||
413  ( error == UNKNOWN_MODBUS_EXCEPTION_ERROR ) ) &&
414  ( strstr( owner_device->_px_device->model.c_str(), "Series NRX 1150M" ) ||
415  strstr( owner_device->_px_device->model.c_str(), "Series NRX 520M" ) ) )
416  {
417  // A work-around hack for the MCAM.
418  // Following a power-up, the MCAM responds with:
419  // - Modbus Exception Code 02 (MCAM V1.0)
420  // - Modbus Exception Code 04 or 02 (MCAM V0.32)
421  // while it is "learning" the trip unit. This code provides "ride-through"
422  // for that period so that the poll blocks & channels are not dropped.
423  // The period can be up to 12 seconds and affects device profiling speed.
424  max_errors = 100; // Empirically determined for MCAM.
425  }
426  else
427  {
428  max_errors = _max_nack_errors;
429  }
430  break;
431 
432  case err_t::DATA_VALIDITY_ERROR:
433  case err_t::SERIALIZE_ERROR:
434  case err_t::PARSE_ERROR:
435  max_errors = _max_data_validity_errors;
436  break;
437 
438  case err_t::BUFFER_OVERRUN_ERROR:
439  max_errors = 0;
440  break;
441  default:
442  max_errors = 0;
443  break;
444  }
445  return ( max_errors );
446 }

449 {
