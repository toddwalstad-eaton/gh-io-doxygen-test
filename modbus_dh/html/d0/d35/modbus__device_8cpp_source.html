137 transaction_deque_t *CModbusDevice::next_transaction_series()
138 {
139  assert( _transaction_series.empty() );
140 
141  // Check if the device is connected
142  if ( !is_connected() )
143  {
144  return ( &_transaction_series );
145  }
146 
147  // This queues to the _command_deque!
148  if ( _command_deque.empty() )
149  {
150  create_multipart_command_transaction();
151  }
152 
153  // This alternates between available commands and waveform data polls!
154  if ( _command_deque.empty() && ++_waveform_poll_counter & 0x01 )
155  {
156  create_waveform_transaction();
157  }
158 
159  if ( !_command_deque.empty() )
160  {
161  // Dequeue one command block and add the messages to the transaction.
162  const std::shared_ptr<CDataBlock> origin = _command_deque.front();
163  _command_deque.pop_front();
164 
165  if ( origin.get() )
166  {
167  std::shared_ptr<CModbusCommandBlock> mb_cmd_block = px::poly_cast<CModbusCommandBlock>( origin );
168  assert( mb_cmd_block.get() );
169  if ( mb_cmd_block.get() )
170  {
171  std::shared_ptr<CModbusMessage> msg = mb_cmd_block->create_message( origin );
172  msg->set_connection_id( connection_id() );
173  _transaction_series.push_back( msg ); // queue up our one message to process
174  }
175  }
176  }
177 
178  SVR_DEBUG( 8, "%s Using priority entry = %u for block id %" PRIu64 "", port_dev_id_str(), priority(), _highest_priority_block );
179 
180  // Get our CModbusDataBlock pointer, but using the base-est class
181  // of CModbusDataBlock::CDataBlock::CMessageHandler as our "origin"
182  std::shared_ptr<CDataBlock> origin = lookup_block( _highest_priority_block );
183  assert( origin.get() ); // pointer should have valid contents
184  if ( origin.get() )
185  {
186  assert( origin->id() == _highest_priority_block );
187 
188  std::shared_ptr<CModbusDataBlock> mb_data_block = std::dynamic_pointer_cast<CModbusDataBlock>( origin );
189  assert( mb_data_block.get() ); // dynamic cast worked
190  assert( _highest_priority_block == mb_data_block->id() ); // right contents
191  assert( mb_data_block->is_valid() ); // lookup_block() should only return if ->is_valid()
192  SVR_DEBUG( 8, "%s Loading msg from datablock id = %" PRIu64 "", port_dev_id_str(), _highest_priority_block );
193  std::shared_ptr<CModbusMessage> msg = mb_data_block->create_message( origin );
194  msg->set_connection_id( connection_id() );
195  _transaction_series.push_back( msg ); // queue up our one message to process
196  }
197  // else, no messages queued
198 
199  return ( &_transaction_series );

555  // If this isn't a multi-part command, this method returns immediately
556  // with a result code.
557  if ( _device_specific_handler.get() )
558  {
559  err_t multipart_retval = _device_specific_handler->execute_multipart_command( _cds_handle, id(), dev_command );
560  if ( err_t::COMMAND_IN_PROGRESS_ERROR == multipart_retval )
561  {
562  // Busy!
563  SVR_ERR( 1, "%s Modbus handler busy. Command [ %s ] will not be executed: CDS handle %u, Device=%s, Address=%s",
564  port_dev_id_str(), device_command.name.c_str(), device_command.handle,
565  _px_device->model.c_str(), get_formatted_address().c_str() );
566 
567  // Set command status to "failed".
568  command_status = px::device_command_status::failed;
569  command_substatus = px::device_command_substatus::busy;
570  // Update the CDS command status.
571  if ( _cds_handle.get() )
572  {
573  CCommandService::send_command_result( _cds_handle, device_command, command_status, command_substatus, CUtil::get_err_text( multipart_retval ) );
574  }
575  return ( multipart_retval );
576  }
577  }
578 
579  const std::shared_ptr<CDataBlock> blk_command = create_command( dev_command, pchan, retval, xBlock );
580 
581  if ( blk_command.get() && ( err_t::NO_ERROR == retval ) )
582  {
583  _command_deque.push_back( blk_command );
584  SVR_DEBUG( 3, "%s Modbus command [ %s ] queued for execution. Device=%s", port_dev_id_str(),
585  dev_command.name.c_str(), _px_device->model.c_str() );
586 
587  // command execution status was already set to in_progress when it was put into the CDS
588  // command_status = px::device_command_status::in_progress;
589  // And when the response is processed in CDevice::handle_command_completion(),
590  // the CDS will be updated with the command results.
591  retval = err_t::NO_ERROR;
592  }
593  else
594  {
595  SVR_ERR( 1, "%s %s", port_dev_id_str(), CUtil::get_err_text( retval ) );
596  command_status = px::device_command_status::failed;
597  }
598 
599  // Now perform an update of the CDS about the execution status of the command
600  if ( _cds_handle.get() )
601  {
602  // Don't send a CDS device command response if this was an automatic command
603  // or if it is queued up and in_progress
604  if ( device_command.handle && !device_command.is_automatic && ( retval != err_t::NO_ERROR ) )
605  {
606  // Update CDS about the execution status of the command
607  CCommandService::send_command_result( _cds_handle, device_command, command_status, command_substatus, err_msg );
608  }
609 
610  // If this is a single channel write, update the CDS value.
611  // This type of channel will usually be updated on a very low priority (minutes),
612  // so the intent is to provide a timely channel value update for the requester
613  if ( ( pchan != nullptr ) && pchan->is_writable() )
614  {
615  CDevice::execute_command( device_command, false );
616  }
617  }
618  // These elses are extremely unlikely in practice, except during test development
619  else if ( err_t::NO_ERROR == retval )
620  {
621  // Command queued, but CDS not updated
622  SVR_ERR( 1, "%s Device Command sent: Could not get valid CDS handle %s!", port_dev_id_str(), device_command.name.c_str() );
623  }
624  else
625  {
626  // Command not queued and CDS not updated.
627  SVR_ERR( 1, "%s Could not send Device Command: Invalid CDS handle %s!", port_dev_id_str(), device_command.name.c_str() );
628  }
629 
630  return ( retval );
631 }

633 // Called from the Data Block to handle a device communication error.
634 void CModbusDevice::handle_device_error( uint64_t block_id, err_t error )
635 {
636  PUSH_FUNC;
637 
638  if ( ( error != err_t::NO_ERROR ) && ( error != err_t::RECEIVE_TIMEOUT ) && ( error != err_t::BUFFER_OVERRUN_ERROR ) )
639  {
640  SVR_ERR( 3, "Modbus request canceled. Device=%s, Error=%s", _px_device->model.c_str(), CUtil::get_err_text( error ) );
641 
642  // Check for Modbus errors we will trap on during profiling process only
643  if ( !_is_profiling_finalized )
644  {
645  switch ( error )
646  {
647  // For these type of Modbus errors, remove block_id and all dependencies from this device
648  case err_t::ILLEGAL_DATA_ADDRESS:
649  case err_t::SLAVE_DEVICE_FAILURE:
650  // During profiling, don't kill the data block for an MCAM!
651  // For MCAM, the block will be disabled by the Modbus data block.
652  if ( !( strstr( _px_device->model.c_str(), "Series NRX 1150M" ) ||
653  strstr( _px_device->model.c_str(), "Series NRX 520M" ) ) )
654  {
655  std::set<uint64_t> kill_list;
656  kill_list.insert( block_id );
657  SVR_ERR( 5, "Modbus profiling error %s for block id %" PRIu64 ", removing it and all dependencies from this device", CUtil::get_err_text( error ), block_id );
658  remove_blocks_and_children( kill_list );
659  return;
660  }
661  break;
662 
663  default:
664  break;
665  }
666  }
667 
668  // Free all messages in the transaction queue, since we shouldn't proceed
669  _transaction_series.clear();
670  }
671 
672  if ( CDeviceIoStats::DeviceIoStateEnums::IO_STATE_NO_ERROR != _io_stats->get_state() )
673  {
674  handle_io_comm_loss( error );
675  }
676 }

713 // Called on command completion
714 void CModbusDevice::handle_command_completion( uint64_t block_id, const px::string &command_name, px::CDeviceCommand &device_command, const px::asio::io_buf_t &data, err_t error )
715 {
716  // Here we complete the device command at the Modbus protocol level.
717  // Commands which were automatically issued by the Device Handler should not
718  // send a command result to the CDS.
719 
720  PUSH_FUNC;
721  bool is_automatic( false );
722 
723  CCommand *command;
724  command = lookup_command( device_command.name );
725  if ( nullptr != command )
726  {
727  is_automatic = command->device_command->is_automatic;
728  }
729 
730  if ( is_automatic )
731  {
732  // The "automatic" commands usually require special handling.
733  // Handle automatic commands (i.e. those issued by Device Handler).
734  if ( _device_specific_handler.get() )
735  {
736  _device_specific_handler->handle_command_completion( block_id, _cds_handle, device_command, data, error );
737  }
738  }
739 
740  // The setpoints commands require special handling.
741  // If this isn't a setpoints command, this method returns immediately.
742  CModbusDeviceSetpoints::handle_setpoints_command_completion( _cds_handle, id(), device_command, data, error );
743 
744  // Log the command result.
745  if ( ( err_t::NO_ERROR == error ) && ( CDeviceIoStats::DeviceIoStateEnums::IO_STATE_NO_ERROR == _io_stats->get_state() ) )
746  {
747  DEVICELOG_INFO( "Status", "%s%sCommand %s was sent successfully.",
748  get_formatted_name().c_str(),
749  get_formatted_address().c_str(),
750  command_name.c_str() );
751  SVR_DEBUG( 3, "%s Command %s was sent successfully!", port_dev_id_str(), command_name.c_str() );
752  device_command.status = px::device_command_status::succeeded;
753  device_command.substatus = px::device_command_substatus::none;
754  }
755  else
756  {
757  DEVICELOG_ERR( "Status", "%s%sI/O error occurred with command %s (Error=%s)!",
758  get_formatted_name().c_str(),
759  get_formatted_address().c_str(),
760  command_name.c_str(),
761  CUtil::get_err_text( error, true ) );
762  SVR_DEBUG( 1, "%s I/O error occurred with Command: %s (Error=%s)!", port_dev_id_str(), command_name.c_str(), CUtil::get_err_text( error, true ) );
763 
764  switch ( error )
765  {
766  case err_t::ILLEGAL_FUNCTION:
767  case err_t::ILLEGAL_DATA_ADDRESS:
768  case err_t::ILLEGAL_DATA_VALUE:
769  case err_t::SLAVE_DEVICE_FAILURE:
770  case err_t::SLAVE_DEVICE_BUSY:
771  case err_t::NEGATIVE_ACKNOWLEDGE:
772  case err_t::MEMORY_PARITY_ERROR:
773  case err_t::UNKNOWN_MODBUS_EXCEPTION_ERROR:
774  case err_t::GATEWAY_PATH_UNAVAILABLE:
775  // These are the Modbus exception codes (NACKs) that could be returned by the device.
776  // Set the command substatus accordingly.
777  device_command.status = px::device_command_status::failed;
778  device_command.substatus = px::device_command_substatus::nacked;
779  break;
780 
781  default:
782  // All other errors...
783  device_command.status = px::device_command_status::failed;
784  device_command.substatus = px::device_command_substatus::none;
785  break;
786  }
787  }
788 
789  // Finally update CDS as required.
790  if ( _cds_handle.get() && !is_automatic )
791  {
792  SVR_DEBUG( 3, "%s Command %s. Send response to CDS command handle %u.", port_dev_id_str(), command_name.c_str(), device_command.handle );
793  CCommandService::send_command_result( _cds_handle, device_command, CUtil::get_err_text( error ) );
794  }
795 }
