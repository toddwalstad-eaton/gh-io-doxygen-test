96 bool CDeviceSpecificHandler::send_command_result( const std::shared_ptr<ICdsHandle> &cds_handle, px::CDeviceCommand &device_command, const uint8_t progress, const err_t error )
97 {
98  PUSH_FUNC;
99 
100  bool is_completed( false );
101  bool is_cds_update_required( true );
102 
103  switch ( error )
104  {
105  case err_t::NO_ERROR:
106  // Success!
107  SVR_INFO( 3, "Device (id: %" PRIu64 ") Command succeeded.", device_command.device_id );
108  device_command.status = px::device_command_status::succeeded;
109  device_command.substatus = px::device_command_substatus::none;
110  is_completed = true;
111  break;
112  case err_t::UNDETERMINED_RESULT:
113  // Progress update
114  SVR_INFO( 3, "Device (id: %" PRIu64 ") Command in progress. Progress=%u%%", device_command.device_id, progress );
115  device_command.status = px::device_command_status::in_progress;
116  device_command.substatus = px::device_command_substatus::none;
117  is_cds_update_required = false;
118  break;
119  default:
120  // Error!
121  SVR_ERR( 3, "Device (id: %" PRIu64 ") Command failed. Error=%s", device_command.device_id, CUtil::get_err_text( error ) );
122  device_command.status = px::device_command_status::failed;
123  device_command.substatus = px::device_command_substatus::none;
124  is_completed = true;
125  break;
126  } //switch
127 
128  // Update CDS
129  if ( is_cds_update_required )
130  {
131  CCommandService::send_command_result( cds_handle, device_command, CUtil::get_err_text( error ) );
132  }
133 
134  return ( is_completed );
135 }

140 err_t CDeviceSpecificHandler::schedule_waveform_upload( const px::id_t id )
141 {
142  PUSH_FUNC;
143 
144  err_t result( err_t::NO_ERROR );
145 
146  if ( ( WaveformStateEnums::WAVEFORM_HALTED != _waveform_state ) &&
147  _waveform_enabled &&
148  ( 0 != id ) &&
149  _product_type )
150  {
151  auto result = CDeviceSpecificHandler::_waveform_schedule.insert( id );
152 
153  if ( result.second )
154  {
155  // On insertion, initialize state machine
156  _waveform_state = WaveformStateEnums::WAVEFORM_INITIALIZING;
157  SVR_DEBUG( 7, "Device (id: %" PRIu64 ") Waveform acquisition scheduled.", id );
158  }
159  }
160  else
161  {
162  result = err_t::CANNOT_EXECUTE_COMMAND;
163  if ( !_product_type )
164  {
165  SVR_ERR( 9, "Device (id: %" PRIu64 ") Waveform acquisition not supported.", id );
166  }
167  else if ( 0 == id )
168  {
169  SVR_ERR( 7, "Device (id: %" PRIu64 ") Waveform device id is invalid.", id );
170  }
171  else if ( WaveformStateEnums::WAVEFORM_HALTED == _waveform_state )
172  {
173  SVR_ERR( 5, "Device (id: %" PRIu64 ") Waveform acquisition halted.", id );
174  }
175  else if ( !_waveform_enabled )
176  {
177  SVR_ERR( 7, "Device (id: %" PRIu64 ") Waveform acquisition disabled.", id );
178  }
179  }
180  return ( result );
181 }
