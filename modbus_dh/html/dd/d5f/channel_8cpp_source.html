30 {
31  PUSH_FUNC;
32 
33  name = eds_descr.canonical_name;
34 
35  // This will be empty normally, unless eg non-MCL
36  _canonical_name = eds_descr.alt_canonical_name;
37 
38  // First, deal with the equation or DBT mapping (may throw if equation is bad)
39  try

42  }

47 
48  // Transfer our EDS settings of interest

52  if ( eds_descr.ext_data_type )

54  _cmd_ext_type = eds_descr.ext_data_type.get();
55  }

57

59  _rx_timeout = ( ( eds_descr.rx_timeout ) ? eds_descr.rx_timeout.get() : 0 );
60 
61  // If an alternate canonical name is provided, use it:
62  if ( !_canonical_name.empty() && ( _canonical_name != name ) )
63  {
64  // If this is a duplicate channel, preserve the true canonical name of this channel
65  _ch_settings->canonical_name = eds_descr.alt_canonical_name;
66  }
67  else
68  {

70  _ch_settings->canonical_name = eds_descr.canonical_name;
71  }
72

74  if ( eds_descr.cds_type )

77  }

85  }
86  }
87 
88  // If Channel overrides or alternate canonical name are provided, process them:
89  if ( eds_descr.contains_overrides )
90  {
91  copy_overrides( _ch_settings, eds_descr );
92 
93  // Since this channel is not in the CDS yet, test whether
94  // this channel has an initialized Alarm Trigger pointer.
95  if ( check_alarm_trigger && eds_descr.overrides.alarm_trigger )
96  {
97  parent.set_alarm_trigger_present( true );
98  alarm_trigger = eds_descr.overrides.alarm_trigger;
99  }

128 // Update this Channel or Parameter now from its equation or xBlock source data.
129 bool CChannel::update_param( CDevice &device, px::CWallTime &when )
130 {
131  PUSH_FUNC;
132  bool rc = false;
133 
134  if ( _conditional_equation )
135  {
136  try
137  {
138  _conditional_equation->evaluate( *device.equation_handler );
139  _conditional_result = device.get_param_value( _conditional_variable_id, PARAM_OBJ ).as_bool();
140  SVR_DEBUG( 8, "Conditional: fn[%s] = %d", _conditional_equation_text.c_str(), _conditional_result );
141  if ( !_conditional_result )
142  {
143  // Note that you could hit this condition several times for a channel with
144  // a dependency condition on a variable to has not had its sources polled yet (e.g. xModel)
145  SVR_DEBUG( 5, "Channel %s not available based on dependency condition fn[%s]",
146  name.c_str(), _conditional_equation_text.c_str() );
147  return ( rc ); // Stop evaluating if conditional equation returns false
148  }
149  }
150  catch ( const px::eq::not_found & /*ex*/ )
151  {
152  SVR_NOTICE( 2, "Equation not found exception, for channel %s fn[%s]", name.c_str(), _conditional_equation_text.c_str() );
153  return ( rc ); // Stop evaluating if conditional equation fails
154  }
155  catch ( const boost::exception & /*ex*/ )
156  {
157  SVR_NOTICE( 2, "Equation exception, for channel %s fn[%s]", name.c_str(), _conditional_equation_text.c_str() );
158  return ( rc ); // Stop evaluating if conditional equation fails
159  }
160  }
161 
162  // Use get_bit_block_value() if the count is non-zero -- UNLESS it's a binary blob byte array or string
163  std::shared_ptr<px::CValue> block_value;
164  if ( ( _dbt_count == 0 ) || ( value.type() == px::datatype::string ) || ( value.type() == px::datatype::binary ) )
165  {
166  block_value = device.get_block_value( _dbt_xblock_id, _dbt_array_index );
167  }
168  else
169  {
170  block_value = device.get_bit_block_value( _dbt_xblock_id, _dbt_array_index, _dbt_bit_position, _dbt_count );
171  }
172 
173  // If just a direct DBT transform
174  if ( ( block_value.get() ) && !_equation )
175  {
176  if ( value.type() != px::datatype::invalid )
177  {
178  value = block_value->as( value.type() ); // Keep the same type
179  }
180  else
181  {
182  value = *block_value;
183  }
184  rc = true;
185  }
186  else if ( _equation )
187  {
188  try
189  {
190  // Handle the case where the channel data type may not
191  // have been initialized correctly from CDS data.
192  if ( value.type() == px::datatype::invalid )
193  {
194  // First try to get a data type from EDS data.
195  px::datatype param_type = device.get_eds_data_type( name );
196  if ( param_type == px::datatype::invalid )
197  {
198  // Next try getting a data type from a DBT (if possible).
199  if ( block_value.get() )
200  {
201  param_type = block_value->type().val();
202  }
203 
204  // Finally, just default to something
205  if ( param_type == px::datatype::invalid )
206  {
207  // Bit of a hack, but peek at the first character of
208  // the channel name to figure out if it is an Alarm.
209  if ( !name.empty() && ( name.at( 0 ) == 'a' ) )
210  {
211  param_type = px::datatype::boolean;
212  }
213  else
214  {
215  param_type = px::datatype::float32;
216  }
217  }
218  }
219  px::CDataType new_type( param_type );
220  value.to( new_type );
221  SVR_DEBUG( 1, "Data type for channel \"%s\" was invalid, set data type to \"%s\"",
222  name.c_str(), px::PXtype2Protobuf( param_type ).c_str() );
223  }
224  assert( value.type() != px::datatype::invalid );
225 
226  // Preset the name of "this" Channel, for the set() operation's use
227  device.equation_handler->preset_param_name( name );
228 
229  // Evaluate the equation
230  _equation->evaluate( *device.equation_handler );
231 
232  SVR_INFO( 8, "Update from Equation: fn[%s] = %s", _translation_text.c_str(), value.as_string().c_str() );
233  rc = true;
234  }
235  // Catch any toolkit-generated exception, including px::eq::not_found
236  catch ( const px::generic_err &gex )
237  {
238  SVR_WARN( 1, "_equation->evaluate(%s) threw: %s", _translation_text.c_str(), gex.what() );
239  }
240  catch ( const std::exception &ex )
241  {
242  SVR_WARN( 1, "_equation->evaluate(%s) threw std: %s", _translation_text.c_str(), ex.what() );
243  }
244  catch ( ... )
245  {
246  SVR_WARN( 1, "_equation->evaluate(%s) threw unknown exception", _translation_text.c_str() );
247  }
248  }
249  else
250  {
251  // Could be that block has not been polled yet
252  SVR_WARN( 8, "No way found yet to update Channel %s", name.c_str() );
253  }
254 
255  _when = when;
256 
257  // Block channel updates for channels with temp ID (they don't exist in CDS yet)
258  if ( has_temp_id )
259  {
260  rc = false;
261  }
262 
263  return ( rc ); // Indicates whether we are ready for updates to the CDS
264 }
