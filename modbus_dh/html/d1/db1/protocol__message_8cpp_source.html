168  SVR_FDEBUG( 8, "parsing msg(%zu)", _msg_num );
169 
170  // Parse the bytes received so far
171  assert( _message.get() );
172 
173  //assert( _rbuf.begin() >= 0 );
174  assert( _offset ); // Catches clean-up error from partial read and timeout
175 
176  //assert ( _offset < _rbuf.size() ); // detects invalid offsets
177  if ( _offset > _rbuf.size() )
178  {
179  SVR_ERR( 1, "(p%" PRIu64 ":d%" PRIu64 ") Read-buffer _offset %zu too large (> than rbuffer size %zu)",
180  _port_id, device_id(), _offset, _rbuf.size() );
181 
182  if ( _response_handler )
183  {
184  // If have a valid protocol handler, let it do something with this error
185  _response_handler->err_response( err_t::BUFFER_OVERRUN_ERROR, _message.get() );
186  }
187  _message_state = io_message_state::parse_error;
188  return;
189  }
190 
191  px::asio::parse_result_t result; // Do initialization so valgrind does not complain
192  result.first = boost::logic::indeterminate;
193  result.second = _rbuf.begin();
194  result = _message->parse( _rbuf.begin(), _rbuf.begin() + _offset );
195 
196  // First, see if we are done:
197  if ( result.first )
198  {
199  SVR_DEBUG( 8, "(p%" PRIu64 ":d%" PRIu64 ") Read-buffer complete with %zu bytes",
200  _port_id, device_id(), _offset );
201 
202  // _message now contains a valid response object
203  if ( _response_handler )
204  {
205  // If have a valid protocol handler, let it do something with this response
206  _response_handler->respond_from_msg( _message.get() );
207  }
208  _message_state = io_message_state::success;
209  }
210  // Secondly, are we known to have failed (and are thus done) ?
211  else if ( !result.first )
212  {
213  if ( result.second != _rbuf.end() )
214  {

218  }
219

224  }

226  }
