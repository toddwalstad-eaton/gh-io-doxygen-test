561 // Normally only called when something has changed, or on initialization
562 bool CSerialPort::update_device_map( std::shared_ptr<px::CDevice> &parent_device, const char *my_name, const bool device_added )
563 {
564  // Get the port's devices and their associated settings. Rescan the CDS every time here, in case things have changed.
565  bool any_changes = false;
566  std::map<px::shared_string, uint32_t> models_with_changed_eds_checksums;
567  device_config_array_t child_devices;
568  device_config_array_t all_child_devices;
569 
570  // Find child devices one level down from the parent ID
571  child_devices = CDeviceService::find_child_devices( _cds_handle, parent_device->id(), FIND_ONE_LEVEL_DOWN );
572  SVR_FINFO( 1, "my_name %s: Device (ID %" PRIu64 ") has %zu child devices", my_name, parent_device->id(), child_devices.size() );
573 
574  // This is the device add loop (for this port).
575  for ( std::shared_ptr<px::CDevice> &this_device : child_devices )
576  {
577  if ( !this_device.get() )
578  {
579  SVR_WARN( 1, "Skipping slave device with missing info for %s.", my_name );
580  continue;
581  }
582 
583  auto map_entry = _devices.find( this_device->id() );
584  if ( map_entry != _devices.end() )
585  {
586  // We already have this device, so don't do anything new with this device
587  SVR_INFO( 8, "Slave device (ID %" PRIu64 ") already in map for Serial Port %s.", this_device->id(), get_port_name() );
588 
589  // Check if it has any unprocessed subdevice instances (does nothing if none to process)
590  try
591  {
592  map_entry->second->update_subdevice_list( getptr() );
593  }
594  catch ( std::exception &ex )
595  {
596  SVR_FERR( 1, "Exception thrown for map_entry->second->update_subdevice_list: %s", ex.what() );
597  }
598 
599  // Maybe someday: check its change signature and see if has Channel or subDevice changes
600  continue;
601  }
602 
603  try
604  {
605  if ( device_added )
606  {
607  // Log if this device was just added to the CDS
608  unsigned int slave_addr = ( unsigned int )strtoul( this_device->network_address.c_str(), NULL, 0 );
609  size_t device_count = current_device_count() + 1;
610  if ( parent_device->is_submaster )
611  {
612  unsigned int parent_address = ( unsigned int )strtoul( parent_device->network_address.c_str(), NULL, 0 );
613  DEVICELOG_INFO( "Config", "%s,%s,%sSubmaster Device was added. Current port device count = %zu.", this_device->name.c_str(), this_device->canonical_model_name.c_str(), CUtil::get_formatted_address( _serial_port_settings->protocol(), true, parent_address, slave_addr ).c_str(), device_count );
614  }
615  else
616  {
617  DEVICELOG_INFO( "Config", "%s,%s,%sDevice was added. Current port device count = %zu.", this_device->name.c_str(), this_device->canonical_model_name.c_str(), CUtil::get_formatted_address( _serial_port_settings->protocol(), false, 0, slave_addr ).c_str(), device_count );
618  }
619  }
620 
621  // Grab eds and include it when creating device
622  std::shared_ptr<CParsedDeviceEds> device_eds = _eds_settings->parse_device_eds( this_device );
623 
624  if ( !device_eds )
625  {
626  SVR_WARN( 1, "Failed to add slave device (Address %s) to %s due to EDS parsing failure.", this_device->network_address.c_str(), my_name );
627  continue;
628  }
629 
630  // Create and store the device object
631  bool success = create_protocol_device( this_device, device_eds, parent_device );
632 
633  if ( success )
634  {
635  any_changes = true;
636  SVR_INFO( 1, "Added slave device (Address %s) to map for %s.", this_device->network_address.c_str(), my_name );
637  if ( ( device_eds->get_current_eds_checksum() != device_eds->get_previous_eds_checksum() ) &&
638  ( this_device->model.length() > 0 ) )
639  {
640  models_with_changed_eds_checksums.insert( std::make_pair( this_device->model, device_eds->get_current_eds_checksum() ) );
641  }
642  }
643  else
644  {
645  SVR_WARN( 1, "Failed to add slave device (Address %s) to Serial Port %s.", this_device->network_address.c_str(), get_port_name() );
646  }
647  }
648  catch ( generic_err &ex )
649  {
650  SVR_ERR( 1, "Failed to add slave device (Address %s) to map for %s: %s", this_device->network_address.c_str(), my_name, ex.what() );
651  }
652  }
653 
654  // Find all child devices of this Port.
655  // Use an unordered map for better search performance.
656  device_map_t all_child_devices_map;
657  all_child_devices_map = CDeviceService::find_device_map( _cds_handle, get_id(), FIND_ALL_CHILDREN );
658  SVR_FINFO( 1, "my_name %s: Device (ID %" PRIu64 ") has %zu child devices", my_name, get_id(), all_child_devices_map.size() );
659 
660  // Now compare map of devices under this Port against the map of devices
661  // in the CDS and remove any that have gone missing from the CDS.
662  std::vector<px::id_t> delete_these_devices;
663 
664  for ( auto map_entry : _devices )
665  {
666  px::id_t map_id = map_entry.first;
667  SVR_DEBUG( 2, " Start check for DH device (ID %" PRIu64 "), port (ID %" PRIu64 "), parent_device->id() (ID %" PRIu64 "), map_entry.second->get_parent_id() (ID %" PRIu64 ")", map_id, get_id(), parent_device->id(), map_entry.second->get_parent_id() );
668 
669  // Search the CDS map for the device (i.e. map_id)
670  auto dev = all_child_devices_map.find( map_id );
671  // Is the device in the CDS map?
672  if ( dev != all_child_devices_map.end() )
673  {
674  SVR_DEBUG( 2, " CDS device (ID %" PRIu64 ") found!", map_id );
675  }
676  else
677  {
678  // Add this device id for removal.
679  delete_these_devices.push_back( map_id );
680  }
681  }
682 
683  if ( delete_these_devices.size() )
684  {
685  for ( size_t i = 0; i < delete_these_devices.size(); i++ )
686  {
687  SVR_INFO( 1, "Removing missing device (ID %" PRIu64 ") from map for %s.", delete_these_devices[i], my_name );
688  any_changes |= remove_device( delete_these_devices[i], false ); // OR the result (hopefully true) of removal
689  }
690  }
691 
692  for ( const std::pair<px::string, uint32_t> &model_checksum_pair : models_with_changed_eds_checksums )
693  {
694  SVR_INFO( 1, "Updating EDS Checksum Model: %s New Checksum: %u", model_checksum_pair.first.c_str(), model_checksum_pair.second );
695  COobService::set_eds_model_checksum( _cds_handle, model_checksum_pair.first.c_str(), model_checksum_pair.second );
696  }
697 
698  // Now, look for any submasters and give them the same treatment,
699  // scanning for devices to add or remove
700  for ( std::shared_ptr<px::CDevice> &this_device : child_devices )
701  {
702  if ( this_device->is_submaster )
703  {
704  any_changes |= update_device_map( this_device, this_device->name.c_str(), device_added );
705  }
706  }
707 
708  return ( any_changes );
709 }
