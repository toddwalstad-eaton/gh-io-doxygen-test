127 void CIoMessenger::step_message( CIoMessage &msg )
128 {
129  PUSH_FUNC;
130  reset_comm_timer( 0 );
131  SVR_FDEBUG( 8, "step_message from state(%d) msg(%zu)", ( int )msg.get_state(), msg._msg_num );
132 
133  switch ( msg.get_state() )
134  {
135  case io_message_state::start:
136  msg.initialize( get_port_id() );
137  step_message( msg );
138  return;
139 
140  case io_message_state::interframe_delay:
141  do_async_wait( msg.get_interframe_delay(),
142  boost::bind( &CIoMessenger::handle_interframe_delay_done, this, pholder::error ),
143  _interframe_delay_timer );
144  return;
145 
146  case io_message_state::receive_flush:
147  // Call the messenger's receiver flush()
148  flush();
149  msg.on_receive_flush_complete();
150  step_message( msg );
151  return;
152 
153  case io_message_state::send:
154  reset_comm_timer( msg.get_timeout() );
155  {
156  size_t wbuf_size = msg.fill_write_buffer( _perm_wbuf );
157  // SVR_DEBUG(5, "(p%" PRIu64 ":d%" PRIu64 ") Initiating write operation", _port_id, _device_id );
158  boost::asio::const_buffer cbuf( &_perm_wbuf[0], wbuf_size );
159  write( cbuf, msg );
160  }
161  return;
162 
163  case io_message_state::receive:
164  reset_comm_timer( msg.get_timeout() );
165  {
166  boost::asio::mutable_buffer mbuf( boost::asio::buffer( _perm_rbuf ) );
167  read_more( mbuf, msg );
168  }
169  // SVR_DEBUG(5, "(p%" PRIu64 ":d%" PRIu64 ") Attempting to read more serial data", _port_id, _device_id );
170  return;
171 
172  case io_message_state::success:
173  if ( _throttle_msec )
174  {
175  try
176  {
177  /* std::this_thread::sleep_for() also throws exceptions, so we'll stick with boost for now */
178  boost::this_thread::sleep_for( boost::chrono::milliseconds( _throttle_msec ) );
179  }
180  catch ( const boost::exception &ex )
181  {
182  SVR_WARN( 2, "Exception when doing msec sleep: %s", diagnostic_information( ex ).c_str() );
183  }
184  }
185  next_msg();
186  return;
187 
188  case io_message_state::ack_cancel:
189  // Do nothing
190  return;
191 
192  case io_message_state::parse_error:
193  case io_message_state::timeout: // Fall through
194  case io_message_state::comm_error:
195  default:
196  cancel( msg );
197  next_msg();
198  return;
199  }

221 void CIoMessenger::handle_read( const boost::system::error_code &err, size_t nbytes )
222 {
223  PUSH_FUNC;
224  reset_comm_timer( 0 );
225  if ( unlikely( ( _current_xact_series == nullptr ) || _current_xact_series->empty() ) )
226  {
227  // Callback from cancel after read timeout (do nothing)
228  if ( err.value() != 0 )
229  {
230  SVR_FINFO( 4, "%s", err.message().c_str() );
231  }
232  return;

235  {
236  // No Error

240  // Signal the message that bytes are received

242  }
243  break;
244 
245  // Error "operation_aborted"
246  // This is generated by cancel() when a timeout occurs.
247  // Ignore this error
248  case boost::asio::error::operation_aborted:
249  // Callback from cancel hits here
250  SVR_FERR( 2, "Operation Aborted (eg, timeout)" );
251  _active_message->on_cancel_complete();
252  break;
253  // Handle error
254  default:
255  _active_message->on_err( err );
256  break;
257  } //switch
