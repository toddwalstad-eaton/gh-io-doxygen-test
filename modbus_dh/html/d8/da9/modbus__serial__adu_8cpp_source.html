134 bool decode_ADU_size( function_code_t func_code, uint8_t pdu_bytes_to_read, uint16_t received_bytes )
135 {
136  SVR_DEBUG( 6, "decode_ADU_size function code 0x%0X, pdu_bytes_to_read %u, received_bytes %u", func_code, received_bytes, pdu_bytes_to_read );
137 
138  switch ( func_code )
139  {
140  case MB_WRITE_SINGLE_COIL:
141  case MB_WRITE_SINGLE_REGISTER:
142  case MB_WRITE_MULTIPLE_COILS:
143  case MB_WRITE_MULTIPLE_HOLDING_REGISTERS:
144  return ( received_bytes >= BYTES_EXPECTED_WRITE_COIL_OR_REG );
145 
146  break;
147  // We have enough bytes for exception status
148  case MB_READ_EXCEPTION_STATUS:
149  // Read read device id functions require multi-stage packet inspection,
150  // return true here to default to the old hunt and peck CRC check behavior.
151  case MB_READ_DEVICE_IDENTIFICATION:
152  return ( true );
153 
154  break;
155  case MB_GET_EVENT_COUNTER:
156  return ( received_bytes >= BYTES_EXPECTED_GET_EVENT_COUNTER );
157 
158  break;
159  case MB_MASK_WRITE_REGISTER:
160  return ( received_bytes >= BYTES_EXPECTED_MASK_WRITE_REGISTER );
161 
162  break;
163  case MB_READ_DIAGNOSTICS:
164  return ( received_bytes >= BYTES_EXPECTED_READ_DIAGNOSTICS );
165 
166  break;
167  default:
168  return ( received_bytes >= ( MODBUS_SERIAL_ADU_BASE_BYTES_EXPECTED + pdu_bytes_to_read ) );
169 
170  break;
171  }
172 
173  return ( true );
