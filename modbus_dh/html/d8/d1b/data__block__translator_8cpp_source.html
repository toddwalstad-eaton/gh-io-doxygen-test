80 int CDataBlockTranslator::translate( const px::asio::io_buf_t &data, std::vector<std::shared_ptr<px::CValue> > &out )
81 {
82  PUSH_FUNC;
83  out.clear();
84  out.reserve( data.size() );
85 
86  const uint8_t *buf = &data[0];
87  bool has_open_multi_point = false;
88 
89  for ( Mapping mapping : _mappings )
90  {
91  std::shared_ptr<px::CValue> elem( nullptr );
92 
93  if ( likely( data.size() >= ( mapping.byte_offset + mapping.len ) ) )
94  {
95  if ( likely( !mapping.is_bit_mapping ) )
96  {
97  // Handle the length for a conversion to binary:
98  // If converting BINARY_BLOB or MULTIPT_BUFFER to binary, and len = 0 or 1,
99  // then we'll take whatever size buffer we got
100  if ( unlikely( ( mapping.val_type == px::datatype::binary ) &&
101  ( ( mapping.ext_type == ext_data_type_t::EXT_TYPE_MULTIPT_BUFFER ) || ( mapping.ext_type == ext_data_type_t::BINARY_BLOB ) ) &&
102  ( mapping.len <= 1 ) ) )
103  {
104  elem = toolkit::ExtTypeToNewValue( mapping.val_type, mapping.ext_type, buf + mapping.byte_offset, ( data.size() - mapping.byte_offset ) );
105  }
106  // Else, do the normal transform UNLESS this is a MULTIPT_BUFFER and no mappings have been defined for it:
107  else if ( likely( ( mapping.ext_type != ext_data_type_t::EXT_TYPE_MULTIPT_BUFFER ) || ( mapping.len != 0 ) ) )
108  {
109  elem = toolkit::ExtTypeToNewValue( mapping.val_type, mapping.ext_type, buf + mapping.byte_offset, mapping.len );
110  }
111  // Else, an Open-ended multi-point buffer with no mapping transforms defined
112  // ToDo: probably should pack all bytes into one px::CValue as a vector of bytes,
113  // rather than the array of shared pointers to individual bytes, as here.
114  // Would also have to handle that change in CDataBlock::get_block_value()
115  else
116  {
117  has_open_multi_point = true;
118  for ( size_t i = mapping.byte_offset; i < data.size(); i++ )
119  {
120  std::shared_ptr<px::CValue> mval( nullptr );
121  mval = toolkit::ExtTypeToNewValue( mapping.val_type, ext_data_type_t::MODBUS_ONE_BYTE_UNSIGNED, buf + i, 1 );
122  out.push_back( mval );
123  }
124  }
125  }
126  else
127  {
128  // Bit mapping: Mask out bits except the one we are interested in
129  uint8_t *masked = NULL;
130  masked = ( uint8_t * )malloc( mapping.len + 1 );
131  if ( likely( masked != nullptr ) )
132  {
133  masked[0] = *( buf + mapping.byte_offset ) & ( 1 << mapping.bit_offset );
134  elem = toolkit::ExtTypeToNewValue( mapping.val_type, mapping.ext_type, masked, mapping.len );
135  free( masked );
136  }
137  }
138  }
139 
140  if ( elem.get() && mapping.equation )
141  {
142  try
143  {
144  // Now using px::CValue rather than just double for equation value type
145  // We will stick with the trivial equation handler here because it gives us better control
146  // of our "Y=10*X" variables X and Y than the equation handler owned by the device.
147  CPxValueHandler equation_handler;
148  equation_handler.set( "X", *elem );
149  mapping.equation->evaluate( equation_handler );
150  px::CValue x = equation_handler.get( "Y" ).val;
151 
152  SVR_INFO( 8, "Update from Scaling Equation: {%s} fn(%.4f) = %.4f", mapping.scaling_expression.c_str(), elem->as_double(), x.as_double() );
153  *elem = x;
154  }
155  // Can't be this one! catch( const px::eq::not_found& nfex )
156  catch ( const px::generic_err &gex )
157  {
158  SVR_WARN( 2, "DBT mapping equation->evaluate(%s) threw: %s", mapping.scaling_expression.c_str(), gex.what() );
159  elem.reset();
160  }
161  catch ( ... )
162  {
163  SVR_WARN( 2, "DBT mapping equation->evaluate(%s) threw unknown exception", mapping.scaling_expression.c_str() );
164  elem.reset();
165  }
166  }
167 
168  if ( !has_open_multi_point )
169  {
170  out.push_back( elem );
171  if ( !elem.get() )
172  {
173  SVR_WARN( 4, "DBT translation failed for %s", toolkit::ExtTypeEnumToString( mapping.ext_type ) );
174  }
175  }
176  else if ( !out.size() )
177  {
178  SVR_WARN( 4, "DBT translation failed for open-ended multi-point buffer. Type=%s", toolkit::ExtTypeEnumToString( mapping.ext_type ) );
179  }
180  }
181 
182  return ( 0 );
183 }

186 {
187  PUSH_FUNC;

190 
191  if ( !scaling_expression.empty() )
