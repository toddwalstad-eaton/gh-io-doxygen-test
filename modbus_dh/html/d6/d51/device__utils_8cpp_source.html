612 void CDeviceUtil::check_channel_properties( const CDevice &device, const std::shared_ptr<px::CChannelSettings> &ch_settings, px::cds::CSetFieldValue &sfv )
613 {
614  PUSH_FUNC;
615 
616  SVR_DEBUG( 8, " ##################### check_channel_properties: model %s, channel ID %" PRIu64 ", canonical_name %s",
617  device._px_device->model.c_str(), ch_settings->id(), ch_settings->canonical_name.c_str() );
618 
619  // This function will modify the IQ260 Pulse Output/Digital Input Card
620  // mKYZMeter1-4 channels based on the pulse ouput configuration
621  if ( IQ260_KYZ_CARD_NAME == device._px_device->model )
622  {
623  if ( !ch_settings.get() )
624  {
625  SVR_ERR( 2, "check_channel_properties: invalid ch_settings object" );
626  return;
627  }
628 
629  size_t found = ch_settings->canonical_name.find( IQ260_KYZ_METER_CH );
630  if ( found == px::string::npos )
631  {
632  return;
633  }
634 
635  // Get the the mKYZMeter1-4 index from the channel name and convert to int
636  char tmp_kyz = ch_settings->canonical_name.back();
637  if ( !isdigit( tmp_kyz ) )
638  {
639  return;
640  }
641  int kyz = tmp_kyz - '0';
642 
643  // Transform mKYZMeter1-4 index to a xCardConfig block index to get the Pulse Ouput #1-4 Config value
644  unsigned int block_index = 0;
645  switch ( kyz )
646  {
647  case 1:
648  block_index = 1;
649  break;
650  case 2:
651  block_index = 3;
652  break;
653  case 3:
654  block_index = 5;
655  break;
656  case 4:
657  block_index = 7;
658  break;
659  default:
660  return;
661 
662  break;
663  }
664 
665  // Get the "xCardConfig" block: It will hold the configuration for Pulse Ouputs #1-4
666  uint64_t block_id = device.lookup_block_id( IQ260_CARD_CFG );
667 
668  if ( block_id != 0 )
669  {
670  // Get Pulse Ouput #1-4 Config value.
671  uint32_t out_cfg = 0;
672  std::shared_ptr<px::CValue> block_value = device.get_block_value( block_id, block_index );
673  if ( block_value.get() )
674  {
675  out_cfg = block_value->as_uint32();
676  }
677  else
678  {
679  return;
680  }
681 
682  // 1) Decode Source for Pulse Ouput#X data into phase and meter info
683  // 2) Use phase and meter info to generate an index into the kyz_fields table
684  //
685  // -----ppp ----vvvv
686  //
687  // " ppp" (Phase):
688  // 000 = none
689  // 001 = Phase A
690  // 010 = Phase B
691  // 011 = Phase C
692  // 100 = All Phases
693  // 101 = Pulse from EOI(End Of Interval)
694  //
695  // "vvvv"(Value) i.e. Meter Type:
696  // 0000 = none,
697  // 0001 = Wh,
698  // 0010 = +Wh,
699  // 0011 = -Wh,
700  // 0100 = Varh,
701  // 0101 = +Varh,
702  // 0110 = -Varh,
703  // 0111 = VAh,
704  // 1000 = Received Wh,
705  // 1001 = Delivered Wh,
706  // 1010 = Inductive Varh,
707  // 1011 = Capacitive Varh
708  //
709  uint32_t phase = out_cfg >> 8;
710  uint32_t meter = out_cfg & 255;
711  int kyz_index = 0;
712 
713  switch ( phase )
714  {
715  case IQ260_PHASE_A:
716  kyz_index = IQ260_PHASE_A_IDX + meter;
717  break;
718  case IQ260_PHASE_B:
719  kyz_index = IQ260_PHASE_B_IDX + meter;
720  break;
721  case IQ260_PHASE_C:
722  kyz_index = IQ260_PHASE_C_IDX + meter;
723  break;
724  case IQ260_PHASE_ABC:
725  kyz_index = IQ260_PHASE_ABC_IDX + meter;
726  break;
727  case IQ260_EOI:
728  kyz_index = IQ260_EOI_IDX;
729  break;
730  default:
731  kyz_index = IQ260_EOI_IDX + kyz;
732  break;
733  }
734 
735  // Only update channel settings if the "name" strings do not match
736  if ( ( kyz_index < IQ260_KYZ_MAX ) && ( ch_settings->name != kyz_fields[kyz_index].name ) )
737  {
738  px::cds::CSingleField request;
739  request.id = ch_settings->id();
740  request.type = px::cds::field_message_type::channel;
741 
742  // Handle 'name' setting
743  px::CValue new_value1 = kyz_fields[kyz_index].name;
744  request.field = "channel_configuration.name";
745  request.value = new_value1;
746  sfv.fields.insert( request );
747 
748  // Handle 'short_name' setting
749  px::CValue new_value2 = kyz_fields[kyz_index].short_name;
750  request.field = "channel_configuration.short_name";
751  request.value = new_value2;
752  sfv.fields.insert( request );
753 
754  // Handle 'units' setting
755  px::CValue new_value3 = kyz_fields[kyz_index].units;
756  request.field = "channel_configuration.units";
757  request.value = new_value3;
758  sfv.fields.insert( request );
759 
760  // Handle 'abbreviated_units' setting
761  px::CValue new_value4 = kyz_fields[kyz_index].abbr_units;
762  request.field = "channel_configuration.abbreviated_units";
763  request.value = new_value4;
764  sfv.fields.insert( request );
765 
766  SVR_DEBUG( 8, " ##################### Updating the name and units for channel ID %" PRIu64 ", %s is now %s",
767  ch_settings->id(), ch_settings->name.c_str(), kyz_fields[kyz_index].name );
768  }
769  }
770  }
