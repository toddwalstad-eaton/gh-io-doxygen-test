40  const std::shared_ptr<CPort> &port_container,
41  const std::string &path,
42  unsigned id )
43  : CIoMessenger( manager, port_container ),
44  _port_open( false ),
45  _id( id )
46 {
47  PUSH_FUNC;
48  SVR_DEBUG( 2, "Creating serial connection on '%s'", path.c_str() );
49 
50  try
51  {
52  // Create the port object, but without opening it yet
53  // (So we can test whether the open really works)
54  _port.reset( new serial_port( get_io_service() ) );
55  assert( _port.get() ); // pointer must have valid contents
56 
57  // Now test opening the port
58  boost::system::error_code ec;
59  _port->open( path.c_str(), ec );
60  // Note that we can't tell if the port was already open;
61  // the same port can be opened more than once.
62  if ( ( ec == boost::system::errc::success ) && _port->is_open() )
63  {
64  // Set the serial port to its defaults
65  set_baud_rate( px::proto::BAUD_9600 );
66  set_flow_control( px::proto::FLOW_NONE );
67  set_char_size( px::proto::CHAR_SIZE_8 );
68  set_parity( px::proto::PARITY_NONE );
69  set_stop_bits( px::proto::BITS_ONE );
70  _port_open = true;
71  }
72  else
73  {
74  _port.reset(); // release the pointer
75  SVR_ERR( 0, "Opening %s failed: %s", path.c_str(), ec.message().c_str() );
76  }
77  }
78  catch ( const boost::exception &ex )
79  {
80  SVR_ERR( 0, "While opening '%s': %s", path.c_str(), diagnostic_information( ex ).c_str() );
81  assert( false );
82  }
