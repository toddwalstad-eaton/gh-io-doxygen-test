157 {
158  PUSH_FUNC;
159 
160  // Initial value is all-ones
161  rtu_crc_t retval = 0xFFFF;
162 
163  retval = build_crc( retval, adu.slave );
164  retval = build_crc( retval, adu.pdu.function );
165 
166  // Certain Modbus Function Codes require additional protocol data to be in the CRC.
167  if ( MB_READ_DIAGNOSTICS == adu.pdu.function )
168  {
169  // Modbus Function Code 08 requires the Sub-Function code (2-bytes) to be in the CRC.
170  retval = build_crc( retval, ( unsigned char ) ( adu.pdu.sub_function >> 8 ) ); // Hi byte
171  retval = build_crc( retval, ( unsigned char ) ( adu.pdu.sub_function & 0xFF ) ); // Lo byte
172  }
173  for ( auto ch : adu.pdu.data )
174  {
175  retval = build_crc( retval, ch );
176  }
177 
178  SVR_FDEBUG( 9, "Method 2 CRC: 0x%x", retval );
179 
180  return ( retval );
