160 void CTcpProxy::handle_read( const boost::system::error_code &err, const size_t &bytes )
161 {
162  if ( err || ( bytes < MODBUS_TCP_ADU_SIZE ) )
163  {
164  close_local_socket();
165  SVR_FERR( 2, "bytes read %zu, err %s", bytes, err.message().c_str() );
166  }
167  else
168  {
169  // Modbus TCP ADU
170  // 00 01 02 03 04 05 06 07 08 09 0A 0B
171  // Trans | Prot |
172  // Hi Lo | ID | Len | UID | FC | Addr Regs
173  // 00 02 | 00 00 | 00 06 | 36 | 03 | 03 e7 00 06
174  //
175 
176  //SVR_FDEBUG(1, " ");
177  //SVR_DBGDUMP(1, _local_data_buf, bytes);
178 
179  if ( !_local_data_buf[MBAP_HDR_PROT_ID_HI] && !_local_data_buf[MBAP_HDR_PROT_ID_LO] )
180  {
181  // Lock access to the proxy map
182  std::lock_guard<std::mutex> L( _proxy_mutex );
183  auto it_dev = _proxy_devices.find( _local_data_buf[MBAP_HDR_UID] );
184  if ( it_dev != _proxy_devices.end() )
185  {
186  // Check if we have reached the session limit
187  if ( _sessions.size() >= _max_connections )
188  {
189  // We have reached session limit maximum size, send a Modbus BUSY error response, and then close the socket.
190  send_error_response( SLAVE_DEVICE_BUSY, _local_sock, _local_data_buf[MBAP_HDR_UID], _local_data_buf, sizeof( _local_data_buf ) );
191  close_local_socket();
192  SVR_FERR( 2, "Reached active session limit" );
193  }
194  else
195  {
196  SVR_DEBUG( 4, "Start proxy session [%s:%u], uid %u", it_dev->second.upstream_ip.c_str(), it_dev->second.upstream_port, it_dev->second.upstream_uid );
197 
198  // Create a new session
199  std::shared_ptr<CTcpSession> session;
200  session = CTcpSession::create( this, get_io_service(), _local_sock, it_dev->second );
201  // Limit the scope of std::lock_guard instead of using std::unique_lock
202  {
203  // Lock access to the session set
204  std::lock_guard<std::mutex> L( _session_mutex );
205  _sessions.insert( session );
206  }
207  try
208  {
209  session->start( _connect_timeout, _local_data_buf, sizeof( _local_data_buf ), bytes );
210  }
211  catch ( const boost::exception &ex )
212  {
213  // Target is unreachable. Send a Modbus error response, and then close the socket.
214  send_error_response( GATEWAY_PATH_UNAVAILABLE, _local_sock, _local_data_buf[MBAP_HDR_UID], _local_data_buf, sizeof( _local_data_buf ) );
215  close_local_socket();
216  SVR_FERR( 1, "Exception thrown at TCP session start: %s", diagnostic_information( ex ).c_str() );
217  }
218  }
219  }
220  else
221  {
222  // Target is unreachable. Send a Modbus error response, and then close the socket.
223  send_error_response( GATEWAY_PATH_UNAVAILABLE, _local_sock, _local_data_buf[MBAP_HDR_UID], _local_data_buf, sizeof( _local_data_buf ) );
224  close_local_socket();
225  SVR_FERR( 2, "Did not find UID %u in the device list", _local_data_buf[MBAP_HDR_UID] );
226  }
227  }
228  else
229  {
230  // Close socket immediately on a protocol ID failure
231  close_local_socket();
232  SVR_FERR( 2, "Invalid Modbus Protocol ID [%u%u]", _local_data_buf[MBAP_HDR_PROT_ID_HI], _local_data_buf[MBAP_HDR_PROT_ID_LO] );
233  }
234  }
235 
236  // Prepare for the next incoming connection
237  accept();

244  const socket_ptr_t &socket,
245  const unit_id_t &uid,
246  unsigned char *data_buf,
247  const size_t &data_buf_size )
248 {
249  PUSH_FUNC;
250 
251  if ( !socket || !data_buf || ( data_buf_size < MODBUS_TCP_ERROR_LEN ) )
252  {
253  return;
254  }
255 
256  try
257  {
258  // Form an error response Modbus PDU.
259  //
260  // Hi Lo | ID | Len | UID | FC | Data
261  // req: 01 00 | 00 00 | 00 06 | 01 | 03 | 0f a1 00 06
262  // resp: 01 00 | 00 00 | 00 03 | 01 | 83 | 0A
263 
264  data_buf[MBAP_HDR_LEN_HI] = 0x00;
265  data_buf[MBAP_HDR_LEN_LO] = MODBUS_EXCEPTION_LEN;
266  data_buf[MBAP_HDR_UID] = uid;
267  data_buf[MBAP_HDR_FC] = MODBUS_EXCEPTION_BASE + data_buf[MBAP_HDR_FC];
268  data_buf[MBAP_HDR_DATA_START] = error;
269 
270  // Send error repsonse
271  write( *socket, boost::asio::buffer( data_buf, MODBUS_TCP_ERROR_LEN ) );
272  }

275  SVR_FERR( 1, "When sending Modbus TCP error response: %s", e.what() );
