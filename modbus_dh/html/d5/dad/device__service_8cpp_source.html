53  {

56

61 
62  std::shared_ptr<px_oob::SerialPortSettings> p_port_settings;
63  serial_port_config_t pair;
64 
65  if ( port_settings_vec.size() > 0 )
66  {
67  const px_oob::SerialPortSettings sps = port_settings_vec.at( 0 );
68  p_port_settings.reset( new px_oob::SerialPortSettings( sps ) );
69 
70  pair.first = p_dev;
71  pair.second = p_port_settings;
72 
73  //TODO: separate logging threads per port would be a better solution when we have only 1 DH
74  if ( dev_id != 1 )

78  if ( !ident.empty() )
79  {
80  toolkit::CSysLog::set_ident( ident.c_str(), true );
81  }
82  }
83  }
84  else
85  {
86  // ToDo: if no settings, no entry is made
87  // But for now, we press ahead and maybe make some fake ones

89  // pair.second = nothing
90  }
91  port = pair;
92  break;
93  }
94  }
95  }
96  catch ( ... )

98  SVR_FERR( 1, "CDS submit_wait threw an exception!" );

259 void CDeviceService::set_CDS_device_comm_state( const std::shared_ptr<ICdsHandle> &cds_handle,
260  const std::shared_ptr<px::CDevice> &px_device,
261  const bool is_communicating,
262  const bool /*to_syslog*/ )
263 {
264  PUSH_FUNC;
265  assert( px_device.get() );
266  assert( px_device->id() );
267 
268  if ( !cds_handle.get() )
269  {
270  SVR_FERR( 1, "Invalid CDS handle!" );
271  return;
272  }
273 
274  // We only want to selectively set or clear the not_communicating bit flag
275  // depending on the is_communicating arg.
276  // Assume good comm first:
277  px::cds::state_op op = px::cds::state_op::op_clear_flags;
278  if ( !is_communicating )
279  {
280  op = px::cds::state_op::op_set_flags;
281  }
282 
283  px::CBitset<px::realtime_state> requested_flags;
284  requested_flags.reset();
285  // Set this one bit in the Bitset, whether we're clearing or setting the not_communicating state
286  requested_flags.set( px::realtime_state::not_communicating, true );
287 
288  px::cds::CSetDeviceState req;
289  req.devices.push_back( px::cds::CSetSingleDeviceState( px_device->id(), requested_flags, op ) );
290  if ( req.devices.size() == 0 )
291  {
292  SVR_DEBUG( 5, "Error in Device %" PRIu64 "! Cannot set Device comm state to %s", px_device->id(), ( is_communicating ) ? "Good" : "Bad" );
293  }
294  else
295  {
296  try
297  {
298  // We will just "fire and forget" this setting (async, no interest in the return or unlikely error)
299  cds_handle->submit( req );
300 
301  SVR_NOTICE( 1, "Set Device %" PRIu64 " state to %s communicating", px_device->id(), ( is_communicating ? "IS" : "NOT" ) );
302 
303  // Now update our copy of this flag, too:
304  px::CBitset<px::realtime_state> new_state( px_device->get_state() );
305 
306  new_state.set( px::realtime_state::not_communicating, !is_communicating );
307  px_device->set_state( new_state );
308  }
309  catch ( ... )
310  {
311  SVR_FERR( 1, "CDS submit threw an exception!" );
312  }
313  }
