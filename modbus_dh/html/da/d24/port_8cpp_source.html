219 // Poll the devices and get the highest priority transaction series from one.
220 transaction_deque_t *CPort::get_next_transaction_series( CPollClock &tnow )
221 {
222  PUSH_FUNC;
223 
224  // Check if any devices need to be deleted from the _devices map
225  review_devices_validity();
226 
227  // Check if any devices need to be synchronized to the PXG RTC.
228  if ( _stats_timer->check_if_next_interval() )
229  {
230  SVR_INFO( 5, "Crossed stats interval period - write stats file" );
231  log_stats();
232  }
233 
234  dvmap_iter_t pos;
235  // Find the last device polled
236  if ( _last_device == 0 )
237  {
238  // First time or reset due to device delete
239  pos = _devices.begin();
240  if ( pos != _devices.end() )
241  {
242  _last_device = pos->first;
243  }
244  }
245  else
246  {
247  pos = _devices.find( _last_device );
248  // Lookup shouldn't fail, but if it does, start at beginning
249  if ( pos == _devices.end() )
250  {
251  pos = _devices.begin();
252  _last_device = pos->first;
253  }
254  }
255 
256  // Cycle through and update and check all devices, but
257  // count loops and bail out when we've seen them all
258  size_t loops = 0;
259  uint32_t prio = 0;
260  px::id_t dev_idx = 0;
261  do
262  {
263  // Now move to the next device and give it first consideration
264  // (If the last device polled has priority equal to another,
265  // the other will be seen first and win.)
266  if ( pos != _devices.end() )
267  {
268  ++pos;
269  }
270 
271  // Iterate through until we get a good one
272  while ( ( pos == _devices.end() ) || pos->second->test_device_state( px::realtime_state::disabled ) )
273  {
274  // Wrap around if we are at the end of the map
275  if ( pos == _devices.end() )
276  {
277  pos = _devices.begin();
278  }
279  else
280  {
281  ++pos; // It is disabled or to be skipped
282  if ( ++loops > _devices.size() )
283  {
284  break;
285  }
286  }
287  }
288 
289  if ( loops > _devices.size() )
290  {
291  break; // break out of the do-while loop, too
292  }
293 
294  // pos->first is the ID, ->second is unique_ptr to the CDevice object
295  pos->second->update_polling_state( tnow.get_monotonic_poll_time() );
296  auto newprio = pos->second->priority();
297  if ( newprio > prio )
298  {
299  // We have a winner!
300  assert( pos->first > 0 );
301  dev_idx = pos->first;
302  prio = newprio;
303  }
304  }
305  while ( ++loops < _devices.size() );
306 
307  if ( dev_idx > 0 )
308  {
309  return ( _devices[dev_idx]->next_transaction_series() );
310  }
311 
312  return ( nullptr );
313 }

788 void CPort::device_command_handler( std::shared_ptr<px::cds::CSubscriberNotify> &update )
789 {
790  PUSH_FUNC;
791 
792  if ( update->command.empty() )
793  {
794  // This circumstance occurs when a device has been deleted
795  SVR_WARN( 2, "Caught Device Command request on Port '%s', but Command list is empty!", get_port_name() );
796  return;
797  }
798 
799  const px::CDeviceCommand &cds_dev_cmd = update->command;
800  SVR_INFO( 1, "CPort: Caught Device Command '%s' request for Device ID=%" PRIu64 " on Port '%s'. CDS command handle is %u.",
801  cds_dev_cmd.name.c_str(), cds_dev_cmd.device_id, get_port_name(), cds_dev_cmd.handle );
802 
803  if ( !cds_dev_cmd.handle )
804  {
805  SVR_ERR( 1, "Invalid CDS command handle for Command '%s' request on Port '%s'!",
806  cds_dev_cmd.name.c_str(), get_port_name() );
807  return;
808  }
809 
810  // Check if this is a network port command
811  if ( get_id() == cds_dev_cmd.device_id )
812  {
813  err_t error = err_t::NO_ERROR;
814  if ( DEVCMD_RESET_PROD_STATS == cds_dev_cmd.name )
815  {
816  reset_stats();
817  }
818  else if ( DEVCMD_TRIGGER_DATA_LOG == cds_dev_cmd.name )
819  {
820  log_block_stats();
821  }
822  else
823  {
824  error = err_t::COMMAND_NOT_FOUND;
825  }
826 
827  if ( err_t::NO_ERROR == error )
828  {
829  SVR_DEBUG( 3, "Command %s was sent successfully to Port %s", cds_dev_cmd.name.c_str(), get_port_name() );
830  if ( _cds_handle.get() )
831  {
832  CCommandService::send_command_result( _cds_handle, cds_dev_cmd,
833  px::device_command_status::succeeded,
834  px::device_command_substatus::none, NULL );
835  }
836  }
837  else
838  {
839  DEVICELOG_ERR( "Status", "%sUnable to process command request %s. Command was not found!", CUtil::get_no_device_string().c_str(), cds_dev_cmd.name.c_str() );
840  SVR_DEBUG( 1, "Port command %s was not found!", cds_dev_cmd.name.c_str() );
841  // Send error response
842  if ( _cds_handle.get() )
843  {
844  char err_msg[256];
845  snprintf( err_msg, sizeof( err_msg ), "Command %s was not found!", cds_dev_cmd.name.c_str() );
846  CCommandService::send_command_result( _cds_handle, cds_dev_cmd,
847  px::device_command_status::failed,
848  px::device_command_substatus::none, err_msg );
849  }
850  }
851  return;
852  }
853 
854  auto device = _devices.find( cds_dev_cmd.device_id );
855 
856  // Is the device in the list?
857  if ( device == _devices.end() )
858  {
859  // Send error response
860  if ( _cds_handle.get() )
861  {
862  DEVICELOG_ERR( "Status", "%sUnable to process command request %s. Could not find Device ID %" PRIu64 " in the Device list for Port '%s'!",
863  CUtil::get_no_device_string().c_str(), cds_dev_cmd.name.c_str(), cds_dev_cmd.device_id, get_port_name() );
864 
865  SVR_ERR( 1, "device_command_callback: Could not find ID %" PRIu64 " in the Device list for Port '%s'!",
866  cds_dev_cmd.device_id, get_port_name() );
867 
868  char err_msg[256];
869  snprintf( err_msg, sizeof( err_msg ), "Could not find ID %" PRIu64 " in the Device list for Port '%s'!",
870  cds_dev_cmd.device_id, get_port_name() );
871 
872  CCommandService::send_command_result( _cds_handle, cds_dev_cmd,
873  px::device_command_status::failed,
874  px::device_command_substatus::none, err_msg );
875  }
876  return;
877  }
878 
879  // Is the device disabled, not communicating or is polling blocked?
880  if ( device->second->test_device_state( px::realtime_state::disabled ) ||
881  device->second->test_device_state( px::realtime_state::not_communicating ) )
882  {
883  // Send error response
884  if ( _cds_handle.get() )
885  {
886  DEVICELOG_ERR( "Status", "%s%sUnable to process command request %s! Command requests are currently not allowed for this device.",
887  device->second->get_formatted_name().c_str(),
888  device->second->get_formatted_address().c_str(),
889  cds_dev_cmd.name.c_str() );
890  SVR_ERR( 1, "Command requests are currently not allowed for Device ID %" PRIu64 " on Port '%s'!", cds_dev_cmd.device_id, get_port_name() );
891 
892  char err_msg[256];
893  snprintf( err_msg, sizeof( err_msg ), "Command requests are currently not allowed for Device ID %" PRIu64 " on Port '%s'!",
894  cds_dev_cmd.device_id, get_port_name() );
895  CCommandService::send_command_result( _cds_handle, cds_dev_cmd,
896  px::device_command_status::failed,
897  px::device_command_substatus::none, err_msg );
898  }
899  return;
900  }
901 
902  // Dispatch the command to the Device:
903  device->second->execute_command( cds_dev_cmd );
904 }
