71 void CTcpSocket::connect()
72 {
73  PUSH_FUNC;
74 
75  if ( _disabled )
76  {
77  return;
78  }
79 
80 #ifdef PXDASHBOARD
81  // The PXDBP does not require the remote socket address to be on the same subnet.
82 #else
83  // Check if IP address is valid
84  if ( !CModbusUtil::is_ip_in_range( _tcp_addr.ip, _tcp_addr.subnet_addr, _tcp_addr.subnet_mask ) )
85  {
86  _reconnecting = false;
87  reset();
88  return;
89  }
90 #endif
91  // Use raw socket ping attempt to find device first before attempting Boost async_connect().
92  int ping_response = CModbusUtil::ping( _tcp_addr.ip, _connect_timeout );
93  //SVR_FDEBUG(6, "ping_response %d", ping_response);
94  if ( 0 == ping_response )
95  {
96  _reconnecting = false;
97  reset();
98  return;
99  }
100 
101  _reconnecting = true;
102  _sock_state = tcp_sock_state::connecting;
103  _sock.reset( new tcp::socket( _ios ) );
104 
105  SVR_FDEBUG( 6, "Resolving device address %s on remote port %u, using timeout of %u milliseconds ",
106  _tcp_addr.ip.c_str(), _tcp_addr.port, _connect_timeout );
107 
108  // Start connection timer
109  _timer.reset( new boost::asio::steady_timer( _ios ) );
110  try
111  {
112  _timer->expires_from_now( std::chrono::milliseconds( _connect_timeout ) );
113  _timer->async_wait( boost::bind( &CTcpSocket::resolver_timeout, this ) );
114 
115  // Try to Connect to the remote device
116  _sock->async_connect( tcp::endpoint( boost::asio::ip::address::from_string( _tcp_addr.ip ), _tcp_addr.port ),
117  boost::bind( &CTcpSocket::post_connect,
118  this,
119  pholder::error ) );
120  }
121  catch ( const boost::exception &ex )
122  {
123  SVR_NOTICE( 2, "Connection to TCP %s failed: %s", _tcp_addr.ip.c_str(), diagnostic_information( ex ).c_str() );
124  }
125 }
