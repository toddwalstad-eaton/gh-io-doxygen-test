66 bool CModbusTcpSerialProtocolHandler::parse_request( err_t &err_was )
67 {
68  PUSH_FUNC;
69 
70  assert( _adu );
71  assert( _transactor != nullptr );
72  px::asio::CIoTransactor &transactor = *_transactor;
73  px::asio::iobuf_const_pair_t rbuf( transactor.get_rbuf() );
74 
75  //SVR_FDEBUG(4, "Parsing incoming command");
76  //SVR_DBGDUMP(6, &(*rbuf.first), rbuf.second-rbuf.first);
77 
78  // First, check that we have room for this request on our queue
79  if ( _my_port->get_pass_through_queue_size() >= NumberMaxOfServerTransactions )
80  {
81  SVR_NOTICE( 2, "MBTCP request rejected since its queue is full." );
82  // We have to parse through to load the ADU and PDU we need, but ignore result
83  _adu->parse( rbuf.first, rbuf.second );
84  // Clear the read buffer
85  transactor.clear_rbuf( rbuf.second );
86  err_was = err_t::SLAVE_DEVICE_BUSY;
87  err_response( err_was );
88  return ( true );
89  }
90 
91  // Parse the input buffer
92  px::asio::parse_result_t result( _adu->parse( rbuf.first, rbuf.second ) );
93  assert( result.second <= rbuf.second );
94  err_was = _adu->get_parse_err();
95 
96  if ( boost::logic::indeterminate( result.first ) )
97  {
98  // Buffer is incomplete
99  SVR_NOTICE( 5, "MBTCP Buffer incomplete ???" );
100  return ( false );
101  }
102 
103  if ( !result.first )
104  {
105  SVR_DEBUG( 5, "Error (%s) encountered while parsing MBTCP request", CUtil::get_err_text( err_was ) );
106  if ( err_was == err_t::DATA_VALIDITY_ERROR )
107  {
108  // This indicates the one case where we don't reply to a MBTCP request:
109  // The Protocol Identifier wasn't 0, so we don't know how to interpret this request
110  transactor.clear_rbuf( rbuf.second );
111  return ( false ); // so we go back to reading more
112  }
113  else
114  {
115  // The input is no longer necessary; discard [this portion of] the
116  // read buffer
117  transactor.clear_rbuf( result.second );
118  err_response( err_was );
119  return ( true );
120  }
121  }
122 
123  // The input is no longer necessary; discard [this portion of] the
124  // read buffer
125  transactor.clear_rbuf( result.second );
126 
127  return ( true );

142 void CModbusTcpSerialProtocolHandler::execute()
143 {
144  PUSH_FUNC;
145 
146  assert( _adu );
147  assert( _my_port );
148 
149  assert( _transactor != nullptr );
150  px::asio::CIoTransactor &transactor = *_transactor;
151 
152  SVR_FDEBUG( 5, "Executing Modbus/TCP request id %u for slave address %u", _adu->header.tid, ( unsigned )_adu->header.uid );
153 
154  if ( _pass_through_timeout != 0 )
155  {
156  // Start an I/O timer right now, setting a timer at 90% of our
157  // max timeout (to allow for some processing delay)
158  size_t timeout = _pass_through_timeout * 9 / 10;
159  SVR_DEBUG( 3, "Using pass_through_timeout of %zu msec", timeout );
160 
161  timer = px::asio::CIoTimer::create( transactor, timeout );
162  }
163 
164  // Screen out 0 and 248-255 as invalid Modbus addresses
165  // (The Modbus protocol doesn't allow addresses above 247)
166  if ( ( _adu->header.uid < MODBUS_MIN_SLAVE_ADDRESS ) || ( _adu->header.uid > MODBUS_MAX_SLAVE_ADDRESS ) )
167  {
168  SVR_NOTICE( 2, "Received Modbus/TCP request for illegal slave address %u", ( unsigned )_adu->header.uid );
169  err_response( GATEWAY_PATH_UNAVAILABLE );
170  return;
171  }
172 
173  // That's all of the TCP-specific stuff; now we can utilize the
174  // general-purpose request handler
175  handle_request( _adu->pdu, ( px::id_t )_adu->header.uid );
176 }

178 void CModbusTcpSerialProtocolHandler::handle_request( CModbusPdu &pdu, px::id_t dev_id )
179 {
180  PUSH_FUNC;
181 
182  // Build the Modbus serial message we want
183  std::unique_ptr<CModbusSerialAdu> rtu_msg( new CModbusSerialAdu );
184  rtu_msg->slave = _adu->header.uid;
185  rtu_msg->pdu.function = pdu.function;
186  // pdu.payload (CModbusFunction) doesn't work for us (no DataBlock) so just leave the pointer empty
187  rtu_msg->pdu.data = pdu.data;
188 
189  // Create the protocol transaction that binds device ID, handler, msg, and timeout
190  try
191  {
192  std::shared_ptr<CModbusTcpSerialProtocolHandler> mememe = shared_from_this();
193  CModbusProtocolMessage *rtu_transaction =
194  new CModbusProtocolMessage( dev_id, mememe, std::move( rtu_msg ), _pass_through_timeout );
195  // Queue up the request to send to the serial port.
196  transaction_deque_t pt_deque;
197  std::shared_ptr<CIoMessage> entry( rtu_transaction );
198  SVR_INFO( 5, "(p%" PRIu64 ":ptd%" PRIu64 ") MBTCP queued up transaction ID %u function %u for UnitID %u",
199  _my_port->get_id(), dev_id, ( unsigned )_adu->header.tid, ( unsigned )_adu->pdu.function, ( unsigned )_adu->header.uid );
200  pt_deque.push_back( std::move( entry ) );
201  _my_port->add_pass_through_transaction( pt_deque );
202  }
203  catch ( const std::bad_weak_ptr &ex )
204  {
205  SVR_FNOTICE( 1, "Cannot handle request: %s", ex.what() );
206  }
207 }
