133 err_t CWaveformState::read_file( std::unordered_map<px::id_t, std::vector<uint32_t> > &waveform_state_map )
134 {
135  PUSH_FUNC;
136  err_t rc = err_t::NO_ERROR;
137 
138  if ( boost::filesystem::exists( _storage_path ) )
139  {
140  try
141  {
142  boost::filesystem::ifstream fis;
143  fis.open( _storage_path );
144  boost::uintmax_t size = boost::filesystem::file_size( _storage_path );
145 
146  // Allocate memory and read file
147  std::vector<char> content;
148  content.resize( size, 0 );
149  fis.read( ( char * )content.data(), size );
150  fis.close();
151  std::string json( content.data() );
152  waveform_state_map = from_json( json );
153  SVR_DEBUG( 8, "waveform data file exists. waveform_state_map size = %zu", waveform_state_map.size() );
154  }
155  catch ( ... )
156  {
157  // Failed to allocate memory or read file.
158  // Will catch also this exception if the file is zero-length.
159  SVR_ERR( 1, "Caught exception in CWaveformState::read_file" );
160 
161  // Delete existing waveform state file and clear the waveform_state_map
162  waveform_state_map.clear();
163  rc = delete_file();
164  }
165  }
166  else
167  {
168  SVR_DEBUG( 8, "No waveform data file" );
169  waveform_state_map.clear();
170  }
171  return ( rc );
172 }

239 std::unordered_map<px::id_t, std::vector<uint32_t> > CWaveformState::from_json( const std::string &json ) const
240 {
241  PUSH_FUNC;
242 
243  std::unordered_map<px::id_t, std::vector<uint32_t> > wave_map;
244 
245  try
246  {
247  ::json_spirit::Value root_val;
248  ::json_spirit::read_string_or_throw( json, root_val );
249 
250  for ( const auto &top_val : root_val.get_obj() )
251  {
252  if ( "WaveformState" == top_val.name_ )
253  {
254  ::json_spirit::Array device_array = top_val.value_.get_array(); // array of device objects
255 
256  for ( const auto &device_array_pair : device_array )
257  {
258  ::json_spirit::Object device_obj = device_array_pair.get_obj();
259  px::id_t device_id = PX_ID_UNSET;
260  std::vector<uint32_t> wave_event_indices;
261 
262  for ( const ::json_spirit::Pair &device_pair : device_obj )
263  {
264  if ( "DeviceId" == device_pair.name_ )
265  {
266  device_id = ( px::id_t )device_pair.value_.get_uint64();
267  }
268  else if ( "WaveformEventIndex" == device_pair.name_ )
269  {
270  ::json_spirit::Array wave_event_array = device_pair.value_.get_array();
271 
272  for ( const ::json_spirit::Value &wave_event_val : wave_event_array )
273  {
274  wave_event_indices.push_back( ( uint32_t )wave_event_val.get_uint64() );
275  }
276  }
277  }
278  if ( device_id != PX_ID_UNSET )
279  {
280  wave_map.insert( std::make_pair( device_id, wave_event_indices ) );
281  }
282  }
283  }
284  }
285  }
286  catch ( ::json_spirit::Error_position &e )
287  {
288  std::ostringstream ss;
289  ss << e.reason_ << " at line " << e.line_ << ", column " << e.column_;
290  SVR_ERR( 1, "Caught json_spirit exception in CWaveformState::from_json" );
291  throw px::generic_err( ss.str().c_str() );
292  }
293  catch ( const std::exception &ex )
294  {
295  SVR_ERR( 1, "Caught exception in CWaveformState::from_json" );
296  throw px::generic_err( ex.what() );
297  }
298  return ( wave_map );
299 }
