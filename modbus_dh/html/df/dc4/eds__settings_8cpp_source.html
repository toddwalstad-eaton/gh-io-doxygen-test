145 CEdsSettings::checksum_result_t CEdsSettings::calc_eds_file_checksum( const std::string &eds_full_filename )
146 {
147  PUSH_FUNC;
148  boost::crc_32_type retval;
149  uint32_t byte_count( 0 );
150  bool success = false;
151 
152  try
153  {
154  std::ifstream ifs( eds_full_filename.c_str(), std::ios_base::binary );
155 
156  if ( ifs )
157  {
158  do
159  {
160  char buffer[1024];
161 
162  ifs.read( buffer, 1024 );
163  retval.process_bytes( buffer, ifs.gcount() ); // calc the file CRC 1Kb at a time
164  byte_count += ifs.gcount();
165  }
166  while ( ifs );
167  success = true;
168  }
169  else
170  {
171  SVR_ERR( 1, "EDS checksum - Failed to open EDS file: %s", eds_full_filename.c_str() );
172  }
173 
174  SVR_FDEBUG( 8, "EDS checksum - file: %s. Checksum = %u Bytes = %u", eds_full_filename.c_str(), retval.checksum(), byte_count );
175  }
176  catch ( std::exception &e )
177  {
178  SVR_ERR( 1, "EDS checksum Exception: %s", e.what() );
179  }
180  catch ( ... )
181  {
182  SVR_ERR( 1, "EDS checksum Unknown Exception" );
183  }
184 
185  return ( std::make_pair( retval, success ) );
186 }
