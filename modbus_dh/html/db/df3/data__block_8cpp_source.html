131 uint32_t CDataBlock::priority( px::CMonoTime tnow, bool is_profiling )
132 {
133  PUSH_FUNC;
134 
135  int64_t result = _Initial_Priority;
136  int64_t ms_diff = 0;
137 
138  // If we don't want this block polled:
139  if ( _priority_scalar == 0 )
140  {
141  result = 0;
142  }
143  else
144  {
145  // Use PRIORITIZED_POLLING (default) algorithm:
146  // - when it is the selected algorithm; or
147  // - this block's EDS update priority is not highest (i.e. > 1); or
148  // - during profiling.
149  if ( ( PRIORITIZED_POLLING == _polling_algorithm ) || ( 1 != _eds_priority ) || is_profiling )
150  {
151  if ( _last_update.tv_sec > 0 ) // if valid and not first time
152  { // Start with millisecond delta (must be >= 1)
153  ms_diff = std::max( ( int64_t )1, ( 1 + tnow.msec() - _last_update.msec() ) );
154 
155  // Multiply by the priority scalar, but everyone lives large while profiling
156  result = ms_diff * ( ( is_profiling ) ? 100 : _priority_scalar );
157  }
158  // Use DETERMINISTIC_POLLING (e.g. on status channels) polling algorithm when:
159  // - it is the selected algorithm; and
160  // - this block's EDS "UpdatePriority" set to highest (i.e. 1); and
161  // - only when profiling has completed.
162  }
163  else if ( ( _polling_algorithm <= DETERMINISTIC_POLLING_MAX ) && ( 1 == _eds_priority ) )
164  {
165  // At the set polling interval, poll the highest priority block(s),
166  // DETERMINISTIC_POLLING ranges from 1 to N (seconds).
167  // Use _polling_interval_msec member to determine if the polling interval has elapsed.
168  if ( ( tnow.msec() - _last_update.msec() ) > ( _polling_interval_msec ) )
169  {
170  // Force polling of this block.
171  result = _Maximum_Priority;
172  if ( current_priority != _Maximum_Priority )
173  {
174  // Only log this message once per polling interval.
175  SVR_DEBUG( 3, "%s Block[%" PRIu64 "]: Deterministic Poll of Data Block %s", parent_dev_id_str(), _id, _block_name.c_str() );
176  }
177  }
178  else
179  {
180  // Don't poll
181  result = 0;
182  }
183  }
184  else
185  {
186  // ToDo: implement a different polling algorithm, here.
187  // For example JOG_POLLING. Until then, use default PRIORITIZED_POLLING.
188  if ( _last_update.tv_sec > 0 ) // if valid and not first time
189  { // Start with millisecond delta (must be >= 1)
190  ms_diff = std::max( ( int64_t )1, ( 1 + tnow.msec() - _last_update.msec() ) );
191 
192  // Multiply by the priority scalar
193  result = ms_diff * _priority_scalar;
194  }
195  }
196 
197  // Don't exceed max
198  result = std::min( ( int64_t )_Maximum_Priority, result );
199  }
200  // Update the field that remembers what we determined for priority
201  current_priority = ( uint32_t )result;
202  SVR_DEBUG( 9, "%s Block[%" PRIu64 "]: ms diff = %u, current priority = %u", parent_dev_id_str(), _id, ( uint32_t ) ms_diff, current_priority );
203  return ( current_priority );
