396 void CDeviceHandlerCore::execute()
397 {
398  PUSH_FUNC;
399  assert( _options ); // we've had problems long before now if this isn't valid!
400 
401  std::shared_ptr<px::asio::CIoManager> manager;
402  size_t good_count = 0;
403  try
404  {
405  manager.reset( new px::asio::CIoManager( *this, *_cds_asio_handler, _options->watchdog_timeout ) );
406 
407  if ( px::proto::PortProtocolTypes::MODBUS_RTU == _protocol_type )
408  {
409  // Open a serial connection to the serial port
410  // Create the IoActor that will manage this serial port
411  std::shared_ptr<CIoMessenger> pserial( _serial_port->create_connection( manager, _serial_port->get_id() ) );
412  if ( pserial.get() && pserial->is_open() )
413  {
414  manager->add( pserial );
415  good_count++;
416 
417  // Kick it off:
418  pserial->execute();
419  }
420  }
421  else if ( px::proto::PortProtocolTypes::MODBUS_TCP == _protocol_type )
422  {
423  // Create the IoActor that will manage this serial port
424  std::shared_ptr<CIoMessenger> pnetwork( _network_port->create_connection( manager, _network_port->get_id() ) );
425  if ( pnetwork.get() )
426  {
427  manager->add( pnetwork );
428  good_count++;
429 
430  // Kick it off:
431  pnetwork->execute();
432  }
433  }
434  else
435  {
436  throw generic_err( "Couldn't initialize ports and devices" );
437  }
438  }
439  catch ( const boost::system::system_error &e )
440  {
441  // TODO: handle this
442  SVR_ERR( 0, "System error encountered in setting up I/O manager: %s", e.what() );
443  good_count = 0; // ensure that we close out below
444  }
445 
446  if ( good_count == 0 )
447  {
448  // Our attempt to open the serial port(s) failed;
449  // Free up our manager and exit below
450  manager->quit();
451  manager.reset();
452  }
453 
454  if ( manager.get() )
455  {
456  // if we created a valid IO Manager
457  try
458  {
459  SVR_DEBUG( 2, "Entering the run() loop" );
460  // This will block indefinitely, and do all the real work of
461  // the server
462  manager->run();
463  }
464  catch ( const boost::system::system_error &e )
465  {
466  // TODO: handle this
467  SVR_ERR( 0, "System error encountered in I/O manager" );
468  }
469 
470  SVR_DEBUG( 2, "Exiting the run() loop" );
471 
472  // Closeout, if we haven't already, and free our resources.
473  manager->quit();
474  }
475 }
