651 // Normally only called when something has changed, or on initialization
652 bool CModbusTcpPort::update_device_map( std::shared_ptr<px::CDevice> &parent_device, const char *my_name, const bool device_added )
653 {
654  // Get the port's devices and their associated settings. Rescan the CDS every time here, in case things have changed.
655  bool any_changes = false;
656  std::map<px::shared_string, uint32_t> models_with_changed_eds_checksums;
657  device_config_array_t child_devices;
658  device_config_array_t all_child_devices;
659 
660  // Find child devices one level down from the parent ID
661  child_devices = CDeviceService::find_child_devices( _cds_handle, parent_device->id(), FIND_ONE_LEVEL_DOWN );
662  SVR_FINFO( 1, "my_name %s: Device (ID %" PRIu64 ") has %zu child devices", my_name, parent_device->id(), child_devices.size() );
663 
664  // This is the device add loop (for this port).
665  for ( auto &this_device : child_devices )
666  {
667  if ( !this_device.get() )
668  {
669  SVR_WARN( 1, "Skipping slave device with missing info for %s.", my_name );
670  continue;
671  }
672 
673  auto map_entry = _devices.find( this_device->id() );
674  if ( map_entry != _devices.end() )
675  {
676  // We already have this device, so don't do anything new with this device
677  SVR_INFO( 8, "Slave device (ID %" PRIu64 ") already in map for Network Port %s.", this_device->id(), get_port_name() );
678 
679  // Check if it has any unprocessed subdevice instances (does nothing if none to process)
680  try
681  {
682  map_entry->second->update_subdevice_list( getptr() );
683  }
684  catch ( std::exception &ex )
685  {
686  SVR_FERR( 1, "Exception thrown for map_entry->second->update_subdevice_list: %s", ex.what() );
687  }
688 
689  // Maybe someday: check its change signature and see if has Channel or subDevice changes
690  continue;
691  }
692 
693  try
694  {
695  if ( device_added )
696  {
697  // Log if this device was just added to the CDS
698  size_t device_count = current_device_count() + 1;
699  DEVICELOG_INFO( "Config", "%s,%s,%sDevice was added. Current port device count = %zu.",
700  this_device->name.c_str(),
701  this_device->canonical_model_name.c_str(),
702  CUtil::get_formatted_address( false, this_device->network_address, 0 ).c_str(),
703  device_count );
704  }
705 
706  // Grab eds and include it when creating device
707  std::shared_ptr<CParsedDeviceEds> device_eds = _eds_settings->parse_device_eds( this_device );
708 
709  if ( !device_eds )
710  {
711  SVR_WARN( 1, "Failed to add slave device (Address %s) to %s due to EDS parsing failure.", this_device->network_address.c_str(), my_name );
712  continue;
713  }
714 
715  // Create and store the device object
716  bool success = create_protocol_device( this_device, device_eds, parent_device );
717 
718  // Update the port throttle time
719  set_throttle_time();
720 
721  if ( success )
722  {
723  any_changes = true;
724  SVR_INFO( 1, "Added slave device (Address %s) to map for %s.", this_device->network_address.c_str(), my_name );
725  if ( ( device_eds->get_current_eds_checksum() != device_eds->get_previous_eds_checksum() ) &&
726  ( this_device->model.length() > 0 ) )
727  {
728  models_with_changed_eds_checksums.insert( std::make_pair( this_device->model, device_eds->get_current_eds_checksum() ) );
729  }
730  }
731  else
732  {
733  SVR_WARN( 1, "Failed to add slave device (Address %s) to Network Port %s.", this_device->network_address.c_str(), get_port_name() );
734  }
735  }
736  catch ( generic_err &ex )
737  {
738  SVR_ERR( 1, "Failed to add slave device (Address %s) to map for %s: %s", this_device->network_address.c_str(), my_name, ex.what() );
739  }
740  }
741 
742  // Find all child devices of this Port.
743  // Use an unordered map for better search performance.
744  device_map_t all_child_devices_map;
745  all_child_devices_map = CDeviceService::find_device_map( _cds_handle, get_id(), FIND_ALL_CHILDREN );
746  SVR_FINFO( 1, "my_name %s: Device (ID %" PRIu64 ") has %zu child devices", my_name, get_id(), all_child_devices_map.size() );
747 
748  // Now compare map of devices under this Port against the map of devices
749  // in the CDS and remove any that have gone missing from the CDS.
750  std::vector<px::id_t> delete_these_devices;
751 
752  for ( auto map_entry : _devices )
753  {
754  px::id_t map_id = map_entry.first;
755  SVR_DEBUG( 2, " Start check for DH device (ID %" PRIu64 "), port (ID %" PRIu64 "), parent_device->id() (ID %" PRIu64 "), map_entry.second->get_parent_id() (ID %" PRIu64 ")", map_id, get_id(), parent_device->id(), map_entry.second->get_parent_id() );
756 
757  // Search the CDS map for the device (i.e. map_id)
758  auto dev = all_child_devices_map.find( map_id );
759  // Is the device in the CDS map?
760  if ( dev != all_child_devices_map.end() )
761  {
762  SVR_DEBUG( 2, " CDS device (ID %" PRIu64 ") found!", map_id );
763  }
764  else
765  {
766  // Add this device id for removal.
767  delete_these_devices.push_back( map_id );
768  }
769  }
770 
771 
772  if ( delete_these_devices.size() )
773  {
774  for ( size_t i = 0; i < delete_these_devices.size(); i++ )
775  {
776  SVR_INFO( 1, "Removing missing device (ID %" PRIu64 ") from map for %s.", delete_these_devices[i], my_name );
777  any_changes |= remove_device( delete_these_devices[i], false ); // OR the result (hopefully true) of removal
778  }
779  }
780 
781  for ( const std::pair<px::string, uint32_t> &model_checksum_pair : models_with_changed_eds_checksums )
782  {
783  SVR_INFO( 1, "Updating EDS Checksum Model: %s New Checksum: %u", model_checksum_pair.first.c_str(), model_checksum_pair.second );
784  COobService::set_eds_model_checksum( _cds_handle, model_checksum_pair.first.c_str(), model_checksum_pair.second );
785  }
786 
787  return ( any_changes );
788 }
