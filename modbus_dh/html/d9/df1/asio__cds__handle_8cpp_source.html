140 bool CAsioCdsHandle::unsubscribe_device( px::id_t dev_id )
141 {
142  // Get the list of subscriptions for this device, and unsubscribe them
143  auto iter_pair = _device_subs_map.equal_range( dev_id );
144 
145  if ( iter_pair.first == iter_pair.second )
146  {
147  SVR_WARN( 2, "No CDS subscriptions found for Device %ju", dev_id );
148  return ( false );
149  }
150  // Else, iterate through the list of subscriptions
151  px::cds::CSubscribe unsub_req;
152  for ( auto pos = iter_pair.first; pos != iter_pair.second; ++pos )
153  {
154  uint64_t sub_id = pos->second;
155  px::cds::CSubscribeSingle sub_entry;
156  sub_entry.unsubscribe = true;
157  // Just set this field; leave the others at defaults.
158  sub_entry.subscriber_id = sub_id;
159  unsub_req.subscriptions.push_back( sub_entry );
160  // And remove it from our callback map
161  _update_callback_map.erase( sub_id );
162  }
163 
164  // Fire and forget the unsubscribe request
165  px::cds::CClient::get()->submit( unsub_req ); // No callback, no transaction ID returned
166  // Remove this device and all its entries from the subscription map
167  _device_subs_map.erase( dev_id );
168  return ( true );
169 }
