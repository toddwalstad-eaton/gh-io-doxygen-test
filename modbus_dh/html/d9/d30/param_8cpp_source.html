47  if ( eds_descr.command == EDS_EXT_CMD_SPRINTF ) // ignoring dependency issues for SPRINTF variations

52  }
53  else if ( !eds_descr.translation.empty() )

55  try

91  }
92  }

212 unsigned int CParam::parse_num_and_range( const px::string &translation, px::string::size_type start_idx,
213  bool &has_range, unsigned int &second_num )
214 {
215  PUSH_FUNC;
216 
217  px::string::size_type end_idx = 0;
218  if ( ( start_idx == px::string::npos ) || !std::isdigit( translation[start_idx + 1] ) ||
219  ( ( end_idx = translation.find( ']', start_idx ) ) == px::string::npos ) )
220  {
221  std::string msg( "Bad block [] mapping: " );
222  msg += translation.c_str();
223  throw eds_parse_err( msg ); // let the catcher add Param name
224  }
225 
226  // Let's see if there's a range specifier
227  px::string::size_type range_idx = translation.find( ".." );
228 
229  // Get the first numeric value
230  px::string num_text;
231  // Range index could legitimately be past our close bracket
232  if ( ( range_idx == px::string::npos ) || ( range_idx > end_idx ) )
233  {
234  num_text = translation.substr( start_idx + 1, end_idx - start_idx - 1 ); // the normal case
235  has_range = false;
236  second_num = 0;
237  }
238  else
239  {
240  num_text = translation.substr( start_idx + 1, range_idx - start_idx - 1 ); // byte range case
241  }
242 
243  char *end_ptr;
244  unsigned int result = ( unsigned int ) std::strtoul( num_text.c_str(), &end_ptr, 10 ); // decimal-only conversion
245  if ( *end_ptr != ( char ) NULL )
246  {
247  std::string msg( "Bad xBlock[idx]: " );
248  msg += num_text.c_str();
249  throw eds_parse_err( msg );
250  }
251 
252  // Now handle the possible second value
253  if ( ( range_idx != px::string::npos ) && ( range_idx < end_idx ) )
254  {
255  num_text = translation.substr( range_idx + 2, end_idx - range_idx - 2 ); // second num
256  second_num = ( unsigned int ) std::strtoul( num_text.c_str(), &end_ptr, 10 );
257  if ( ( *end_ptr != ( char )NULL ) || ( second_num <= result ) )
258  {
259  std::string msg( "Bad byte range in: " );
260  msg += translation.c_str();
261  throw eds_parse_err( msg );
262  }
263  // else, indicate that we have a second numeric value to return
264  has_range = true;
265  }
266  return ( result );
267 }

269 bool CParam::parse_data_block_info( const px::string &expression,
270  const CDevice &parent,
271  px::datatype /* vtype */,
272  data_block_translation_info_t &data_block_info,
273  px::datatype &cds_type )
274 {
275  PUSH_FUNC;
276 
277  bool success( true );
278  bool has_range( false );
279  unsigned int second_num( 0 );
280 
281  data_block_info.xblock_expression = expression;
282  data_block_info.dbt_array_index = data_block_info.dbt_bit_position = data_block_info.dbt_count = 0;
283  data_block_info.dbt_xblock_id = 0;
284 
285  px::string::size_type start_idx = expression.find( '[' );
286  data_block_info.dbt_array_index = parse_num_and_range( expression, start_idx, has_range, second_num );
287 
288  // (Will have thrown if start_idx was npos, and for other faults)
289  // Ask our parent device to lookup the xBlock mapping for its ID
290  data_block_info.dbt_xblock_id = parent.lookup_block_id( expression.substr( 0, start_idx ) );
291 
292  // Now we can finally get the data type
293  px::datatype tmp_cds_type = parent.lookup_dbt_cds_type( data_block_info.dbt_xblock_id, data_block_info.dbt_array_index );
294 
295  if ( cds_type == px::datatype::invalid )
296  {
297  cds_type = tmp_cds_type;
298  data_block_info.data_type = tmp_cds_type;
299  }
300  else
301  {
302  data_block_info.data_type = cds_type;
303  }
304 
305  if ( data_block_info.dbt_xblock_id == 0 )
306  {
307  // Maybe this is ordinary ? Eg, the xBlock failed a dependency check, so this param can't exist
308  std::string msg( "No block found: " );
309  msg += expression.substr( 0, start_idx ).c_str();
310  throw eds_parse_err( msg ); // let the catcher add Param name
311  }
312 
313  /* Mappings of interest: array_index bit_position count
314  * xBlock5[2] (reference location) 2 0 0 (regular value reference)
315  * xBlock4[3][7] (bit reference) 3 7 1 (single bit)
316  * xBlock4[2][4..7] (bit range) 2 4 4 (bit range, masked and down-shifted)
317  * xBlock7[8..15] (blob or string type) 8 0 8 (byte range)
318  * xBlock7[8] (blob or string type) 8 0 1 (one byte/char */
319  // Let's see if there's a range specifier or bit reference
320  start_idx = expression.find( '[', start_idx + 1 ); // look for second brackets for bit references
321 
322  // Is this a binary (blob) or string type ?
323  if ( ( px::datatype::string == cds_type ) || ( px::datatype::binary == cds_type ) )
324  {
325  if ( !has_range )
326  {
327  data_block_info.dbt_count = 1; // nothing more than one char/byte specified here
328  }
329  else
330  {
331  // else, save _dbt_count
332  data_block_info.dbt_count = second_num - data_block_info.dbt_array_index + 1;
333  }
334  }
335  // Otherwise, is this a bit reference (with second set of brackets) ?
336  else if ( start_idx != px::string::npos )
337  {
338  // start_idx is now position of second open bracket '['
339  data_block_info.dbt_bit_position = parse_num_and_range( expression, start_idx, has_range, second_num );
340  if ( !has_range )
341  {
342  // Single bit fetches always default to Boolean
343  cds_type = px::datatype::boolean;
344  data_block_info.data_type = px::datatype::boolean;
345  data_block_info.dbt_count = 1; // nothing more than one bit specified here
346  }
347  else
348  {
349  // else, save bit range size in _dbt_count
350  data_block_info.dbt_count = second_num - data_block_info.dbt_bit_position + 1;
351  }
352  }
353 
354  return ( success );
355 }
