139 bool CModbusSerialDevice::create_waveform_transaction()
140 {
141  PUSH_FUNC;
142 
143  bool ret_val( false );
144  err_t result( err_t::UNDETERMINED_RESULT );
145 
146  // Device Waveform Acquisition
147  if ( priority() && _device_specific_handler.get() ) // Device not Disabled
148  {
149  if ( CDeviceIoStats::DeviceIoStateEnums::IO_STATE_NO_ERROR == _io_stats->get_state() )
150  {
151  // The out params for the waveform command
152  px::CDeviceCommand device_command;
153  unsigned int number_of_registers( 0 );
154  px::string values = "";
155  eds_xBlock_entry_t xBlock;
156 
157  // Update the waveform_disabled member for this device handler.
158  _device_specific_handler->set_waveform_enabled( _px_device->waveform_enabled && !unprofiled_blocks_remain() );
159 
160  // Set the correct waveform alarm channel ID.
161  _device_specific_handler->set_waveform_channel_id( get_waveform_channel_id() );
162 
163  // Create the waveform command.
164  // The handler will return the required Modbus device command.
165  result = _device_specific_handler->get_waveform_command( _cds_handle, id(), device_command, number_of_registers, values );
166 
167  if ( ( NO_ERROR == result ) && device_command.device_id )
168  {
169  if ( number_of_registers && ( "" != values ) )
170  {
171  // Initialize the xBlock data using the command data as required by the waveform command.
172  xBlock.nObjects = number_of_registers;
173  xBlock.value = values;
174  }
175 
176  const std::shared_ptr<CDataBlock> origin = create_command( device_command, nullptr, result, xBlock );
177 
178  if ( origin.get() )
179  {
180  std::shared_ptr<CModbusCommandBlock> modbus_block = px::poly_cast<CModbusCommandBlock>( origin );
181 
182  assert( modbus_block.get() );
183  if ( modbus_block.get() )
184  {
185  // Add the waveform command to the command deque
186  _command_deque.push_back( origin );
187  ret_val = true;
188  } //if
189  } //if
190  else
191  {
192  // Creation failed for some reason (i.e. command does not exist in EDS).
193  // Halt waveforms!!
194  SVR_ERR( 1, "%s Could not create Device Command '%s' for %s (Error=%s)!", port_dev_id_str(),
195  device_command.name.c_str(), _px_device->model.c_str(), CUtil::get_err_text( err_t::COMMAND_NOT_FOUND, true ) );
196  _device_specific_handler->halt_waveform_command();
197  }
198  } //if
199  }
200  }
201  else if ( _device_specific_handler.get() )
202  {
203  // Update the waveform_disabled member for this device handler.
204  _device_specific_handler->set_waveform_enabled( false );
205  }
206  return ( ret_val );
207 }
