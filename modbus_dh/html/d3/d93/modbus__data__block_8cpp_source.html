57 static void get_write_data_for_dev( modbus_func_code_t func_code, px::string value, px::asio::io_buf_t &data_buf )
58 {
59  data_buf.clear();
60 
61  try
62  {
63  switch ( func_code )
64  {
65  case modbus_func_code_t::MB_WRITE_SINGLE_REGISTER:
66  {
67  uint16_t data = ( uint16_t )std::stoul( px::to_basicstr( value ) );
68  data_buf.push_back( ( unsigned char )( ( data >> 8 ) & 0xFF ) );
69  data_buf.push_back( ( unsigned char )( data & 0xFF ) );
70  }
71  break;
72  case modbus_func_code_t::MB_WRITE_MULTIPLE_HOLDING_REGISTERS:
73  {
74  std::vector<px::string> data_vec = px::CUtil::split_string( value, ';' );
75  if ( data_vec.size() )
76  {
77  for ( size_t j = 0; j < data_vec.size(); j++ )
78  {
79  uint16_t data = ( uint16_t )std::stoul( px::to_basicstr( data_vec[j] ) );
80  data_buf.push_back( ( unsigned char )( ( data >> 8 ) & 0xFF ) );
81  data_buf.push_back( ( unsigned char )( data & 0xFF ) );
82  }
83  }
84  else
85  {
86  data_buf.push_back( 0x00 );
87  data_buf.push_back( 0x00 );
88  }
89  }
90  break;
91  case modbus_func_code_t::MB_READ_COILS:
92  case modbus_func_code_t::MB_READ_DISCRETE_INPUTS:
93  case modbus_func_code_t::MB_READ_INPUT_REGISTERS:
94  case modbus_func_code_t::MB_READ_HOLDING_REGISTERS:
95  // There are no values associated with a read command.
96  break;
97  default:
98  data_buf.push_back( 0x00 );

100  break;
101  }
102  }
103  catch ( ... )

305 void CModbusDataBlock::adjust_timeouts( uint16_t rx_timeout )
306 {
307  PUSH_FUNC;
308 
309  _timeout = rx_timeout;
310  // Make sure block timeout value never goes
311  // below a predefined minimum value.
312  if ( _timeout < MIN_BLOCK_TIMEOUT )
313  {
314  SVR_WARN( 1, "%s: Rx timeout (%zu) is too small. Reset value to: %zu msec", _block_name.c_str(), _timeout, MIN_BLOCK_TIMEOUT );
315  _timeout = MIN_BLOCK_TIMEOUT;
316  }
317 
318  // Reassess the algorithm for calculating the secondary timeout.
319  // The resultant secondary timeout value should be large enough (plus a
320  // safety margin) so that max loading on the blue box does not cause
321  // Comm errors at higher baud rates.
322 
323  // Secondary timeout value to avoid CDS related comm timeouts.
324  _secondary_timeout = _timeout / 2;
325  if ( _secondary_timeout < 100 )
326  {
327  SVR_WARN( 2, "%s: Rx secondary timeout (%zu) is too small. Reset value to: 100 msec", _block_name.c_str(), _secondary_timeout );
328  _secondary_timeout = 100;
329  }
330 
331  // Interframe delay floor - 13000us was empirically determined for IQ260 @ 57.6 kbaud
332  if ( _interframe_delay < MIN_BLOCK_INTERFRAME_DELAY )
333  {
334  SVR_WARN( 1, "%s: Interframe delay (%u) is too small. Reset value to: %zu msec", _block_name.c_str(), _interframe_delay, MIN_BLOCK_INTERFRAME_DELAY );
335  _interframe_delay = MIN_BLOCK_INTERFRAME_DELAY;
336  }
337 
338  SVR_DEBUG( 5, " ModbusDataBlock primary timeout: %zu msec", _timeout );
339  SVR_DEBUG( 5, " ModbusDataBlock secondary timeout: %zu msec", _secondary_timeout );
340  SVR_DEBUG( 5, " ModbusDataBlock interframe delay: %u usec", _interframe_delay );
