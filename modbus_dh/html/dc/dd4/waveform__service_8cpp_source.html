178 std::time_t get_event_time_t( const WAVEFORM_FILE_HEADER_STRUC *hdr )
179 {
180  struct tm event_tm;
181  std::time_t event_time_t;
182  uint8_t year;
183 
184  // Waveform event date & time are in binary coded decimal format (BCD).
185  event_tm.tm_mon = BCD_TO_DECIMAL( hdr->bWaveformEventDate[0] ) - 1; // 0-11 month range
186  event_tm.tm_mday = BCD_TO_DECIMAL( hdr->bWaveformEventDate[1] );
187 
188  // Splitting 2-day year at 70: 70-99 = 1970-1999, 0-69 = 2000-2069
189  year = BCD_TO_DECIMAL( hdr->bWaveformEventDate[2] );
190  event_tm.tm_year = ( year < 70 ) ? year + 100 : year;
191 
192  event_tm.tm_hour = BCD_TO_DECIMAL( hdr->bWaveformEventTime[0] );
193  event_tm.tm_min = BCD_TO_DECIMAL( hdr->bWaveformEventTime[1] );
194  event_tm.tm_sec = BCD_TO_DECIMAL( hdr->bWaveformEventTime[2] );
195  event_tm.tm_isdst = -1;
196  event_time_t = mktime( &event_tm );
197 
198  // Failsafe check - Make sure the waveform capture time
199  // falls within the Linux epoch boundary
200  if ( ( event_time_t > 0 ) && ( event_time_t < std::numeric_limits<time_t>::max() ) )
201  {
202  // set event_time_t to seconds since epoch for event date + time
203  return ( event_time_t );
204  }
205 
206  // If not, set the the waveform capture time to now
207  SVR_ERR( 1, "Waveform event time is outside epoch limits. Invalid sec value is %" PRIu64 "", ( uint64_t )event_time_t );
208  return ( std::time( NULL ) );
209 }

250 void CWaveformService::cleanup_waveform_space( const std::shared_ptr<ICdsHandle> &cds_handle )
251 {
252  PUSH_FUNC;
253 
254  if ( !cds_handle.get() )
255  {
256  SVR_ERR( 4, "Invalid CDS handle" );
257  return;
258  }
259 
260  // Get current list of TGZ files
261  std::unordered_set<std::string> tgz_files = get_files_from_dir( _waveform_export_path.c_str(), EXT_TGZ, false );
262  std::unordered_set<uint32_t> id_list;
263  std::unordered_set<uint32_t> deleted_id_list;
264 
265  if ( tgz_files.empty() )
266  {
267  return;
268  }
269 
270  // Get the disk space used.
271  size_t total_usage = 0;
272  size_t file_count = 0;
273  total_usage = get_directory_size( _waveform_export_path, file_count );
274 
275  // Check space used versus the limit.
276  size_t rolloff = 0;
277  uint32_t delta = 0;
278  if ( total_usage > _waveform_log_capacity )
279  {
280  // Compute delta between usage and the capacity limit.
281  // This is the number of bytes we need to recover by rolling off files
282  delta = total_usage - _waveform_log_capacity;
283  SVR_DEBUG( 3, "Disk space for directory: %s [Used=%zu, Limit=%zu, Files=%zu]", _waveform_export_path.string().c_str(), total_usage, _waveform_log_capacity, file_count );
284  }
285 
286  bool setOOB = false;
287  ::px::proto::toolkitStruct oobInfo;
288  ::px::proto::WaveformList *new_wave = oobInfo.mutable_waveform_list();
289 
290  px::cds::CGetOOBData goob_req;
291  goob_req.tags.push_back( px::string( WAV_OOB ) );
292 
293  std::shared_ptr<px::cds::CGetOOBData> resp =
294  std::dynamic_pointer_cast<px::cds::CGetOOBData>( cds_handle->submit_wait( goob_req ) );
295  if ( resp.get() )
296  {
297  for ( px::cds::CGetOOBDataUnit goob_unit : resp->data )
298  {
299  if ( goob_unit.valid && goob_unit.data.has_waveform_list() )
300  {
301  // Check if we are over the max waveform capacity limit.
302  for ( const auto &old_entry : goob_unit.data.waveform_list().entries() )
303  {
304  if ( delta > 0 )
305  {
306  // Drop this waveform list entry.
307  setOOB = true;
308  if ( deleted_id_list.find( old_entry.device_id() ) == deleted_id_list.end() )
309  {
310  // The find() keeps from repeatedly logging the message.
311  deleted_id_list.insert( old_entry.device_id() );
312  SVR_DEBUG( 5, "cleanup_waveform_space: Add device_id %u to the deleted_id_list", old_entry.device_id() );
313  }
314 
315  // This file will remain in tgz_files and will be deleted.
316  // Accumulate file size into rolloff amount until it exceeds the delta.
317  boost::filesystem::path old_entry_path( old_entry.file() );
318  rolloff += fs::file_size( to_export_path( old_entry_path.filename() ) );
319  if ( rolloff > delta )
320  {
321  delta = 0; // Stop rolloff.
322  }
323  }
324  else
325  {
326  if ( old_entry.has_file() )
327  {
328  fs::path wave_file = old_entry.file();
329  auto test_wave = tgz_files.find( wave_file.filename().c_str() );
330  if ( test_wave != tgz_files.end() )
331  {
332  // This file is erased from tgz_files and will be deleted.
333  tgz_files.erase( test_wave );
334  SVR_DEBUG( 5, "Do not remove wave file: %s", wave_file.filename().c_str() );
335  }
336 
337  if ( id_list.find( old_entry.device_id() ) == id_list.end() )
338  {
339  // The find() keeps from repeatedly logging the message.
340  id_list.insert( old_entry.device_id() );
341  SVR_DEBUG( 5, "cleanup_waveform_space: Add device_id %u to the id_list", old_entry.device_id() );
342  }
343 
344  new_wave->add_entries()->CopyFrom( old_entry );
345  }
346  }
347  }
348  break;
349  }
350  }
351  }
352 
353  if ( setOOB )
354  {
355  px::cds::CSetOOBData soob;
356  soob.data.push_back( px::cds::CSetOOBDataUnit( WAV_OOB, oobInfo, px::cds::persistence::save_always ) );
357  cds_handle->submit_wait( soob );
358  }
359 
360  if ( !deleted_id_list.empty() )
361  {
362  for ( auto &deleted_id : deleted_id_list )
363  {
364  if ( id_list.find( deleted_id ) == id_list.end() )
365  {
366  SVR_DEBUG( 5, "cleanup_waveform_space: set_waveform_available false for device_id %u", deleted_id );
367  CWaveformService::set_waveform_available( cds_handle, ( px::id_t )deleted_id, false );
368  }
369  }
370  }
371 
372  if ( !tgz_files.empty() )
373  {
374  boost::system::error_code rm_ec;
375  for ( auto &tgz_file : tgz_files )
376  {
377  fs::path bad_file = to_export_path( tgz_file );
378  fs::remove( bad_file, rm_ec );
379  SVR_DEBUG( 3, "Removed waveform file: %s", bad_file.c_str() );
380  }
381  total_usage = get_directory_size( _waveform_export_path, file_count );
382  SVR_DEBUG( 3, "Disk space for directory: %s [Used=%zu, Limit=%zu, Files=%zu]", _waveform_export_path.string().c_str(), total_usage, _waveform_log_capacity, file_count );
383  }
384 }

574  // is run, but sometimes this does not occur.
575  osw_path.replace_extension( EXT_TGZ );
576  fs::remove( osw_path, rm_ec );
577  //}
578  }
579 
580  if ( err_t::NO_ERROR != result )
581  {

584  // cycles without touching the CDS and triggering a watchdog timeout.

586  cds_handle->submit( ping_msg ); // Fire and forget
587  }
588 
589  return ( result );

592 err_t CWaveformService::convert_osw( const fs::path &osw_path,

664 err_t CWaveformService::add_waveform_to_cds( const std::shared_ptr<ICdsHandle> &cds_handle,
665  const fs::path &tgz_path,
666  const std::string cause,
667  const px::id_t parent_id,
668  const px::id_t device_id,
669  const px::CWallTime &capture_time,
670  const px::CWallTime &receive_time )
671 {
672  PUSH_FUNC;
673 
674  if ( !cds_handle.get() )
675  {
676  SVR_ERR( 1, "Invalid CDS handle" );
677  return ( err_t::CANNOT_EXECUTE_COMMAND );
678  }
679 
680  // Get the disk space used.
681  size_t file_count = 0;
682  fs::path directory = ( ( tgz_path.has_filename() ) ? tgz_path.parent_path() : tgz_path );
683  size_t total_usage = get_directory_size( tgz_path, file_count );
684  // Todo: Should we check for 0 return, meaning the function failed? Or could that mean "empty" directory?
685 
686  // Check space used versus the limit.
687  size_t rolloff = 0;
688  uint32_t delta = 0;
689  if ( total_usage > _waveform_log_capacity )
690  {
691  // Compute delta between usage and the capacity limit.
692  // This is the number of bytes we need to recover by rolling off files
693  delta = total_usage - _waveform_log_capacity;
694  SVR_DEBUG( 3, "Disk space for directory: %s [Used=%zu, Limit=%zu]", directory.string().c_str(), total_usage, _waveform_log_capacity );
695  }
696 
697  if ( !CWaveformService::is_space_available( tgz_path, 128000 ) )
698  {
699  SVR_ERR( 1, "Error! Insufficient waveform file space - %s", tgz_path.c_str() );
700 
701  px::string px_string = tgz_path.c_str();
702  fs::path new_path = px_string.c_str();
703  SVR_ERR( 1, "Insufficient waveform file space - %s", new_path.c_str() );
704 
705  return ( err_t::MEMORY_ERROR ); // @TODO: need something better? - Yes, how would the user know?
706  }
707 
708  // Server warning when capacity limit is approached/exceeded.
709  float percent_used;
710  percent_used = ( ( float ) total_usage / ( float ) _waveform_log_capacity ) * 100.0;
711  if ( percent_used >= 80.0 )
712  {
713  if ( ( _percent_capacity_used < 100.0 ) && ( percent_used >= 100.0 ) )
714  {
715  SVR_CRIT( 1, "Waveform file space limit exceeded: file rolloff will commence!" );
716  SVR_CRIT( 1, "Disk space for directory: %s [Used=%zu, Limit=%zu]", directory.string().c_str(), total_usage, _waveform_log_capacity );
717  _percent_capacity_used = percent_used;
718  }
719  else if ( _percent_capacity_used < 80.0 )
720  {
721  SVR_WARN( 1, "Approaching waveform file space limit!" );
722  SVR_WARN( 1, "Disk space for directory: %s [Used=%zu, Limit=%zu]", directory.string().c_str(), total_usage, _waveform_log_capacity );
723  _percent_capacity_used = percent_used;
724  }
725  }
726  else if ( percent_used <= 50.0 )
727  {
728  // Reset
729  _percent_capacity_used = percent_used;
730  }
731 
732 
733  // WaveformListEntry:
734  // optional string file = 1; // (required) The Waveform file name.
735  // optional string cause = 2; // (required) Cause of the Waveform capture.
736  // optional uint32 parent_id = 3; // (required) Parent device or port ID
737  // optional uint32 device_id = 4; // (required) Device ID
738  // optional Timestamp capture_time = 5; // (required) Represents the UTC time when the waveform was captured.
739  // optional Timestamp receive_time = 6; // (required) Represents the UTC time when the waveform was received (now)
740 
741  // List of files to delete if we are over the max waveform count
742  std::unordered_set<std::string> tgz_files;
743  std::unordered_set<uint32_t> id_list;
744  std::unordered_set<uint32_t> deleted_id_list;
745 
746  ::px::proto::toolkitStruct oobInfo;
747  ::px::proto::WaveformList *new_wave = oobInfo.mutable_waveform_list();
748 
749  px::cds::CGetOOBData goob_req;
750  goob_req.tags.push_back( px::string( WAV_OOB ) );
751 
752  std::shared_ptr<px::cds::CGetOOBData> resp =
753  std::dynamic_pointer_cast<px::cds::CGetOOBData>( cds_handle->submit_wait( goob_req ) );
754  if ( resp.get() )
755  {
756  for ( px::cds::CGetOOBDataUnit goob_unit : resp->data )
757  {
758  if ( goob_unit.valid && goob_unit.data.has_waveform_list() )
759  {
760  // Check if we are over the max waveform limit.
761 
762  // Copy over anything that does not match new one (prevent dups)
763  for ( const auto &old_entry : goob_unit.data.waveform_list().entries() )
764  {
765  boost::filesystem::path old_entry_path( old_entry.file() );
766  if ( delta > 0 )
767  {
768  // Drop this waveform list entry and delete (rolloff) its associated tgz file.
769  // Insert this file in tgz_files so it will be deleted.
770  tgz_files.insert( old_entry_path.filename().c_str() );
771  if ( deleted_id_list.find( old_entry.device_id() ) == deleted_id_list.end() )
772  {
773  // The find() keeps from repeatedly logging the message.
774  deleted_id_list.insert( old_entry.device_id() );
775  SVR_DEBUG( 5, "add_waveform_to_cds: Add device_id %u to the deleted_id_list", old_entry.device_id() );
776  }
777 
778  // Accumulate fize size to rolloff amount until it exceeds the delta.
779  rolloff += fs::file_size( to_export_path( old_entry_path.filename() ) );
780  if ( rolloff > delta )
781  {
782  delta = 0; // Stop rolloff.
783  }
784  }
785  else
786  {
787  if ( old_entry_path.filename() != tgz_path.filename() )
788  {
789  if ( id_list.find( old_entry.device_id() ) == id_list.end() )
790  {
791  // The find() keeps from repeatedly logging the message.
792  id_list.insert( old_entry.device_id() );
793  SVR_DEBUG( 5, "add_waveform_to_cds: Add device_id %u to the id_list", old_entry.device_id() );
794  }
795  new_wave->add_entries()->CopyFrom( old_entry );
796  }
797  }
798  }
799  break; // done with goob_unit : for (px::cds::CGetOOBDataUnit goob_unit : resp->data)
800  }
801  }
802  }
803 
804  px::proto::WaveformListEntry *wave_entry = new_wave->add_entries();
805  wave_entry->set_file( to_www_path( tgz_path ) );
806  wave_entry->set_cause( cause );
807  wave_entry->set_parent_id( parent_id );
808  wave_entry->set_device_id( device_id );
809 
810  px::proto::Timestamp *proto_capture_time = wave_entry->mutable_capture_time();
811  // Failsafe check - Make sure the waveform capture time
812  // falls within the Linux epoch boundary
813  if ( ( capture_time.tv_sec > 0 ) && ( capture_time.tv_sec < std::numeric_limits<time_t>::max() ) )
814  {
815  proto_capture_time->set_sec( capture_time.tv_sec );
816  proto_capture_time->set_nsec( capture_time.tv_nsec );
817  }
818  else
819  {
820  // If epoch limit error, use the receive time as the waveform capture time
821  proto_capture_time->set_sec( receive_time.tv_sec );
822  proto_capture_time->set_nsec( receive_time.tv_nsec );
823  SVR_ERR( 1, "Waveform receive time is outside epoch limits. Invalid sec value is %" PRIu64 "", ( uint64_t )capture_time.tv_sec );
824  }
825 
826  px::proto::Timestamp *proto_receive_time = wave_entry->mutable_receive_time();
827  proto_receive_time->set_sec( receive_time.tv_sec );
828  proto_receive_time->set_nsec( receive_time.tv_nsec );
829 
830  px::cds::CSetOOBData soob;
831  soob.data.push_back( px::cds::CSetOOBDataUnit( WAV_OOB, oobInfo, px::cds::persistence::save_always ) );
832  std::shared_ptr<px::cds::CRequestResponse> req_resp = cds_handle->submit_wait( soob );
833  CWaveformService::set_waveform_available( cds_handle, device_id, true );
834 
835  if ( !deleted_id_list.empty() )
836  {
837  for ( auto &deleted_id : deleted_id_list )
838  {
839  if ( id_list.find( deleted_id ) == id_list.end() )
840  {
841  SVR_DEBUG( 5, "add_waveform_to_cds: set_waveform_available false for device_id %u", deleted_id );
842  CWaveformService::set_waveform_available( cds_handle, ( px::id_t )deleted_id, false );
843  }
844  }
845  }
846 
847  if ( !tgz_files.empty() )
848  {
849  boost::system::error_code rm_ec;
850  for ( auto &tgz_file : tgz_files )
851  {
852  fs::path bad_file = to_export_path( tgz_file );
853  fs::remove( bad_file, rm_ec );
854  SVR_DEBUG( 3, "Removed rolloff waveform file: %s", bad_file.c_str() );
855  }
856 
857  total_usage = get_directory_size( tgz_path, file_count );
858  SVR_DEBUG( 3, "Disk space for directory: %s [Used=%zu, Limit=%zu, Files=%zu]", directory.string().c_str(), total_usage, _waveform_log_capacity, file_count );
859  }
860 
861  return ( err_t::NO_ERROR );
862 }

1116 void CWaveformService::package_comtrade_files( const std::vector<fs::path> &path,
1117  const fs::path &tgz_path,
1118  boost::system::error_code &ec )
1119 {
1120  PUSH_FUNC;
1121 
1122  std::string loose_file_list;
1123  for ( const auto &_path : path )
1124  {
1125  loose_file_list += _path.filename().string() + " ";
1126  }
1127 
1128  fs::path work_dir = path[0].parent_path();
1129  fs::path tgz_tmp_path = work_dir;
1130  tgz_tmp_path /= tgz_path.filename();
1131 
1132  std::string cmd;
1133 
1134  // TGZ file output is user 'device_handler' when called from DH & user 'root' for unit test code
1135  cmd += "cd " + work_dir.string();
1136  cmd += " && tar czf " + tgz_tmp_path.string() + " " + loose_file_list;
1137  cmd += " && chmod a=r " + tgz_tmp_path.string();
1138  cmd += " && mv -f " + tgz_tmp_path.string() + " " + tgz_path.string();
1139 
1140  SVR_DEBUG( 5, "COMTRADE Packaging Cmd: %s", cmd.c_str() );
1141  int rc = std::system( cmd.c_str() );
