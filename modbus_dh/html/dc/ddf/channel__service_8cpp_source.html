167  channel_settings.display_order, channel_settings.multistate_map.get().size() );
168  }
169 
170  if ( !( channel_settings == *pchannel->_ch_settings ) )
171  {
172  update_req.requested_channel_updates.push_back( *pchannel->_ch_settings );
173  SVR_DEBUG( 8, " Updating settings for Channel %" PRIu64 "", pchannel->id() );

182  }
183
