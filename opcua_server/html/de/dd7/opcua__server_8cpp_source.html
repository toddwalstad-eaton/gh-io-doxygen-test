219 void COPCUaServer::create_map( px::id_t start_cds_id )
220 {
221  PUSH_FUNC;
222 
223  SVR_DEBUG( 3, "=====\r\nBrowsing the CDS device subtree from root device ID %d.",
224  ( int )start_cds_id );
225  static std::string cds_device_tree_str = "";
226 
227  static unsigned short depth_level = 0;
228  /* Limit recursion depth. */
229  if ( ++depth_level < _options->max_cds_browse_depth )
230  {
231  SVR_DEBUG( 6, "Browsing CDS level (intent depth) %d of %d.", depth_level,
232  _options->max_cds_browse_depth );
233  std::vector<px::id_t> ids;
234  ids.push_back( start_cds_id );
235  std::vector<px::CDevice> devices = _cds->get_devices_wait( ids );
236 
237  for ( auto &device : devices ) // access by reference to avoid copying
238  {
239  /* Create new device mapping. */
240  std::set<px::id_t>::size_type subdev_cnt = device.sub_devices.size();
241  std::set<px::id_t>::size_type ch_cnt = device.channels.size();
242 
243  cds_device_tree_str += cds_device_tree_to_str( device, depth_level );
244  try
245  {
246  UA_NodeId opc_dev_id = create_object_node( device );
247  update_mapping( device.id(), opc_dev_id );
248  }
249  catch ( COpen62541Wrapper::node_add_err )
250  {
251  SVR_NOTICE( 1, "OPC UA node creation failed. Skipping all its "
252  "channels and continuing for next device." );
253  continue;
254  }
255 
256  /* No channels to browse for this device. Go for next device. */
257  SVR_DEBUG( 3, "This device has %d channels to process.", ch_cnt );
258  if ( ch_cnt != 0 )
259  {
260  SVR_DEBUG( 6, "Getting channles info." );
261  std::vector<px::id_t> ids;
262  ids.assign( device.channels.begin(), device.channels.end() );
263  std::vector<px::cds::CGetSingleChannel> channels =
264  _cds->get_channels_wait( ids );
265 
266  for ( auto &channel :
267  channels ) // access by reference to avoid copying
268  {
269  try
270  {
271  /* Create new channel mapping. */
272  UA_NodeId opc_ch_id = create_variable_node( channel );
273  SVR_DEBUG( 3, "Registering OPC UA callbacks." );
274  _opcua_server->register_value_node_callbacks( opc_ch_id, _opc_value_callbacks );
275  cds_device_tree_str += cds_channel_tree_to_str( channel, depth_level );
276  px::id_t cds_id = channel.id;
277  update_mapping( cds_id, opc_ch_id );
278  }
279  catch ( COpen62541Wrapper::node_add_err )
280  {
281  SVR_NOTICE( 1, "OPC UA variable node creation failed. "
282  "Continuing for next channel." );
283  continue;
284  }
285  catch ( opcua::utils::exceptions::conv_err )
286  {
287  SVR_NOTICE( 1, "Value data type could not be converted. "
288  "OPC UA Variable node not created. "
289  "Continuing for next channel." );
290  continue;
291  }
292  catch ( opcua::utils::exceptions::not_implemented_err )
293  {
294  SVR_NOTICE( 1, "Value data type conversion required "
295  "feature that is not yet implemented. "
296  "OPC UA Variable node not created. "
297  "Continuing for next channel." );
298  continue;
299  }
300  catch ( opcua::server::COpen62541Wrapper::node_mandatory_err )
301  {
302  SVR_NOTICE( 1, "OPC UA variable node could not be set "
303  "as mandatory. Left as optional node." );
304  }
305  }
306  }
307 
308  /* Go for other subdevices */
309  if ( subdev_cnt > 0 )
310  {
311  SVR_DEBUG( 3, "This device has %d subdevices to process recursivelly.", subdev_cnt );
312  for ( auto &dev_id : device.sub_devices )
313  {
314  create_map( dev_id );
315  }
316  }
317  }
318  }
319  else
320  {
321  SVR_WARN( 0, "CDS Browsing indent depth exceeded. To fetch more"
322  "devices and channels, increase the maximal depth level"
323  " (-mdl parameter)." );
324  throw COPCUaServer::recursion_depth();
325  }
326  depth_level--;
327  SVR_DEBUG( 6, "=====\r\nGoing up in the browsing to depth %d.", depth_level );
328  /* Finishing the browsing */
329  if ( depth_level == 0 )
330  {
331  SVR_DEBUG( 3, "Browsing finished." );
332  SVR_DEBUG( 5, "CDS DeviceTree browsed:\r\n%s", cds_device_tree_str.c_str() );
333  cds_device_tree_str.clear();
334  SVR_DEBUG( 6, "Mapping created: %s.", _map_of_dev_ids->to_string().c_str() );
335  }
