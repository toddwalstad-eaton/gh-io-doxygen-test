874 struct tm *internal_gmtime( uint64_t epochSeconds )
875 {
876  unsigned int yearTable[2][12] = {
877  { 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 }, { 31, 29, 31, 30, 31, 30, 31,
878  31, 30, 31, 30, 31 }
879  };
880  static struct tm br_time;
881  register struct tm *gmTime = &br_time;
882  uint64_t dayclock, dayno;
883  int year = EPOCH_YR;
884 
885  dayclock = epochSeconds % SECS_DAY;
886  dayno = epochSeconds / SECS_DAY;
887 
888  gmTime->tm_sec = dayclock % 60;
889  gmTime->tm_min = ( dayclock % 3600 ) / 60;
890  gmTime->tm_hour = dayclock / 3600;
891  gmTime->tm_wday = ( dayno + 4 ) % 7;/* day 0 was a thursday */
892  while ( dayno >= YEARSIZE( year ) )
893  {
894  dayno -= YEARSIZE( year );
895  year++;
896  }
897  gmTime->tm_year = year - YEAR0;
898 
899  gmTime->tm_yday = dayno;
900 
901  gmTime->tm_mon = 0;
902  while ( dayno >= yearTable[LEAPYEAR( year )][gmTime->tm_mon] )
903  {
904  dayno -= yearTable[LEAPYEAR( year )][gmTime->tm_mon];
905  gmTime->tm_mon++;
906  }
907 
908  gmTime->tm_mday = dayno + 1;
909  gmTime->tm_isdst = 0;
910 
911  return ( gmTime );
912 }
