767 std::string CertmanagerService::sendCommandToServer( Commands &cmd )
768 {
769  //add info in cmd
770  cmd.m_user = Tools::getUserName();
771  cmd.m_pid = std::to_string( Tools::getPid() );
772 
773  //get the the paths
774  //FIFO in and lock files are created by the daemon
775  //FIFO out file is created by the service when it wants to communicate
776  std::string pathFifoOut = DaemonConfiguration::getNamePipesDirectory() + "/" + m_serviceConfig->getId() + "/" + m_serviceConfig->getId() + ".out";
777  std::string pathFifoIn = DaemonConfiguration::getNamePipesDirectory() + "/" + m_serviceConfig->getId() + "/" + m_serviceConfig->getId() + ".in";
778  std::string pathFifoLock = DaemonConfiguration::getNamePipesDirectory() + "/" + m_serviceConfig->getId() + "/" + m_serviceConfig->getId() + ".lock";
779 
780  //try to open lock the file
781  errno = 0;
782  int lock = open( pathFifoLock.c_str(), O_WRONLY );
783  if ( errno == ENOENT )
784  {
785  debugCert( "Failed to open the file '" + pathFifoLock + "'. Looks like the 'certmanagd' service is not running." );
786  throw CertmanagerApiException( DAEMON_OFF );
787  }
788  if ( lock == -1 )
789  {
790  debugCert( std::string( "Error: FIFO lock file open error " ) + strerror( errno ) );
791  throw CertmanagerApiException( ACCESS_DENIED );
792  }
793 
794  //try to lock the file
795  if ( flock( lock, LOCK_EX | LOCK_NB ) == -1 )
796  {
797  debugCert( std::string( "Error: file already locked : " ) + strerror( errno ) );
798  close( lock );
799  throw CertmanagerApiException( API_LOCKED );
800  }
801 
802  //create the fifo out
803  if ( mkfifo( pathFifoOut.c_str(), 0760 ) != 0 )
804  {
