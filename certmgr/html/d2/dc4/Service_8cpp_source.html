790

799  Log::write( Severity::ERROR,

804 
805  ssize_t bytes;
806  char buffer[1024];
807 
808  std::string message;
809

815  message += std::string( buffer );

817  }
818

821 
822  debugCert( "Received: " + message );

824  {
825  continue;
826  }
827 
828  //create the command structure
829  try
830  {
831  //lock
832  service->m_serviceBusy.lock();
833 
834  Commands cmd( message );
835 
836  //try to exec the command
837  service->execCmd( cmd );
838  }
839  catch ( const std::exception &e )
840  {
841  const char *excp_error = "Unknown error";
842  if ( e.what() != nullptr )
843  {
844  excp_error = e.what();
845  }
846  Log::write( Severity::ERROR, "std::exception caught while creating command structure for service '" + service->m_config.getId() + "' " + excp_error );
847  }
848  catch ( ... )
849  {
850  Log::write( Severity::ERROR, "Unknown exception caught while creating command structure for service '" + service->m_config.getId() + "'." );
851  }
852 
853  //unlock or ensure it's unlock
854  service->m_serviceBusy.unlock();
855  }

900 void *Service::timer( void *p_service )
901 {
902  //timer always run until we get a new certificate
903  //timer notify when the certificate will expire, from when it expire or when it will became valide.
904 
905  Log::open();
906 
907  Service *service = static_cast<Service *>( p_service );
908 
909  try
910  {
911  while ( true )
912  {
913  //wait until next time instant
914  // certmanager checks the certificate validity daily at given point in time (on or before 00:00:00 UTC) and renews the certificate if it is invalid
915  // It reads the time offset from service config file to calculate time instant
916  std::chrono::system_clock::time_point tp = std::chrono::system_clock::now();
917  std::chrono::system_clock::duration dtn = tp.time_since_epoch();
918  int now = dtn.count() * std::chrono::system_clock::period::num / std::chrono::system_clock::period::den;
919 
920  //time until midnight (86400s = 1day)
921  int sleeping_time = 86400 - ( now % 86400 );
922 
923  sleeping_time = ( sleeping_time - service->m_config.getCertDailyExpiryChkTimeOffset() );
924  if ( sleeping_time <= 0 )
925  {
926  sleeping_time = sleeping_time + 86400;
927  }
928 
929  //wait until the next time instant
930  if ( sleep( sleeping_time ) == 0 )
931  {
932  //we run the check
933  if ( service->m_config.isServer() )
934  {
935  const char *err = "Unknown";
936  std::unique_lock<std::mutex> l( service->m_serviceBusy );
937  try
938  {
939  service->m_server->checkTimeValidity();
940  }
941  catch ( const CertmanagerDaemonException &e )
942  {
943  if ( e.what() != NULL )
944  {
945  err = e.what();
946  }

950  {
951  if ( e.what() != NULL )

954  }

956  }

958  {
959  Log::write( Severity::ERROR, "Unknown exception during timer thread execution for service '" + service->m_config.getId() + "'." );
960  }
961  }
962  }
963  }
964  }
