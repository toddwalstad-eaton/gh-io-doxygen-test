848 bool set_cert_time( const uint64_t &validFromSeconds, const uint64_t &validUntilSeconds, const uint64_t &validity,
849  X509_T &crtObj, X509_T &caObj )
850 {
851  std::chrono::time_point<std::chrono::system_clock> now = std::chrono::system_clock::now();
852  uint64_t epochSeconds = std::chrono::time_point_cast<std::chrono::seconds>( now ).time_since_epoch().count();
853 
854  // set time
855  const ASN1_TIME *crtValidFrom = X509_get_notBefore( *crtObj );
856  const ASN1_TIME *crtValidUntil = X509_get_notAfter( *crtObj );
857 
858  X509_gmtime_adj( const_cast<ASN1_TIME *>( crtValidFrom ), validFromSeconds - epochSeconds );
859  X509_gmtime_adj( const_cast<ASN1_TIME *>( crtValidUntil ), validUntilSeconds - epochSeconds );
860 
861  //Get the CA certificate validity start date and compare with currently set client certificate start date
862  const ASN1_TIME *caValidFrom = X509_get_notBefore( *caObj );
863  uint64_t caValidNotBefore = asn1datetime_isodatetime( X509_get0_notBefore( *caObj ) );
864  uint64_t clientValidNotBefore = asn1datetime_isodatetime( X509_get0_notBefore( *crtObj ) );
865 
866  //Get CA certificate validity end date and compare with currently set client certificate end date
867  const ASN1_TIME *caValidUntil = X509_get0_notAfter( *caObj );
868  uint64_t caValidNotAfter = asn1datetime_isodatetime( X509_get0_notAfter( *caObj ) );
869  uint64_t clientValidNotAfter = asn1datetime_isodatetime( X509_get0_notAfter( *crtObj ) );
870 
871  // If client certificate start date is going beyond CA start date, set the start date to CA certificate start date
872  if ( caValidNotBefore > clientValidNotBefore )
873  {
874  X509_set1_notBefore( *crtObj, caValidFrom );
875  //set the client certificate, new end date based on updated start date and certificate validity
876  X509_set1_notAfter( *crtObj, ASN1_TIME_set( NULL, clientValidNotBefore + validity ) );
877  }
878 
879  // If client end date is going beyond CA end date, set the end date to CA certificate end date
880  // If CA certificate validity is lower than the client certificate, then set CA validity to client validity
881  if ( caValidNotAfter < clientValidNotAfter )
882  {
883  X509_set1_notAfter( *crtObj, caValidUntil );
884  }
885 
886  // If client certificate end date is going beyond client certificate start date, then avoid creating a certificate
887  //refresh the date
888  clientValidNotBefore = asn1datetime_isodatetime( X509_get0_notBefore( *crtObj ) );
889  clientValidNotAfter = asn1datetime_isodatetime( X509_get0_notAfter( *crtObj ) );
890 
891  if ( clientValidNotAfter < clientValidNotBefore )
892  {
893  Log::write( certmanager::Severity::ERROR,
894  "Certificate creation aborted because end date is set prior to the start date" );
895  return ( false );
896  }
897  return ( true );
898 }
