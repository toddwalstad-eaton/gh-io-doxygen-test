1264 void Server::setValidSeconds( bool selfsigned, uint64_t &validFromSeconds, uint64_t &validUntilSeconds,
1265  uint64_t &validity )
1266 {
1267  std::chrono::time_point<std::chrono::system_clock> now = std::chrono::system_clock::now();
1268 
1269  //Get current epoch sesonds
1270  uint64_t epochSeconds = std::chrono::time_point_cast<std::chrono::seconds>( now ).time_since_epoch().count();
1271 
1272  if ( selfsigned )
1273  {
1274  //Based on the offset and validity duration calculate valid from valid utill
1275  validFromSeconds = epochSeconds - ( SECOND_IN_DAY * static_cast<uint64_t>( m_config.getSelfSignedValidityDurationOffset() ) );
1276  validUntilSeconds = validFromSeconds + ( SECOND_IN_DAY * static_cast<uint64_t>( m_config.getSelfSignedValidityDuration() ) );
1277  }
1278  else
1279  {
1280  //Based on the offset and validity duration calculate valid from valid utill
1281  validFromSeconds = epochSeconds - ( SECOND_IN_DAY * static_cast<uint64_t>( m_config.getCASignedValidityDurationOffset() ) );
1282  validUntilSeconds = validFromSeconds + ( SECOND_IN_DAY * static_cast<uint64_t>( m_config.getCASignedValidityDuration() ) );
1283  }
1284 
1285  validFromSeconds -= validFromSeconds % ( SECOND_IN_DAY );
1286  validUntilSeconds -= validUntilSeconds % ( SECOND_IN_DAY );
1287  validity = validUntilSeconds - validFromSeconds;
1288 
1289  if ( validUntilSeconds <= validFromSeconds )
1290  {
1291  throw CertmanagerDaemonException( "Requested valid until '" + std::string( std::asctime( std::gmtime( ( const time_t * )&validUntilSeconds ) ) )
1292  + "' is not higher than valid from time '" + std::string( std::asctime( std::gmtime( ( const time_t * )&validFromSeconds ) ) )
1293  + "'." );
1294  }
1295 
1296  // Block creating certificates beyond the defined date period
1297  if ( ( validFromSeconds < MIN_ALLOWED_SET_DATE_SECS ) || ( validUntilSeconds < MIN_ALLOWED_SET_DATE_SECS ) )
1298  {
1299  uint64_t min_allowed_epochSeconds = MIN_ALLOWED_SET_DATE_SECS;
1300  throw CertmanagerDaemonException( "Requested valid from time '" + std::string( std::asctime( std::gmtime( ( const time_t * )&validFromSeconds ) ) )
1301  + "' is beyond the allowed lower limit of time '" + std::string( std::asctime( std::gmtime( ( const time_t * )&min_allowed_epochSeconds ) ) )
1302  + "'." );
1303  }
1304 }
