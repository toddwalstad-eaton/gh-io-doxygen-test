70  poll_descriptors[1].events = POLLIN | POLLRDHUP;
71  poll_descriptors[1].revents = 0;
72  const char *beginning_of_write;
73  size_t bytes_remaining;
74 
75  if ( str_to_write.empty() )
76  {
77  return;
78  }
79  beginning_of_write = str_to_write.c_str();
80  bytes_remaining = str_to_write.size();
81 
82  while ( ( bytes_remaining > 0 ) &&
83  ( poll( poll_descriptors, 2, -1 ) >= 0 ) )
84  {
85  if ( ( poll_descriptors[0].revents & POLLOUT ) != 0 )
86  {
87  rc = write( process_fd, beginning_of_write, bytes_remaining );
88  if ( rc > 0 )
89  {
90  bytes_remaining -= rc;
91  beginning_of_write += rc;
92  }
93  else if ( rc < 0 )
94  {
95  break; // some error
96  }
97  else
98  {
99  sleep( 1 ); // safety in case of continuous error
100  }
101  }
102  if ( ( poll_descriptors[1].revents != 0 ) || ( ( poll_descriptors[0].revents & ( POLLHUP | POLLERR | POLLNVAL ) ) != 0 ) )
103  {
104  // Main thread asked us to stop
105  break;
106  }
107  poll_descriptors[0].revents = 0;
108  poll_descriptors[1].revents = 0;
109  }
