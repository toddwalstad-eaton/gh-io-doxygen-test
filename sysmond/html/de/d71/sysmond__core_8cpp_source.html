138 {
139  PUSH_FUNC;
140 
141  while ( !want_to_stop )

146

149  // we will just wait for one watchdog-pat-interval time.

169 #endif
170  status_chanid = cds->get_status_channel_id();
171  if ( likely( cds && ( status_chanid != PX_ID_UNSET ) ) )
172  {
173  cds->update_status_channel( status_chanid, true );
174  }

177  }
178

180  {

183  next_scan_time = px::CMonoTime::now().tv_sec;
184  SYSLOG_DEBUG( 5, "Monotonic time now: %ld", next_scan_time );

186  SYSLOG_DEBUG( 5, "Monotonic time of next scan: %ld", next_scan_time );
