686 //
687 //Function scans all overlay file system directiries
688 //
689 bool CScanner::CheckOverlayFSIntegrity()
690 {
691  std::list<OFS> mnts;
692 
693  try
694  {
695  etn::CLinuxUtil::get_overlayfs_mounts( mnts );
696  }
697  catch ( const std::runtime_error &e )
698  {
699  std::string msg = std::string( "Error checking for overlay filesystems: " ) + std::string( e.what() );
700  log_failure( msg );
701  return ( false );
702  }
703  /* opening each overlay FS directory in /proc/mount */
704  for ( auto it = mnts.begin(); it != mnts.end(); it++ )
705  {
706  SYSLOG_INFO( 4, "Scanning overlayfs mounted on %s", it->mount_dir.c_str() );
707  // For each executable file in lower directory, is the mounted copy the same?
708  std::set<px::CRecursiveFind::file_info_t> lowerFileList;
709  getAllFilesInDir( lowerFileList, it->lower_dir, _exclude );
710  std::string lowerFilePath;
711  std::unordered_set<std::string> lowerFileSet;
712  for ( auto file_it : lowerFileList )
713  {
714  struct stat fileStatUpper, fileStatLower, fileStatMounted;
715  lowerFilePath = std::get<px::CRecursiveFind::GET_FILEPATH>( file_it );
716 
717  std::string mountFilePath = getCorrespondingPath( lowerFilePath, it->lower_dir, it->mount_dir );
718  // Collect info on original read-only copy of the file
719  SYSLOG_DEBUG( 9, "Checking file: %s (reference: %s)", mountFilePath.c_str(), lowerFilePath.c_str() );
720  if ( stat( lowerFilePath.c_str(), &fileStatLower ) != 0 )
721  {
722  // File returned in list but stat() failed, lower dirs are read-only so it shouldn't
723  // have disappeared, and otherwise this should be impossible.
724  std::string msg = std::string( "stat() failed on " ) + lowerFilePath +
725  std::string( " (error " ) + std::to_string( errno ) + std::string( "), should not be possible" );
726  log_failure( msg );
727  return ( false );
728  }
729  // Don't try to deal with things that aren't regular files
730  if ( !S_ISREG( fileStatLower.st_mode ) )
731  {
732  SYSLOG_DEBUG( 5, "Skipping %s since it's not a regular file", mountFilePath.c_str() );
733  continue;
734  }
735  // Populate as we go, for later check of added files
736  lowerFileSet.insert( lowerFilePath );
737  // Mode bits for most of the time, access() as insurance
738  // in case we are able to execute via ACLs or something
739  bool lower_file_exec = ( ( fileStatLower.st_mode & ( S_IXUSR | S_IXGRP | S_IXOTH ) ) != 0 ) ||
740  ( access( lowerFilePath.c_str(), X_OK ) == 0 );
741 
742  // Collect info on the live (as-mounted) copy of the file
743  if ( stat( mountFilePath.c_str(), &fileStatMounted ) != 0 )
744  {
745  // File may have been deleted. If not executable, this is OK.
746  if ( lower_file_exec )
747  {
748  std::string msg = std::string( "Executable file " ) + mountFilePath +
749  std::string( " has been deleted or rendered inaccessible" );
750  log_failure( msg );
751  return ( false );
752  }
753  else
754  {
755  continue;
756  }
757  }
758  bool mounted_file_exec = ( ( fileStatMounted.st_mode & ( S_IXUSR | S_IXGRP | S_IXOTH ) ) != 0 ) ||
759  ( access( mountFilePath.c_str(), X_OK ) == 0 );
760 
761  if ( mounted_file_exec && ( !lower_file_exec ) )
762  {
763  std::string msg = std::string( "File " ) + mountFilePath + std::string( " was changed to be executable" );
764  log_failure( msg );
765  return ( false );
766  }
767  if ( lower_file_exec && ( !mounted_file_exec ) )
768  {
769  std::string msg = std::string( "File " ) + mountFilePath + std::string( " was changed to be non-executable" );
770  log_failure( msg );
771  return ( false );
772  }
773  // If mounted file is not a regular file, we have replaced a regular file with something else like
774  // a socket or something - shenanigans. Symlinks are OK but won't be S_IFLNK because we intentionally
775  // did not use lstat().
776  if ( ( fileStatMounted.st_mode & S_IFMT ) != ( fileStatLower.st_mode & S_IFMT ) )
777  {
778  std::string msg = std::string( "File " ) + mountFilePath + std::string( " has replaced a regular file with something else, suspicious" );
779  log_failure( msg );
780  return ( false );
781  }
782  if ( !lower_file_exec )
783  {
784  SYSLOG_DEBUG( 9, "Skipping file, not executable" );
785  continue;
786  }
787  // Free checks: permissions and size
788  if ( fileStatMounted.st_mode != fileStatLower.st_mode )
789  {
790  std::string msg = std::string( "File " ) + mountFilePath + std::string( " changed permissions" );
791  log_failure( msg );
792  return ( false );
793  }
794  if ( fileStatMounted.st_size != fileStatLower.st_size )
795  {
796  std::string msg = std::string( "File " ) + mountFilePath + std::string( " changed size" );
797  log_failure( msg );
798  return ( false );
799  }
800 
801  // Before comparing, see if anything exists for this file in the upper directory.
802  // If not, there's no way for them to differ, so we can skip
803  std::string upperFilePath = getCorrespondingPath( mountFilePath, it->mount_dir, it->upper_dir );
804  if ( stat( upperFilePath.c_str(), &fileStatUpper ) != 0 )
805  {
806  SYSLOG_DEBUG( 8, "File %s does not have upper-directory counterpart, can't differ",
807  mountFilePath.c_str() );
808  continue;
809  }
810 
811  if ( !compareFile_Contents( lowerFilePath, mountFilePath ) )
812  {
813  std::string msg = std::string( "The contents of executable file " ) + mountFilePath + std::string( " have changed" );
814  log_failure( msg );
815  return ( false );
816  }
817  } // End iteration over each file in upper directory
818 
819  SYSLOG_DEBUG( 9, "Checking for new executable files" );
820  std::set<px::CRecursiveFind::file_info_t> mountFileList;
821  getAllFilesInDir( mountFileList, it->mount_dir, _exclude );
822  std::string mountFilePath;
823  for ( auto file_it : mountFileList )
824  {
825  mountFilePath = std::get<px::CRecursiveFind::GET_FILEPATH>( file_it );
826  std::string lowerFilePath = getCorrespondingPath( mountFilePath, it->mount_dir, it->lower_dir );
827  if ( lowerFileSet.count( lowerFilePath ) != 0 )
828  {
829  SYSLOG_DEBUG( 9, "Skipping %s, it's not new", mountFilePath.c_str() );
830  continue;
831  }
832  struct stat fileStatMounted;
833  if ( stat( mountFilePath.c_str(), &fileStatMounted ) != 0 )
834  {
835  // This is a writable filesystem so the file may have disappeared since we listed it.
836  SYSLOG_INFO( 2, "Skipping %s, listed but can't stat, errno %d", mountFilePath.c_str(), errno );
837  continue;
838  }
839  if ( !S_ISREG( fileStatMounted.st_mode ) )
840  {
841  SYSLOG_DEBUG( 5, "Skipping %s since it's not a regular file", mountFilePath.c_str() );
842  continue;
843  }
844  bool mounted_file_exec = ( ( fileStatMounted.st_mode & ( S_IXUSR | S_IXGRP | S_IXOTH ) ) != 0 ) ||
845  ( access( mountFilePath.c_str(), X_OK ) == 0 );
846  if ( mounted_file_exec )
847  {
848  std::string msg = std::string( "Executable file " ) + mountFilePath + std::string( " has been added" );
849  log_failure( msg );
850  return ( false );
851  }
852  } // End iteration over each file in lower directory
853  SYSLOG_DEBUG( 2, "Overlayfs integrity check passed for mount %s", it->mount_dir.c_str() );
854  }// End iteration over each overlayfs mount
855  SYSLOG_DEBUG( 2, "Overlayfs integrity check passed for all mounts" );
856  return ( true );
857 }

859 // Launches the thread that does the integrity scanner operations.
860 bool CScanner::launch_scanner_thread()
861 {
862  PUSH_FUNC;
863 
864  want_thread_stop = false;
865  bool launch_result = false;
866 
867  try
868  {
869  /* Launch the scanner task in a separate thread using std::async and std::future
870  * for the results that we will wait on.
871  * Uncrustify really messes up the appearance of this lambda:
872  * [this] { return ( net_diag_tests() ); } */
873  scanner_future_result = std::async( std::launch::async, [this] {
874  return ( scanner_task() );
875  } );
876  SVR_INFO( 2, "Launched the integrity scanner thread" );
877  launch_result = true;
878  scanner_thread_is_launched = true;
879  }
880  catch ( const std::exception &ex )
881  {
882  // Caught either std::system_error or std::bad_alloc; equally bad to us.
883  SVR_ERR( 0, "Failed to launch integrity scanner thread: %s", ex.what() );
884  }
885 
886  return ( launch_result );

889 // Utility to check to see if the integrity scanner task (thread) is done.
890 bool CScanner::check_is_done()
891 {
892  PUSH_FUNC;
893 
894  // This function should not be called unless we have launched the scanner task thread,
895  // and are waiting on this future.
896  if ( !scanner_future_result.valid() )
897  {
898  // But if already done, will just provide the warning message and signal "Done".
899  SVR_WARN( 1, "Found that the scanner thread was already Done." );
900  return ( true );
901  }
902 
903  SVR_DEBUG( 9, "Checking to see if the scanner thread is done..." );
904  // We use wait_for( 1s ) here to test to see if the future has been
905  // set by the completion of the thread, with that very short timeout
906  std::future_status status = scanner_future_result.wait_for( std::chrono::seconds( 1 ) );
907  if ( status == std::future_status::ready )
908  {
909  SVR_DEBUG( 5, "Found that the scanner thread is done." );
910  return ( true );
911  }
912  else
913  {
914  SVR_DEBUG( 9, "Found that the scanner thread is Not Done yet..." );
915  return ( false );
916  }
917 }
