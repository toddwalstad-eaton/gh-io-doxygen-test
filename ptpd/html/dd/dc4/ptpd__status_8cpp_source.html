81 bool CPtpdStatus::pmc_resp_parser( std::vector<std::string> &pmc_resp, mngmt_field_t &pmc_field_map )
82 {
83  bool success = false;
84 
85  for ( auto &line : pmc_resp )
86  {
87  // Ignore the line starting with two tabs, as it does not contain management fields.
88  if ( line.rfind( "\t\t", 0 ) == std::string::npos )
89  {
90  continue;
91  }
92  //
93  // Remove any tabs, leading and trailing spaces from a string
94  // Also remove spaces between words, but keep single space between them.
95  //
96  line.erase( std::remove( line.begin(), line.end(), '\t' ), line.end() );
97  line = std::regex_replace( line, std::regex( "^ +| +$|( ) +" ), "$1" );
98 
99  // Get key and value.
100  // i.e. line : "stepRemoved 1"
101  // key_field : "stepRemoved"
102  // value_field : "1"
103  size_t pos = line.find( ' ', 0 );
104  std::string key_field = line.substr( 0, pos );
105  std::string value_field = line.substr( pos + 1 );
106 
107  // Search for a field as a key in a map,and assigned above processed value to searched key
108  mngmt_field_t::iterator it = pmc_field_map.find( key_field );
109  if ( it != pmc_field_map.end() )
110  {
111  //
112  // On success, PMC never miss any single field in output
113  // Its OK to not associate success flag with each field.
114  //
115  success = true;
116  it->second = value_field;
117  std::string field_value = "Management field - " + it->first + " : " + it->second;
118  SYSLOG_DEBUG( 4, field_value.c_str() );
119  }
120  }
121 
122  return ( success );
123 }
