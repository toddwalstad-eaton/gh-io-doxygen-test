296 bool CSnmpdConfCdsHandler::update_usmuser_oobs( const std::unordered_map<px::string, px::string> &list )
297 {
298  PUSH_FUNC;
299  bool rv = true;
300 
301  // Get existing list to compare against
302  std::unordered_map<px::string, px::string> existing_list;
303  if ( !get_usmuser_oobs( existing_list ) )
304  {
305  SYSLOG_WARN( 0, "Cannot get list of usmUser OOBs before updating." );
306  return ( false );
307  }
308  px::cds::CSetOOBData so;
309  px::cds::CClearOOBData co;
310  for ( auto u : list )
311  {
312  if ( ( existing_list.count( u.first ) != 0 ) &&
313  ( existing_list[u.first] == u.second ) )
314  {
315  SYSLOG_DEBUG( 8, "User %s usmUser line unchanged", u.first.c_str() );
316  continue;
317  }
318  px::proto::toolkitStruct this_oob;
319  this_oob.mutable_generic()->set_stringdata( px::to_basicstr( u.second ) );
320  this_oob.add_read_permission_required( px::proto::Permission::SECURITY_READ );
321  this_oob.add_read_permission_required( px::proto::Permission::INTERNAL_APP );
322  this_oob.add_read_permission_required( px::proto::Permission::SECURITY_ADMIN );
323  this_oob.add_write_permission_required( px::proto::Permission::SECURITY_ADMIN );
324  this_oob.add_write_permission_required( px::proto::Permission::INTERNAL_APP );
325  this_oob.add_delete_or_permission_permission_required( px::proto::Permission::CONFIG_WRITE );
326  this_oob.add_delete_or_permission_permission_required( px::proto::Permission::INTERNAL_APP );
327  this_oob.add_delete_or_permission_permission_required( px::proto::Permission::SECURITY_ADMIN );
328  so.data.emplace_back( _usmuser_tag_prefix + u.first, this_oob, px::cds::persistence::save_always );
329  SYSLOG_DEBUG( 8, "Processed new or changed usmUser line for %s", u.first.c_str() );
330  } // End loop over 'list'
331  if ( !so.data.empty() )
332  {
333  std::shared_ptr<px::cds::CSetOOBData> sr = std::dynamic_pointer_cast<px::cds::CSetOOBData>( submit_wait( so ) );
334  if ( !sr || ( sr->status != px::cds::RequestStatus::SUCCEEDED ) )
335  {
336  SYSLOG_WARN( 0, "Error updating usmUser OOB objects" );
337  rv = false;
338  }
339  }
340  for ( auto u : existing_list )
341  {
342  if ( list.count( u.first ) == 0 )
343  {
344  SYSLOG_DEBUG( 8, "Will remove usmUser OOB data for user %s",
345  u.first.c_str() );
346  co.tags.push_back( _usmuser_tag_prefix + u.first );
347  }
348  }
349  if ( !co.tags.empty() )
350  {
351  std::shared_ptr<px::cds::CClearOOBData> cr = std::dynamic_pointer_cast<px::cds::CClearOOBData>( submit_wait( co ) );
352  if ( !cr || ( cr->status != px::cds::RequestStatus::SUCCEEDED ) )
353  {
354  SYSLOG_WARN( 0, "Error removing usmUser OOB objects" );
355  rv = false;
356  }
357  }
358 
359  return ( rv );
360 }
