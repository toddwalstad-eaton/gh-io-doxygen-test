63 static px::string escape( const px::string &str, bool bracket_quotes )
64 {
65  px::string rv;
66 
67  rv.reserve( str.length() + 2 );
68  if ( bracket_quotes )
69  {
70  rv.push_back( '\'' );
71  }
72  for ( size_t i = 0; i < str.length(); i++ )
73  {
74  if ( ( str.at( i ) == '\'' ) || ( str.at( i ) == '"' ) )
75  {
76  rv.push_back( '\\' );
77  }
78  rv.push_back( str.at( i ) );
79  }

82  rv.push_back( '\'' );
83  }
84  return ( rv );

543 {

547 
548  full_path += "/snmpd.conf";

551 
552  // Read the file into memory
553  input_file.open( file_path.c_str(), std::ios_base::in );
554  if ( input_file.fail() )
555  {
556  SYSLOG_DEBUG( 2, "SMNP userconf %s does not exist or cannot be read. Creating a fresh one.", file_path.c_str() );
557  }
558  else
559  {
560  for ( std::string thisline; std::getline( input_file, thisline );)
561  {
562  existing_userconf.push_back( px::to_pxstr( thisline ) );
563  }
564  input_file.close();
565  }
566 
567  std::vector<px::string> user_names;
568 
569  // Open the file in text mode, erase whatever's there
570  output_file.open( file_path.c_str(), std::ios_base::out | std::ios_base::trunc );
571 
572  if ( output_file.fail() )
573  {
574  SYSLOG_ERR( 0, "Failed to open SNMP userconf %s for output, new passphrases are lost. Error: %s", file_path.c_str(), strerror( errno ) );
575  return;
576  }
577  else
578  {
579  SYSLOG_INFO( 2, "Opened file %s for output", file_path.c_str() );
580  // First, write out existing file, except for usmUser lines
581  for ( const px::string &l : existing_userconf )
582  {
583  if ( l.substr( 0, 8 ) != "usmUser " )
584  {
585  output_file << l << "\n";
586  }
587  }
588  existing_userconf.clear();
589  for ( auto p : g_passphrases )
590  {
591  if ( ( !p.second.has_username() ) ||
592  ( p.second.username().empty() ) )
593  {
594  SYSLOG_WARN( 1, "Invalid passphrase OOB, no username. Skipping." );
595  continue;
596  }
597  bool has_auth = false;
598  bool has_priv = false;
599  if ( p.second.has_authentication_passphrase() &&
600  ( p.second.authentication_passphrase().length() >= 8 ) )
601  {
602  output_file << "createUser " << escape( p.second.username(), WANT_QUOTES ) <<
603  " SHA " << escape( p.second.authentication_passphrase(), WANT_QUOTES );
604  has_auth = true;
605  if ( p.second.has_encryption_passphrase() &&
606  ( p.second.encryption_passphrase().length() >= 8 ) )
607  {
608  output_file << " AES " << escape( p.second.encryption_passphrase(), WANT_QUOTES );
609  has_priv = true;
610  }
611  else
612  {
613  SYSLOG_DEBUG( 3, "User %s encryption passphrase unspecified or invalid; using authentication passphrase for encryption",
614  p.second.username().c_str() );
615  output_file << " AES " << escape( p.second.authentication_passphrase(), WANT_QUOTES );
616  has_priv = true;
617  }
618  output_file << "\n";
619  }
620  else
621  {
622  SYSLOG_WARN( 0, "User %s authentication passphrase unspecified or invalid; SNMPv3 for that user will not work", p.second.username().c_str() );
623  }
624  user_names.push_back( px::to_pxstr( p.second.username() ) );
625  g_cds_hndlr->update_has_authpriv( g_snmp_user_settings,
626  px::to_pxstr( p.second.username() ),

652 static px::string extract_user_from_usmuser_line( const px::string &line )
653 {
654  PUSH_FUNC;
655 
656  std::vector<px::string> words = px::tokenize( line );
657  if ( words.size() < 5 )
658  {
659  return ( "" );
660  }
661  if ( words[4].size() < 2 )
662  {
663  return ( words[4] );
664  }
665  if ( ( words[4].at( 0 ) == '0' ) && ( words[4].at( 1 ) == 'x' ) )
666  {
667  SYSLOG_DEBUG( 8, "Parsing usmUser user from hex of %s", words[4].c_str() );
668  px::string un;
669  std::vector<unsigned char> rv = px::hex_2_bytes( words[4].substr( 2 ) );
670  un.reserve( rv.size() );
671  for ( auto c : rv )
672  {
673  un.push_back( c );
674  }
675  SYSLOG_DEBUG( 8, "Parsed to %s", un.c_str() );
676  return ( un );
677  }
678  return ( words[4] );
679 }

685 int do_userscan()
686 {
687  PUSH_FUNC;
688 
689  std::unordered_map<px::string, px::string> new_list;
690  px::string file_path = g_options->snmpd_userconf_path + "/snmpd.conf";
691  std::ifstream instream;
692  instream.open( file_path.c_str(), std::ios_base::in );
693  if ( instream.fail() )
694  {
695  SYSLOG_ERR( 0, "Error opening userconf file %s", file_path.c_str() );
696  return ( 1 );
697  }
698  for ( std::string thisline; std::getline( instream, thisline );)
699  {
700  if ( thisline.substr( 0, 8 ) != "usmUser " )
701  {
702  continue;
703  }
704  px::string this_user = extract_user_from_usmuser_line( px::to_pxstr( thisline ) );
705  if ( !this_user.empty() )
706  {
707  if ( thisline.back() == '\n' )
708  {
709  thisline.pop_back();
710  }
711  SYSLOG_DEBUG( 5, "User scan got usmUser line for %s", this_user.c_str() );
712  new_list[this_user] = px::to_pxstr( thisline );
713  }
714  else
715  {
716  SYSLOG_WARN( 0, "User scan saw usmUser line but could not parse a user: [%s]", thisline.c_str() );
717  }
718  }
719  instream.close();
720 
721  px::proto::SnmpUserSettings current;
722 
723  g_cds_hndlr->get_snmp_user_oob( current );
724 
725  std::vector<px::string> existing_v3user_names;
726  for ( int i = 0; i < current.v3_users_size(); i++ )
727  {
728  if ( current.v3_users( i ).has_username() )
729  {
730  existing_v3user_names.push_back( current.v3_users( i ).username().c_str() );
731  }
732  }
733 
734  std::vector<px::string>::iterator iter;
735  auto it = new_list.begin();
736 
737  while ( it != new_list.end() )
738  {
739  iter = find( existing_v3user_names.begin(), existing_v3user_names.end(), it->first.c_str() );
740  //If the user from new_list is not found in existing v3 users, then remove it from the list
741  if ( iter == existing_v3user_names.end() )
742  {
743  SYSLOG_INFO( 1, "Removing usmUser OOB for %s user", it->first.c_str() );
744  it = new_list.erase( it );
745  }
746  else
747  {
748  ++it;
749  }
750  }
751  if ( !g_cds_hndlr->update_usmuser_oobs( new_list ) )
752  {
753  SYSLOG_ERR( 0, "User scan had error updating OOBs" );
754  return ( 1 );
755  }
756 
757  return ( 0 );
758 }
