65 static int jsmn_parse_primitive( jsmn_parser *parser, const char *js,
66  size_t len, jsmntok_t *tokens, size_t num_tokens )
67 {
68  jsmntok_t *token;
69  int start;
70 
71  start = parser->pos;
72 
73  for (; parser->pos < len && js[parser->pos] != '\0'; parser->pos++ )
74  {
75  switch ( js[parser->pos] )
76  {
77 #ifndef JSMN_STRICT
78  /* In strict mode primitive must be followed by "," or "}" or "]" */
79  case ':':
80 #endif
81  case '\t':
82  case '\r':
83  case '\n':
84  case ' ':
85  case ',':
86  case ']':
87  case '}':
88  goto found;
89  }
90  if ( ( js[parser->pos] < 32 ) || ( js[parser->pos] >= 127 ) )
91  {
92  parser->pos = start;
93  return ( JSMN_ERROR_INVAL );
94  }
95  }
96 #ifdef JSMN_STRICT
97  /* In strict mode primitive must be followed by a comma/object/array */
98  parser->pos = start;
99  return ( JSMN_ERROR_PART );
100 
101 #endif
102 
103 found:
104  if ( tokens == NULL )
105  {
106  parser->pos--;
107  return ( 0 );
108  }
109  token = jsmn_alloc_token( parser, tokens, num_tokens );
110  if ( token == NULL )
111  {
112  parser->pos = start;
113  return ( JSMN_ERROR_NOMEM );
114  }
115  jsmn_fill_token( token, JSMN_PRIMITIVE, start, parser->pos );
116 #ifdef JSMN_PARENT_LINKS
117  token->parent = parser->toksuper;
118 #endif
119  parser->pos--;
120  return ( 0 );
121 }

126 static int jsmn_parse_string( jsmn_parser *parser, const char *js,
127  size_t len, jsmntok_t *tokens, size_t num_tokens )
128 {
129  jsmntok_t *token;
130 
131  int start = parser->pos;
132 
133  parser->pos++;
134 
135  /* Skip starting quote */
136  for (; parser->pos < len && js[parser->pos] != '\0'; parser->pos++ )
137  {
138  char c = js[parser->pos];
139 
140  /* Quote: end of string */
141  if ( c == '\"' )
142  {
143  if ( tokens == NULL )
144  {
145  return ( 0 );
146  }
147  token = jsmn_alloc_token( parser, tokens, num_tokens );
148  if ( token == NULL )
149  {
150  parser->pos = start;
151  return ( JSMN_ERROR_NOMEM );
152  }
153  jsmn_fill_token( token, JSMN_STRING, start + 1, parser->pos );
154 #ifdef JSMN_PARENT_LINKS
155  token->parent = parser->toksuper;
156 #endif
157  return ( 0 );
158  }
159 
160  /* Backslash: Quoted symbol expected */
161  if ( ( c == '\\' ) && ( parser->pos + 1 < len ) )
162  {
163  int i;
164  parser->pos++;
165  switch ( js[parser->pos] )
166  {
167  /* Allowed escaped symbols */
168  case '\"':
169  case '/':
170  case '\\':
171  case 'b':
172  case 'f':
173  case 'r':
174  case 'n':
175  case 't':
176  break;
177  /* Allows escaped symbol \uXXXX */
178  case 'u':
179  parser->pos++;
180  for ( i = 0; i < 4 && parser->pos < len && js[parser->pos] != '\0'; i++ )
181  {
182  /* If it isn't a hex character we have an error */
183  if ( !( ( ( js[parser->pos] >= 48 ) && ( js[parser->pos] <= 57 ) ) || /* 0-9 */
184  ( ( js[parser->pos] >= 65 ) && ( js[parser->pos] <= 70 ) ) || /* A-F */
185  ( ( js[parser->pos] >= 97 ) && ( js[parser->pos] <= 102 ) ) ) ) /* a-f */
186  {
187  parser->pos = start;
188  return ( JSMN_ERROR_INVAL );
189  }
190  parser->pos++;
191  }
192  parser->pos--;
193  break;
194  /* Unexpected symbol */
195  default:
196  parser->pos = start;
197  return ( JSMN_ERROR_INVAL );
198  }
199  }
200  }
201  parser->pos = start;
202  return ( JSMN_ERROR_PART );
203 }

222  jsmntok_t *tokens, unsigned int num_tokens )
223 {
224  int r;
225  int i;
226  jsmntok_t *token;
227  int count = parser->toknext;
228 
229  for (; parser->pos < len && js[parser->pos] != '\0'; parser->pos++ )
230  {
231  char c;
232  jsmntype_t type;
233 
234  c = js[parser->pos];
235  switch ( c )
236  {
237  case '{':
238  case '[':
239  count++;
240  if ( tokens == NULL )
241  {
242  break;
243  }
244  token = jsmn_alloc_token( parser, tokens, num_tokens );
245  if ( token == NULL )
246  {
247  return ( JSMN_ERROR_NOMEM );
248  }
249  if ( parser->toksuper != -1 )
250  {
251  tokens[parser->toksuper].size++;
252 #ifdef JSMN_PARENT_LINKS
253  token->parent = parser->toksuper;
254 #endif
255  }
256  token->type = ( c == '{' ? JSMN_OBJECT : JSMN_ARRAY );
257  token->start = parser->pos;
258  parser->toksuper = parser->toknext - 1;
259  break;
260  case '}':
261  case ']':
262  if ( tokens == NULL )
263  {
264  break;
265  }
266  type = ( c == '}' ? JSMN_OBJECT : JSMN_ARRAY );
267 #ifdef JSMN_PARENT_LINKS
268  if ( parser->toknext < 1 )
269  {
270  return ( JSMN_ERROR_INVAL );
271  }
272  token = &tokens[parser->toknext - 1];
273  for (;;)
274  {
275  if ( ( token->start != -1 ) && ( token->end == -1 ) )
276  {
277  if ( token->type != type )
278  {
279  return ( JSMN_ERROR_INVAL );
280  }
281  token->end = parser->pos + 1;
282  parser->toksuper = token->parent;
283  break;
284  }
285  if ( token->parent == -1 )
286  {
287  break;
288  }
289  token = &tokens[token->parent];
290  }
291 #else
292  for ( i = parser->toknext - 1; i >= 0; i-- )
293  {
294  token = &tokens[i];
295  if ( ( token->start != -1 ) && ( token->end == -1 ) )
296  {
297  if ( token->type != type )
298  {
299  return ( JSMN_ERROR_INVAL );
300  }
301  parser->toksuper = -1;
302  token->end = parser->pos + 1;
303  break;
304  }
305  }
306  /* Error if unmatched closing bracket */
307  if ( i == -1 )
308  {
309  return ( JSMN_ERROR_INVAL );
310  }
311  for (; i >= 0; i-- )
312  {
313  token = &tokens[i];
314  if ( ( token->start != -1 ) && ( token->end == -1 ) )
315  {
316  parser->toksuper = i;
317  break;
318  }
319  }
320 #endif
321  break;
322  case '\"':
323  r = jsmn_parse_string( parser, js, len, tokens, num_tokens );
324  if ( r < 0 )
325  {
326  return ( r );
327  }
328  count++;
329  if ( ( parser->toksuper != -1 ) && ( tokens != NULL ) )
330  {
331  tokens[parser->toksuper].size++;
332  }
333  break;
334  case '\t':
335  case '\r':
336  case '\n':
337  case ' ':
338  break;
339  case ':':
340  parser->toksuper = parser->toknext - 1;
341  break;
342  case ',':
343  if ( ( tokens != NULL ) && ( parser->toksuper != -1 ) &&
344  ( tokens[parser->toksuper].type != JSMN_ARRAY ) &&
345  ( tokens[parser->toksuper].type != JSMN_OBJECT ) )
346  {
347 #ifdef JSMN_PARENT_LINKS
348  parser->toksuper = tokens[parser->toksuper].parent;
349 #else
350  for ( i = parser->toknext - 1; i >= 0; i-- )
351  {
352  if ( ( tokens[i].type == JSMN_ARRAY ) || ( tokens[i].type == JSMN_OBJECT ) )
353  {
354  if ( ( tokens[i].start != -1 ) && ( tokens[i].end == -1 ) )
355  {
356  parser->toksuper = i;
357  break;
358  }
359  }
360  }
361 #endif
362  }
363  break;
364 #ifdef JSMN_STRICT
365  /* In strict mode primitives are: numbers and booleans */
366  case '-':
367  case '0':
368  case '1':
369  case '2':
370  case '3':
371  case '4':
372  case '5':
373  case '6':
374  case '7':
375  case '8':
376  case '9':
377  case 't':
378  case 'f':
379  case 'n':
380  /* And they must not be keys of the object */
381  if ( ( tokens != NULL ) && ( parser->toksuper != -1 ) )
382  {
383  jsmntok_t *t = &tokens[parser->toksuper];
384  if ( ( t->type == JSMN_OBJECT ) ||
385  ( ( t->type == JSMN_STRING ) && ( t->size != 0 ) ) )
386  {
387  return ( JSMN_ERROR_INVAL );
388  }
389  }
390 #else
391  /* In non-strict mode every unquoted value is a primitive */
392  default:
393 #endif
394  r = jsmn_parse_primitive( parser, js, len, tokens, num_tokens );
395  if ( r < 0 )
396  {
397  return ( r );
398  }
399  count++;
400  if ( ( parser->toksuper != -1 ) && ( tokens != NULL ) )
401  {
402  tokens[parser->toksuper].size++;
403  }
404  break;
405 
406 #ifdef JSMN_STRICT
407  /* Unexpected char in strict mode */
408  default:
409  return ( JSMN_ERROR_INVAL );
410 
411 #endif
412  }
413  }
414 
415  if ( tokens != NULL )
416  {
417  for ( i = parser->toknext - 1; i >= 0; i-- )
418  {
419  /* Unmatched opened object or array */
420  if ( ( tokens[i].start != -1 ) && ( tokens[i].end == -1 ) )
421  {
422  return ( JSMN_ERROR_PART );
423  }
424  }
425  }
426 
427  return ( count );
428 }
