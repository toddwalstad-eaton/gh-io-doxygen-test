741  b_file->is_timed_out = true;
742  // IOTHUB_CLIENT_FILE_UPLOAD_GET_DATA_ABORT to indicate that the upload shall not continue
743  return ( IOTHUB_CLIENT_FILE_UPLOAD_GET_DATA_ABORT );
744  }
745 
746  if ( result == FILE_UPLOAD_OK )
747  {
748  if ( ( data != NULL ) && ( size != NULL ) )
749  {
750  if ( b_file->current_block < total_full_blocks )
751  {
752  // This is the full block with block_size bytes of data
753  *data = ( const unsigned char * )( ( b_file->source ) + ( b_file->current_block * block_size ) );
754  *size = block_size;
755  Log_Debug( "blob_getDataCallback : sending full block %d", b_file->current_block );
756  b_file->current_block++;
757  }
758  else if ( ( b_file->current_block == total_full_blocks ) && ( ( b_file->size % block_size ) != 0 ) )
759  {
760  // This is the last block in file which is less than block_size
761  *data = ( const unsigned char * )( ( b_file->source ) + ( b_file->current_block * block_size ) );
762  *size = ( b_file->size % block_size );
763  Log_Debug( "blob_getDataCallback : sending partial block %d", b_file->current_block );
764  b_file->current_block++;
765  }
766  else
767  {
768  // Setting data to NULL and/or passing size as zero indicates the upload is completed.
769  *data = NULL;
770  *size = 0;
771  Log_Debug( "blob_getDataCallback : Total blocks sent = %d", b_file->current_block );
772  }
773  }
774  else
775  {
776  // The last call to this callback is to indicate the result of uploading the previous data block provided.
777  // Note: In this last call, data and size pointers are NULL.
778  Log_Info( "blob_getDataCallback : Last call to getDataCallback. Total blocks sent = %d", b_file->current_block );
779  }
780  }
781  else
782  {
783  Log_Error( "blob_getDataCallback : Received unexpected result. Total blocks sent = %zu\n", b_file->current_block );
784  return ( IOTHUB_CLIENT_FILE_UPLOAD_GET_DATA_ABORT );
785  }
786 
787  // IOTHUB_CLIENT_FILE_UPLOAD_GET_DATA_OK to indicate that the upload shall continue.
788  return ( IOTHUB_CLIENT_FILE_UPLOAD_GET_DATA_OK );
789 }
