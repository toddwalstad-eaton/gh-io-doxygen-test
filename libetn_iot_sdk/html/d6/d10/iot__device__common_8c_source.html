1124  singlylinkedlist_destroy( commandResponse.list_2 );

1253 {

1255

1257

1262  {
1263  connection->connectionStatus = IOT_OPENED;
1264  }
1265  else if ( reasonCode == IOTHUB_CLIENT_CONNECTION_NO_NETWORK )

1268  * Azure code realizes that the attempt to open failed. */
1269  connection->connectionStatus = IOT_CLOSED;
1270  Log_Info( "Got low-level network loss indication." );
1271  }
1272  else if ( reasonCode == IOTHUB_CLIENT_CONNECTION_EXPIRED_SAS_TOKEN )
1273  {
1274  // Let the Azure code manage the reconnection, so no callback here
1275  // (Don't frighten the natives)
1276  /* The desire here was to let Azure manage the reconnection for us.
1277  * do_callback = false;
1278  * connection->connectionStatus = IOT_CONNECTING; -- will have to experiment and see whether to do this or do nothing
1279  * Log_Debug( "Letting Azure reconnect, after SAS Token epxiration" );
1280  * However, it turns out that this doesn't happen (yet - maybe someday).
1281  *
1282  * So I looked into arranging an under-the-hood reconnect here, without
1283  * alerting the IoT Agent process, but could not pull it off:
1284  * - The disconnect and connect function allocate and free structures,
1285  * which would make it impossible to let the IoT Agent get the updated "handle".
1286  * - The destroy function frees enqued messages, which we had wanted to avoid.
1287  * - All of these structures involve opaque pointers that we can't decode here.
1288  * - No suitable "just reconnect" Azure method exists that we could use.
1289  * - I did try to just call the DoWork function, but it did not work (reconnect on its own).
1290  * So we have to do the other alternative: declare the connection Closed, and start over.
1291  * Though that will still mean that we will lose any messages that were queued up. */
1292  connection->connectionStatus = IOT_CLOSED;
1293  Log_Info( "Got SAS Token epxiration notice; must close and reconnect." );
1294  }
1295  else
1296  {
1297  connection->connectionStatus = IOT_CLOSED;
1298  }
1299  connection->connectionStatusCode = statusCode;
1300  connection->connectionReasonCode = reasonCode;
1301 
1302  /* Invoke connection status callback (if registered)
1303  * Invoking the callback needs to be run on a separate thread */
1304  if ( do_callback && ( connection->connectionStatusCallback != NULL ) )
1305  {
1306 #ifndef PX_GREEN_IOT_DEVICE
1307  THREAD_HANDLE handle = NULL;
1308  if ( ThreadAPI_Create( &handle, ( THREAD_START_FUNC )connectionStatusChangedThread, connection ) != THREADAPI_OK )
