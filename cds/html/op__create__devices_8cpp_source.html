23  {
24  throw px::generic_err( "Invalid API message type given to COpCreateDevices" );
25  }

27  if ( unlikely( in == nullptr ) )
28  {
29  return;
30  }
31  msg->status = px::cds::RequestStatus::EMPTY_RESULT;
32  std::unique_ptr<CThreadPersistThis> persist_cmd;
33  persist_cmd.reset( new CThreadPersistThis );
34  std::set<px::id_t> pending_devices;
35 
36  try
37  {
38  std::vector<px::CDevice>::iterator it = in->devices.begin();
39  while ( it != in->devices.end() )
40  {
41  // No passing in subdevices or channels
42  if ( ( !( it->sub_devices.empty() ) ) ||
43  ( !( it->channels.empty() ) ) )
44  {
45  CDS_WARN( 1, "COpCreateDevices: cannot process device %zu with Subdevices or Channels", ( size_t ) it->id() );
46  it = in->devices.erase( it );
47 
48  AUDIT_LOG_MSG( AUDIT_LOG_DEVICE, LOG_WARNING, cee_info,
49  "Device Create for [%zu:%s] failed as it has sub-devices or channels in it",
50  ( size_t ) it->id(), it->name.c_str() );
51  msg->status = px::cds::RequestStatus::FAILED;
52  continue;
53  }
54 
55  // Can only create a Publisher device if the list is otherwise empty
56  if ( unlikely( it->role == px::device_role::publisher ) )
57  {
58  if ( !( cds.devices.empty() ) )
59  {
60  CDS_WARN( 1, "COpCreateDevices: cannot process Publisher device since we have devices." );
61  it = in->devices.erase( it );
62 
63  AUDIT_LOG_MSG( AUDIT_LOG_DEVICE, LOG_WARNING, cee_info,
64  "Device Create for [%zu:%s] failed as a publisher already exists",
65  ( size_t ) it->id(), it->name.c_str() );
66  msg->status = px::cds::RequestStatus::FAILED;
67  continue;
68  }
69  }
70  else
71  {
72  // If not the publisher, parent must be specified and exist either in CDS list or in the pending list that is to be created
73  if ( ( !( it->parent ) ) ||
74  ( ( cds.devices.count( *( it->parent ) ) == 0 ) && ( pending_devices.count( *( it->parent ) ) == 0 ) ) )
75  {
76  it = in->devices.erase( it );
77  AUDIT_LOG_MSG( AUDIT_LOG_DEVICE, LOG_WARNING, cee_info,
78  "Device Create for [%zu:%s] failed as the parent is not specified/parent ID is not present in cds", ( size_t ) it->id(), it->name.c_str() );
79  msg->status = px::cds::RequestStatus::BAD_PARAMETER;
80  continue;
81  }
82  }
83 
84  // If a port, parent must be the publisher
85  if ( unlikely( it->role == px::device_role::port ) )
86  {
87  if ( *( it->parent ) != cds.publisher_id )
88  {
89  CDS_WARN( 1, "COpCreateDevices: Port device %zu must have publisher as parent", ( size_t ) it->id() );
90  it = in->devices.erase( it );
91 
92  AUDIT_LOG_MSG( AUDIT_LOG_DEVICE, LOG_WARNING, cee_info,
93  "Device Create for [%zu:%s] failed as the parent is not the publisher",
94  ( size_t ) it->id(), it->name.c_str() );
95  msg->status = px::cds::RequestStatus::FAILED;
96  continue;
97  }
98  }
99 
100  // ID must not already exist (very unlikely but worth checking)
101  if ( unlikely( ( cds.devices.count( it->id() ) != 0 ) || ( pending_devices.count( it->id() ) != 0 ) ) )
102  {
103  AUDIT_LOG_MSG( AUDIT_LOG_DEVICE, LOG_WARNING, cee_info,
104  "Device Create for [%zu:%s] failed as the ID is already present in cds",
105  ( size_t ) it->id(), it->name.c_str() );
106  msg->status = px::cds::RequestStatus::FAILED;
107  // while in other cases, just throw out the problematic device, why stop here?
108  throw px::generic_err( "Attempt to create device with already-existing ID, somehow" );
109  }
110 
111  // Device's state timestamp is now (state set now)
112  it->state_timestamp = px::CWallTime::now();
113  pending_devices.insert( it->id() );
114  it++;
115  } // End loop over all passed-in devices
116 
117  cee_info._cee_status = CEE_STS_success;
118  // At this point, all devices left have assigned IDs and are valid to create
119  // Put them in map
120  for ( unsigned int i = 0; i < in->devices.size(); i++ )
121  {
122  if ( unlikely( in->devices[i].role == px::device_role::publisher ) )
123  {
124  cds.publisher_id = in->devices[i].id();
125  }
126  else
127  {
128  // Devices neither "publisher" nor "port" are created in the
129  // not_communicating state
130  if ( ( in->devices[i].role != px::device_role::port ) &&
131  ( in->devices[i].role != px::device_role::publisher ) )
132  {
133  px::CBitset<px::realtime_state> st = in->devices[i].get_own_state();
134  st.set( px::realtime_state::not_communicating, true );
135  in->devices[i].set_state( st );
136  }
137  // Inherit state from parent
138  in->devices[i].set_inherited_state( cds.devices[*( in->devices[i].parent )].get_state() );
139  // Add to parent's subdevices
140  cds.devices[*( in->devices[i].parent )].sub_devices.insert( in->devices[i].id() );
141  persist_cmd->devices[*( in->devices[i].parent )] = cds.devices[*( in->devices[i].parent )];
142 
143  if ( ( in->devices[i].guid.size() > 0 ) &&
144  ( in->devices[i].guid.size() < 32 ) )
145  {
146  // Convert raw guid bytes to the ASCII format expected in the CDS
147  std::vector<unsigned char> ascii_guid;
148  ascii_guid = px::vec_bytes_2_hex( in->devices[i].guid );
149  if ( ascii_guid.size() )
150  {
151  in->devices[i].guid = ascii_guid;
152  }
153  }
154 
155  AUDIT_LOG_MSG( AUDIT_LOG_DEVICE, LOG_INFO, cee_info, "Device Create: [%zu:%s]",
156  ( size_t ) in->devices[i].id(),
157  in->devices[i].name.c_str() );
158  }
159  std::pair<px::id_t, px::CDevice> dev_pair( in->devices[i].id(), in->devices[i] );
160  cds.devices.insert( dev_pair );
161  persist_cmd->devices.insert( dev_pair );
162 
163  //add to existing recursive subscribers
164  if ( in->devices[i].parent )
165  {
166  for ( auto stp : cds.subs.list_device_recursive_subscription_types() )
167  {
168  std::vector<uint64_t> _subs;
169  px::sub::CItemId event( *( in->devices[i].parent ), stp );
170  cds.subs.get_subscribers( event, _subs );
171  //set the correct id and register
172  event._id = in->devices[i].id();
173  for ( auto &listener: _subs )
174  {
175  cds.subs.add_subscription( listener, event );
176  }
177  }
178  }
179 
180  // notify subscribers of settings/state change
181  // unlikely that anyone's already subscribed to this device
182  // except via a recursive subscription to the parent
183  // (how else did they get the ID?) but it's cheap to cover these cases
184  for ( auto stp : cds.subs.list_device_direct_subscription_types() )
185  {
186  std::vector<uint64_t> _subs;
187  px::sub::CItemId event( in->devices[i].id(), stp );
188  cds.subs.get_subscribers( event, _subs );
189  for ( auto &listener: _subs )
190  {
191  if ( stp == px::sub::subscription_type::device_settings )
192  {
193  ops.insert( new COpEventNotify( listener, event, in->devices[i] ) );
194  }
195  else if ( stp == px::sub::subscription_type::device_state )
196  {
197  ops.insert( new COpEventNotify( listener, event, in->devices[i].get_state() ) );
198  }
199  }
200  }
201 
202  // Notify any subscribers to device_create_recursive or device_state_recursive that it's created
203  std::vector<uint64_t> _subs;
204  px::sub::CItemId event( in->devices[i].id(), px::sub::subscription_type::device_create_recursive );
205  cds.subs.get_subscribers( event, _subs );
206  for ( auto &listener: _subs )
207  {
208  ops.insert( new COpEventNotify( listener, event, in->devices[i] ) );
209  }
210  _subs.clear();
211  px::sub::CItemId sevent( in->devices[i].id(), px::sub::subscription_type::device_state_recursive );
212  cds.subs.get_subscribers( sevent, _subs );
213  for ( auto &listener: _subs )
214  {
215  ops.insert( new COpEventNotify( listener, sevent, in->devices[i].get_state() ) );
216  }
217  } // End iteration over all new devices
218  }
219  catch ( const px::generic_err &e )
220  {
221  CDS_ERR( 0, "Error in COpCreateDevices: %s", e.what() );
222  }
223  catch ( const std::bad_alloc &e )
224  {
225  msg->status = px::cds::RequestStatus::FAILED;
226  CDS_ERR( 0, "Out of memory in COpCreateDevices" );
227  }
228  catch ( ... )
229  {
230  msg->status = px::cds::RequestStatus::FAILED;
231  CDS_ERR( 0, "Unknown error in COpCreateDevices" );
232  }
233 
234  // There're 2 choices, 1) not SUCCEEDED when one device creation failed,
235  // or 2) SUCCEEDED when at least one device creation succeeded.
236  // For now, keep the old logic, choice 2. In another word, only all fail,
237  // return the last failure code
238  // For mercury, always one device at a time, the 2 choices are the same
239  if ( in->devices.size() > 0 )
240  {
241  msg->status = px::cds::RequestStatus::SUCCEEDED;
242  }
243 
244  // Return response to client
245  if ( !( px::cds::CServer::get()->send_response( msg ) ) )
246  {
247  CDS_ERR( 0, "Lost response to Create Device to a full queue" );
248  }
249 
250  // Enqueue op to update parent devices' change signatures
251  COpBumpChangeSignature *bumper = new COpBumpChangeSignature();
252  for ( auto &this_dev : in->devices )
253  {
254  if ( this_dev.parent )
255  {
256  if ( bumper->devs_to_bump.count( *( this_dev.parent ) ) == 0 )
257  {
258  bumper->devs_to_bump.insert( *( this_dev.parent ) );
259  }
260  }
261  }
262  ops.insert( bumper );
263 
264  // Tell persistence thread to save the affected devices
265  if ( likely( cds.persistence_thread ) )
266  {
267  cds.persistence_thread->send( std::move( persist_cmd ) );
268  }
269  else
270  {
271  CDS_ERR( 0, "Attempt to save created devices but persistence thread not running" );
272  }
