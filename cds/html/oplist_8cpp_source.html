141 bool COpList::execute( std::shared_ptr<px::cds::CRequestResponse> msg,
142  CDataStore &cds )
143 {
144  PUSH_FUNC;
145  if ( _pending_buckets.empty() )
146  {
147  return ( false );
148  }
149 
150  pending_buckets_t::iterator ibucket = _pending_buckets.begin();
151  size_t prio = *ibucket;
152 
153  assert( prio < _buckets.size() );
154  assert( _buckets[prio] );
155 
156  try
157  {
158  bool has_more = _buckets[prio]->execute( msg, *this, cds );
159  if ( !has_more )
160  {
161  _pending_buckets.erase( ibucket );
162  }
163  }
164  catch ( const px::generic_err &x )
165  {
166  CDS_ERR( 0, "Exception executing CDS operation: %s", x.what() );
167  }
168 
169  return ( true );
170 }
