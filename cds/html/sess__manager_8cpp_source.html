125  user_perms = _auth_mgr->AuthenticateUserPlainText( username, service, password );
126  bool user_exists = _auth_mgr->GetUserInfo( username, user );// can't fail if Authenticate succeeded
127  if ( user_perms == empty_perms )
128  {
129  cee_info._cee_status = CEE_STS_failure;
130  if ( user_exists )
131  {

135  }

139  }

143  }

145  }
146  else

149  }
150  return ( token );
151  }
152  // If there's a limit, count the number of logged-in sessions this user has,
153  // fail if exceeded. Don't count subsessions toward this limit.
154  uint32_t effective_limit = _concurrent_session_limit;
155  if ( user.passwd_policy.max_concurrent_sessions )
156  {
157  effective_limit = *user.passwd_policy.max_concurrent_sessions;
158  }
159  if ( effective_limit != 0 )
160  {
161  uint32_t session_count = 0;
162  {
163  std::lock_guard<std::mutex> lock{_mutex};
164  for ( auto i : _sessions )
165  {
166  // If parent session ID isn't set it means this is a parent session and is a
167  // countable session.
168  if ( ( i.second.GetUserName() == username ) && ( i.second.GetParentSessionId() == "" ) )
169  {
170  session_count++;
171  }
172  }
173  } // end lock scope
174  if ( ( effective_limit > 0 ) && ( session_count >= effective_limit ) )
175  {
176  cee_info._cee_status = CEE_STS_failure;
177  AUDIT_LOG_MSG( AUDIT_LOG_SESSION, LOG_WARNING, cee_info, "Login %s to %s failed, session limit exceeded", username.c_str(), service.c_str() );
178  // Don't update lastlog for this
179  // ugly: flag value
180  return ( "Limit Exceeded" );// random ones are longer, won't get mistaken
181  }
182  }
183  token = session_token_rand();
184  uint32_t timeout = user.services[service]; // also guaranteed present because of auth success
185  if ( timeout == UINT32_MAX )
186  {
187  timeout = 0;// different values for 'unlimited'
188  }
189  // Setup the Info for Audit Logging
190  AUDIT_LOG_MSG( AUDIT_LOG_SESSION, LOG_INFO, cee_info, "Login %s to %s", username.c_str(), service.c_str() );
191  {
192  std::lock_guard<std::mutex> lock{_mutex};
193  _sessions.emplace( std::make_pair( token, CSession( username, service, user_perms, client, timeout, 0, ip ) ) );
194  _handles[token] = _next_handle;
195  _next_handle++;
196  if ( unlikely( _next_handle == 0 ) )
197  {
198  _next_handle++;
199  }
200  }
201  update_lastlog( username, false, ip );
202  return ( token );
203 }

361 
362  std::lock_guard<std::mutex> lock{_mutex};
363  sess_map_iter s = _sessions.find( session_token );
364  if ( s == _sessions.end() )
365  {

367  }

373  AUDIT_LOG_MSG( AUDIT_LOG_SESSION, LOG_INFO, cee_info,

375  remove_session( s, session_token, px::sub::subscription_type::session_timeout );
376  return ( false );
377  }

379  {
380  std::lock_guard<std::mutex> conn_lock{px::cds::CServer::get()->conn_list_lock};
381  auto c = px::cds::CServer::get()->connections.find( s->second.GetClientID() );
382  if ( unlikely( c == px::cds::CServer::get()->connections.end() ) )
383  {
384  px::string username = s->second.GetUserName();
385  px::string service = s->second.GetService();
386  toolkit::CCeeLogMsg cee_info( CEE_ACT_disconnect, CEE_STS_success, username, s->second.GetIP(), APP_NAME );
387  AUDIT_LOG_MSG( AUDIT_LOG_SESSION, LOG_INFO, cee_info,
388  "Logout %s from %s via client disconnect", username.c_str(), service.c_str() );
389  // this reason value means "don't send a message"
390  remove_session( s, session_token, px::sub::subscription_type::max_event_type );

392  }

482 px::string CSessManager::session_token_rand()
483 {
484  PUSH_FUNC;
485 
486  std::vector<uint8_t> random_bytes;
487  int ret_val = 0;
488 
489  random_bytes.resize( SESSMGR_TOKEN_BYTES );
490 
491  // First try /dev/urandom because that never blocks.
492  // This way we don't delay startup.
493  int randfd = open( "/dev/urandom", O_RDONLY );
494  if ( randfd >= 0 )
495  {
496  // Don't iterate the read, if for some unlikely reason we don't get
497  // all the bytes we want, fall back to RAND_bytes().
498  int read_bytes = read( randfd, &random_bytes[0], random_bytes.size() );
499  if ( read_bytes == random_bytes.size() )
500  {
501  ( void ) close( randfd );
502  return ( px::CBase64Converter::uri_encode( random_bytes ) );
503  }
504  ( void ) close( randfd );
505  }
506  ret_val = RAND_bytes( ( unsigned char * )( &random_bytes[0] ), random_bytes.size() );
507  if ( ( ret_val == 0 ) || ( ret_val == -1 ) )
508  {
509  CDS_ERR( 0, "RAND_bytes failed and returned error :%d", ret_val );
510  return ( px::string( "NonRandomToken" ) );
511  }
512  return ( px::CBase64Converter::uri_encode( random_bytes ) );
513 }

621 void CSessManager::remove_session( sess_map_iter &map_pair, const px::string &session_token,
622  px::sub::subscription_type reason )
623 {
624  PUSH_FUNC;
625 
626  // If this session has subsessions, remove them first
627  // Note that GetSubsessions() returns a copy of its set of tokens,
628  // so we are OK to remove them from the parent session
629  for ( const px::string &subsess_tok : map_pair->second.GetSubsessions() )
630  {
631  sess_map_iter sub_s = _sessions.find( subsess_tok );
632  if ( sub_s != _sessions.end() )
633  {
634  px::string username = sub_s->second.GetUserName();
635  px::string service = sub_s->second.GetService();
636  toolkit::CCeeLogMsg cee_info( CEE_ACT_logout, CEE_STS_success, username, sub_s->second.GetIP(), APP_NAME );
637  AUDIT_LOG_MSG( AUDIT_LOG_SESSION, LOG_INFO, cee_info,
638  "Logout sub-user %s from %s", username.c_str(), service.c_str() );
639  remove_session( sub_s, subsess_tok, reason );
640  }
641  }
642 
643  // On the other hand, if this is a subsession, remove us from our parent's list of tokens
644  const px::string &parent_tok = map_pair->second.GetParentSessionId();
645  if ( !parent_tok.empty() )
646  {
647  sess_map_iter parent_s = _sessions.find( parent_tok );
648  if ( parent_s != _sessions.end() )
649  {
650  parent_s->second.RemoveSubsession( session_token );
651  /* Again, this is OK when remove_session() is called recursively,
652  * because the loop above gets a copy of the set of tokens that this was just removed from.
653  * And if remove_session() is called as a one-off for a subsession, this is also still OK.
654  */
655  }
656  }
657 
658  // Now, in all cases, remove ourself from both of these maps
659  _sessions.erase( session_token );
660  _handles.erase( session_token );
661 
662  // Notify client this session was terminated by server initiated events
663  if ( ( reason > px::sub::subscription_type::max_subscription_type ) &&
664  ( reason < px::sub::subscription_type::max_event_type ) )
665  {
666  std::shared_ptr<px::cds::CSubscriberNotify> msg( new px::cds::CSubscriberNotify() );
667  msg->callback_handle = px::cds::CallbackHandle( CALLBACK_UNSOLICITED );
668  msg->subscriber_id = 0;
669  msg->transaction_id = 0;
670  msg->status = px::cds::RequestStatus::SUCCEEDED;
671  msg->client_id = map_pair->second.GetClientID();
672  msg->update_sessions.insert( session_token );
673  msg->event = reason;
674  if ( !( px::cds::CServer::get()->send_response( msg ) ) )
675  {
676  CDS_NOTICE( 0, "Lost session term event notify to a full queue" );
677  }
678  }
679 }

713 void CSessManager::update_lastlog( const px::string &username, bool failed,
714  const px::string &ip )
715 {
716  PUSH_FUNC;
717 
718  std::unique_lock<std::mutex> _lock{_lastlog_mutex};
719  auto i = _lastlog.find( username );
720  if ( i == _lastlog.end() )
721  {
722  CLastLog new_entry;
723  if ( failed )
724  {
725  new_entry.failures = 1;
726  }
727  else
728  {
729  new_entry.last_login_time = px::CWallTime::now();
730  new_entry.last_login_host = ip;
731  new_entry.failures = 0;
732  }
733  _lastlog.insert( std::make_pair( username, new_entry ) );
734  return;
735  }
736  if ( failed )
737  {
738  i->second.failures++;
739  }
740  else
741  {
742  i->second.failures = 0;
743  i->second.last_login_time = px::CWallTime::now();
744  i->second.last_login_host = ip;
745  }
746  if ( unlikely( _db_path.empty() ) )
747  {
748  CDS_DEBUG( 8, "Not saving last-login info to DB, path not set" );
749  return;
750  }
751  try
752  {
753  px::db::CKeyValueStore db( _db_path, false, false );
754  px::db::CBlobHolder key_blob, val_blob;
755  px::tk_proto::storage::LoginHistory vp;
756  i->second.to( vp );
757  key_blob.append( username.data(), username.size() );
758  val_blob.reset( vp.ByteSize() );
759  vp.SerializeWithCachedSizesToArray( val_blob.get_mutable_ptr() );
760  px::db::CBatch wb;
761  wb.add( std::move( key_blob ), std::move( val_blob ) );
762  db.write( wb, false /* no sync, db is about to close anyway */ );
763  CDS_DEBUG( 8, "Saved last-log info to DB for user %s",
764  username.c_str() );
765  }
766  catch ( ... )
767  {
768  CDS_ERR( 0, "Could not open last-log DB to store info" );
769  return;
770  }
771 }

773 void CSessManager::LoadLastLog( const px::string &db_path )
774 {
775  PUSH_FUNC;
776 
777  _db_path = db_path;
778  px::db::CKeyValueStore db( db_path, true /* create */, false /* don't compress */ );
779  uint32_t schema_ver = px::db::GetSchemaVersion( db, 1 );
780  if ( schema_ver != 0 )
781  {
782  CDS_ERR( 0, "Not loading last-login data, schema version %u not supported", schema_ver );
783  return;
784  }
785  px::db::CIterator i = db.iterator( false /* no caching */ );
786  i.first();
787  if ( likely( i.valid() ) )
788  {
789  if ( likely( ( i.key_size() == 1 ) && ( i.key_data()[0] == 0 ) ) )
790  {
791  ++i;// skip schema version
792  }
793  }
794  std::unique_lock<std::mutex> lock{_lastlog_mutex};
795  _lastlog.clear();
796  std::set<px::string> invalid_users;
797  while ( i.valid() )
798  {
799  px::string username( ( const char * )i.key_data(), ( size_t )i.key_size() );
800  px::tk_proto::storage::LoginHistory vp;
801  vp.ParseFromArray( i.value_data(), i.value_size() );
802  CLastLog ll( vp );
803  px::CUser dummy;
804  if ( !_auth_mgr->GetUserInfo( username, dummy ) )
805  {
806  CDS_DEBUG( 2, "Skipping and removing last-login information for nonexistent user %s",
807  username.c_str() );
808  invalid_users.insert( username );
809  ++i;
810  continue;
811  }
812  CDS_DEBUG( 8, "Loading last-login information for user %s",
813  username.c_str() );
814  _lastlog.insert( std::make_pair( username, ll ) );
815  ++i;
816  } // End loop over DB
817  px::db::CBatch del_batch;
818  for ( auto u : invalid_users )
819  {
820  px::db::CBlobHolder key_blob;
821  key_blob.append( u.data(), u.size() );
822  del_batch.remove( std::move( key_blob ) );
823  }
824  if ( !invalid_users.empty() )
825  {
826  CDS_DEBUG( 2, "Removing nonexistent users from last-login DB" );
827  db.write( del_batch, true );
828  }
829 }
