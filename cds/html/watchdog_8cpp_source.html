34 void CWatchdog::watchdog_thread_func()
35 {
36  // Set up syslog object for this thread
37  try
38  {
39  _thr_local_syslog.reset( new toolkit::CSysLog() );
40  }
41  catch ( const std::exception & )
42  {
43  // can't log here, logging not working
44  // just carry on
45  }
46  SYSLOG_VERBOSITY = _cds_verb_array[CDS_LOG_WATCHDOG];
47  px::CUtil::set_thread_name( "Sys Watchdog" );
48 
49  while ( 1 )
50  {
51  // Figure out when the next timeout is
52  px::CMonoTime next_timeout;
53  px::CMonoTime now = px::CMonoTime::now();
54  _clients_lock.lock();
55  for ( auto c : _clients )
56  {
57  if ( c.second.deadline.valid() &&
58  ( !next_timeout.valid() || ( c.second.deadline < next_timeout ) ) )
59  {
60  next_timeout = c.second.deadline;
61  }
62  }
63  _clients_lock.unlock();
64  if ( _hw_dog_wait != 0 )
65  {
66  /* If the client-based deadline is later than our hw watchdog wait after "now",
67  * then use the necessary and sooner hw ping time as our deadline. */
68  if ( ( !next_timeout.valid() ) ||
69  ( next_timeout.tv_sec > ( now.tv_sec + _hw_dog_wait ) ) )
70  {
71  next_timeout.tv_sec = now.tv_sec + _hw_dog_wait;
72  }
73  }
74  // Minimum sleep: 1 second
75  if ( next_timeout.valid() && ( next_timeout.tv_sec <= now.tv_sec ) )
76  {
77  next_timeout.tv_sec = now.tv_sec + 1;
78  }
79  // Wait until that timeout, or until the list of clients changes
80  if ( next_timeout.valid() )
81  {
82  if ( next_timeout > now )
83  {
84  CDS_LOG_DEBUG( CDS_LOG_WATCHDOG, 9, "Watchdog thread waiting %ld seconds",
85  next_timeout.tv_sec - now.tv_sec );
86  // Since pthread_cond_timedwait() unlocks and then locks the mutex, it is good form to lock it
87  // before calling pthread_cond_timedwait(), and unlock it after.
88  // Ignoring the unlikely error returns here.
89  pthread_mutex_lock( &_client_list_cond_mutex );
90  ( void ) pthread_cond_timedwait( &_client_list_cond, &_client_list_cond_mutex,
91  &next_timeout );
92  pthread_mutex_unlock( &_client_list_cond_mutex );
93  }
94  else
95  {
96  sleep( 1 );
97  }
98  }
99  else
100  {
101  // No clients or hardware watchdog - wait forever for client
102  // list to change
103  CDS_LOG_INFO( CDS_LOG_WATCHDOG, 1, "Watchdog thread waiting forever for first client (no hw wd)" );
104  pthread_mutex_lock( &_client_list_cond_mutex );
105  pthread_cond_wait( &_client_list_cond, &_client_list_cond_mutex );
106  pthread_mutex_unlock( &_client_list_cond_mutex );
107  CDS_LOG_INFO( CDS_LOG_WATCHDOG, 1, "Waiting Watchdog thread awoken" );
108  }
109 
110  // We woke up from sleep. First, check if CDS wants this thread to exit
111  if ( _bg_exit_please )
112  {
113  CDS_LOG_DEBUG( CDS_LOG_WATCHDOG, 1, "Watchdog thread exiting" );
114  break;
115  }
116  // Pat the hardware watchdog, if open
117  if ( _hw_dog_fd != -1 )
118  {
119  ( void ) write( _hw_dog_fd, "\0", 1 );
120  }
121 
122  // See if any clients timed out
123  CWatchdogState timed_out_client;
124  bool timed_out_client_found = false;
125  px::CMonoTime check_time = px::CMonoTime::now();
126  _clients_lock.lock();
127  for ( auto c : _clients )
128  {
129  if ( c.second.deadline < check_time )
130  {
131  timed_out_client_found = true;
132  // Copy so we can release the lock
133  timed_out_client = c.second;
134  break;
135  }
136  }
137  _clients_lock.unlock();
138  if ( unlikely( timed_out_client_found ) )
139  {
140  // See if the client has gone away, clear if so and it's auto-clearing
141  px::cds::CServer *pcds;
142  try
143  {
144  pcds = px::cds::CServer::get();
145  }
146  catch ( const px::generic_err & )
147  {
148  CDS_FERR( 0, "Very unlikely failure of CServer::get()" );
149  continue; // skip the below
150  }
151  pcds->conn_list_lock.lock();
152  bool client_found = ( pcds->connections.count( timed_out_client.client_id ) != 0 );
153  pcds->conn_list_lock.unlock();
154  if ( timed_out_client.auto_clear )
155  {
156  if ( !client_found )
157  {
158  CDS_LOG_DEBUG( CDS_LOG_WATCHDOG, 7, "Auto-clearing client %hu (%s)",
159  timed_out_client.client_id,
160  timed_out_client.client_name.c_str() );
161  _clients_lock.lock();
162  _clients.erase( timed_out_client.client_id );
163  _clients_lock.unlock();
164  continue;
165  }
166  }
167  CDS_CRIT( 0, "Client %hu (%s) timed out its watchdog after %u seconds of no messages, wdg ping timeout time is %lld but current time is %lld",
168  timed_out_client.client_id, timed_out_client.client_name.c_str(),
169  timed_out_client.timeout, next_timeout.tv_sec, check_time );
170  // In case we don't reboot, update this client so we don't see this
171  // message again, at least until another timeout passes
172  ping( timed_out_client.client_id );
173 
174  if ( _reboot )
175  {
176  CDS_ERR( 0, "Will reboot system" );
177  reboot();
178  }
179  else
180  {
181  if ( !client_found )
182  {
183  CDS_ERR( 0, "Not rebooting system. Client %hu (%s) connection no longer present, clearing its watchdog",
184  timed_out_client.client_id, timed_out_client.client_name.c_str() );
185  _clients_lock.lock();
186  _clients.erase( timed_out_client.client_id );
187  _clients_lock.unlock();
188  }
189  else
190  {
191  CDS_ERR( 0, "Not rebooting system. Client %hu (%s) connection still present, may time out again.",
192  timed_out_client.client_id, timed_out_client.client_name.c_str() );
193  }
194  }
195  }
196  } // end while forever loop
197 }

199 void CWatchdog::reboot()
200 {
201  // Flush disk writes, to be nice
202  sync();
203  gid_t egid = getegid();
204  uid_t euid = geteuid();
205  //raise permission level for the reboot operation
206  try
207  {
208  px::CUtil::set_egid( "root" );
209  px::CUtil::set_euid( "root" );
210  }
211  catch ( const px::generic_err &x )
212  {
213  // Failed to set user ID
214  CDS_ERR( 0, "GID/UID set to root to execute reboot operation failed. Err: %s.", x.what() );
215  if ( setegid( egid ) != 0 )
216  {
217  CDS_ERR( 0, "set GID back to original failed errno:%s", strerror( errno ) );
218  }
219  if ( seteuid( euid ) != 0 )
220  {
221  CDS_ERR( 0, "set UID back to original failed errno:%s", strerror( errno ) );
222  }
223  return;
224  }
225  // Tell 'init' we'd like to reboot
226  kill( 1, SIGTERM );
227  if ( ( setegid( egid ) != 0 ) || ( seteuid( euid ) != 0 ) )
228  {
229  CDS_ERR( 0, "Failed to restore GID/UID after executing reboot command. Err: %u.", errno );
230  return;
231  }
232  // Wait here forever, also providing us a backup: if the
233  // hardware watchdog is active, it'll trigger if we don't reboot
234  while ( 1 )
235  {
236  sleep( 86400 );
237  }
238 }

240 void CWatchdog::resume()
241 {
242  // Start hardware watchdog, if asked to
243  if ( _reboot && ( !( _hw_dog_dev.empty() ) ) )
244  {
245  if ( _hw_dog_fd == -1 )
246  {
247  // Eg, "/dev/watchdog"
248  try
249  {
250  _hw_dog_fd = open( _hw_dog_dev.c_str(), O_WRONLY );
251  }
252  catch ( const px::generic_err &x )
253  {
254  // Failed to open the hardware watchdog driver
255  SYSLOG_FERR( 0, "Couldn't open watchdog driver file %s.", _hw_dog_dev.c_str() );
256  CDS_ERR( 0, "Failed to open watchdog driver file %s with error: %s.", _hw_dog_dev.c_str(), x.what() );
257  return;
258  }
259  if ( _hw_dog_fd == -1 )
260  {
261  CDS_LOG_WARN( CDS_LOG_WATCHDOG, 0, "Could not enable hardware watchdog. Err: %d", errno );
262  // Will still spawn the background thread below; needs to be there for
263  // the dead-cds-client watchdog detection as well.
264  }
265  else
266  {
267  // Default: If we can't find out via ioctl, pat every second
268  _hw_dog_wait = 1;
269  unsigned long p = 0;
270  int irc = ioctl( _hw_dog_fd, WDIOC_GETTIMEOUT, &p );
271  if ( irc == 0 )
272  {
273  _hw_dog_wait = p / 4; // Pat at 1/4 of timeout
274  if ( _hw_dog_wait == 0 )
275  {
276  _hw_dog_wait = 1;
277  }
278  }
279  CDS_LOG_DEBUG( CDS_LOG_WATCHDOG, 3, "Hardware watchdog enabled, will pat every %u seconds", _hw_dog_wait );
280  CDS_INFO( 0, "Hardware watchdog enabled, will pat every %u seconds.", _hw_dog_wait );
281  }
282  }
283  }
284  // Start background thread to pat the watchdog, check clients
285  _bg_exit_please = false;
286  _bg_thread = std::thread( &CWatchdog::watchdog_thread_func, this );
287 }

289 void CWatchdog::pause()
290 {
291  // Tell the background thread we'd like for it to exit
292  _bg_exit_please = true;
293  // Choosing not to lock the mutex here; not required.
294  pthread_cond_broadcast( &_client_list_cond );
295 
296  // Close the hardware watchdog
297  if ( _hw_dog_fd != -1 )
298  {
299  ( void ) write( _hw_dog_fd, "V", 1 ); // Watchdog 'magic close' so as not to
300  // reboot system because we went away

302  // so this probably has no effect
303  close( _hw_dog_fd );
304  _hw_dog_fd = -1;
305  }
306
