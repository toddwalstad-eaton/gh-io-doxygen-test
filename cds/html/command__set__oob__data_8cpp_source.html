79  bool perms_change = false;
80  bool data_change = true;
81  px::CBitset<px::permissions> old_rd_perms, old_dp_perms,

83  px::CBitset<px::permissions> old_wr_perms = oob_effective_perms( cur_oob.write_permission_required(), px::permissions::config_write );
84  old_dp_perms = oob_effective_perms( cur_oob.delete_or_permission_permission_required(), px::permissions::config_write );
85  // First, see if this is new. If so we need CONFIG_WRITE permission.
86  if ( !exists )
87  {
88  perms_good = req_perms.get( px::permissions::config_write );
89  data_change = false;
90  CDS_DEBUG( 8, "Set OOB: New object, create permission %s",
91  perms_good ? "good" : "bad" );
92  }
93  else
94  {
95  // Determine effective permission sets, see if permissions are
96  // being changed
97  if ( ( cur_oob.has_allow_unauthenticated_read() != new_oob.has_allow_unauthenticated_read() ) ||
98  ( cur_oob.has_allow_unauthenticated_read() && new_oob.has_allow_unauthenticated_read() &&
99  ( cur_oob.allow_unauthenticated_read() != new_oob.allow_unauthenticated_read() ) ) )
100  {
101  CDS_DEBUG( 8, "Change to allow_unauthenticated_read" );
102  perms_change = true;
103  }
104  else if ( ( cur_oob.has_log_changes() != new_oob.has_log_changes() ) ||
105  ( cur_oob.has_log_changes() && new_oob.has_log_changes() &&
106  ( cur_oob.log_changes() != new_oob.log_changes() ) ) )
107  {
108  CDS_DEBUG( 8, "Change to log_changes flag" );
109  perms_change = true;
110  }
111  else
112  {
113  old_rd_perms = oob_effective_perms( cur_oob.read_permission_required(), px::permissions::config_read );
114  new_rd_perms = oob_effective_perms( new_oob.read_permission_required(), px::permissions::config_read );
115  new_wr_perms = oob_effective_perms( new_oob.write_permission_required(), px::permissions::config_write );
116  new_dp_perms = oob_effective_perms( new_oob.delete_or_permission_permission_required(), px::permissions::config_write );
117  if ( ( old_rd_perms != new_rd_perms ) ||
118  ( old_wr_perms != new_wr_perms ) ||
119  ( old_dp_perms != new_dp_perms ) )
120  {
121  CDS_DEBUG( 8, "Change to effective permissions" );
122  perms_change = true;
123  }
124  }
125  if ( perms_change )
126  {
127  if ( !px::permission::perms_overlap( req_perms, old_dp_perms ) )
128  {
129  CDS_DEBUG( 8, "Will deny change to permissions" );
130  perms_good = false;
131  }
132  }
133  // Clone the permissions so data-only comparison below works
134  cur_oob.mutable_read_permission_required()->Clear();
135  cur_oob.mutable_write_permission_required()->Clear();
136  cur_oob.mutable_delete_or_permission_permission_required()->Clear();
137  cur_oob.mutable_read_permission_required()->CopyFrom( new_oob.read_permission_required() );
138  cur_oob.mutable_write_permission_required()->CopyFrom( new_oob.write_permission_required() );
139  cur_oob.mutable_delete_or_permission_permission_required()->CopyFrom( new_oob.delete_or_permission_permission_required() );
140  cur_oob.clear_allow_unauthenticated_read();
141  if ( new_oob.has_allow_unauthenticated_read() )
142  {
143  cur_oob.set_allow_unauthenticated_read( new_oob.allow_unauthenticated_read() );
144  }
145 
146  if ( ( cur_oob.ByteSize() > 0 ) && ( cur_oob.ByteSize() == new_oob.ByteSize() ) )
147  {
148  std::shared_ptr<unsigned char> cur_oob_data( new unsigned char[cur_oob.GetCachedSize()] );
149  std::shared_ptr<unsigned char> new_oob_data( new unsigned char[new_oob.GetCachedSize()] );
150  cur_oob.SerializeWithCachedSizesToArray( cur_oob_data.get() );
151  new_oob.SerializeWithCachedSizesToArray( new_oob_data.get() );

153  {
154  data_change = false;
155  }
156  }
157  // Deny request even if data is same, to avoid info leak
158  if ( ( !perms_change ) && ( !px::permission::perms_overlap( new_wr_perms, req_perms ) ) )
159  {
160  CDS_DEBUG( 8, "Denying OOB set of non-permission fields of OOB" );

281  }

288  respond_success();
