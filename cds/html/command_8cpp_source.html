360 px::cds::RequestStatus CCommand::do_oob_data_validation( CDataStore &cds, const bool &bypass_validation,
361  const px::string &tag, px::proto::toolkitStruct &oob, const std::string &cur_field )
362 {
363  std::string tag_str = px::to_basicstr( tag );
364 
366  if ( !bypass_validation )
367  {
369  std::weak_ptr<px::sub::CValidationHook> wptr_validation_object;
370 
371  if ( cds.subs.get_validationhook_object( tag_str, wptr_validation_object ) )
372  {
374  if ( !wptr_validation_object.expired() )
375  {
376  px::sub::ValidationStatus validation_status;
377  CDS_DEBUG( 8, "Data validation started for OOB \"%s\"", tag_str.c_str() );
379  auto sptr_validation_object = wptr_validation_object.lock();
380  try
381  {
382  validation_status = sptr_validation_object->Valid( oob, cur_field );
383  }
384  catch ( const px::generic_err &gx )
385  {
386  CDS_WARN( 1, "Data validation failed for OOB tag \"%s\" with error:%s ", tag_str.c_str(), gx.what() );
387  return ( px::cds::RequestStatus::BAD_PARAMETER );
388  }
389  if ( validation_status != px::sub::ValidationStatus::VALIDATION_SUCCESS )
390  {
391  CDS_WARN( 1, "Data validation failed for OOB tag \"%s\"", tag_str.c_str() );
392  return ( px::cds::RequestStatus::BAD_PARAMETER );
393  }
394  }
395  else
396  {
398  CDS_WARN( 1, "Data validation object expired for OOB tag \"%s\"", tag_str.c_str() );
399  return ( px::cds::RequestStatus::BAD_PARAMETER );
400  }
401  }
402  else
403  {
404  CDS_DEBUG( 2, "Data validation object not found for OOB tag \"%s\"", tag_str.c_str() );
405  }
406  }
407  else
408  {
409  CDS_DEBUG( 2, "Validation bypass flag is set for OOB tag \"%s\"", tag_str.c_str() );
410  }
411  return ( px::cds::RequestStatus::SUCCEEDED );
412 }
