19  while ( devices.begin() != devices.end() )
20  {

33  // Tell persistence thread to remove this device from the DB
34  unpersist_cmd->devices.insert( i );
35  // Remove this device from the to-do list
36  devices.erase( i );
37  // Delete each sub-device
38  if ( sub_devices.size() != 0 )
39  {
40  COpDeleteDevices *op = new COpDeleteDevices();
41  for ( px::id_t sd : sub_devices )
42  {
43  op->devices.insert( sd );
44  }
45  ops.insert( op ); // makes unique_ptr which will delete() for us
46  }
47  // Delete its channels
48  if ( channels.size() != 0 )
49  {

51  for ( px::id_t ch : channels )
52  {
53  op->channels.insert( ch );
54  }
55  ops.insert( op );
56  }
57 
58  // Notify any subscribers (of any kind) to this device that it's deleted
59  // and unsubscribe
60  for ( auto stp : cds.subs.list_device_subscription_types() )
61  {
62  if ( stp == px::sub::subscription_type::device_create_recursive )
63  {
64  continue;
65  }
66  std::vector<uint64_t> _subs;
67  px::sub::CItemId event( i, stp );

69  for ( auto &listener: _subs )

72  ops.insert( op );
73  cds.subs.remove_subscription( listener, event );
74  }
75  }
76  } // End iteration over each device to delete
77  if ( likely( cds.persistence_thread ) )
78  {

80  }
