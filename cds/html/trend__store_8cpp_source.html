321  update->actual ? boost::lexical_cast<std::string>( ( float ) ( *update->actual ) ).c_str() : "",

371  if ( stage_db )

373  stage_db->write( batch, db_sync );
374  batch.clear();

436 void CTrendStore::load_stage_dbs()
437 {
438  assert( _db_path.size() != 0 );
439  path folder( std::string( _db_path.c_str() ) );
440  if ( !exists( folder ) || !is_directory( folder ) )
441  {
442  std::string err = "Invalid db folder: " + px::to_basicstr( _db_path );
443  CDS_FERR( 0, "%s", err.c_str() );
444  throw px::generic_err( err.c_str() );
445  }
446  // Make a list of all the existing trend stage DBs
447  std::set<uint32_t> stage_indices;
448  std::set<uint32_t> obsolete_stage_indices;
449  directory_iterator it_end;
450  for ( directory_iterator it( folder ); it != it_end; ++it )
451  {
452  if ( is_directory( *it ) )
453  {
454  auto name = it->path().leaf().string();
455  if ( name.compare( 0, _stage_db_prefix.size(), _stage_db_prefix.c_str() ) == 0 )
456  {
457  auto postfix = name.substr( _stage_db_prefix.size() );
458  if ( postfix.end() == std::find_if( postfix.begin(), postfix.end(),
459  [] ( unsigned char c ) -> bool {
460  return ( !isdigit( c ) );
461  } ) )
462  {
463  // make sure the direcotry name is stage_xxxx,
464  // xxxx is a number and there is nothing else.
465  char *endptr;
466  uint32_t stage_index = ( uint32_t )std::strtol( postfix.c_str(), &endptr, 10 );
467  if ( stage_index > 0 )
468  {
469  stage_indices.insert( stage_index );
470  }
471  }
472  }
473  else if ( name.compare( 0, _obsolete_stage_db_prefix.size(), _obsolete_stage_db_prefix.c_str() ) == 0 )
474  {
475  auto postfix = name.substr( _obsolete_stage_db_prefix.size() );
476  if ( postfix.end() == std::find_if( postfix.begin(), postfix.end(),
477  [] ( unsigned char c ) -> bool {
478  return ( !isdigit( c ) );
479  } ) )
480  {
481  char *endptr;
482  uint32_t stage_index = ( uint32_t )std::strtol( postfix.c_str(), &endptr, 10 );
483  if ( stage_index > 0 )
484  {
485  obsolete_stage_indices.insert( stage_index );
486  }
487  }
488  }
489  }
490  }
491 
492  // Take the most recent stage DBs.
493  uint32_t num_stage_dbs = 0;
494  for ( auto it = stage_indices.rbegin(); it != stage_indices.rend(); it++ )
495  {
496  if ( num_stage_dbs > _max_stage_dbs_atstart )
497  {
498  try
499  {
500  boost::filesystem::rename( path( get_stage_db_path( *it ).c_str() ),
501  path( get_obsolete_stage_db_path( *it ).c_str() ) );
502  }
503  catch ( ... )
504  {
505  CDS_ERR( 0, "Error renaming trend stage DB %u into obsolete area", *it );
506  }
507  obsolete_stage_indices.insert( *it );
508  }
509  else
510  {
511  // make sure the index is a positive number >= 1.
512  px::string stage_db_path = get_stage_db_path( *it );
513  std::shared_ptr<px::db::CKeyValueStore> stage_db;
514  try
515  {
516  stage_db.reset( new px::db::CKeyValueStore( stage_db_path, false, true ) );
517  if ( !stage_db || !stage_db.get() || px::db::IsTrendStageDbTombstoned( *stage_db ) )
518  {
519  // check the tombstone
520  stage_db.reset();
521  }
522  }
523  catch ( const px::db_corrupt &err )
524  {
525  stage_db.reset();
526  CDS_ERR( 0, "The trend stage db %u is corrupted. %s", *it, err.what() );
527  }
528  catch ( px::generic_err &err )
529  {
530  stage_db.reset();
531  CDS_ERR( 0, "The trend stage db %u cannot be opened. %s", *it, err.what() );
532  }
533  if ( unlikely( !stage_db ) )
534  {
535  // delete the faulted stage db.
536  try
537  {
538  boost::filesystem::remove_all( path( stage_db_path.c_str() ) );
539  }
540  catch ( const boost::filesystem::filesystem_error &fs_err )
541  {
542  CDS_ERR( 0, "Failed to delete the bad trend stage db folder %s. %s", stage_db_path.c_str(), fs_err.what() );
543  }
544  catch ( ... )
545  {
546  CDS_ERR( 0, "Failed to delete the bad trend stage db folder %s.", stage_db_path.c_str() );
547  }
548  }
549  // add the valid stage_db to the trend store.
550  if ( stage_db )
551  {
552  std::lock_guard<std::mutex> lock{_mutex};
553  _stage_dbs.insert( std::make_pair( *it, stage_db ) );
554  num_stage_dbs++;
555  }
556  }
557  }
558 
559  // Start the background task to deal with the obsolete DBs.
560  bool need_obsolete_move = false;
561  if ( obsolete_stage_indices.size() > 0 )
562  {
563  std::lock_guard<std::mutex> olock( _mutex );
564  for ( uint32_t id : obsolete_stage_indices )
565  {
566  _obsolete_move_indices.push_front( id );
567  }
568  need_obsolete_move = !_obsolete_move_indices.empty();
569  }
570  if ( need_obsolete_move && likely( !( _cds->exit_bg_tasks ) ) )
571  {
572  CDS_INFO( 0, "Too many trend stage DBs. Moving %zu of them to bucket DB in background.", _obsolete_move_indices.size() );
573  poke_obsolete_move_task();
574  }

652 std::tuple<bool, std::vector<uint32_t>, double> CTrendStore::move_stage_to_bucket( uint32_t cutoff )
653 {
654  // See log.hpp
655  // the live_stage_index and live_stage_index-1 will be preserved.
656  // All earlier stages will be moved to the bucket.
657  try
658  {
659  _thr_local_syslog.reset( new toolkit::CSysLog );
660  }
661  catch ( const std::exception & )
662  {
663  // ignore
664  }
665  SYSLOG_VERBOSITY = _cds_verb_array[CDS_LOG_TREND];
666 
667  CDS_DEBUG( 3, "async move stage to bucket, prior to day %u", cutoff );
668 
669  // Make sure another of these is not in progress
670  bool result_successful = true;
671  double result_seconds = 0;
672  std::vector<uint32_t> result_indexes;
673 
674  std::stringstream ss;
675  auto t0 = std::chrono::high_resolution_clock::now();
676  try
677  {
678  // Step 1: get list of stage DBs that are to be moved (<=live_stage_index-1)
679  std::vector<uint32_t> move_list;
680  { // lock scope
681  std::vector<uint32_t> del_list; // a list of invalid db indexes.
682  std::lock_guard<std::mutex> lock{_mutex};
683  for ( const auto &pair : _stage_dbs )
684  {
685  if ( !pair.second || !pair.second.get() )
686  {
687  // clean up invalid map
688  del_list.emplace_back( pair.first );
689  }
690  if ( pair.first >= cutoff )
691  {
692  break;
693  }
694  move_list.emplace_back( pair.first );
695  }
696  // delete the invalid ones
697  if ( del_list.size() > 0 )
698  {
699  for ( const auto stage_index : del_list )
700  {
701  auto it = _stage_dbs.find( stage_index );
702  if ( it != _stage_dbs.end() )
703  {
704  _stage_dbs.erase( it );
705  }
706  }
707  del_list.clear();
708  }
709  }
710 
711  // Step 2: Move and delete the stage dbs one by one.
712  // OK for redundant data to momentarily exist between the end of the
713  // above call and the below erase() call; if a query hits both, the
714  // way it merges the data should not cause duplicate results.
715  // simple_wait_timer(50); // for testing

717  {

759  }
760  catch ( px::generic_err &err )

763  ( err.what() ? err.what() : "" ) << " ";
764  }
765  }
766 
767  stage_db.reset();
768  ss << " moved,";
769  // throw px::generic_err("Seed failure on purpose for testing.");
770  // only delete the stage db if it was moved successfully.
771  leveldb::Status status;
772  { // lock scope
773  std::lock_guard<std::mutex> lock{_mutex};

775  if ( it != _stage_dbs.end() )
776  {
777  _stage_dbs.erase( it );
778  }
779  }
780  status = px::db::DestroyDb( get_stage_db_path( stage_index ) );
781  result_indexes.emplace_back( stage_index );
782  ss << " deleted, ";
783  if ( !status.ok() )
784  {
785  ss << status.ToString() << ", ";
786  }
787  // if the destroy failed, the db is removed from the CDS, and will be re-deleted when CDS restarts.
788  }
789  catch ( const px::db_corrupt &err )

791  result_successful = false;

795  cds_main_loop->update_system_channel( SystemChannel::DB_CORRUPT, px::CValue( true ) );
796  }
797  }
798  catch ( px::generic_err &err )
799  {

802  }
803  }

810  }
811  catch ( const px::db_corrupt &err )

966 void CTrendStore::query_db( std::shared_ptr<px::db::CKeyValueStore> db, bool is_bucket,
967  px::CChannelTrends &trends,
968  const px::CWallTime &start_time,
969  const px::CWallTime &end_time,
970  const unsigned int request_period,
971  const bool include_avg, const bool include_min,
972  const bool include_max, const bool include_actual,
973  const unsigned int row_count_limit,
974  bool &more_data ) const
975 {
976  more_data = false;
977  px::db::CIterator i = db->iterator( false );
978  i.first();
979  uint32_t start_min = 0, end_min = 0, start_bucketday_min = 0, end_bucketday_min = 0;
980 
981  bool on_interval = false;
982  px::db::get_normalized_interval_timestamp( start_time, 1, start_min, on_interval );
983  px::db::get_normalized_interval_timestamp( start_time, 1440, start_bucketday_min, on_interval );
984 
985  if ( likely( end_time.valid() && ( end_time.tv_sec != 0 ) ) )
986  {
987  px::db::get_normalized_interval_timestamp( end_time, 1, end_min, on_interval );
988  px::db::get_normalized_interval_timestamp( end_time, 1440, end_bucketday_min, on_interval );
989  }
990  else
991  {
992  end_min = 0xFFFFFFFFUL;
993  end_bucketday_min = 0xFFFFFFFFUL;
994  }
995 
996  // Go to first item in result set
997  px::db::SeekTrendIterator( i, trends.channel_id, start_min );
998  if ( !i.valid() )
999  {
1000  // No results
1001  return;
1002  }
1003  // store the trends read from the db as intermediate data, in case aggregation is needed.
1004  std::vector<px::CChannelTrend> in_progress_trends;
1005  px::CTrendBucketUpdate agg_bucket;
1006  for (; i.valid(); ++i )
1007  {
1008  uint32_t this_cid = 0;
1009  uint32_t this_min = 0;
1010  if ( unlikely( _cds->exit_bg_tasks ) )
1011  {
1012  break;
1013  }
1014  if ( unlikely( !px::db::ReadTrendKey( i, this_cid, this_min ) ) )
1015  {
1016  // ignore invalid key, and move on.
1017  continue;
1018  }
1019  if ( this_cid != trends.channel_id )
1020  {
1021  // different channel, we are done with this channel.
1022  break;
1023  }
1024  int read_period = 0;
1025  unsigned int return_value_count = 0, row_count = 0;
1026  bool is_double = false;
1027  in_progress_trends.clear();
1028  if ( is_bucket )
1029  {
1030  // when querying daily data from the bucket, break if end_min is in that bucket (only return the day before this bucket)
1031  if ( this_min > end_bucketday_min )
1032  {
1033  // this bucket contains data beyond the end_time already
1034  break;
1035  }
1036  if ( request_period == 1440 )
1037  {
1038  if ( ( this_min == end_bucketday_min ) && ( end_min < end_bucketday_min ) )
1039  {
1040  break;
1041  }
1042  }
1043  px::db::ReadTrendsFromBucketRow( i, in_progress_trends, this_min,
1044  start_min, end_min, request_period,
1045  include_avg, include_min, include_max,
1046  include_actual, read_period, is_double,
1047  0, row_count, return_value_count, false, false );
1048  if ( read_period <= 0 ) // Invalid read period can cause division by zero exception
1049  {
1050  continue;
1051  }
1052  }
1053  else
1054  {
1055  if ( ( this_min < start_min ) || ( this_min > end_min ) )
1056  {
1057  // the time is beyond the query range, done with this stage db.
1058  break;
1059  }
1060  read_period = 5;
1061  px::db::ReadTrendsFromStageRow( i, in_progress_trends,
1062  include_avg, include_min, include_max,
1063  include_actual, is_double,
1064  row_count, return_value_count );
1065  }
1066  if ( in_progress_trends.size() > 0 )
1067  {
1068  if ( read_period >= ( int )request_period )
1069  {
1070  // No aggregation needed
1071  trends.trends.insert( trends.trends.end(),
1072  in_progress_trends.begin(), in_progress_trends.end() );
1073  in_progress_trends.clear();
1074  }
1075  else
1076  {
1077  if ( unlikely( agg_bucket.get_expected_count() == 0 ) )
1078  {
1079  // first time through this channel - init
1080  bool on_interval = false;
1081  uint32_t aligned_time;
1082  px::db::get_normalized_interval_timestamp( in_progress_trends[0].time, request_period,
1083  aligned_time, on_interval );
1084  agg_bucket.reset_data();
1085  agg_bucket.initialize_bucket( request_period, read_period, aligned_time );
1086  }
1087  for ( auto t : in_progress_trends )
1088  {
1089  if ( ( t.time > agg_bucket.end_time ) || ( t.time < agg_bucket.start_time ) )
1090  {
1091  // Finished the prior aggregation, if any
1092  if ( agg_bucket.get_traits_count() > 0 )
1093  {
1094  double avg, min, max, actual;
1095  uint32_t min_offset, max_offset;
1096  int cnt_avg, cnt_min, cnt_max, cnt_actual, cnt_traits;
1097  agg_bucket.get_aggregated_values( avg, min, min_offset,
1098  max, max_offset, actual );
1099  agg_bucket.get_actual_values_count( cnt_avg, cnt_min,
1100  cnt_max, cnt_actual,
1101  cnt_traits );
1102  trends.trends.emplace_back( agg_bucket.end_time, ( cnt_avg != 0 ), avg, ( cnt_min != 0 ), min, min_offset, ( cnt_max != 0 ), max, max_offset, ( cnt_actual != 0 ), actual, ( uint16_t )request_period );
1103  }
1104  agg_bucket.reset_data();
1105  bool on_interval = false;
1106  uint32_t aligned_time;
1107  px::db::get_normalized_interval_timestamp( t.time, request_period,
1108  aligned_time, on_interval );
1109  agg_bucket.initialize_bucket( request_period, read_period, aligned_time );
1110  }
1111  if ( t.min )
1112  {
1113  agg_bucket.add_update( t.time, *( t.min ), px::trend_trait::min, t.min_time_offset );
1114  }
1115  if ( t.max )
1116  {
1117  agg_bucket.add_update( t.time, *( t.max ), px::trend_trait::max, t.max_time_offset );
1118  }
1119  if ( t.avg )
1120  {
1121  agg_bucket.add_update( t.time, *( t.avg ), px::trend_trait::avg, 0 );
1122  }
1123  if ( t.actual )
1124  {
1125  agg_bucket.add_update( t.time, *( t.actual ), px::trend_trait::actual, 0 );
1126  }
1127  } // End loop over read trend values
1128  } // End if (aggregation needed)
1129  // only promote the data type to double, if we do not know the type yet.
1130  if ( trends.data_type == px::datatype::invalid )
1131  {
1132  trends.data_type = is_double ? px::datatype::float64 : px::datatype::float32;
1133  }
1134  }
1135  if ( unlikely( ( row_count_limit != 0 ) && ( trends.trends.size() >= row_count_limit ) ) )
1136  {
1137  // too many data
1138  break;
1139  }
1140  } // End DB iteration
1141  if ( ( agg_bucket.get_expected_count() != 0 ) &&
1142  ( agg_bucket.get_traits_count() > 0 ) )
1143  {
1144  // Have a last aggregated trend value to push on
1145  double avg, min, max, actual;
1146  uint32_t min_offset, max_offset;
1147  int cnt_avg, cnt_min, cnt_max, cnt_actual, cnt_traits;
1148  agg_bucket.get_aggregated_values( avg, min, min_offset,
1149  max, max_offset, actual );
1150  agg_bucket.get_actual_values_count( cnt_avg, cnt_min,
1151  cnt_max, cnt_actual,
1152  cnt_traits );
1153  trends.trends.emplace_back( agg_bucket.end_time, ( cnt_avg != 0 ), avg, ( cnt_min != 0 ), min, min_offset, ( cnt_max != 0 ), max, max_offset, ( cnt_actual != 0 ), actual, ( uint16_t )request_period );
1154  }
1155  if ( unlikely( ( row_count_limit != 0 ) && ( trends.trends.size() > row_count_limit ) ) )
1156  {
1157  // too many data, truncate the newer data.
1158  more_data = true;
1159  trends.trends.resize( row_count_limit );
1160  }
1161 }

1163 unsigned int CTrendStore::merge_query_results( px::CChannelTrends &dst,
1164  const px::CChannelTrends &src ) const
1165 {
1166  unsigned int removed = 0;
1167 
1168  if ( ( src.channel_id != dst.channel_id ) || src.trends.empty() )
1169  {
1170  return ( removed ); // nothing to merge in this channel
1171  }
1172  if ( dst.trends.empty() )
1173  {
1174  // Special cases: dst empty for this channel (shouldn't happen)
1175  dst.trends = src.trends;
1176  return ( removed );
1177  }
1178  const std::vector<px::CChannelTrend> *s = &( src.trends );
1179  std::vector<px::CChannelTrend> *d = &( dst.trends );
1180  // Determine if forward or backwards
1181  bool forwards = dst.is_time_ascending() || src.is_time_ascending();
1182  // Special case: no overlap, just append
1183  if ( ( ( d->back().time < s->at( 0 ).time ) && forwards ) ||
1184  ( ( d->back().time > s->at( 0 ).time ) && ( !forwards ) ) )
1185  {
1186  d->insert( d->end(), s->begin(), s->end() );
1187  return ( removed );
1188  }
1189  // Find spot where dst and src start overlapping
1190  std::vector<px::CChannelTrend> residue;
1191  size_t spot = d->size() - 1;
1192  while ( spot > 0 )
1193  {
1194  if ( ( ( d->at( spot ).time < s->at( 0 ).time ) && forwards ) ||
1195  ( ( d->at( spot ).time > s->at( 0 ).time ) && ( !forwards ) ) )
1196  {
1197  break;
1198  }
1199  spot--;
1200  }
1201  // Remove elements that overlap
1202  residue.insert( residue.end(), d->begin() + spot, d->end() );
1203  d->resize( spot );
1204  d->reserve( d->size() + residue.size() + s->size() );
1205  // merge
1206  size_t rp = 0, sp = 0;
1207  while ( ( rp < residue.size() ) && ( sp < s->size() ) )
1208  {
1209  if ( residue.at( rp ).time < s->at( sp ).time )
1210  {
1211  if ( forwards )
1212  {
1213  d->insert( d->end(), residue.at( rp ) );
1214  rp++;
1215  }
1216  else
1217  {
1218  d->insert( d->end(), s->at( sp ) );
1219  sp++;
1220  }
1221  }
1222  else if ( residue.at( rp ).time > s->at( sp ).time )
1223  {
1224  if ( forwards )
1225  {

2423  uint32_t min_free_space_mb = 0, step_days = 0;

2425

2427  px::CWallTime now = px::CWallTime::now();

2429  bool on_interval = false;
2430  px::db::get_normalized_interval_timestamp( now, daily, tomorrow, on_interval );
2431

2433  if ( ( time_threshold_15min > 0 ) && ( time_threshold_15min < earliest_time ) )
2434  {
2435  earliest_time = time_threshold_15min;
2436  }
2437  if ( ( time_threshold_5min > 0 ) && ( time_threshold_5min < earliest_time ) )
2438  {
2439  earliest_time = time_threshold_5min;
2440  }
2441  if ( earliest_time > 0 )
2442  {
2443  earliest_time--; // px::db::DeleteTrends uses <=, not <
2444  }
2445  CDS_NOTICE( 1, "Trend bucket cleanup starting" );
2446  size_t num_trended_channels;

2448  std::unordered_set<uint32_t> trended_channels;
2449  get_trended_channels( trended_channels );
2450  num_trended_channels = trended_channels.size();
2451 
2452  // First, date-based levelup/rolloff.
2453  if ( ( time_threshold_5min != 0 ) || ( time_threshold_15min != 0 ) ||
2454  ( time_threshold_60min != 0 ) )
2455  {
2456  CDS_DEBUG( 2, "Trend bucket cleanup: date-based (5min:%u, 15min:%u, 60min:%u)",
2457  time_threshold_5min, time_threshold_15min, time_threshold_60min );
2458  ( void )levelup_bucket( time_threshold_5min, time_threshold_15min,
2459  time_threshold_60min, num_trended_channels );
2460  }
2461 
2462  // Next, delete the rows with ealier timestamp, or from removed/untrended
2463  // channels. Most of the time this has no effect.
2464  CDS_DEBUG( 2, "Trend bucket cleanup: deleting trends older than %u or not from %zu currently-trended channels",
2465  earliest_time, trended_channels.size() );
2466  if ( !px::db::DeleteTrends( *_bucket_db, earliest_time, trended_channels,
2467  _cds->exit_bg_tasks ) )
2468  {
2469  if ( likely( cds_main_loop ) )
2470  {
2471  cds_main_loop->update_system_channel( SystemChannel::DB_CORRUPT, px::CValue( true ) );
2472  }
2473  CDS_ERR( 0, "failed to cleanup the database during the shrink" );
2474  return;
2475  }
2476  } // de-scope trended_channels list
2477 
2478  // Next, space-based levelup/rolloff, if needed.
2479  uint64_t fs = get_trend_bucket_disk_free_space();
2480  if ( ( fs >> 20 ) >= min_free_space_mb )
2481  {
2482  CDS_NOTICE( 1, "%ju MB free space, space-based levelup/rolloff not needed, bucket cleanup complete",
2483  fs >> 20 );
2484  return;
2485  }
2486  uint64_t sz = get_trend_bucket_disk_usage();
2487  if ( unlikely( ( ( sz >> 20 ) + ( fs >> 20 ) ) < ( min_free_space_mb * 1.01 ) ) )
2488  {
2489  CDS_WARN( 0, "Removing entire trend history (%ju MB) to free up disk space",
2490  sz >> 20 );
2491  px::db::DeleteTrends( *_bucket_db, 0, UINT32_MAX, _cds->exit_bg_tasks );
2492  _bucket_db->compact();
2493  return;
2494  }
2495  bool done = false;
2496  unsigned int iter_count = 0;// infinite loop guard just in case
2497  while ( ( !done ) && ( !_cds->exit_bg_tasks ) && ( iter_count < 20 ) )
2498  {
2499  // Determine the oldest age of 5, 15 and 60 minute data.
2500  // Re-done every iteration, since previous rolloff might have
2501  // made some other channel the 'oldest' since last time.
2502  uint32_t oldest_60min = tomorrow, oldest_15min = tomorrow,
2503  oldest_5min = tomorrow;
2504  {
2505  px::db::CIterator i = _bucket_db->iterator( false );
2506  i.first();
2507  while ( i.valid() && ( !_cds->exit_bg_tasks ) )
2508  {
2509  uint32_t chanid, time;
2510  if ( ( !ReadTrendKey( i, chanid, time ) ) || ( i.value_size() == 0 ) ||
2511  ( ( ( i.value_data()[0] ) >> 4 ) != 0 ) )
2512  {
2513  ++i;
2514  continue;
2515  }
2516  switch ( i.value_data()[0] & 0xF )
2517  {
2518  case 0:
2519  if ( time < oldest_5min )
2520  {
2521  oldest_5min = time;
2522  }
2523  SeekTrendIterator( i, chanid + 1, 0 );
2524  continue;
2525  break;
2526  case 1:
2527  if ( time < oldest_15min )
2528  {
2529  oldest_15min = time;
2530  }
2531  break;
2532  case 2:
2533  if ( time < oldest_60min )
2534  {
2535  oldest_60min = time;
2536  }
2537  break;
2538  default:
2539  ++i;
2540  continue;
2541  }
2542  ++i;
2543  } // end finding oldest times
2544  } // iterator de-scope
2545  CDS_DEBUG( 5, "Space-based rolloff: Current oldest 5min: %u, 15min: %u, 60min: %u",
2546  oldest_5min, oldest_15min, oldest_60min );
2547  uint32_t thresh_5min, thresh_15min, thresh_60min;
2548  if ( ( oldest_60min == tomorrow ) && ( oldest_15min == tomorrow ) &&
2549  ( oldest_5min != tomorrow ) )
2550  {
2551  // Special case for pure space-based rolloff; if no 60min or 15min
2552  // data in oldest channel, this is our first levelup. In that
2553  // case, do 1/4 of data 5min, 1/4 15min, rest 60min.
2554  thresh_60min = 0;
2555  thresh_15min = ( tomorrow - oldest_5min ) / 2 + oldest_5min;
2556  thresh_5min = ( tomorrow - oldest_5min ) * 3 / 4 + oldest_5min;
2557  CDS_NOTICE( 1, "First levelup is space-based. 5min threshold: %u, 15min: %u", thresh_5min, thresh_15min );
2558  }
2559  else
2560  {
2561  thresh_5min = oldest_5min + step_days;
2562  thresh_15min = oldest_15min + step_days;
2563  thresh_60min = oldest_60min + step_days;
2564  if ( thresh_5min > tomorrow )
2565  {
2566  thresh_5min = tomorrow;
2567  }
2568  if ( thresh_15min > tomorrow )
2569  {
2570  thresh_15min = tomorrow;
2571  }
2572  if ( thresh_60min > tomorrow )
2573  {
2574  thresh_60min = tomorrow;
2575  }
2576  }
2577 
2578  CDS_DEBUG( 5, "Space-based levelup/rolloff: 5min to %u, 15min to %u, 60min to %u",
2579  thresh_5min, thresh_15min, thresh_60min );
2580  iter_count++;
2581  ( void )levelup_bucket( thresh_5min, thresh_15min, thresh_60min, num_trended_channels );
2582  fs = get_trend_bucket_disk_free_space();
2583  if ( ( fs >> 20 ) >= ( min_free_space_mb * 1.01 ) )
2584  {
2585  CDS_NOTICE( 1, "%ju MB free space, space-based levelup/rolloff complete",
2586  fs >> 20 );
2587  done = true;
2588  }
2589  else
2590  {
2591  CDS_DEBUG( 5, "%ju MB free space, space-based levelup/rolloff will retry",
2592  fs >> 20 );
2593  }
2594  } // end loop for repeating levelup/rolloff
2595 }
