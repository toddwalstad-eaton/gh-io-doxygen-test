87  for ( const px::CParameterInfo &out_param : parameters )

89  // First attempt: see if it's a defined output parameter, and check against its metadata:
90  bool found = false;
91  for ( px::CParameterInfo &cmd_param : tmp->parameters )
92  {
93  if ( cmd_param.name == out_param.name )
94  {
95  found = true;
96  if ( !cmd_param.is_in )
97  {
98  if ( out_param.value.is_valid() )
99  {

103  CDS_INFO( 5, "Set command %s output param %s to %s.", tmp->name.c_str(),
104  cmd_param.name.c_str(), cmd_param.value.as_string().c_str() );
105  }
106  }
107  else
108  {
109  CDS_WARN( 1, "Cannot set command %s input param %s (not output param).", tmp->name.c_str(),
110  cmd_param.name.c_str() );
111  }
112  }
113  }
114  if ( !found )
115  {
116  // Second attempt: Warn, but proceed and allow this anonymous output parameter as long as it is well defined
117  CDS_WARN( 2, "Command %s has no param named %s.", tmp->name.c_str(),
118  out_param.name.c_str() );
119  if ( !out_param.is_in && out_param.value.is_valid() )
120  {
121  // We won't double-check the type or other info;
122  // will just use pass along whatever it happens to be and hope for the best
123  tmp->parameters.push_back( out_param );
124  CDS_INFO( 5, "Set command %s anonymous output param %s to %s.", tmp->name.c_str(),
125  out_param.name.c_str(), out_param.value.as_string().c_str() );
126  }
127  }
128  }
129  }
130  // else: ignore the set.
131  else
