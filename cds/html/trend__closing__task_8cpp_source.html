21 void TrendIntervalClosingTask::execute()
22 {
23  // Initialize this thread's syslog pointer, if the first time
24  if ( !_thr_local_syslog.get() )
25  {
26  _thr_local_syslog.reset( new toolkit::CSysLog( "cds" ) );
27  }
28 
29  // To avoid issues with the map of devices changing during the somewhat extensive time we spend
30  // processing all devices, first build a sorted list (set) of the Device IDs
31  std::set<px::id_t> sorted_device_ids;
32  for ( const auto &map_pair : cds_datastore->devices )
33  {
34  sorted_device_ids.insert( map_pair.first );
35  }
36 
37  // Now give our task a lower-than default priority of 8, where 0 is higher
38  // Leave the real-time priority's priority level at 0.
39  px::CUtil::set_thread_prio( px::CRealtimePriority( "normal", "0", NICE_PRIORITY ) );
40 
41  // Do the work here
42  // First, get the normalized trend time in UTC minutes (since the 1970 epoch)
43  uint32_t ended_interval_mins;
44  const uint32_t aligned_minutes = TREND_INTERVAL_MINUTES;
45  bool on_interval = false;
46  const px::CWallTime tnow( px::CWallTime::now() );
47  px::db::get_normalized_interval_timestamp( tnow, aligned_minutes, ended_interval_mins, on_interval );
48  // Since this task is executing into the next trend interval, the value we just got for
49  // ended_interval_mins is later than we want; adjust it back to the interval just ended
50  ended_interval_mins -= TREND_INTERVAL_MINUTES;
51 
52  // Now process our sorted list of device IDs
53  for ( const px::id_t &dev_id : sorted_device_ids )
54  {
55  if ( cds_datastore->exit_bg_tasks )
56  {
57  break; // We are done here
58  }
59 
60  auto map_pos = cds_datastore->devices.find( dev_id );
61  if ( map_pos == cds_datastore->devices.end() )
62  {
63  continue; // no longer in the map
64  }
65  const px::CDevice &device = map_pos->second;
66 
67  // If the device is disabled or not communicating, skip it
68  px::CBitset<px::realtime_state> dev_states = device.get_state();
69  if ( ( dev_states.get( px::realtime_state::disabled ) == true ) ||
70  ( dev_states.get( px::realtime_state::not_communicating ) == true ) )
71  {
72  continue;
73  }
74 
75  // Call the function to do the work
76  // To avoid issues with the list of channels changing while we are
77  // processing all channels, intentionally copy the (sorted) list of channel IDs
78  // as the argument here instead of providing a reference
79  process_device_channels( device.channels, ended_interval_mins );
80  }
81 
82  // Restore the priority levels.
83  px::CUtil::set_thread_prio( px::CRealtimePriority( "normal", "0", "0" ) );
84 }

87 void TrendIntervalClosingTask::process_device_channels( std::set<px::id_t> channel_ids, uint32_t ended_interval_mins )
88 {
89  uint32_t next_interval_mins = ended_interval_mins + TREND_INTERVAL_MINUTES;
90  std::vector<std::shared_ptr<px::CTrendStageUpdate> > trends;
91 
92  trends.clear();
93 
94  for ( const px::id_t &channel_id : channel_ids )
95  {
96  if ( cds_datastore->exit_bg_tasks )
97  {
98  return; // We are done here
99  }
100 
101  auto ch_map_pos = cds_datastore->channels.find( channel_id );
102  if ( ch_map_pos == cds_datastore->channels.end() )
103  {
104  continue; // not found
105  }
106 
107  // Now we get to the heart of it; close the interval in the channel's accumulator
108  // Doing like CCommandUpdateChannelValue::execute() does
109  const px::CChannel &channel_data = ch_map_pos->second;
110  if ( channel_data.trend_accumulator )
111  {
112  // trend_accumulator would be empty if this channel weren't trended.
113  std::shared_ptr<px::CTrendStageUpdate> trend_update;
114  trend_update = channel_data.trend_accumulator->close_interval( ended_interval_mins, next_interval_mins );
115  if ( trend_update )
116  {
117  // Only contains an update if close_interval() just closed a still-open interval
118  // See if it has any traits that we actually want trended
119  bool has_avg = channel_data.channel_settings->flags.get( px::channel_flag::has_avg );
120  bool has_min = channel_data.channel_settings->flags.get( px::channel_flag::has_min );
121  bool has_max = channel_data.channel_settings->flags.get( px::channel_flag::has_max );
122  bool has_actual = channel_data.channel_settings->flags.get( px::channel_flag::has_actual );
123  if ( likely( has_avg || has_min || has_max || has_actual ) )
124  {
125  // push update to the stage
126  trend_update->channel_id = ( uint32_t )channel_id;
127  // Don't log traits we aren't configured to trend
128  if ( !has_max )
129  {
130  trend_update->max.reset();
131  }
132  if ( !has_min )
133  {
134  trend_update->min.reset();
135  }
136  if ( !has_avg )
137  {
138  trend_update->avg.reset();
139  }
140  if ( !has_actual )
141  {
142  trend_update->actual.reset();
143  }
144  trends.emplace_back( trend_update );
145  }
146  }
147  }
148  }
149 
150  // push updates to the stage
151  if ( !trends.empty() )
152  {
153  // Dispatch the updates to the trend stage thread
154  if ( likely( cds_datastore->trend_stage_thread ) )
155  {
156  // CDS_NOTICE( 0, "Closing trends for %zu channels at %u minutes", trends.size(), ended_interval_mins );
157  std::unique_ptr<CThreadUpdateTrendStage> persist_cmd;
158  persist_cmd.reset( new CThreadUpdateTrendStage );
159  persist_cmd->req.swap( trends );
160  cds_datastore->trend_stage_thread->send( std::move( persist_cmd ) );
161  }
162  else
163  {
164  CDS_ERR( 0, "Tried to close trends but the trend stage thread not running" );
165  }
166  }
167  else
168  {
169  // CDS_NOTICE( 0, "No trends to Close at %u minutes", ended_interval_mins );
170  }
171 }
