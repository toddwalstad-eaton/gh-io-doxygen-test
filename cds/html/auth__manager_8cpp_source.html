610  const px::string &password ) const
611 {
612  std::vector<uint8_t> rv, used_salt;
613 
614  rv.resize( 29 + 31 + 1 ); // prefix + salt + hash + null terminator
615  used_salt.resize( 30 );

617  {
618  for ( size_t i = 0; i < 29; i++ )
619  {

845 void CAuthManager::update_pw_expiration_time( px::CUser &user )
846 {
847  if ( user.pw_expiration_time_fixed )
848  {
849  return;
850  }
851  if ( user.last_pw_change != px::CWallTime() )
852  {
853  uint8_t pw_exp_days = px::CUserPolicy::default_pw_exp_days();
854  if ( user.passwd_policy.password_expiration_days )
855  {
856  pw_exp_days = *user.passwd_policy.password_expiration_days;
857  }
858  else if ( _gbl_passwd_policy.password_expiration_days )
859  {
860  pw_exp_days = *_gbl_passwd_policy.password_expiration_days;
861  }

865  px::CWallTime now = px::CWallTime::now();
866  px::CWallTime new_expiration_time = px::CWallTime( ( user.last_pw_change.tv_sec +
867  pw_exp_days * _secs_per_day ),
868  user.last_pw_change.tv_nsec );

870  {
871  user.pw_expiration_time = new_expiration_time;
872  }

874  {

890  {
891  grace_days = 1; // failsafe
892  }

894  user.id.c_str(), grace_days );
895  user.pw_expiration_time = px::CWallTime( now.tv_sec + grace_days * _secs_per_day, 0 );
896  }
897  else
898  {
899  user.pw_expiration_time = new_expiration_time;
900  }
901  }
902  }
903  else
904  {
905  user.pw_expiration_time = px::CWallTime();
