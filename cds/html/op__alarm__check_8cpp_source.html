11 bool COpAlarmCheck::trigger_tripped( const px::CAlarmTrigger &at,
12  px::CValue cur_val, uint32_t cur_level, px::CValue prev_val ) const
13 {
14  if ( unlikely( cur_val.type() == px::datatype::invalid ) )
15  {
16  return ( false );
17  }
18  if ( ( at.trigger_type != px::alarm_trigger_type::change_of_value ) && unlikely( at.threshold.type() != cur_val.type() ) )
19  {
20  cur_val.to( at.threshold.type() );
21  }
22  // Hysteresis only applies if this trigger level or a higher one is
23  // currently triggered
24  bool have_hysteresis = false;
25  if ( ( at.trigger_type != px::alarm_trigger_type::on_value ) &&
26  ( at.trigger_type != px::alarm_trigger_type::change_of_value ) &&
27  ( at.level <= cur_level ) &&
28  ( at.hysteresis.type() != px::datatype::invalid ) )
29  {
30  have_hysteresis = true;
31  }
32 
33  if ( at.trigger_type == px::alarm_trigger_type::on_value )
34  {
35  return ( cur_val == at.threshold );
36  }
37 
38  if ( at.trigger_type == px::alarm_trigger_type::change_of_value )
39  {
40  if ( ( ( prev_val == px::CValue() ) && ( prev_val.type() != px::datatype::boolean ) || ( prev_val.type() == px::datatype::invalid ) ) || ( ( cur_val == px::CValue() ) && ( cur_val.type() != px::datatype::boolean ) || ( cur_val.type() == px::datatype::invalid ) ) )
41  {
42  CDS_FNOTICE( 4, "Invalid channel value observed while processing change_of_value alarm" );
43  return ( false );
44  }
45  return ( prev_val != cur_val );
46  }
47 
48  px::CValue bound_val( at.threshold );
49 
50  switch ( at.trigger_type.val() )
51  {
52  case px::alarm_trigger_type::upper_bound:
53  if ( have_hysteresis )
54  {
55  bound_val -= at.hysteresis;
56  }
57  return ( cur_val >= bound_val );
58 
59  case px::alarm_trigger_type::lower_bound:
60  if ( have_hysteresis )
61  {
62  bound_val += at.hysteresis;
63  }
64  return ( cur_val <= bound_val );
65 
66  case px::alarm_trigger_type::in_between:
67  {
68  px::CValue bound_val2( at.threshold2 );
69  if ( have_hysteresis )
70  {
71  bound_val += at.hysteresis;
72  bound_val2 -= at.hysteresis;
73  }
74  return ( ( cur_val <= bound_val ) && ( cur_val >= bound_val2 ) );
75  }
76  default:
77  return ( false );
78  }
79 }

81 static px::CAlarmUpdate make_update( px::alarm_trigger_type trigger_type,
82  const px::CAlarmTrigger *trigger,
83  const px::CChannel *chan,
84  const px::CDevice *dev,
85  const CChannelAlarmState &alarm_state,
86  CDataStore &cds )
87 {
88  px::CAlarmUpdate rv;
89 
90  if ( unlikely( chan == nullptr ) )
91  {
92  return ( rv );
93  }
94 
95  bool is_bound = ( ( trigger_type == px::alarm_trigger_type::upper_bound ) ||
96  ( trigger_type == px::alarm_trigger_type::lower_bound ) );
97 
98  // New alarm opening?
99  if ( ( trigger != nullptr ) &&
100  ( ( ( trigger_type == px::alarm_trigger_type::in_between ) && ( alarm_state.between_id == 0 ) ) ||
101  ( is_bound && ( alarm_state.bound_id == 0 ) ) ||
102  ( ( trigger_type == px::alarm_trigger_type::on_value ) && ( alarm_state.value_id == 0 ) ) ||
103  ( ( trigger_type == px::alarm_trigger_type::change_of_value ) && ( alarm_state.cov_id == 0 ) ) ) )
104  {
105  // Leave ID/sequence/GC unassigned. Set new fields.
106  rv.latching = trigger->latching;
107  rv.closed = false;
108  rv.ack_level = 0;
109  rv.trigger_condition_cleared = false;
110  rv.first_occur_time = chan->timestamp;
111  rv.trigger = *trigger;
112  }
113  else
114  {
115  // Start with the most-recent update to that alarm
116  bool alarm_found = false;
117  if ( is_bound )
118  {
119  alarm_found = cds.open_alarms_store.GetAlarmByID( alarm_state.bound_id, rv );
120  }
121  else if ( trigger_type == px::alarm_trigger_type::on_value )
122  {
123  alarm_found = cds.open_alarms_store.GetAlarmByID( alarm_state.value_id, rv );
124  }
125  else if ( trigger_type == px::alarm_trigger_type::change_of_value )
126  {
127  alarm_found = cds.open_alarms_store.GetAlarmByID( alarm_state.cov_id, rv );
128  }
129  else if ( trigger_type == px::alarm_trigger_type::in_between )
130  {
131  alarm_found = cds.open_alarms_store.GetAlarmByID( alarm_state.between_id, rv );
132  }
133  if ( unlikely( !alarm_found ) )
134  {
135  CDS_FERR( 0, "Alarm referenced in channel alarm state but not open. type %u, channel %" \
136  PRIu64 ", bound/val/between IDs %" PRIu64 "/%" PRIu64 "/%" PRIu64,
137  ( unsigned int )trigger_type, chan->id(), alarm_state.bound_id, alarm_state.value_id,
138  alarm_state.between_id );
139  return ( rv );
140  }
141  // Clear fields specific to the prior update
142  rv.user = "";
143  rv.notes = "";
144  rv.waveform_uri = "";
145  rv.global_count = 0;// System will assign one
146  // Update fields
147  rv.index++;
148  // Is trigger cleared?
149  if ( trigger == nullptr )
150  {
151  rv.trigger_condition_cleared = true;
152  CDS_DEBUG( 4, "Alarm %ju condition cleared", rv.id );
153  if ( ( !( rv.latching ) ) || ( rv.ack_level >= rv.max_reached_trigger_level ) )
154  {
155  // Close this alarm
156  rv.closed = true;
157  CDS_DEBUG( 4, "Closing alarm %ju", rv.id );
158  }
159  }
160  else
161  {
162  rv.trigger_condition_cleared = false;
163  }
164  }
165  // Update fields common to new or update
166  rv.channel_id = chan->id();
167  rv.device_id = chan->parent();
168  if ( trigger != nullptr )
169  {
170  rv.trigger = *trigger;
171  if ( trigger->level > rv.max_reached_trigger_level )
172  {
173  rv.max_reached_trigger_level = trigger->level;
174  }
175  }
176  rv.timestamp = chan->timestamp;
177  rv.value = chan->value;
178  if ( likely( dev != nullptr ) )
179  {
180  rv.this_device_state = dev->get_state();
181  }
182  rv.this_channel_state = chan->get_state();
183  return ( rv );

186 void COpAlarmCheck::execute( std::shared_ptr<px::cds::CRequestResponse>/*msg*/,
187  COpList &ops, CDataStore &cds )
188 {
189  COpAlarmUpdate *upd_op = new COpAlarmUpdate();
190 
191  if ( upd_op == NULL )
192  {
193  CDS_DEBUG( 9, "Memory allocation failed for COpAlarmUpdate" );
194  return;
195  }
196 
197  CDS_DEBUG( 9, "Checking alarms on channel %ju", channel_id );
198  if ( unlikely( cds.channels.count( channel_id ) == 0 ) )
199  {
200  delete upd_op;
201  return;
202  }
203  px::CChannel *cur_chan = &( cds.channels[channel_id] );
204  const px::CValue &cur_val = cur_chan->value;
205  CChannelAlarmState alarm_state = cds.open_alarms_store.GetAlarmState( channel_id );
206  // Find the highest-level tripped trigger for each type
207  const px::CAlarmTrigger *onval_trigger = nullptr;
208  const px::CAlarmTrigger *bound_trigger = nullptr;
209  const px::CAlarmTrigger *ib_trigger = nullptr;
210  const px::CAlarmTrigger *cov_trigger = nullptr;
211  uint32_t bound_highest_level = 0;
212  uint32_t bound_cur_level = alarm_state.bound_level;
213  uint32_t ib_highest_level = 0;
214  uint32_t ib_cur_level = alarm_state.between_level;
215 
216  for ( const px::CAlarmTrigger &at : cur_chan->alarm_settings )
217  {
218  if ( unlikely( at.trigger_type == px::alarm_trigger_type::invalid ) )
219  {
220  continue;
221  }
222  if ( at.trigger_type == px::alarm_trigger_type::on_value )
223  {
224  if ( ( onval_trigger == nullptr ) &&// There can be only one tripped at a time
225  ( trigger_tripped( at, cur_val, 0 ) ) )
226  {
227  onval_trigger = &at;
228  }
229  }
230  else if ( at.trigger_type == px::alarm_trigger_type::change_of_value )
231  {
232  if ( ( cov_trigger == nullptr ) &&
233  ( trigger_tripped( at, cur_val, 0, cds.open_alarms_store.get_ch_val( channel_id ) ) ) )
234  {
235  cov_trigger = &at;
236  }
237  // Now update ch_val with current channel value after checking trigger condition
238  cds.open_alarms_store.update_ch_val( channel_id, cur_val );
239  }
240  else if ( at.trigger_type == px::alarm_trigger_type::in_between )
241  {
242  if ( trigger_tripped( at, cur_val, ib_cur_level ) )
243  {
244  if ( at.level > ib_highest_level )
245  {
246  ib_highest_level = at.level;
247  ib_trigger = &at;
248  }
249  }
250  }
251  else if ( ( alarm_state.bound_trigger ) &&
252  ( alarm_state.bound_trigger->trigger_type != at.trigger_type ) )
253  {
254  // Hysteresis never applies when the bounds are different
255  if ( trigger_tripped( at, cur_val, 0 ) )
256  {
257  if ( at.level > bound_highest_level )
258  {
259  bound_highest_level = at.level;
260  bound_trigger = &at;
261  }
262  }
263  }
264  else
265  {
266  if ( trigger_tripped( at, cur_val, bound_cur_level ) )
267  {
268  if ( at.level > bound_highest_level )
269  {
270  bound_highest_level = at.level;
271  bound_trigger = &at;
272  }
273  }
274  }
275  }
276 
277  px::CDevice *dev = nullptr;
278  if ( likely( cds.devices.count( cur_chan->parent() ) != 0 ) )
279  {
280  dev = &( cds.devices[cur_chan->parent()] );
281  }
282  // On-value: Post an update if we went from something to nothing,
283  // nothing to something, or something to something else
284  if ( ( ( onval_trigger != nullptr ) != ( alarm_state.value_trigger.is_initialized() ) ) ||
285  ( alarm_state.value_trigger && ( onval_trigger != nullptr ) &&
286  ( *onval_trigger != *( alarm_state.value_trigger ) ) ) )
287  {
288  upd_op->updates.push_back( make_update( px::alarm_trigger_type::on_value, onval_trigger, cur_chan, dev, alarm_state, cds ) );
289  }
290  // Change_of_value: Post an update if the current channel value is not equal to the last channel value
291  if ( ( ( cov_trigger != nullptr ) != ( alarm_state.cov_trigger.is_initialized() ) ) ||
292  ( alarm_state.cov_trigger && ( cov_trigger != nullptr ) &&
293  ( *cov_trigger != *( alarm_state.cov_trigger ) ) ) )
294  {
295  upd_op->updates.push_back( make_update( px::alarm_trigger_type::change_of_value, cov_trigger, cur_chan, dev, alarm_state, cds ) );
296  }
297  // UB/LB: Post an update if we went from something to nothing, nothing to
298  // something, something to something else of higher level, or something to
299  // something at the other bound.
300  if ( ( ( bound_highest_level == 0 ) != ( bound_cur_level == 0 ) ) ||
301  ( alarm_state.bound_trigger && ( bound_trigger != nullptr ) &&
302  ( ( bound_highest_level > bound_cur_level ) || ( bound_trigger->trigger_type != alarm_state.bound_trigger->trigger_type ) ) ) )
303  {
304  upd_op->updates.push_back( make_update( px::alarm_trigger_type::upper_bound, bound_trigger, cur_chan, dev, alarm_state, cds ) );
305  }
306  // In-between: As UB/LB
307  if ( ( ( ib_highest_level == 0 ) != ( ib_cur_level == 0 ) ) ||
308  ( alarm_state.between_trigger && ( ib_trigger != nullptr ) &&
309  ( ib_highest_level > ib_cur_level ) ) )
310  {
311  upd_op->updates.push_back( make_update( px::alarm_trigger_type::in_between, ib_trigger, cur_chan, dev, alarm_state, cds ) );
312  }
313  // Submit operation for any alarm updates we generated
314  if ( upd_op->updates.size() != 0 )
315  {
316  CDS_DEBUG( 5, "Posting %zu alarm updates for channel %ju",
317  upd_op->updates.size(), channel_id );
318  ops.insert( upd_op );
319  }
320  else
321  {
322  delete upd_op;
323  }
324 }
