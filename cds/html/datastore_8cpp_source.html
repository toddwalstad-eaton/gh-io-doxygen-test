267 // Attempt DB repair at given stage
268 static void db_repair( const char *name, const px::string &path,
269  const char *exception_err, int stage )
270 {
271  if ( stage == 0 )
272  {
273  // Repair and retry
274  CDS_ERR( 0, "%s database corrupt, attempting repair. Data may be lost. DB error: %s", name, exception_err );
275  px::db::CKeyValueStore::repair( path );
276  }
277  else if ( stage == 1 )
278  {
279  // Repair failed, try clearing
280  CDS_ERR( 0, "%s database still corrupt after repair. Clearing all %s data and retrying. DB error: %s", name, name, exception_err );
281  boost::filesystem::remove_all( px::to_basicstr( path ) );
282  }
283  else if ( stage == 2 )
284  {
285  CDS_ERR( 0, "%s database corrupt, repair and clearing didn't work. Giving up. DB error: %s", name, exception_err );
286  throw px::generic_err();
287  }

295  // Transferred to open_alarms_store in setup stage

300  auth_mgr.reset( new CAuthManager( options->db_path, "/etc/passwd", perm_conf_dir, *this ) );
301 
302  // How to set aDataStoreCorrupt when finished loading
303  bool some_db_corrupt = auth_mgr->DBCorruptionDetected();
304 
305  // Set auth mgr's hashing strength per our command-line options.
306  auth_mgr->SetHashStrength( options->password_hash_strength );
307  CDS_DEBUG( 5, "Auth DB load finished" );
308 
309  // Open the MCL, if this fails new will throw
310  px::string mcl_path = options->mcl_db_path;
311 
312  try
313  {
314  mcl_db = new px::db::CKeyValueStore( mcl_path, false, true );
315  }
316  catch ( const px::generic_err &e )
317  {
318  CDS_FERR( 0, "Could not open MCL database at %s", mcl_path.c_str() );
319  throw px::generic_err();
320  }
321 
322  px::db::CIterator new_mcl_iter = mcl_db->iterator( false );
323  mcl_iter.reset( new px::db::CIterator( std::move( new_mcl_iter ) ) );
324  CDS_DEBUG( 5, "MCL DB open finished" );
325 
326  // Open OOB database, initialize internal copy
327  px::string oob_path = options->db_path + "/" + px::db::oob_db_name;
328  px::db::CKeyValueStore *oob_db = nullptr;
329  int stage = 0;
330  do
331  {
332  if ( oob_db != nullptr )
333  {
334  delete oob_db;
335  oob_db = nullptr;
336  }
337  try
338  {
339  CDS_DEBUG( 8, "OOB DB open, stage %d", stage );
340  oob_db = new px::db::CKeyValueStore( oob_path, true, false );
341  if ( oob_db != nullptr )
342  {
343  px::db::MigrateOOBDB( *oob_db );
344  oob_lock.lock();
345  px::db::FetchOOBs( *oob_db, oob );
346  oob_lock.unlock();
347  stage = 3;
348  }
349  }
350  catch ( const px::generic_err &e ) // also catches db_corrupt
351  {
352  if ( oob_db != nullptr )
353  {
354  delete oob_db;
355  oob_db = nullptr;
356  }
357  some_db_corrupt = true;
358  db_repair( "OOB", oob_path, e.what(), stage );
359  stage++;
360  }
361  }
362  while ( stage < 3 );
363 
364  // Open device database, initialize internal copy
365  // The "internal" databases (oob, device, channel) will be passed to
366  // CThreadPersistence and managed from there.
367  px::string devices_path = options->db_path + "/" + px::db::device_db_name;
368  px::db::CKeyValueStore *device_db = nullptr;
369 
370  stage = 0;
371  do
372  {
373  if ( device_db != nullptr )
374  {
375  delete device_db;
376  device_db = nullptr;
377  }
378  try
379  {
380  CDS_DEBUG( 8, "Device DB open, stage %d", stage );
381  device_db = new px::db::CKeyValueStore( devices_path, true, false );
382  if ( device_db != nullptr )
383  {
384  px::db::MigrateDeviceDB( *device_db );
385  px::db::FetchDevices( *device_db, devices );
386  stage = 3;
387  }
388  }
389  catch ( const px::generic_err &e )
390  {
391  if ( device_db != nullptr )
392  {
393  delete device_db;
394  device_db = nullptr;
395  }
396  some_db_corrupt = true;
397  db_repair( "Devices", devices_path, e.what(), stage );
398  stage++;
399  }
400  }
401  while ( stage < 3 );
402 
403  // Find and cache the publisher's ID
404  for ( std::map<px::id_t, px::CDevice>::const_iterator it = devices.begin();
405  it != devices.end(); it++ )
406  {
407  if ( it->second.role == px::device_role::publisher )
408  {
409  publisher_id = it->second.id();
410  if ( unlikely( publisher_id != 1 ) )
411  {
412  CDS_WARN( 0, "Publisher device ID is %ju, not 1. Problems may result.", publisher_id );
413  }
414  break;
415  }
416  }
417 
418  // Open channel database, initialize internal copy
419  px::string channels_path = options->db_path + "/" + px::db::channel_db_name;
420  px::db::CKeyValueStore *channel_db = nullptr;
421 
422  stage = 0;
423  do
424  {
425  if ( channel_db != nullptr )
426  {
427  delete channel_db;
428  channel_db = nullptr;
429  }
430  try
431  {
432  CDS_DEBUG( 8, "Channel DB open, stage %d", stage );
433  channel_db = new px::db::CKeyValueStore( channels_path, true, false );
434  px::db::MigrateChannelDB( *channel_db, *mcl_iter );
435  px::db::FetchChannels( *channel_db, *mcl_iter, channels, devices );
436  stage = 3;
437  }
438  catch ( const px::generic_err &e )
439  {
440  if ( channel_db != nullptr )
441  {
442  delete channel_db;
443  channel_db = nullptr;
444  }
445  some_db_corrupt = true;
446  db_repair( "Channels", channels_path, e.what(), stage );
447  stage++;
448  }
449  }
450  while ( stage < 3 );
451 
452  // Open alarm and alarm index DBs. If alarm DB corrupt, try to recover.
453  // If index DB corrupt, delete. Schedule a rebuild once that is implemented.
454  alarm_db = nullptr;
455  alarm_index_db = nullptr;
456  px::string alarm_path = options->db_path + "/" + px::db::alarm_db_name;
457  px::string alarm_index_path = options->db_path + "/" +
458  px::db::alarm_index_db_name;
459  stage = -1; // First try only deleting index DB
460  do
461  {
462  try
463  {
464  CDS_DEBUG( 8, "Alarm & index DB open, stage %d", stage );
465  alarm_db = new px::db::CKeyValueStore( alarm_path, true, false );
466  alarm_index_db = new px::db::CKeyValueStore( alarm_index_path, true, false );
467  if ( ( alarm_db != nullptr ) && ( alarm_index_db != nullptr ) )
468  {
469  px::db::MigrateAlarmDB( *alarm_db, channels, devices );
470  px::db::MigrateAlarmIndexDB( *alarm_index_db );
471  // Finish any in-progress alarm deletions that got interrupted
472  std::unordered_set<px::id_t> empty_list;
473  px::db::DeleteAlarms( *alarm_db, *alarm_index_db, empty_list, exit_bg_tasks );
474  // initialize active alarm list
475  CDS_DEBUG( 8, "Loading active alarms" );
476  px::db::LoadActiveAlarms( *alarm_db, *alarm_index_db, alarms_from_db );
477  stage = 3;
478  }
479  }
480  catch ( const px::generic_err &e )
481  {
482  if ( alarm_db != nullptr )
483  {
484  delete alarm_db;
485  alarm_db = nullptr;
486  }
487  if ( alarm_index_db != nullptr )
488  {
489  delete alarm_index_db;
490  alarm_index_db = nullptr;
491  }
492  boost::filesystem::remove_all( px::to_basicstr( alarm_index_path ) );
493  if ( stage >= 0 )
494  {
495  some_db_corrupt = true;
496  db_repair( "Alarms", alarm_path, e.what(), stage );
497  }
498  stage++;
499  }
500  }
501  while ( stage < 3 );
502 
503  // Open ID-tracking DB. Double-check against other DBs.
504  // If corrupt, just regenerate from other DBs.
505  px::string id_path = options->db_path + "/" + px::db::id_db_name;
506  id_db = nullptr;
507  stage = 0;
508  do
509  {
510  try
511  {
512  CDS_DEBUG( 8, "ID DB open, stage %d", stage );
513  id_db = new px::db::CKeyValueStore( id_path, true, false );
514  if ( id_db != nullptr )
515  {
516  px::db::MigrateIDDB( *id_db );
517  px::db::CheckIDDB( *id_db, *device_db, *channel_db, *alarm_db );
518  stage = 2;
519  }
520  }
521  catch ( const px::generic_err &e )
522  {
523  if ( stage == 0 )
524  {
525  if ( id_db != nullptr )
526  {
527  delete id_db;
528  id_db = nullptr;
529  }
530  boost::filesystem::remove_all( px::to_basicstr( id_path ) );
531  stage = 1;
532  CDS_INFO( 1, "ID-tracking database corrupt, rebuilding" );
533  continue;
534  }
535  else
536  {
537  CDS_ERR( 0, "Cannot initialize ID-tracking DB, cannot continue" );
538  throw;
539  }
540  }
541  }
542  while ( stage < 2 );
543  // Allocate a hunk of global counts and alarm IDs.
544  std::vector<px::id_t> new_aid = px::db::AllocateAlarmIDs( *id_db,
545  INIT_ALARM_ID_ALLOCATION );
546  if ( likely( new_aid.size() == INIT_ALARM_ID_ALLOCATION ) )
547  {
548  last_used_alarm_id = new_aid[0];
549  last_allocated_alarm_id = new_aid.back();
550  }
551  else
552  {
553  delete oob_db;
554  oob_db = nullptr;
555  delete device_db;
556  device_db = nullptr;
557  delete channel_db;
558  channel_db = nullptr;
559  throw px::generic_err( "Could not allocate initial alarm IDs" );
560  }
561 
562  std::vector<uint64_t> new_gc = px::db::AllocateGlobalCounts( *id_db,
563  INIT_GLOBAL_COUNT_ALLOCATION );
564  if ( likely( new_gc.size() == INIT_GLOBAL_COUNT_ALLOCATION ) )
565  {
566  last_used_global_count = new_gc[0];
567  last_allocated_global_count = new_gc.back();
568  }
569  else
570  {
571  delete oob_db;
572  oob_db = nullptr;
573  delete device_db;
574  device_db = nullptr;
575  delete channel_db;
576  channel_db = nullptr;
577  throw px::generic_err( "Could not allocate initial alarm global counts" );
578  }
579 
580  // initialize trend db
581  CDS_DEBUG( 8, "Opening trend store" );
582  trend_store.reset( new CTrendStore( options->db_path, this ) );
583 
584  // Start the persistence thread
585  persistence_thread.reset( new CThreadPersistence( exit_bg_tasks, device_db, channel_db, oob_db, alarm_db, alarm_index_db, mcl_db, options->db_path ) );
586 
587  // Start alarms thread
588  alarm_thread.reset( new CThreadAlarm( exit_bg_tasks, alarm_db, alarm_index_db ) );
589  delete_alarms_thread.reset( new CThreadBase( exit_bg_tasks, "DeleteAlarm", boost::none, 1024 ) );
590 
591  // Check device DB to see if it's empty. If so, load in the publisher
592  // definition, so the tree has some root.
593  if ( unlikely( publisher_id == UINT64_MAX ) )
594  {
595  CDS_INFO( 0, "Creating publisher device from publisher.json..." );
596  std::shared_ptr<px::proto::Device> pub_proto;
597  std::vector<std::shared_ptr<google::protobuf::Message> > protos;
598  try
599  {
600  px::string pub_filename = options->config_path + "/publisher.json";
601  px::string pub_json = px::CUtil::read_file_to_string( pub_filename );
602  px::string conv;
603  if ( pub_json.empty() )
604  {
605  CDS_WARN( 0, "JSON parse error creating publisher device: %s (failed to read file contents of %s) ",
606  conv.c_str(), pub_filename.c_str() );
607  throw px::generic_err();
608  }
609  conv = px::JSON2Protobuf( pub_json, protos );
610  if ( conv != "" )
611  {
612  CDS_WARN( 0, "JSON parse error creating publisher device: %s",
613  conv.c_str() );
614  throw px::generic_err();
615  }
616  if ( ( protos.size() != 1 ) ||
617  ( ( pub_proto = std::dynamic_pointer_cast<px::proto::Device>
618  ( protos[0] ) ) == nullptr ) )
619  {
620  CDS_WARN( 0, "Publisher creation failed parsing protobuf.json" );
621  throw px::generic_err();
622  }
623  }
624  catch ( ... )
625  {
626  CDS_WARN( 0, "Could not create publisher device from publisher.json - creating generic publisher device at root" );
627  pub_proto = std::shared_ptr<px::proto::Device>( new px::proto::Device() );
628  pub_proto->set_role( "publisher" );
629  pub_proto->set_model( "PX-Toolkit" );
630  pub_proto->set_name( "PowerXpert Toolkit Device" );
631  }
632  pub_proto->set_id( 1 );
633  px::CDevice pub( *pub_proto );
634  std::shared_ptr<px::cds::CCreateDevice> msg( new px::cds::CCreateDevice() );
635  pub.role = px::device_role::publisher;
636  msg->devices.push_back( pub );
637  COpList ops;
638  COpCreateDevices op_create;
639  op_create.execute( msg, ops, *this );
640  if ( msg->devices.size() == 1 )
641  {
642  publisher_id = msg->devices[0].id();
643  }
644  else
645  {
646  CDS_FERR( 0, "Publisher device creation failed" );
647  throw px::generic_err();
648  }
649  } // end create publisher
650 
651  // Devices' "parent" members are not persisted since they're not part
652  // of Mercury, and are calculatable. Calculate them.
653  // Also propagate inheritable state downward.
654  for ( auto &dev : devices )
655  {
656  // For non-"port"/"publisher" devices, set not_communicating
657  if ( ( dev.second.role != px::device_role::publisher ) &&
