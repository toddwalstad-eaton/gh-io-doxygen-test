74 CThreadBaseQueue::cmdlist_t CThreadBaseQueue::remove_impl( const px::CTimestamp &timeout,
75  queue_lock_t *plock )
76 {
77  PUSH_FUNC;
78 
79  boost::system_time tout;
80  if ( timeout )
81  {
82  tout = boost::get_system_time() + boost::posix_time::seconds( timeout.tv_sec ) +
83  boost::posix_time::microsec( timeout.tv_nsec / 1000 );
84  }
85 
86  queue_lock_t lock( _mutex, boost::defer_lock_t() );
87 
88  if ( plock == NULL )
89  {
90  plock = &lock;
91 
92  // Grab the lock
93  lock.lock();
94  }
95 
96  assert( plock != NULL );
97  assert( plock->owns_lock() );
98 
99  while ( _queue.empty() )
100  {
101  if ( timeout )
102  {
103  if ( !_cond.timed_wait( *plock, tout ) )
104  {
105  throw timed_out();
106  }
107  }
108  else
109  {
110  _cond.wait( *plock );
111  }
112  }
113 
114  bool was_full = full_queue();
115 
116  cmdlist_t retval;
117  retval.swap( _queue );
118 
119  if ( unlikely( was_full ) )
120  {
121  _cond.notify_one();
122  }
123 
124  return ( retval );
125 }
