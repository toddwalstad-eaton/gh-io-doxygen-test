416 // Gets the greatest verbosity setting from the various verbosity options.
417 unsigned int COptions::get_max_verbosity()
418 {
419  unsigned int max_verbosity = verbosity;
420 
421  if ( main_verbosity )
422  {
423  max_verbosity = std::max( max_verbosity, *( main_verbosity ) );
424  }
425 
426  if ( subsys_verbosity[CDS_LOG_DEFAULT] )
427  {
428  max_verbosity = std::max( max_verbosity, *( subsys_verbosity[CDS_LOG_DEFAULT] ) );
429  }
430  return ( max_verbosity );
431 }
