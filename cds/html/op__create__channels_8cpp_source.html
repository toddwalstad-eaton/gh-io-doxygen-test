18 void COpCreateChannels::execute( std::shared_ptr<px::cds::CRequestResponse> msg,
19  COpList &ops, CDataStore &cds )
20 {
21  if ( typeid( *( msg.get() ) ) != typeid( px::cds::CCreateChannel ) )
22  {
23  throw px::generic_err( "Invalid API message type given to COpCreateChannels" );
24  }
25  px::cds::CCreateChannel *in = px::poly_cast<px::cds::CCreateChannel *>( msg.get() );
26  if ( unlikely( in == nullptr ) )
27  {
28  return;
29  }
30  // Remove the vector of assigned IDs (courtesy of COpAssignIDs) from the API message, use it as we go
31  // Swap instead of copying for efficiency
32  std::vector<px::CChannel> id_list;
33  id_list.swap( in->created_channels ); // in->created_channels is now empty
34 
35  std::vector<px::cds::CCreateSingleChannel>::const_iterator cur_req;
36  std::unique_ptr<CThreadPersistThis> persist_cmd;
37  persist_cmd.reset( new CThreadPersistThis );
38  msg->status = px::cds::RequestStatus::SUCCEEDED;
39  try
40  {
41  for ( cur_req = in->requests.begin(); cur_req != in->requests.end(); cur_req++ )
42  {
43  // Validation. Skip this request if anything is wrong.
44  // Parent device must exist
45  if ( cds.devices.count( cur_req->parent ) == 0 )
46  {
47  CDS_WARN( 1, "No such parent Device %" PRIu64 "; skipping", cur_req->parent );
48  continue;
49  }
50  if ( cur_req->mcl_name.size() < 2 )
51  {
52  // Can't proceed without some name for this Channel
53  CDS_WARN( 1, "No MCL name given for Channel; skipping" );
54  continue;
55  }
56  px::CDevice *this_parent = &( cds.devices[cur_req->parent] );
57  // Passed-in channel name must not already exist in this device's
58  // channels
59  bool found_dupe = false;
60  px::CDevice::chlist_t::const_iterator ch;
61  for ( ch = this_parent->channels.begin();
62  ch != this_parent->channels.end(); ch++ )
63  {
64  if ( cds.channels.count( *ch ) != 0 )
65  {
66  if ( cds.channels[*ch].channel_settings->canonical_name == cur_req->mcl_name )
67  {
68  found_dupe = true;
69  break;
70  }
71  }
72  }
73  if ( found_dupe )
74  {
75  CDS_WARN( 2, "Attempted to create duplicate Channel %s; skipping", cur_req->mcl_name.c_str() );
76  continue;
77  }
78 
79  /* Now for the Channel ID: two possibilities:
80  * If use_given_ids is set, then id_list[] contains the requested ID
81  * Else id_list[] contains an arbitrary list of IDs, we will just take the last assigned
82  *
83  * There is one sticky point for use_given_ids == true: the id_list[] may be shorter
84  * than the requested channels, if no ID was provided or the Channel already existed.
85  */
86  px::id_t new_id = cur_req->id;
87  if ( in->use_given_ids )
88  {
89  auto found = id_list.begin();
90  for (; found != id_list.end(); ++found )
91  {
92  if ( found->id() == new_id )
93  {
94  break;
95  }
96  }
97  if ( found == id_list.end() )
98  {
99  CDS_WARN( 1, "No ID or pre-existing Channel %s; skipping", cur_req->mcl_name.c_str() );
100  continue;
101  }
102  // else, new_id holds the correct value
103  }
104  else
105  {
106  new_id = id_list.back().id();
107  id_list.pop_back();
108  }
109  if ( unlikely( cds.channels.count( new_id ) != 0 ) )
110  {
111  throw px::generic_err( "Attempt to create channel with already-existing ID, somehow" );
112  }
113  px::CChannel new_chan;
114  // If in_mcl is false, no need to look anything up; just create with
115  // no settings, let client set them
116  if ( cur_req->in_mcl == false )
117  {
118  new_chan.set_ids( new_id, this_parent->id() );
119  new_chan.channel_settings->canonical_name = cur_req->mcl_name;
120  new_chan.channel_settings->flags.set( px::channel_flag::is_custom, true );
121  }
122  else
123  {
124  new_chan = px::CChannel( px::db::FetchMCLEntry( *( cds.mcl_iter ), cur_req->mcl_name ) );
125  if ( new_chan.channel_settings->canonical_name != cur_req->mcl_name )
126  {
127  // Not found in databases, skip
128  CDS_WARN( 1, "Did not find Channel %s in MCL; skipping", cur_req->mcl_name.c_str() );
129  continue;
130  }
131  new_chan.set_ids( new_id, this_parent->id() );
132  }
133  in->created_channels.push_back( new_chan );
134  // Add to master list of channels
135  std::pair<px::id_t, px::CChannel> ch_pair( new_id, new_chan );
136  cds.channels.insert( ch_pair );
137  // Add to parent's list of channels
138  this_parent->channels.insert( new_id );
139  // Add to parent's list of channel names
140  this_parent->channel_names.emplace( new_chan.channel_settings->canonical_name, new_id );
141  // Add to list of channels to save to persistent storage
142  persist_cmd->channels.insert( ch_pair );
143  // Save parent device as well
144  persist_cmd->devices[this_parent->id()] = *this_parent;
145 
146  // If parent device has recursive subscriptions, subscribe
147  // this channel as well
148  for ( auto stp : cds.subs.list_channel_recursive_subscription_types() )
149  {
150  std::vector<uint64_t> _subs;
151  px::sub::CItemId event( this_parent->id(), stp );
152  cds.subs.get_subscribers( event, _subs );
153  event._id = new_id;
154  for ( auto &listener: _subs )
155  {
156  cds.subs.add_subscription( listener, event );
157  }
158  }
159  // The parent device's channel list has changed; notify its
160  // subscribers
161  std::vector<uint64_t> parent_subs;
162  px::sub::CItemId parent_event( this_parent->id(), px::sub::subscription_type::device_settings );
163  cds.subs.get_subscribers( parent_event, parent_subs );
164  for ( auto &listener: parent_subs )
165  {
166  ops.insert( new COpEventNotify( listener, parent_event, *this_parent ) );
167  }
168 
169  // notify subscribers of settings change
170  // unlikely that anyone's already subscribed to this channel
171  // (how did they get the ID?) but it's cheap to cover these cases
172  for ( auto i : cds.subs.list_channel_direct_subscription_types() )
173  {
174  std::vector<uint64_t> _subs;
175  px::sub::CItemId event( new_id, i );
176  cds.subs.get_subscribers( event, _subs );
177  for ( auto &listener: _subs )
178  {
179  ops.insert( new COpEventNotify( listener, event, new_chan ) );
180  }
181  }
182  } // End iteration over every request
183  }
184  catch ( const px::generic_err &e )
185  {
186  msg->status = px::cds::RequestStatus::FAILED;
187  CDS_ERR( 0, "Error in COpCreateChannels: %s", e.what() );
188  }
189  catch ( const std::bad_alloc &e )
190  {
191  msg->status = px::cds::RequestStatus::FAILED;
192  CDS_ERR( 0, "Out of memory in COpCreateChannels" );
193  }
194  catch ( ... )
195  {
196  msg->status = px::cds::RequestStatus::FAILED;
197  CDS_ERR( 0, "Unknown error in COpCreateChannels" );
198  }
199  // Return response to client
200  if ( !( px::cds::CServer::get()->send_response( msg ) ) )
201  {
202  CDS_ERR( 0, "Lost response to Create Channel to a full queue" );
203  }
204 
205  // Enqueue op to update parent devices' change signatures
206  COpBumpChangeSignature *bumper = new COpBumpChangeSignature();
207  for ( auto &this_chan : in->created_channels )
208  {
209  if ( bumper->devs_to_bump.count( this_chan.parent() ) == 0 )
210  {
211  bumper->devs_to_bump.insert( this_chan.parent() );
212  }
213  }
214  ops.insert( bumper );
215 
216  // Tell persistence thread to save the successfully-created channels
217  if ( likely( cds.persistence_thread ) )
218  {
219  cds.persistence_thread->send( std::move( persist_cmd ) );
220  }
221  else
222  {
223  CDS_ERR( 0, "Attempt to save created channels but persistence thread not running" );
224  }
225 }
