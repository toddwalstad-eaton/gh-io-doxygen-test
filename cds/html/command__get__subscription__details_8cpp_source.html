65 void CCommandGetSubscriptionDetails::execute( COpList & /*ops*/, CDataStore &cds )
66 {
67  std::shared_ptr<px::cds::CGetSubscriptionDetails> cds_req =
68  std::dynamic_pointer_cast<px::cds::CGetSubscriptionDetails>( req );
69 
70  if ( unlikely( req->callback_handle == CALLBACK_NONE ) )
71  {
72  CDS_WARN( 0, "GetSubscriptionDetails called with no callback, response will be lost" );
73  }
74 
75  cds_req->status = px::cds::RequestStatus::SUCCEEDED;
76 
77  CDS_DEBUG( 3, "Execute GetSubscriptionDetails, client %u, subscription id %" PRIu64 ".",
78  cds_req->client_id, cds_req->subscription_id );
79 
80  // Check if ID given belongs to this client. If not, check for
81  // 'troubleshoot' privilege.
82  bool mine = false;
83  std::shared_ptr<px::cds::CServerConnection> conn;
84  px::cds::CServer::get()->conn_list_lock.lock();
85  auto it = px::cds::CServer::get()->connections.find( cds_req->client_id );
86  if ( it != px::cds::CServer::get()->connections.end() )
87  {
88  conn = it->second;
89  }
90  px::cds::CServer::get()->conn_list_lock.unlock();
91  if ( conn )
92  {
93  for ( uint64_t sid : conn->subscriber_ids )
94  {
95  if ( sid == cds_req->subscription_id )
96  {
97  mine = true;
98  break;
99  }
100  }
101  }
102  if ( ( !mine ) && ( !check_permission( cds, px::permissions::troubleshoot ) ) )
103  {
104  return; // check_permission already responded 'permission denied'
105  }
106 
107  const auto &subscriber_info = cds.subs.get_subscriber_info( cds_req->subscription_id );
108  if ( subscriber_info.is_initialized() )
109  {
110  auto &items = subscriber_info.get().get_items();
111  if ( items.size() > 0 )
112  {
113  //auto hash_subscription_type = [](const px::sub::subscription_type& subtype) { return subtype.
114  std::set<px::sub::subscription_type> sub_types;
115  std::vector<px::id_t> invalid_ids;
116  std::vector<px::string> tags;
117  for ( auto &item : items )
118  {
119  sub_types.insert( item._type );
120  // channel or device
121  auto id = item._id;
122  if ( id != PX_ID_UNSET )
123  {
124  if ( cds.channels.find( id ) != cds.channels.end() )
125  {
126  cds_req->channel_ids.push_back( id );
127  }
128  else if ( cds.devices.find( id ) != cds.devices.end() )
129  {
130  cds_req->device_ids.push_back( id );
131  }
132  else
133  {
134  invalid_ids.push_back( id );
135  }
136  }
137  else if ( !item._tag.empty() )
138  {
139  tags.push_back( item._tag );
140  }
141  }
142  // the performance is not a concern in this case,
143  // because this method is only used for diagnostic and won't be called too often.
144  std::stringstream s;
145  if ( sub_types.size() > 0 )
146  {
147  s << "type: ";
148  for ( auto sub_type : sub_types )
149  {
150  s << get_subscription_type( sub_type ) << " ";
151  }
152  }
153  if ( invalid_ids.size() > 0 )
154  {
155  s << "invalid device/channel id: ";
156  for ( auto invalid_id : invalid_ids )
157  {
158  s << std::to_string( invalid_id ) << " ";
159  }
160  }
161  if ( tags.size() > 0 )
162  {
163  s << "tags: ";
164  for ( auto tag : tags )
165  {
166  s << tag << " ";
167  }
168  }
169  auto str = s.str();
170  if ( str.size() > 0 )
171  {
172  cds_req->detail = px::to_pxstr( str );
173  }
174  }
175  }
176  respond();
177 }
