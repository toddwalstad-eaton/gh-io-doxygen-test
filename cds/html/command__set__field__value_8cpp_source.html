586  }
587  }

590  CDS_ERR( 3, "OOB SetFieldValue() failed with error: %s", gx.what() );
591  req->status = px::cds::BAD_PARAMETER;
592  AUDIT_LOG_MSG( AUDIT_LOG_CONFIG, LOG_INFO, cee_info,
593  "SetFieldValue() Failed to update OOB '%s' due to invalid argument", fulltag.c_str() );
594  }
595  }
596  else
597  {
598  // No such OOB
599  CDS_ERR( 3, "OOB SetFieldValue() failed to find OOB %s", fulltag.c_str() );
600  AUDIT_LOG_MSG( AUDIT_LOG_CONFIG, LOG_INFO, cee_info,
601  "SetFieldValue() Failed to find OOB '%s'", fulltag.c_str() );
602  req->status = px::cds::BAD_PARAMETER;
603  }
604  }
605  break;
606  case px::cds::field_message_type::command:
607  {
608  auto cmd = px::CDeviceCommand( cur->id, cur->field );
609  // Change the CEE Action for commands
610  cee_info._cee_action = CEE_ACT_execute;
611 
612  // Make sure we have a proper device ID
613  auto it = cds.devices.find( cmd.device_id );
614  if ( it == cds.devices.end() )
615  {
616  cur->value = px::CValue();
617  req->status = px::cds::BAD_PARAMETER;
618  AUDIT_LOG_MSG( AUDIT_LOG_COMMAND, LOG_NOTICE, cee_info,
619  "Requesting Command [%s] from Invalid Device [%ju]",
620  !cmd.name.empty() ? cmd.name.c_str() : "unknown", cmd.device_id );
621  break;
622  }
623 
624  // Make sure the device has commands
625  auto &cmds = it->second.device_commands;
626  if ( cmds.size() <= 0 )
627  {
628  set_req->status = req_status::FAILED;
629  cur->value = px::CValue();
630  req->status = px::cds::BAD_PARAMETER;
631  AUDIT_LOG_MSG( AUDIT_LOG_COMMAND, LOG_NOTICE, cee_info,
632  "Requesting Invalid/Unsupported Device Command:[%s]",
633  !cmd.name.empty() ? cmd.name.c_str() : "unknown" );
634  break;
635  }
636 
637  // Make sure we are asking a valid command
638  const auto &c_it = cmds.find( cmd.name );
639  if ( c_it == cmds.end() )
640  {
641  // the command is not supported
642  cur->value = px::CValue();
643  req->status = px::cds::BAD_PARAMETER;
644  AUDIT_LOG_MSG( AUDIT_LOG_COMMAND, LOG_NOTICE, cee_info,
645  "Requesting Invalid/Unsupported Device Command:[%s]",
646  !cmd.name.empty() ? cmd.name.c_str() : "unknown" );
647  break;
648  }
649 
650  if ( !px::permission::perms_overlap( perms, c_it->second.permissions ) )
651  {
652  AUDIT_LOG_MSG( AUDIT_LOG_COMMAND, LOG_WARNING, cee_info,
653  "Unauthorized set-field device command %s", cmd.name.c_str() );
654  req->status = px::cds::PERMISSION_DENIED;
655  continue;
656  }
657 
658  // issue the command
659  cmd.permissions = c_it->second.permissions;
660  cmd.status = px::device_command_status::in_progress;
661  cmd.handle = cds.device_commands_store->
662  add( std::make_shared<px::CDeviceCommand>( cmd ) );
663  cur->value = cmd.handle;
664  set_req->status = req_status::SUCCEEDED;
665 
666  // Change the CEE Status for this one success case
667  cee_info._cee_status = CEE_STS_success;
668  AUDIT_LOG_MSG( AUDIT_LOG_COMMAND, LOG_INFO, cee_info,
669  "Requesting Device Command: [%s]", cmd.name.c_str() );
670 
671  // notify subscribers to execute the command.
672  std::vector<uint64_t> subs;
673  px::sub::CItemId event( cmd.device_id,
674  px::sub::subscription_type::device_command );
675 
676  cds.subs.get_subscribers( event, subs );
677  if ( subs.size() > 0 )
678  {
679  for ( const auto &sub: subs )
680  {
681  ops.insert( new COpEventNotify( sub, event, cmd ) );
682  }
683  }
684  else
685  {
686  // nobody handles the execution
687  set_req->status = req_status::NOT_ABLE_TO_EXECUTE;
688  cur->value = px::CValue();
689  break;
690  }
691  }
692  break;
693  default:
694  break;
695  }
696  } // End loop over each request
697 
698  // Persist any Config changes that occurred
699  if ( likely( cds.persistence_thread ) )
700  {
701  if ( ( !persist_cmd->oob.empty() ) || ( !persist_cmd->channels.empty() ) ||
702  ( !persist_cmd->devices.empty() ) )
703  {
704  cds.persistence_thread->send( std::move( persist_cmd ) );
705  }
706  if ( !( unpersist_cmd->oob.empty() ) )
707  {
708  cds.persistence_thread->send( std::move( unpersist_cmd ) );
709  }
710  }
711  else
712  {
713  CDS_ERR( 0, "Attempt to save Config data but persistence thread not running" );
714  req->status = px::cds::RequestStatus::FAILED;
715  }
716 
717  // Trigger change signature bumps as needed
718  if ( change_signature_change_device_ids.size() > 0 )
719  {
720  auto *bumper = new COpBumpChangeSignature;
721  for ( const auto device_id : change_signature_change_device_ids )
722  {
723  bumper->devs_to_bump.insert( device_id );
724  }
725  ops.insert( bumper );
726  }
727 
728  // Only respond once we have sent any Config changes to be persisted.
729  // Doesn't guarantee persistence was successful.
730  respond();
731 }
