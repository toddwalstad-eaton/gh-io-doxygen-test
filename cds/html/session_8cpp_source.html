61 time_t CSession::GetTimeLeft() const
62 {
63  if ( !_timeout && !_hard_timeout )
64  {
65  return ( -1 );
66  }
67  time_t tnow_secs = px::CMonoTime::now().tv_sec;
68  time_t left_secs = 0; // assume expired, to start
69  if ( _timeout )
70  {
71  if ( _expiration.tv_sec > tnow_secs )
72  {
73  left_secs = _expiration.tv_sec - tnow_secs;
74  }
75  else
76  {
77  return ( 0 ); // already timed out
78  }
79  }
80 
81  if ( _hard_timeout )
82  {
83  if ( _hard_expiration.tv_sec > tnow_secs )
84  {
85  time_t thard_secs = _hard_expiration.tv_sec - tnow_secs;
86  // Take the min() of the two times
87  if ( ( left_secs == 0 ) || ( left_secs > thard_secs ) )
88  {
89  left_secs = thard_secs;
90  }
91  }
92  else
93  {
94  left_secs = 0;
95  }
96  }
97 
98  return ( left_secs );
99 }
