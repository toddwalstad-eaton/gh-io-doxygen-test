11  COpList &ops, CDataStore &cds )
12 {

14 
15  // Go through devs_to_bump, put together a list that includes all
16  // ancestors
17  for ( auto &this_dev : devs_to_bump )
18  {
19  px::id_t cur_id = this_dev;
20  while ( 1 )
21  {
22  if ( unlikely( cds.devices.count( cur_id ) == 0 ) )
23  {
24  break;
25  }
26  if ( full_bump_list.count( cur_id ) != 0 )
27  {
28  break; // Already have this one (and therefore all its
29  // ancestors)
30  }
31  CDS_DEBUG( 9, "Change sig bump: Will bump %ju", cur_id );
32  full_bump_list.insert( cur_id );
33  if ( !( cds.devices[cur_id].parent ) )
34  {
35  CDS_DEBUG( 9, "Change sig bump: %ju top of tree", cur_id );
36  break; // Hit top of tree
37  }
38  cur_id = *( cds.devices[cur_id].parent );
39  }
40  }
41 
42  // Now we have a de-duped list; do the bumping
43  // These are the same kind of events, hence merged together. Operation is executed only
44  // on the first device in the full_bump_list. The first one is random for unsorted_set,
45  // but we'd rather operate on the last inserted, the first ancestor.
46  // But that should always be 1, right?
47  std::unique_ptr<CThreadPersistThis> persist_cmd( new CThreadPersistThis );
48  for ( auto &this_dev : full_bump_list )
49  {
50  CDS_DEBUG( 9, "Change sig bump: Bumping %ju", this_dev );
51  ( cds.devices[this_dev].change_signature )++;
52 
53  std::vector<uint64_t> _subs;
54  px::sub::CItemId event( this_dev,
55  px::sub::subscription_type::change_signature );
56  cds.subs.get_subscribers( event, _subs );
57  for ( auto &listener: _subs )
58  {
59  ops.insert( new COpEventNotify( listener,
60  event,
61  cds.devices[this_dev].change_signature ) );
62  }
63 
64  persist_cmd->devices[this_dev] = cds.devices[this_dev];
65  }
66  if ( likely( cds.persistence_thread ) )
67  {
68  cds.persistence_thread->send( std::move( persist_cmd ) );
69  }
70  else
71  {
72  CDS_ERR( 0, "Attempt to save device change-signature updates but persistence thread not running" );
73  }
74 }
